14  /**
15   * This class enables easy running of a shared WComponent instance in a Jetty servlet container for development and
16   * testing purposes.
17   * <p>
18   * You need to set the class name of the WComponent you want to run. Do this by setting the parameter
19   * "bordertech.wcomponents.lde.component.to.launch" in your "local_app.properties" file. E.g.
20   *
21   * <pre>
22   * ui.web.component.to.launch = com.github.bordertech.wcomponents.examples.picker.ExamplePicker
23   * </pre>
24   *
25   * @author Martin Shevchenko
26   * @since 1.0.0
27   */
28  public class PlainLauncher extends TestServlet {
29  
30  	/**
31  	 * The logger instance for this class.
32  	 */
33  	private static final Log LOG = LogFactory.getLog(PlainLauncher.class);
34  
35  	/**
36  	 * The {@link Config configuration} property key for which component to launch.
37  	 */
38  	public static final String COMPONENT_TO_LAUNCH_PARAM_KEY = ConfigurationProperties.LDE_PLAINLAUNCHER_COMPONENT_TO_LAUNCH;
39  
40  	/**
41  	 * The {@link Config configuration} property key for whether to display the memory profile.
42  	 */
43  	protected static final String SHOW_MEMORY_PROFILE_PARAM_KEY = ConfigurationProperties.LDE_SHOW_MEMORY_PROFILE;
44  
45  	/**
46  	 * The singleton instance of the UI which is being run by the PlainLauncher.
47  	 */
48  	private WApplication sharedUI;
49  
50  	/**
51  	 * The fully qualified name of the WComponent class which is being served as the UI.
52  	 */
53  	private String uiClassName;
54  
55  	/**
56  	 * This method has been overridden to load a WComponent from parameters.
57  	 *
58  	 * @param httpServletRequest the servlet request being handled.
59  	 * @return the top-level WComponent for this servlet.
60  	 */
61  	@Override
62  	public synchronized WComponent getUI(final Object httpServletRequest) {
63  		String configuredUIClassName = getComponentToLaunchClassName();
64  
65  		if (sharedUI == null || !Util.equals(configuredUIClassName, uiClassName)) {
66  			uiClassName = configuredUIClassName;
67  			WComponent ui = createUI();
68  
69  			if (ui instanceof WApplication) {
70  				sharedUI = (WApplication) ui;
71  			} else {
72  				LOG.warn(
73  						"Top-level component should be a WApplication."
74  						+ " Creating WApplication wrapper...");
75  
76  				sharedUI = new WApplication();
77  				ui.setLocked(false);
78  				sharedUI.add(ui);
79  				sharedUI.setLocked(true);
80  			}
81  
82  			if (ConfigurationProperties.getLdeServerShowMemoryProfile()) {
83  				ProfileContainer profiler = new ProfileContainer();
84  
85  				sharedUI.setLocked(false);
86  				sharedUI.add(profiler);
87  				sharedUI.setLocked(true);
88  			}
89  		}
90  
91  		return sharedUI;
92  	}
93  
94  	/**
95  	 * Creates the UI which the launcher displays. If there is misconfiguration or error, a UI containing an error
96  	 * message is returned.
97  	 *
98  	 * @return the UI which the launcher displays.
99  	 */
100 	protected WComponent createUI() {
101 		// Check if the parameter COMPONENT_TO_LAUNCH_PARAM_KEY has been
102 		// configured with the name of a component to launch.
103 
104 		WComponent sharedApp;
105 
106 		uiClassName = getComponentToLaunchClassName();
107 
108 		if (uiClassName == null) {
109 			sharedApp = new WText(
110 					"You need to set the class name of the WComponent you want to run.<br />"
111 					+ "Do this by setting the parameter \""
112 					+ COMPONENT_TO_LAUNCH_PARAM_KEY
113 					+ "\" in your \"local_app.properties\" file.<br />"
114 					+ "Eg.  <code>" + COMPONENT_TO_LAUNCH_PARAM_KEY
115 					+ "=com.github.bordertech.wcomponents.examples.picker.ExamplePicker</code>");
116 
117 			((WText) sharedApp).setEncodeText(false);
118 		} else {
119 			UIRegistry registry = UIRegistry.getInstance();
120 			sharedApp = registry.getUI(uiClassName);
121 
122 			if (sharedApp == null) {
123 				sharedApp = new WText(
124 						"Unable to load the component \""
125 						+ uiClassName
126 						+ "\".<br />"
127 						+ "Either the component does not exist as a resource in the classpath,"
128 						+ " or is not a WComponent.<br />"
129 						+ "Check that the parameter \""
130 						+ COMPONENT_TO_LAUNCH_PARAM_KEY
131 						+ "\" is set correctly.");
132 
133 				((WText) sharedApp).setEncodeText(false);
134 			}
135 		}
136 
137 		return sharedApp;
138 	}
139 
140 	/**
141 	 * @return the class name of the component to launch.
142 	 */
143 	protected String getComponentToLaunchClassName() {
144 		return ConfigurationProperties.getLdePlainLauncherComponentToLaunch();
145 	}
146 
147 	/**
148 	 * The entry point when the launcher is run as a java application.
149 	 *
150 	 * @param args command-line arguments, ignored.
151 	 * @throws Exception on error
152 	 */
153 	public static void main(final String[] args) throws Exception {
154 		PlainLauncher launcher = new PlainLauncher();
155 		launcher.run();
156 	}
157 }
