55  	/**
56  	 * Creates a StandaloneLauncher.
57  	 */
58  	private StandaloneLauncher() {
59  		final JFrame frame = new JFrame("WComponent stand-alone LDE");
60  		frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
61  
62  		JPanel content = new JPanel();
63  		content.setLayout(new BorderLayout());
64  		frame.setContentPane(content);
65  
66  		content.add(new JLabel("Log:"), BorderLayout.NORTH);
67  
68  		log.setColumns(120);
69  		log.setRows(30);
70  		log.setFont(new Font("Monospaced", Font.PLAIN, 12));
71  		content.add(new JScrollPane(log, JScrollPane.VERTICAL_SCROLLBAR_ALWAYS,
72  				JScrollPane.HORIZONTAL_SCROLLBAR_AS_NEEDED), BorderLayout.CENTER);
73  
74  		JPanel buttonPanel = new JPanel();
75  		buttonPanel.setLayout(new FlowLayout(FlowLayout.LEFT, 5, 0));
76  		content.add(buttonPanel, BorderLayout.SOUTH);
77  
78  		JButton button = new JButton("Clear log");
79  		buttonPanel.add(button);
80  		button.addActionListener(new ActionListener() {
81  			@Override
82  			public void actionPerformed(final ActionEvent event) {
83  				log.setText("");
84  			}
85  		});
86  
87  		button = new JButton("Launch browser");
88  		buttonPanel.add(button);
89  		button.addActionListener(new ActionListener() {
90  			@Override
91  			public void actionPerformed(final ActionEvent event) {
92  				// Use of reflections here is required as we still need to compile under java 1.5.
93  				//Desktop.getDesktop().open(index);
94  				try {
95  					Class<?> clazz = Class.forName("java.awt.Desktop");
96  					Object instance = clazz.getMethod("getDesktop").invoke(null);
97  					clazz.getMethod("browse", URI.class).invoke(instance, new URI(
98  							getInstance().launcher.getUrl()));
99  				} catch (ClassNotFoundException e) {
100 					//Desktop isn't available in Java 1.5, so we'll just use a
101 					// windows specific cmdline for now
102 					String cmd = "rundll32 url.dll,FileProtocolHandler "
103 							+ getInstance().launcher.getUrl();
104 
105 					try {
106 						Runtime.getRuntime().exec(cmd);
107 					} catch (Exception e2) {
108 						LogFactory.getLog(getClass()).error("Failed to launch browser", e);
109 					}
110 				} catch (Exception e) {
111 					LogFactory.getLog(getClass()).error("Failed to launch browser", e);
112 				}
113 			}
114 		});
115 
116 		button = new JButton("Exit");
117 		buttonPanel.add(button);
118 		button.addActionListener(new ActionListener() {
119 			@Override
120 			public void actionPerformed(final ActionEvent event) {
121 				// Trigger closing of the frame rather than forcefully terminating the VM.
122 				// This allows clean-up routines registered in window listeners to run.
123 				WindowEvent closeEvent = new WindowEvent(frame, WindowEvent.WINDOW_CLOSING);
124 				Toolkit.getDefaultToolkit().getSystemEventQueue().postEvent(closeEvent);
125 			}
126 		});
127 
128 		frame.pack();
129 		frame.setVisible(true);
130 	}

139 	/**
140 	 * Logs a message to the console (text area).
141 	 *
142 	 * @param message the message to log.
143 	 */
144 	public void log(final String message) {
145 		if (!SwingUtilities.isEventDispatchThread()) {
146 			SwingUtilities.invokeLater(new Runnable() {
147 				@Override
148 				public void run() {
149 					log(message);
150 				}
151 			});
152 		} else {
153 			log.append(message);
154 		}
155 	}

157 	/**
158 	 * The entry point when the launcher is run as a java application.
159 	 *
160 	 * @param args command-line arguments, ignored.
161 	 * @throws Exception on error
162 	 */
163 	public static void main(final String[] args) throws Exception {
164 		// Set the logger to use the text area logger
165 		System.setProperty("org.apache.commons.logging.Log",
166 				"com.github.bordertech.wcomponents.lde.StandaloneLauncher$TextAreaLogger");
167 
168 		// Set the port number to a random port
169 		Configuration internalWComponentConfig = Config.getInstance();
170 		CompositeConfiguration config = new CompositeConfiguration(new MapConfiguration(
171 				new HashMap<String, Object>()));
172 		config.addConfiguration(internalWComponentConfig); // Internal WComponent config next
173 		config.setProperty(ConfigurationProperties.LDE_SERVER_PORT, 0);
174 		Config.setConfiguration(config);
175 
176 		getInstance().launcher.run();
177 		getInstance().log("LDE now running on " + getInstance().launcher.getUrl() + '\n');
178 	}

194 	/**
195 	 * A commons-logging Log implementation which logs to the StandaloneLauncher log.
196 	 */
197 	public static final class TextAreaLogger implements Log {
198 
199 		/**
200 		 * The TRACE log-level should be used to output even more fine-grained information than DEBUG.
201 		 */
202 		public static final int TRACE = 0;
203 		/**
204 		 * The DEBUG log-level should be used for fine-grained information useful for debugging an application.
205 		 */
206 		public static final int DEBUG = 1;
207 		/**
208 		 * The INFO log-level is used to output application processing at a course level.
209 		 */
210 		public static final int INFO = 2;
211 		/**
212 		 * The WARN log-level is used to log information about potentially damaging situations.
213 		 */
214 		public static final int WARN = 3;
215 		/**
216 		 * The ERROR log-level is used to log application errors where the application can recover.
217 		 */
218 		public static final int ERROR = 4;
219 		/**
220 		 * The FATAL log-level is used to log application errors where the application can not recover.
221 		 */
222 		public static final int FATAL = 5;
223 
224 		/**
225 		 * The current log level.
226 		 */
227 		private int logLevel = DEBUG;
228 
229 		/**
230 		 * Creates a TextAreaLogger.
231 		 *
232 		 * @param name the logger name.
233 		 */
234 		public TextAreaLogger(final String name) {
235 			// Warning, commons logging requires this (currently unused) constructor.
236 		}
237 
238 		/**
239 		 * Sets the log level.
240 		 *
241 		 * @param level the new log level to set.
242 		 */
243 		public void setLogLevel(final int level) {
244 			logLevel = level;
245 		}
246 
247 		/**
248 		 * Logs a message.
249 		 *
250 		 * @param message the message to log.
251 		 * @param throwable the throwable which caused the message to be logged (if applicable).
252 		 * @param level the logging level to log the message at.
253 		 */
254 		private void log(final Object message, final Throwable throwable, final int level) {
255 			if (level < logLevel) {
256 				return;
257 			}
258 
259 			if (message != null) {
260 				String prefix = new SimpleDateFormat("dd/MM/yy HH:mm:ss").format(new Date()) + ": ";
261 				StandaloneLauncher.getInstance().log(prefix + message.toString() + '\n');
262 			}
263 
264 			if (throwable != null) {
265 				StringWriter writer = new StringWriter();
266 				PrintWriter printWriter = new PrintWriter(writer);
267 				throwable.printStackTrace(printWriter);
268 				printWriter.close();
269 
270 				StandaloneLauncher.getInstance().log(writer.toString());
271 			}
272 		}
