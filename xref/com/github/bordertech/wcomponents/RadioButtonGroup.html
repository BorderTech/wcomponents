4   import org.apache.commons.logging.Log;
5   import org.apache.commons.logging.LogFactory;

7   /**
8    * <p>
9    * This class is used to create a group of radio buttons.
10   * </p>
11   * <p>
12   * A {@link WRadioButton} can only be created by using the {@link #addRadioButton(Object)} method or
13   * {@link #addRadioButton()} method. Call one of these methods to get an instance of a radio button and then add the
14   * radio button to the required location in the UI component tree.
15   * </p>
16   * <p>
17   * If the value of the radio button is known when creating its instance, then use the {@link #addRadioButton(Object)}.
18   * </p>
19   * <p>
20   * If a radio button needs to be used in a repeating component and will have different values depending on its context,
21   * then use the {@link #addRadioButton()}. Radio buttons used with a {@link WRepeater} will get their value from the
22   * "bean" associated to the row in the repeater. By default, the bean property for the radio button will be set to ".",
23   * but this can be overridden by calling {@link #setBeanProperty(String)} on the radio button instance.
24   * </p>
25   * <p>
26   * For the radio button group to work correctly, it is important that each radio button in the group has a unique value.
27   * The radio button group uses the {@link String} representation of the radio button's value to identify which button
28   * has been selected. As the string representation of the radio button's value is sent to the client, be mindful that it
29   * should not be too large.
30   * </p>
31   *
32   * @author Jonathan Austin
33   * @since 1.0.0
34   */
35  public class RadioButtonGroup extends AbstractInput implements AjaxTrigger, SubordinateTrigger, MultiInputComponent {
36  
37  	/**
38  	 * The logger instance for this class.
39  	 */
40  	private static final Log LOG = LogFactory.getLog(RadioButtonGroup.class);
41  
42  	/**
43  	 * {@inheritDoc}
44  	 */
45  	@Override
46  	public String getValue() {
47  		Object data = getData();
48  		if (data == null) {
49  			return null;
50  		}
51  		// An empty string is treated as null
52  		return Util.empty(data.toString()) ? null : data.toString();
53  	}
54  
55  	/**
56  	 * This method will only processes a request where the group is on the request and has no value. If the group has no
57  	 * value, then none of the group's radio buttons will be triggered to process the request.
58  	 *
59  	 * @param request the request being responded to.
60  	 * @return true if the group has changed, otherwise false
61  	 */
62  	@Override
63  	protected boolean doHandleRequest(final Request request) {
64  		// Check if the group has a value on the Request
65  		if (request.getParameter(getId()) != null) {
66  			// Allow the handle request to be processed by the radio buttons
67  			return false;
68  		}
69  
70  		// If no value, then clear the current value (if required)
71  		if (getValue() != null) {
72  			setData(null);
73  			return true;
74  		}
75  
76  		return false;
77  	}
78  
79  	/**
80  	 * This method is to be only called by the {@link WRadioButton} that has its value on the request.
81  	 *
82  	 * @param request the request being responded to.
83  	 * @return true if the group has changed, otherwise false
84  	 */
85  	boolean handleButtonOnRequest(final Request request) {
86  		// Only process on a POST
87  		if (!"POST".equals(request.getMethod())) {
88  			LOG.warn("RadioButton on a request that is not a POST. Will be ignored.");
89  			return false;
90  		}
91  
92  		String value = getRequestValue(request);
93  		String current = getValue();
94  
95  		boolean changed = !Util.equals(value, current);
96  
97  		if (changed) {
98  			setChangedInLastRequest(true);
99  			setData(value);
100 			doHandleChanged();
101 		}
102 
103 		return changed;
104 	}

106 	/**
107 	 * The radio button group does not set focus. It allows the radio buttons to set their own focus if they are
108 	 * selected.
109 	 */
110 	@Override
111 	public void setFocussed() {
112 		// Do Nothing - Allow radio buttons to set the focus if they are selected
113 	}
114 
115 	/**
116 	 * {@inheritDoc}
117 	 */
118 	@Override
119 	public String getRequestValue(final Request request) {
120 		if (isPresent(request)) {
121 			String value = request.getParameter(getId());
122 			// Treat empty as null
123 			return Util.empty(value) ? null : value;
124 		} else {
125 			return getValue();
126 		}
127 	}
128 
129 	/**
130 	 * {@inheritDoc}
131 	 */
132 	@Override
133 	protected boolean isPresent(final Request request) {
134 		return request.getParameter(getId() + "-h") != null;
135 	}

137 	/**
138 	 * Retrieves the selected value for this group.
139 	 *
140 	 * @return the value of the selected radio button, or null if there is no selection.
141 	 */
142 	public String getSelectedValue() {
143 		return getValue();
144 	}
145 
146 	/**
147 	 * Sets the selected value for this group.
148 	 *
149 	 * @param value the selected value, or null to clear the selection.
150 	 */
151 	public void setSelectedValue(final String value) {
152 		setData(value);
153 	}

178 		return super.isSubmitOnChange();
179 	}

181 	/**
182 	 * This method will add a radio button to the group with the given value. The value must be unique for the group.
183 	 * <p>
184 	 * The radio button returned by this method must be added to the required location in the UI Component tree.
185 	 * </p>
186 	 * <p>
187 	 * The radio button group uses the {@link String} representation of the radio button's value to identify which
188 	 * button has been selected. As the string representation of the radio button's value is sent to the client, be
189 	 * mindful that it should not be too large.
190 	 * </p>
191 	 *
192 	 * @param value a unique value for the radio button.
193 	 * @return the radio button that was added to the group
194 	 */
195 	public WRadioButton addRadioButton(final Object value) {
196 		WRadioButton radioButton = new WRadioButton(this);
197 		radioButton.setData(value);
198 		return radioButton;
199 	}
200 
201 	/**
202 	 * This method will add a radio button to the group.
203 	 * <p>
204 	 * Unlike {{@link #addRadioButton(Object)}, which requires a value to be passed in, this method has no value passed
205 	 * in for the radio button, as it is expected the radio button will be used with a {@link WRepeater} and will get
206 	 * its value from a bean. The bean property will default to ".", but can be set to the appropriate bean property
207 	 * after being returned.
208 	 * </p>
209 	 * <p>
210 	 * The radio button returned by this method must be added to the required location in the UI Component tree.
211 	 * </p>
212 	 * <p>
213 	 * The radio button group uses the {@link String} representation of the radio button's value to identify which
214 	 * button has been selected. As the string representation of the radio button's value is sent to the client, be
215 	 * mindful that it should not be too large.
216 	 * </p>
217 	 *
218 	 * @return the radio button that was added to the group
219 	 */
220 	public WRadioButton addRadioButton() {
221 		WRadioButton radioButton = new WRadioButton(this);
222 		radioButton.setBeanProperty(".");
223 		return radioButton;
224 	}
