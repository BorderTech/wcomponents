57  	 * Returns a string value of the selected item for this users session. If multiple selections have been made, this
58  	 * will be a comma separated list of string values. If no value is selected, null is returned.
59  	 *
60  	 * @return the selected item value as a rendered String
61  	 */
62  	@Override
63  	public String getValueAsString() {
64  		List<?> selected = getValue();
65  
66  		if (selected == null || selected.isEmpty()) {
67  			return null;
68  		}
69  
70  		StringBuffer stringValues = new StringBuffer();
71  
72  		for (int i = 0; i < selected.size(); i++) {
73  			if (i > 0) {
74  				stringValues.append(", ");
75  			}
76  			stringValues.append(optionToString(selected.get(i)));
77  		}
78  
79  		return stringValues.toString();
80  	}

83  	 * @return true if nothing selected or if a selection is required and only the null option has been selected.
84  	 */
85  	@Override
86  	public boolean isEmpty() {
87  		List<?> selected = getValue();
88  		if (selected == null || selected.isEmpty()) {
89  			return true;
90  		}
91  
92  		// If selection required, check at least one option is selected that is not null and not empty
93  		if (!isAllowNoSelection()) {
94  			for (Object option : selected) {
95  				// Check option is null or empty
96  				boolean isNull = option == null ? true : option.toString().length() == 0;
97  				if (!isNull) {
98  					return false;
99  				}
100 			}
101 			return true;
102 		}
103 
104 		return false;
105 	}

171 	/**
172 	 * Returns a {@link List} of the selected options. If no options have been selected, then it returns an empty list.
173 	 * <p>
174 	 * As getValue calls {@link #getData()} for the currently selected options, it usually expects getData to return
175 	 * null (for no selection) or a {@link List} of selected options. If the data returned by getData is not null and is
176 	 * not a List, then setData will either (1) if the data is an array, convert the array to a List or (2) create a
177 	 * List and add the data as the selected option.
178 	 * </p>
179 	 * <p>
180 	 * getValue will verify the selected option/s are valid. If a selected option does not exist, then it will throw an
181 	 * {@link IllegalArgumentException}.
182 	 * </p>
183 	 *
184 	 * @return the selected options in the given UI context.
185 	 */
186 	@Override
187 	public List<?> getValue() {
188 		// Convert data to a list (if necessary)
189 		List<?> data = convertDataToList(getData());
190 
191 		// Validate the selected options (allow handle invalid options)
192 		List<?> validOptions = findValidOptions(getOptions(), data, true);
193 
194 		if (validOptions == null || validOptions.isEmpty()) {
195 			return NO_SELECTION;
196 		}
197 		return validOptions;
198 	}

200 	/**
201 	 * {@inheritDoc}
202 	 * <p>
203 	 * getData expects the data to be either null (for no selection) or a {@link List} of selected options.
204 	 * </p>
205 	 * <p>
206 	 * getData will check that if no options have been selected (ie data is null) and the list component requires an
207 	 * option to be selected (ie {@link #isAllowNoSelection()} is false), it will return a list containing the first
208 	 * option as the default selected option. If the data is an empty list or an empty array, then it will be treated
209 	 * the same as there being no selection.
210 	 * </p>
211 	 */
212 	@Override
213 	public Object getData() {
214 		Object data = super.getData();
215 
216 		// Treat "empty" the same as null (ie no selection)
217 		boolean empty = false;
218 		if (data == null) {
219 			empty = true;
220 		} else if (data instanceof List<?>) {
221 			empty = ((List<?>) data).isEmpty();
222 		} else if (data instanceof Object[]) {
223 			empty = ((Object[]) data).length == 0;
224 		}
225 
226 		// Check if we need to default to the first option
227 		if (empty && !isAllowNoSelection()) {
228 			List<?> options = getOptions();
229 			if (options != null && !options.isEmpty()) {
230 				// Check if NULL is an option
231 				if (SelectListUtil.containsOption(options, null)) {
232 					return Arrays.asList(new Object[]{null});
233 				} else {
234 					// Use the first option
235 					Object firstOption = SelectListUtil.getFirstOption(options);
236 					return Arrays.asList(new Object[]{firstOption});
237 				}
238 			}
239 		}
240 
241 		return data;
242 	}

245 	 * Sets the data that this component displays/edits.
246 	 * <p>
247 	 * setData expects the data being passed in to be either null (for no selection) or a {@link List} of selected
248 	 * options. If an empty list is passed in, then it will be treated the same as null.
249 	 * </p>
250 	 * <p>
251 	 * If the data is not null and is not a List, then setData will either (1) if the data is an array, convert the
252 	 * array to a List or (2) create a List and add the data as the selected option.
253 	 * </p>
254 	 * <p>
255 	 * setData will verify the selected option/s are valid. If a selected option does not exist, then it will throw an
256 	 * {@link IllegalArgumentException}.
257 	 * </p>
258 	 * <p>
259 	 * If the list component requires a selection (ie {@link #isAllowNoSelection()} is false) and the data being set is
260 	 * null and null is not a valid option, then setData will throw an {@link IllegalArgumentException}.
261 	 * </p>
262 	 *
263 	 * @param data the data to set which is usually a List containing the selected options or null for no selection.
264 	 * @throws IllegalArgumentException if the data Object is not null but the options List is null or empty.
265 	 */
266 	@Override
267 	public void setData(final Object data) {
268 		List<?> options = getOptions();
269 		if (!isEditable() && data != null && (options == null || options.isEmpty())) {
270 			throw new IllegalStateException(
271 					"Should not set selections on a list component with no options.");
272 		}
273 
274 		// Convert data to a list (if necessary)
275 		List<?> selected = convertDataToList(data);
276 
277 		// Validate the selected options
278 		List<?> validOptions = findValidOptions(options, selected, false);
279 		super.setData(validOptions);
280 	}

282 	/**
283 	 * Convert the data to a list (if necessary).
284 	 *
285 	 * @param data the data to convert to a list
286 	 * @return the data converted to a list
287 	 */
288 	private List<?> convertDataToList(final Object data) {
289 		if (data == null) {
290 			return null;
291 		} else if (data instanceof List) {
292 			return (List<?>) data;
293 		} else if (data instanceof Object[]) {
294 			return Arrays.asList((Object[]) data);
295 		} else {
296 			return Arrays.asList(new Object[]{data});
297 		}
298 	}

300 	/**
301 	 * Find the valid options for the selected data, allowing for option/code and legacy matching.
302 	 *
303 	 * @param options the list of options
304 	 * @param selected the options to search for
305 	 * @param handleInvalid true if allow handle invalid options
306 	 * @return the valid option
307 	 */
308 	private List<?> findValidOptions(final List<?> options, final List<?> selected,
309 			final boolean handleInvalid) {
310 		// No selection made
311 		if (selected == null || selected.isEmpty()) {
312 			// No Selection required, or no options to select
313 			if (isAllowNoSelection() || options == null || options.isEmpty()) {
314 				return null;
315 			} else { // Selection is required
316 				// Check if NULL is an option
317 				if (selected == null && SelectListUtil.containsOption(options, null)) {
318 					return null;
319 				}
320 
321 				// Must have a selected option
322 				throw new IllegalStateException(
323 						"A valid option must be selected for lists that require a selected option.");
324 			}
325 		}
326 
327 		// Check we have valid options (If options are matched via Option/Code or Legacy Matching, then return a list of
328 		// the actual options)
329 		List<Object> validSelections = new ArrayList<>();
330 
331 		for (Object selectedOption : selected) {
332 			boolean found = false;
333 
334 			if (options != null) {
335 				if (selectedOption == null) {
336 					found = SelectListUtil.containsOption(options, null);
337 					if (found) {
338 						validSelections.add(null);
339 					}
340 				} else {
341 					Object option = SelectListUtil.getOptionWithMatching(options, selectedOption);
342 					if (option != null) {
343 						found = true;
344 						validSelections.add(option);
345 					}
346 				}
347 			}
348 
349 			if (!found) {
350 				if (isEditable() && selectedOption instanceof String) {
351 					validSelections.add(selectedOption);
352 				} else if (handleInvalid) { // Handle invalid option
353 					List<?> valid = doHandleInvalidOption(selectedOption);
354 					setSelected(valid);
355 					return valid;
356 				} else {
357 					throw new IllegalStateException("The selection \"" + selectedOption
358 							+ "\" is not an available option.");
359 				}
360 			}
361 		}
362 
363 		return validSelections;
364 
365 	}

373 	 * This method can be overridden to handle an invalid option without throwing the exception. It must return a list

376 	 *
377 	 * @param invalidOption the invalid option
378 	 * @return the list of valid options to be set as the selected options

380 	protected List<?> doHandleInvalidOption(final Object invalidOption) {
381 		throw new IllegalStateException("The selection \"" + invalidOption
382 				+ "\" is invalid. It is not an available option.");
383 	}

388 	 * {@inheritDoc}
389 	 */
390 	@Override
391 	protected boolean doHandleRequest(final Request request) {
392 		// First we need to figure out if the new selections are the same as the
393 		// previous selections.
394 		final List<?> newSelections = getRequestValue(request);
395 		List<?> priorSelections = getValue();
396 
397 		boolean changed = !selectionsEqual(newSelections, priorSelections);
398 
399 		if (changed) {
400 			setData(newSelections);
401 		}
402 
403 		return changed;
404 	}

418 	/**
419 	 * Determines which selections have been added in the given request.
420 	 *
421 	 * @param request the current request
422 	 * @return a list of selections that have been added in the given request.
423 	 */
424 	protected List<?> getNewSelections(final Request request) {
425 		String[] paramValues = request.getParameterValues(getId());
426 		if (paramValues == null || paramValues.length == 0) {
427 			return NO_SELECTION;
428 		}
429 
430 		List<String> values = Arrays.asList(paramValues);
431 		List<Object> newSelections = new ArrayList<>(values.size());
432 
433 		// Figure out which options have been selected.
434 		List<?> options = getOptions();
435 		if (options == null || options.isEmpty()) {
436 			if (!isEditable()) {
437 				// User could not have made a selection.
438 				return NO_SELECTION;
439 			}
440 			options = Collections.EMPTY_LIST;
441 		}
442 
443 		for (Object value : values) {
444 			boolean found = false;
445 			int optionIndex = 0;
446 
447 			for (Object option : options) {
448 				if (option instanceof OptionGroup) {
449 					List<?> groupOptions = ((OptionGroup) option).getOptions();
450 					if (groupOptions != null) {
451 						for (Object nestedOption : groupOptions) {
452 							if (value.equals(optionToCode(nestedOption, optionIndex++))) {
453 								newSelections.add(nestedOption);
454 								found = true;
455 								break;
456 							}
457 						}
458 					}
459 				} else if (value.equals(optionToCode(option, optionIndex++))) {
460 					newSelections.add(option);
461 					found = true;
462 					break;
463 				}
464 			}
465 
466 			if (!found) {
467 				if (isEditable()) {
468 					newSelections.add(value);
469 				} else {
470 					LOG.warn(
471 							"Option \"" + value + "\" on the request is not a valid option. Will be ignored.");
472 				}
473 			}
474 		}
475 
476 		// If no valid options found, then return the current settings
477 		if (newSelections.isEmpty()) {
478 			LOG.warn("No options on the request are valid. Will be ignored.");
479 			return getValue();
480 		}
481 
482 		// If must have selection and more than 1 option selected, remove the "null" entry if it was selected.
483 		if (!isAllowNoSelection() && newSelections.size() > 1) {
484 			List<Object> filtered = new ArrayList<>();
485 			Object nullOption = null;
486 			for (Object option : newSelections) {
487 				// Check option is null or empty
488 				boolean isNull = option == null ? true : option.toString().length() == 0;
489 				if (isNull) {
490 					// Hold the option as it could be "null" or "empty"
491 					nullOption = option;
492 				} else {
493 					filtered.add(option);
494 				}
495 			}
496 			// In the case where only null options were selected, then add one nullOption
497 			if (filtered.isEmpty()) {
498 				filtered.add(nullOption);
499 			}
500 			return filtered;
501 		} else {
502 			return newSelections;
503 		}
504 	}

506 	/**
507 	 * Selection lists are considered equal if they have the same items (order is not important). An empty list is
508 	 * considered equal to a null list.
509 	 *
510 	 * @param list1 the first list to check.
511 	 * @param list2 the second list to check.
512 	 * @return true if the lists are equal, false otherwise.
513 	 */
514 	private boolean selectionsEqual(final List<?> list1, final List<?> list2) {
515 		if (isSelectionOrderable()) {
516 			return Util.equals(list1, list2);
517 		}
518 
519 		// Empty or null lists
520 		if ((list1 == null || list1.isEmpty()) && (list2 == null || list2.isEmpty())) {
521 			return true;
522 		}
523 
524 		// Same size and contain same entries
525 		return list1 != null && list2 != null && list1.size() == list2.size() && list1.
526 				containsAll(list2);
527 	}

529 	/**
530 	 * Override WInput's validateComponent to perform further validation on the options selected.
531 	 *
532 	 * @param diags the list into which any validation diagnostics are added.
533 	 */
534 	@Override
535 	protected void validateComponent(final List<Diagnostic> diags) {
536 		super.validateComponent(diags);
537 
538 		List<?> selected = getValue();
539 
540 		// Only validate max and min if options have been selected
541 		if (!selected.isEmpty()) {
542 			int value = selected.size();
543 			int min = getMinSelect();
544 			int max = getMaxSelect();
545 
546 			if (min > 0 && value < min) {
547 				diags.add(
548 						createErrorDiagnostic(InternalMessages.DEFAULT_VALIDATION_ERROR_MIN_SELECT,
549 								this, min));
550 			}
551 
552 			if (max > 0 && value > max) {
553 				diags.add(
554 						createErrorDiagnostic(InternalMessages.DEFAULT_VALIDATION_ERROR_MAX_SELECT,
555 								this, max));
556 			}
557 		}
558 	}
