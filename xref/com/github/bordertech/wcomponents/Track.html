3   /**
4    * A track provides additional information relating to the video, for example subtitles. Note that all themes support
5    * tracks in WebVTT format. These files are encoded as UTF-8 and must be served with MIME type "text/vtt". Themes may
6    * provide additional support for different track formats.
7    *
8    * @author Yiannis Paschalidis
9    * @since 1.0.0
10   */
11  public interface Track extends ContentStreamAccess {
12  
13  	/**
14  	 * Describes the kind of track. This is used by the client to determine how the track should be handled.
15  	 */
16  	enum Kind {
17  		/**
18  		 * Transcription or translation of the dialogue, suitable for when the sound is available but not understood
19  		 * (e.g. because the user does not understand the language of the media resource's audio track). Overlaid on the
20  		 * video.
21  		 */
22  		SUBTITLES,
23  		/**
24  		 * Transcription or translation of the dialogue, sound effects, relevant musical cues, and other relevant audio
25  		 * information, suitable for when sound is unavailable or not clearly audible (e.g. because it is muted,
26  		 * drowned-out by ambient noise, or because the user is deaf). Overlaid on the video; labeled as appropriate for
27  		 * the hard-of-hearing.
28  		 */
29  		CAPTIONS,
30  		/**
31  		 * Textual descriptions of the video component of the media resource, intended for audio synthesis when the
32  		 * visual component is obscured, unavailable, or not usable (e.g. because the user is interacting with the
33  		 * application without a screen while driving, or because the user is blind). Synthesized as audio.
34  		 */
35  		DESCRIPTIONS,
36  		/**
37  		 * Chapter titles, intended to be used for navigating the media resource. Displayed as an interactive
38  		 * (potentially nested) list in the user agent's interface.
39  		 */
40  		CHAPTERS,
41  		/**
42  		 * Tracks intended for use from script. Not displayed by the user agent.
43  		 */
44  		METADATA
45  	}
46  
47  	/**
48  	 * @return the ISO 639-1 language code for this track.
49  	 */
50  	String getLanguage();
51  
52  	/**
53  	 * @return the type of this track.
54  	 */
55  	Kind getKind();
56  }
