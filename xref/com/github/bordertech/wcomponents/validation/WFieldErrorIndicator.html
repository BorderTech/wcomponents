3   import com.github.bordertech.wcomponents.WComponent;
4   import java.util.List;
5   
6   /**
7    * <p>
8    * This component is used to render errors relating to an individual field. Generally, this is placed next to the input
9    * field to help highlight the individual fields that are in error. An instance of the indicator is automatically
10   * provided when adding a field to a WFieldLayout.</p>
11   *
12   * @author Adam Millard
13   * @deprecated 1.4.0 This component has never been supported in the front end code. It has never worked and is incompatible with WValidationErrors.
14   * It will be deleted asap.
15   */
16  public class WFieldErrorIndicator extends AbstractWFieldIndicator {
17  
18  	/**
19  	 * Creates an WFieldErrorIndicator which is associated with a field.
20  	 *
21  	 * @param relatedField The related field.
22  	 */
23  	public WFieldErrorIndicator(final WComponent relatedField) {
24  		super(relatedField);
25  	}
26  
27  	/**
28  	 * Calls {@link #showIndicatorsForComponent(List, int)} with a {@link Diagnostic#ERROR}.
29  	 *
30  	 * @param diags A List of Diagnostic objects.
31  	 */
32  	@Override
33  	protected void showErrorIndicatorsForComponent(final List<Diagnostic> diags) {
34  		showIndicatorsForComponent(diags, Diagnostic.ERROR);
35  	}
36  
37  	/**
38  	 * Return the field indicator type of error.
39  	 *
40  	 * @return the field indicator type
41  	 */
42  	@Override
43  	public FieldIndicatorType getFieldIndicatorType() {
44  		return FieldIndicatorType.ERROR;
45  	}
46  }
