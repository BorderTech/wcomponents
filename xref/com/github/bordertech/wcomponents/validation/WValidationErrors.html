12  /**
13   * <p>
14   * This component is used to render an error message box (commonly near the top of an application's UI) which contains
15   * all supplied validation errors.
16   * </p>
17   * <p>
18   * This provides (in conjuntion with {@link WFieldErrorIndicator} components) the ability to link error messages with
19   * the input field that is in error.
20   * </p>
21   * <p>
22   * NOTE: The collection of errors passed to this component must be implementations of {@link Diagnostic}.
23   * </p>
24   *
25   * @see Diagnostic
26   * @see com.github.bordertech.wcomponents.WMessages
27   *
28   * @author Adam Millard
29   * @author Mark Reeves
30   * @since 1.0.0
31   */
32  public class WValidationErrors extends AbstractWComponent {
33  
34  	/**
35  	 * Clears the errors for a given user.
36  	 */
37  	public void clearErrors() {
38  		getOrCreateComponentModel().errors.clear();
39  	}
40  
41  	/**
42  	 * Sets the errors for a given user.
43  	 *
44  	 * @param errors the errors to set.
45  	 */
46  	public void setErrors(final List<Diagnostic> errors) {
47  		if (errors != null) {
48  			ValidationErrorsModel model = getOrCreateComponentModel();
49  
50  			for (Diagnostic error : errors) {
51  				if (error.getSeverity() == Diagnostic.ERROR) {
52  					model.errors.add(error);
53  				}
54  			}
55  		}
56  	}
57  
58  	/**
59  	 * Retrieves the errors for a given user.
60  	 *
61  	 * @return the list of errors for the given user.
62  	 */
63  	public List<Diagnostic> getErrors() {
64  		return Collections.unmodifiableList(getComponentModel().errors);
65  	}
66  
67  	/**
68  	 * Indicates whether there are any validation errors.
69  	 *
70  	 * @return true if there are validation errors, false otherwise.
71  	 */
72  	public boolean hasErrors() {
73  		ValidationErrorsModel model = getComponentModel();
74  		return !model.errors.isEmpty();
75  	}
76  
77  	/**
78  	 * Groups the errors by their source field.
79  	 *
80  	 * @return a list of grouped errors.
81  	 */
82  	public List<GroupedDiagnositcs> getGroupedErrors() {
83  		List<GroupedDiagnositcs> grouped = new ArrayList<>();
84  		Diagnostic previousError = null;
85  		GroupedDiagnositcs group = null;
86  
87  		for (Diagnostic theError : getErrors()) {
88  			boolean isNewField = ((previousError == null) || (previousError.getContext() != theError.
89  					getContext())
90  					|| (previousError.getComponent() != theError.getComponent()));
91  
92  			if (group == null || isNewField) {
93  				group = new GroupedDiagnositcs();
94  				grouped.add(group);
95  			}
96  
97  			group.addDiagnostic(theError);
98  
99  			previousError = theError;
100 		}
101 
102 		return Collections.unmodifiableList(grouped);
103 	}
104 
105 	/**
106 	 * Groups a set of related diagnostics.
107 	 */
108 	public static class GroupedDiagnositcs {
109 
110 		/**
111 		 * The grouped diagnostics.
112 		 */
113 		private final List<Diagnostic> diags = new ArrayList<>();
114 
115 		/**
116 		 * Adds a diagnostic to this group.
117 		 *
118 		 * @param diag the diagnostic to add.
119 		 */
120 		public void addDiagnostic(final Diagnostic diag) {
121 			diags.add(diag);
122 		}
123 
124 		/**
125 		 * @return the diagnostics contained in this group.
126 		 */
127 		public List<Diagnostic> getDiagnostics() {
128 			return Collections.unmodifiableList(diags);
129 		}
130 	}
131 
132 	/**
133 	 * @return a String representation of this component, for debugging purposes.
134 	 */
135 	@Override

137 		return getClass().getSimpleName() + '(' + getErrors() + ')';
138 	}

143 	 * Sets the message box title.
144 	 *
145 	 * @param title the message box title to set, using {@link MessageFormat} syntax.
146 	 * @param args optional arguments for the message format string.
147 	 */
148 	public void setTitleText(final String title, final Serializable... args) {
149 		ValidationErrorsModel model = getOrCreateComponentModel();
150 		model.title = I18nUtilities.asMessage(title, args);
151 	}
152 
153 	/**
154 	 * @return the message box title.
155 	 */
156 	public String getTitleText() {
157 		return I18nUtilities.format(null, getComponentModel().title);
158 	}
