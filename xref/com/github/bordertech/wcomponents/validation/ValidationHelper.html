3   import java.util.ArrayList;
4   import java.util.List;

6   /**
7    * Provides methods to assist with validation.
8    *
9    * @author James Gifford
10   * @since 1.0.0
11   */
12  public final class ValidationHelper {
13  
14  	/**
15  	 * Prevent instantiation of this class.
16  	 */
17  	private ValidationHelper() {
18  	}
19  
20  	/**
21  	 * Extract diagnostics with a given severity from a list of Diagnostic objects. This is useful for picking errors
22  	 * out from a list of diagnostics, for example.
23  	 *
24  	 * @param diagnostics the list of diagnostics to look through.
25  	 * @param severity the severity of diagnostics to extract.
26  	 * @return a list of diagnostics with the given severity, may be empty.
27  	 */
28  	public static List<Diagnostic> extractDiagnostics(final List<Diagnostic> diagnostics,
29  			final int severity) {
30  		ArrayList<Diagnostic> extract = new ArrayList<>();
31  
32  		for (Diagnostic diagnostic : diagnostics) {
33  			if (diagnostic.getSeverity() == severity) {
34  				extract.add(diagnostic);
35  			}
36  		}
37  
38  		return extract;
39  	}
40  }
