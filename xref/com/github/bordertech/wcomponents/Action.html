7    * submit first causes the entire WComponent tree to be updated via their handleRequest methods. Next, the action will

9    *
10   * @author Martin Shevchenko
11   * @since 1.0.0
12   */
13  public interface Action extends Serializable {
14  
15  	/**
16  	 * This method is invoked when an action occurs.
17  	 *
18  	 * @param event details about the event that occured.
19  	 */
20  	void execute(ActionEvent event);
21  }
