1   package com.github.bordertech.wcomponents;

3   import java.io.Serializable;
4   
5   /**
6    * <p>
7    * This interface enables access to arbitrary document content such as a PDF. It could be fetched from the database or
8    * it could be generated on the fly.</p>
9    *
10   * <p>
11   * For larger content, consider using {@link ContentStreamAccess}, as it does not require the entire binary content to
12   * be held in memory.</p>
13   *
14   * @author Martin Shevchenko
15   * @since 1.0.0
16   */
17  public interface ContentAccess extends Serializable {
18  
19  	/**
20  	 * The bytes that make up the document content.
21  	 *
22  	 * @return the bytes
23  	 */
24  	byte[] getBytes();
25  
26  	/**
27  	 * Some text that describes the document content. This text could be the document filename or title, for instance.
28  	 *
29  	 * @return a short document description
30  	 */
31  	String getDescription();
32  
33  	/**
34  	 * The mime type of the document. For example, "image/jpeg", "image/gif", "application/pdf" etc
35  	 *
36  	 * @return the mime type.
37  	 */
38  	String getMimeType();
39  }
