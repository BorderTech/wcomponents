189 	/**
190 	 * Adds a runnable to the list of runnables to be invoked later.
191 	 *
192 	 * @param uic the UIContext to invoke the runnable in.
193 	 * @param runnable the runnable to add
194 	 */
195 	@Override
196 	public void invokeLater(final UIContext uic, final Runnable runnable) {
197 		if (invokeLaterRunnables == null) {
198 			invokeLaterRunnables = new ArrayList<>();
199 		}
200 
201 		invokeLaterRunnables.add(new UIContextImplRunnable(uic, runnable));
202 	}
203 
204 	/**
205 	 * Runs the runnables that were added using {@link #invokeLater(Runnable)}.
206 	 */
207 	@Override
208 	public void doInvokeLaters() {
209 		if (invokeLaterRunnables == null) {
210 			return;
211 		}
212 
213 		// The Runnables we are about to run may add their own invoke later
214 		// runnables, so
215 		// loop to make sure we process them all.
216 		while (!invokeLaterRunnables.isEmpty()) {
217 			List<UIContextImplRunnable> runnables = new ArrayList<>();
218 			runnables.addAll(invokeLaterRunnables);
219 			invokeLaterRunnables.clear();
220 
221 			for (UIContextImplRunnable run : runnables) {
222 				UIContextHolder.pushContext(run.getContext());
223 
224 				try {
225 					run.getRunnable().run();
226 				} finally {
227 					UIContextHolder.popContext();
228 				}
229 			}
230 		}
231 	}

266 	/**
267 	 * @return the unique id of the focused component for this UI.
268 	 */
269 	@Override
270 	public String getFocussedId() {
271 		// No focused
272 		if (focussed == null) {
273 			return null;
274 		}
275 
276 		// Check there is an active context and a UI set
277 		UIContext cuic = UIContextHolder.getCurrent();
278 		if (cuic == null || cuic.getUI() == null) {
279 			return null;
280 		}
281 
282 		UIContext fuic = (focussedUIC == null) ? this : focussedUIC;
283 		boolean differentFocusContext = fuic != cuic;
284 
285 		if (differentFocusContext) {
286 			UIContextHolder.pushContext(fuic);
287 		}
288 
289 		String id;
290 
291 		try {
292 			if (focussed.isHidden()) {
293 				return null;
294 			} else if (focussed instanceof Input && ((Input) focussed).isReadOnly()) {
295 				return null;
296 			} else if (focussed instanceof Disableable && ((Disableable) focussed).isDisabled()) {
297 				return null;
298 			}
299 
300 			id = focussed.getId();
301 		} finally {
302 			if (differentFocusContext) {
303 				UIContextHolder.popContext();
304 			}
305 		}
306 
307 		// Check id is focusable
308 		WComponent root = cuic.getUI();
309 		if (TreeUtil.isIdFocusable(root, id)) {
310 			return id;
311 		}
312 
313 		return null;
314 	}

395 	/**
396 	 * Reserved for internal framework use. Retrieves a scratch area, where data can be temporarily stored. WComponents
397 	 * must not rely on data being available in the scratch area after each phase.
398 	 *
399 	 * @param component the component to retrieve the scratch map for.
400 	 * @return the scratch map for the given component.
401 	 */
402 	@Override
403 	public Map<Object, Object> getScratchMap(final WComponent component) {
404 		if (scratchMaps == null) {
405 			scratchMaps = new HashMap<>();
406 		}
407 
408 		Map<Object, Object> componentScratchMap = scratchMaps.get(component);
409 
410 		if (componentScratchMap == null) {
411 			componentScratchMap = new HashMap<>(2);
412 			scratchMaps.put(component, componentScratchMap);
413 		}
414 
415 		return componentScratchMap;
416 	}
