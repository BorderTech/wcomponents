185 	/**
186 	 * Adds a runnable to the list of runnables to be invoked later.
187 	 *
188 	 * @param uic the UIContext to invoke the runnable in.
189 	 * @param runnable the runnable to add
190 	 */
191 	@Override
192 	public void invokeLater(final UIContext uic, final Runnable runnable) {
193 		if (invokeLaterRunnables == null) {
194 			invokeLaterRunnables = new ArrayList<>();
195 		}
196 
197 		invokeLaterRunnables.add(new Duplet<>(uic, runnable));
198 	}
199 
200 	/**
201 	 * Runs the runnables that were added using {@link #invokeLater(Runnable)}.
202 	 */
203 	@Override
204 	public void doInvokeLaters() {
205 		if (invokeLaterRunnables == null) {
206 			return;
207 		}
208 
209 		// The Runnables we are about to run may add their own invoke later
210 		// runnables, so
211 		// loop to make sure we process them all.
212 		while (!invokeLaterRunnables.isEmpty()) {
213 			List<Duplet<UIContext, Runnable>> runnables = new ArrayList<>();
214 			runnables.addAll(invokeLaterRunnables);
215 			invokeLaterRunnables.clear();
216 
217 			for (Duplet<UIContext, Runnable> run : runnables) {
218 				UIContextHolder.pushContext(run.getFirst());
219 
220 				try {
221 					run.getSecond().run();
222 				} finally {
223 					UIContextHolder.popContext();
224 				}
225 			}
226 		}
227 	}

262 	/**
263 	 * @return the unique id of the focused component for this UI.
264 	 */
265 	@Override
266 	public String getFocussedId() {
267 		// No focused
268 		if (focussed == null) {
269 			return null;
270 		}
271 
272 		// Check there is an active context and a UI set
273 		UIContext cuic = UIContextHolder.getCurrent();
274 		if (cuic == null || cuic.getUI() == null) {
275 			return null;
276 		}
277 
278 		UIContext fuic = (focussedUIC == null) ? this : focussedUIC;
279 		boolean differentFocusContext = fuic != cuic;
280 
281 		if (differentFocusContext) {
282 			UIContextHolder.pushContext(fuic);
283 		}
284 
285 		String id;
286 
287 		try {
288 			if (focussed.isHidden()) {
289 				return null;
290 			} else if (focussed instanceof Input && ((Input) focussed).isReadOnly()) {
291 				return null;
292 			} else if (focussed instanceof Disableable && ((Disableable) focussed).isDisabled()) {
293 				return null;
294 			}
295 
296 			id = focussed.getId();
297 		} finally {
298 			if (differentFocusContext) {
299 				UIContextHolder.popContext();
300 			}
301 		}
302 
303 		// Check id is focusable
304 		WComponent root = cuic.getUI();
305 		if (TreeUtil.isIdFocusable(root, id)) {
306 			return id;
307 		}
308 
309 		return null;
310 	}

391 	/**
392 	 * Reserved for internal framework use. Retrieves a scratch area, where data can be temporarily stored. WComponents
393 	 * must not rely on data being available in the scratch area after each phase.
394 	 *
395 	 * @param component the component to retrieve the scratch map for.
396 	 * @return the scratch map for the given component.
397 	 */
398 	@Override
399 	public Map<Object, Object> getScratchMap(final WComponent component) {
400 		if (scratchMaps == null) {
401 			scratchMaps = new HashMap<>();
402 		}
403 
404 		Map<Object, Object> componentScratchMap = scratchMaps.get(component);
405 
406 		if (componentScratchMap == null) {
407 			componentScratchMap = new HashMap<>(2);
408 			scratchMaps.put(component, componentScratchMap);
409 		}
410 
411 		return componentScratchMap;
412 	}
