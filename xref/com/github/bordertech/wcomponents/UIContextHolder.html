3   import com.github.bordertech.wcomponents.util.DebugUtil;
4   import java.util.ArrayList;
5   import java.util.Collections;
6   import java.util.List;
7   import java.util.Map;
8   import java.util.Stack;
9   import java.util.WeakHashMap;
10  import org.apache.commons.logging.Log;
11  import org.apache.commons.logging.LogFactory;
12  
13  /**
14   * <p>
15   * This class holds the current (active) UIContext.</p>
16   *
17   * <p>
18   * WComponents store user (session) component models within the UIContext, and the UIContextHolder provides a central
19   * location for components to read from. It is the responsibility of the WComponent container (e.g. Servlet) to
20   * initialise the holder at the start of each WComponent lifecycle, and clean it up afterwards.</p>
21   *
22   * <p>
23   * Applications should not normally need to use this class directly. If application code needs access to the UIContext,
24   * it must only call {@link #getCurrent()}. Do not call any other method within this class.</p>
25   *
26   * <p>
27   * The UIContextHolder uses a stack internally, to allow for components which utilise "SubUIContexts", for repeated
28   * data.</p>
29   *
30   * @author Yiannis Paschalidis
31   * @since 1.0.0
32   */
33  public final class UIContextHolder {
34  
35  	/**
36  	 * The logger instance for this class.
37  	 */
38  	private static final Log LOG = LogFactory.getLog(UIContextHolder.class);
39  
40  	/**
41  	 * Hide the constructor as there are no instance methods.
42  	 */
43  	private UIContextHolder() {
44  		// Do Nothing
45  	}
46  
47  	/**
48  	 * The UIContexts are stored on a thread-local. This is safe, as WComponent request processing and rendering are
49  	 * single-threaded per user.
50  	 */
51  	private static final ThreadLocal<Stack<UIContext>> CONTEXT_STACK = new ThreadLocal<>();
52  
53  	/**
54  	 * This is used for debugging only. It contains all UIContexts which are in an active thread.
55  	 */
56  	//TODO: JDK1.6+ Set<UIContxt> allContexts = Collections.newSetFromMap(new WeakHashMap<UIContext, Boolean>());
57  	private static final Map<UIContext, ?> ALL_ACTIVE_CONTEXTS = Collections.synchronizedMap(
58  			new WeakHashMap<UIContext, Object>());
59  
60  	/**
61  	 * Pushes a context onto the top of stack. This method is called by internal WComponent containers (e.g. WServlet)
62  	 * at the start of processing a request. and by components which repeat content.
63  	 *
64  	 * @param uic the UIContext to push.
65  	 */
66  	public static void pushContext(final UIContext uic) {
67  		if (DebugUtil.isDebugFeaturesEnabled()) {
68  			ALL_ACTIVE_CONTEXTS.put(getPrimaryUIContext(uic), null);
69  		}
70  
71  		getStack().push(uic);
72  	}
73  
74  	/**
75  	 * For debugging purposes only - returns all active UIContexts.
76  	 *
77  	 * @return all active UIContexts within the VM.
78  	 */
79  	static List<UIContext> getActiveContexts() {
80  		if (DebugUtil.isDebugFeaturesEnabled()) {
81  			return new ArrayList<>(ALL_ACTIVE_CONTEXTS.keySet());
82  		}
83  
84  		return null;
85  	}
86  
87  	/**
88  	 * Removes a context from the top of the stack. This method is only called by components which repeat content.
89  	 *
90  	 * @return the UIContext which was removed.
91  	 */
92  	public static UIContext popContext() {
93  		return getStack().pop();
94  	}
95  
96  	/**
97  	 * Retrieves the current effective UIContext. Note that this method will return null if called from outside normal
98  	 * request processing, for example during the intial application UI construction.
99  	 *
100 	 * @return the current effective UIContext.
101 	 */
102 	public static UIContext getCurrent() {
103 		Stack<UIContext> stack = CONTEXT_STACK.get();
104 
105 		if (stack == null || stack.isEmpty()) {
106 			return null;
107 		}
108 
109 		return getStack().peek();
110 	}
111 
112 	/**
113 	 * Retrieves the internal stack, creating it if necessary.
114 	 *
115 	 * @return the internal stack
116 	 */
117 	private static Stack<UIContext> getStack() {
118 		Stack<UIContext> stack = CONTEXT_STACK.get();
119 
120 		if (stack == null) {
121 			stack = new Stack<>();
122 			CONTEXT_STACK.set(stack);
123 		}
124 
125 		return stack;
126 	}
127 
128 	/**
129 	 * @return the primary context for the current context
130 	 */
131 	public static UIContext getCurrentPrimaryUIContext() {
132 		return getPrimaryUIContext(getCurrent());
133 	}
134 
135 	/**
136 	 * A utility function to iterate to the primary (top most) context and return it.
137 	 *
138 	 * @param uic the UIContext to retrieve the primary context for.
139 	 * @return the primary context for the given context.
140 	 */
141 	public static UIContext getPrimaryUIContext(final UIContext uic) {
142 		if (uic == null) {
143 			return null;
144 		}
145 
146 		UIContext primary = null;
147 		UIContext current = uic;
148 
149 		while (primary == null) {
150 			if (current instanceof UIContextDelegate) {
151 				UIContext backing = ((UIContextDelegate) current).getBacking();
152 				if (backing != null) {
153 					current = backing;
154 				} else {
155 					// This case should probably never happen.
156 					primary = current;
157 					LOG.warn("UIContextDelegate found without a backing context");
158 				}
159 			} else {
160 				primary = current;
161 			}
162 		}
163 
164 		return primary;
165 	}
166 
167 	/**
168 	 * Clears the UIContext stack. This method is called by internal WComponent containers (e.g. WServlet) after a
169 	 * request has been processed.
170 	 */
171 	public static void reset() {
172 		if (DebugUtil.isDebugFeaturesEnabled()) {
173 			UIContext context = getCurrent();
174 			ALL_ACTIVE_CONTEXTS.remove(context);
175 		}
176 
177 		CONTEXT_STACK.remove();
178 	}
179 }
