120 	 * Retrieves the code for the given option. Will return null if there is no matching option.
121 	 *
122 	 * @param option the option
123 	 * @param index the index of the option in the list.
124 	 * @return the code for the given option, or null if there is no matching option.
125 	 */
126 	protected String optionToCode(final Object option, final int index) {
127 		if (index < 0) {
128 			List<?> options = getOptions();
129 
130 			if (options == null || options.isEmpty()) {
131 				Integrity.issue(this, "No options available, so cannot convert the option \""
132 						+ option + "\" to a code.");
133 			} else {
134 				StringBuffer message = new StringBuffer();
135 				message.append("The option \"").append(option).append(
136 						"\" is not one of the available options.");
137 				Object firstOption = SelectListUtil.getFirstOption(options);
138 
139 				if (firstOption != null && option != null && firstOption.getClass() != option.
140 						getClass()) {
141 					message.append(" The options in this list component are of type \"");
142 					message.append(firstOption.getClass().getName())
143 							.append("\", the selection you supplied is of type \"");
144 					message.append(option.getClass().getName()).append("\".");
145 				}
146 
147 				Integrity.issue(this, message.toString());
148 			}
149 
150 			return null;
151 		} else if (option instanceof Option) {
152 			Option opt = (Option) option;
153 			return opt.getCode() == null ? "" : opt.getCode();
154 		} else {
155 			String code = APPLICATION_LOOKUP_TABLE.getCode(getLookupTable(), option);
156 
157 			if (code == null) {
158 				return String.valueOf(index + 1);
159 			} else {
160 				return code;
161 			}
162 		}
163 	}

165 	/**
166 	 * Retrieves the index of the given option. The index is not necessarily the index of the option in the options
167 	 * list, as there may be options nested in OptionGroups.
168 	 *
169 	 * @param option the option
170 	 * @return the index of the given option, or -1 if there is no matching option.
171 	 */
172 	protected int getOptionIndex(final Object option) {
173 		int optionCount = 0;
174 		List<?> options = getOptions();
175 
176 		if (options != null) {
177 			for (Object obj : getOptions()) {
178 				if (obj instanceof OptionGroup) {
179 					List<?> groupOptions = ((OptionGroup) obj).getOptions();
180 
181 					int groupIndex = groupOptions.indexOf(option);
182 
183 					if (groupIndex != -1) {
184 						return optionCount + groupIndex;
185 					}
186 
187 					optionCount += groupOptions.size();
188 				} else if (Util.equals(option, obj)) {
189 					return optionCount;
190 				} else {
191 					optionCount++;
192 				}
193 			}
194 		}
195 
196 		return -1;
197 	}

347 	 * Indicates whether this list was present in the request.
348 	 * </p>

351 	 * in the request.

353 	 *
354 	 * @param request the request being responded to.
355 	 * @return true if this list was present in the request, false if not.
356 	 */
357 	@Override
358 	protected boolean isPresent(final Request request) {
359 		if (isAllowNoSelection()) {
360 			String id = getId();
361 			return request.getParameter(id + "-h") != null;
362 		} else {
363 			return super.isPresent(request);
364 		}
365 	}

368 	 * Override preparePaintComponent to register an AJAX operation if this list is AJAX enabled.
369 	 *
370 	 * @param request the request being responded to.
371 	 */
372 	@Override
373 	protected void preparePaintComponent(final Request request) {
374 		super.preparePaintComponent(request);
375 
376 		if (isAjax() && UIContextHolder.getCurrent().getUI() != null) {
377 			AjaxTarget target = getAjaxTarget();
378 			AjaxHelper.registerComponent(target.getId(), getId());
379 		}
380 
381 		String cacheKey = getListCacheKey();
382 
383 		if (cacheKey != null) {
384 			LookupTableHelper.registerList(cacheKey, request);
385 		}
386 	}

426 	 * Retrieves the description for the given option. Intended for use by Renderers.
427 	 *
428 	 * @param option the option to retrieve the description for.
429 	 * @param index the option index.
430 	 * @return the description for the given option.
431 	 */
432 	public String getDesc(final Object option, final int index) {
433 		String desc = "";
434 
435 		if (option instanceof Option) {
436 			String optDesc = ((Option) option).getDesc();
437 
438 			if (optDesc != null) {
439 				desc = optDesc;
440 			}
441 		} else {
442 			String tableDesc = APPLICATION_LOOKUP_TABLE.getDescription(getLookupTable(), option);
443 
444 			if (tableDesc != null) {
445 				desc = tableDesc;
446 			} else if (option != null) {
447 				desc = option.toString();
448 			}
449 		}
450 
451 		return I18nUtilities.format(null, desc);
452 	}

467 	 */
468 	public static class SelectionModel extends InputModel {
469 
470 		/**
471 		 * The options for this list.
472 		 */
473 		private List<?> options;
474 
475 		/**
476 		 * The name of the lookup table which will be used to obtain the list options.
477 		 */
478 		private Object lookupTable;
479 
480 		/**
481 		 * Indicates whether the user is able to enter in an arbitrary value, rather than having to pick one from the
482 		 * list.
483 		 */
484 		private boolean editable;
485 
486 		/**
487 		 * The target component to repaint (via AJAX).
488 		 */
489 		private AjaxTarget ajaxTarget;
490 
491 		/**
492 		 * @return Returns the options.
493 		 */
494 		private List<?> getOptions() {
495 			return options == null ? null : Collections.unmodifiableList(options);
496 		}
497 
498 		/**
499 		 * @param options The options to set.
500 		 */
501 		private void setOptions(final List<?> options) {
502 			this.options = options;
503 			lookupTable = null;
504 		}
505 
506 		/**
507 		 * @param lookupTable The lookup table name to set.
508 		 */
509 		private void setLookupTable(final Object lookupTable) {
510 			this.lookupTable = lookupTable;
511 			options = null;
512 		}
513 
514 		/**
515 		 * @return Returns the lookupTable.
516 		 */
517 		private Object getLookupTable() {
518 			return lookupTable;
