95  	/**
96  	 * Adds a term to this definition list. If there is an existing term, the component is added to the list of data for
97  	 * the term.
98  	 *
99  	 * @param term the term to add.
100 	 * @param data the term data.
101 	 */
102 	public void addTerm(final String term, final WComponent... data) {
103 		for (WComponent component : data) {
104 			if (component != null) {
105 				content.add(component, term);
106 			}
107 		}
108 
109 		// If the term doesn't exist, we may need to add a dummy component
110 		if (getComponentsForTerm(term).isEmpty()) {
111 			content.add(new DefaultWComponent(), term);
112 		}
113 	}
114 
115 	/**
116 	 * {@inheritDoc}
117 	 */
118 	@Override
119 	public void remove(final WComponent child) {
120 		content.remove(child);
121 	}

123 	/**
124 	 * Removes a term from the definition list. All data (components) for the term is removed.
125 	 *
126 	 * @param term the term to remove.
127 	 */
128 	public void removeTerm(final String term) {
129 		for (WComponent child : getComponentsForTerm(term)) {
130 			content.remove(child);
131 		}
132 	}

134 	/**
135 	 * Groups a definition list's child components by their term for rendering.
136 	 *
137 	 * @return a list of this definition list's children grouped by their terms.
138 	 */
139 	public List<Duplet<String, ArrayList<WComponent>>> getTerms() {
140 		Map<String, Duplet<String, ArrayList<WComponent>>> componentsByTerm = new HashMap<>();
141 		List<Duplet<String, ArrayList<WComponent>>> result = new ArrayList<>();
142 
143 		List<WComponent> childList = content.getComponentModel().getChildren();
144 
145 		if (childList != null) {
146 			for (int i = 0; i < childList.size(); i++) {
147 				WComponent child = childList.get(i);
148 				String term = child.getTag();
149 
150 				Duplet<String, ArrayList<WComponent>> termComponents = componentsByTerm.get(term);
151 
152 				if (termComponents == null) {
153 					termComponents = new Duplet<>(term,
154 							new ArrayList<WComponent>());
155 					componentsByTerm.put(term, termComponents);
156 					result.add(termComponents);
157 				}
158 
159 				termComponents.getSecond().add(child);
160 			}
161 		}
162 
163 		return result;
164 	}
165 
166 	/**
167 	 * Retrieves the components for the given term.
168 	 *
169 	 * @param term the term of the children to be retrieved.
170 	 * @return the child components for the given term, may be empty.
171 	 */
172 	private List<WComponent> getComponentsForTerm(final String term) {
173 		List<WComponent> childList = content.getComponentModel().getChildren();
174 		List<WComponent> result = new ArrayList<>();
175 
176 		if (childList != null) {
177 			for (int i = 0; i < childList.size(); i++) {
178 				WComponent child = childList.get(i);
179 
180 				if (term.equals(child.getTag())) {
181 					result.add(child);
182 				}
183 			}
184 		}
185 
186 		return result;
187 	}
