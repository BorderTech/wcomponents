3   /**
4    * AbstractTransientContextContainer ensures that all child components have transient state - ie will have their UI
5    * contexts reset after each request.
6    *
7    * The intended usage pattern is to set an attribute in this component's UIContext in the handleRequest(Request) method
8    * or from within an action. The attribute can then be used in the {@link #setupData()} method to read in the data (e.g.
9    * from the application cache) and pass it to the children.
10   *
11   * @author Yiannis Paschalidis
12   * @since 1.0.0
13   */
14  public abstract class AbstractTransientDataContainer extends AbstractMutableContainer {
15  
16  	/**
17  	 * After the component has been painted, the UIContexts for this component and all its children are cleared.
18  	 *
19  	 * @param renderContext the renderContext to send output to.
20  	 */
21  	@Override
22  	protected void afterPaint(final RenderContext renderContext) {
23  		super.afterPaint(renderContext);
24  
25  		final int size = getChildCount();
26  
27  		for (int i = 0; i < size; i++) {
28  			WComponent child = getChildAt(i);
29  			child.reset();
30  		}
31  	}
32  
33  	/**
34  	 * Prepares the component for painting. The {@link #setupData()} method is called before preparePaint is called for
35  	 * the child components.
36  	 *
37  	 * @param request the request being responded to.
38  	 */
39  	@Override
40  	public final void preparePaintComponent(final Request request) {
41  		setupData();
42  		super.preparePaintComponent(request);
43  	}
44  
45  	/**
46  	 * Subclasses must implement this method to ensure that all child components are set up correctly before they are
47  	 * painted.
48  	 */
49  	public abstract void setupData();
50  }
