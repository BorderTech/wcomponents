9   /**
10   * <p>
11   * The WAjaxControl links an AJAX trigger component with one or more AJAX target components. Creating an AJAX trigger
12   * will result in an AJAX request being made on the 'onChange' event of the trigger element. The implication of this is
13   * that an AJAX trigger component <em>should</em> have an Action attached (usually via setActionOnChanged).
14   * </p>
15   * <p>
16   * During an AJAX request the whole UI tree is serviced in the action phase but only the 'target' components of this
17   * control will be painted in the response.
18   * </p>
19   *
20   * @author Christina Harris
21   * @author Mark Reeves
22   * @since 1.0.0
23   */
24  public class WAjaxControl extends AbstractWComponent {
25

27  	 * The component that will trigger the AJAX request.
28  	 */

32  	 * Create a WAjaxControl with a specified trigger component.

34  	 * @param trigger the WComponent that will fire the AJAX request on change
35  	 */
36  	public WAjaxControl(final AjaxTrigger trigger) {
37  		this.trigger = trigger;

40  	/**
41  	 * Create a WAjaxControl with specified trigger and target components.
42  	 *
43  	 * @param trigger the WComponent that will fire the AJAX request on change
44  	 * @param target the WComponent to be re-painted as a result of the AJAX request
45  	 */
46  	public WAjaxControl(final AjaxTrigger trigger, final AjaxTarget target) {
47  		this(trigger);
48  
49  		List<AjaxTarget> targets = new ArrayList<>();
50  		targets.add(target);
51  		getComponentModel().targets = targets;
52  	}

58  	 * @param targets the WComponents to be re-painted in the AJAX response
59  	 */
60  	public WAjaxControl(final AjaxTrigger trigger, final AjaxTarget[] targets) {
61  		this(trigger);
62  
63  		List<AjaxTarget> targetList = new ArrayList<>();
64  		targetList.addAll(Arrays.asList(targets));
65  		getComponentModel().targets = targetList;
66  	}

69  	 * Create a WAjaxControl with specified trigger component and a List of target components.

99  	/**
100 	 * Add a list of target components that should be targets for this AJAX request.
101 	 *
102 	 * @param targets the components that will be repainted for the AJAX request
103 	 */
104 	public void addTargets(final List<? extends AjaxTarget> targets) {
105 		if (targets != null) {
106 			for (AjaxTarget target : targets) {
107 				this.addTarget(target);
108 			}
109 		}
110 	}

112 	/**
113 	 * Add a single target WComponent to this AJAX control.
114 	 *
115 	 * @param target a WComponent to be repainted
116 	 */
117 	public void addTarget(final AjaxTarget target) {
118 		AjaxControlModel model = getOrCreateComponentModel();
119 
120 		if (model.targets == null) {
121 			model.targets = new ArrayList<>();
122 		}
123 
124 		model.targets.add(target);
125 		MemoryUtil.checkSize(model.targets.size(), this.getClass().getSimpleName());
126 	}

157 	 * Get the indicator of whether a trigger may fire once or an unlimited number of times. If loadCount &gt; 0 the
158 	 * WAjaxControl may <strong>only fire once per page load</strong>. See
159 	 * <a href="https://github.com/BorderTech/wcomponents/issues/495">#495</a>.

161 	 * @return 1 if the trigger may only load once, otherwise -1
162 	 * @deprecated 1.2.0 use {@link #isLoadOnce()}.

168 	/**
169 	 * <p>
170 	 * Get the target WComponents that will be repainted as a consequence of the AJAX request.
171 	 * </p>
172 	 * <p>
173 	 * When the AJAX request is triggered only the target component(s) will be re-painted. An empty list is returned if
174 	 * no targets have been defined.
175 	 * </p>
176 	 *
177 	 * @return the target regions that are repainted in the AJAX response
178 	 */
179 	public List<AjaxTarget> getTargets() {
180 		List<AjaxTarget> targets = getComponentModel().targets;
181 
182 		if (targets == null) {
183 			return Collections.emptyList();
184 		}
185 
186 		return Collections.unmodifiableList(targets);
187 	}

189 	/**
190 	 * <p>
191 	 * Get the target WComponents as an array.
192 	 * </p>
193 	 * <p>
194 	 * When the AJAX request is triggered only the target component(s) will be re-painted. If no targets have been
195 	 * registered then an empty array is returned.
196 	 * </p>
197 	 *
198 	 * @return an array of AJAX target components
199 	 */
200 	public WComponent[] getTargetsArray() {
201 		List<AjaxTarget> targets = getTargets();
202 		return targets.toArray(new WComponent[targets.size()]);
203 	}

216 	/**
217 	 * <p>
218 	 * Set a delay period, in milliseconds, between the WAjaxControl being rendered in the view and it being
219 	 * <em>automatically</em> triggered. A WAjaxControl with a delay &gt; 0 will result in a request being made without
220 	 * a change to any {@link AjaxTrigger} component.
221 	 * </p>
222 	 * <p>
223 	 * The use of a delay may be useful for setting up a trigger which polls for changes in a part of a UI. See
224 	 * {@link com.github.bordertech.wcomponents.WAjaxPollingRegion}.
225 	 * </p>
226 	 * <p>
227 	 * If the trigger is part of a polling region (therefore it is itself updated and potentially re-triggers itself
228 	 * possibly many times) then the delay <strong>must not</strong> be less than 334.</p>
229 	 *
230 	 * @param delay the delay after page load before AJAX control triggered
231 	 */
232 	public void setDelay(final int delay) {
233 		getOrCreateComponentModel().delay = delay;
234 	}

236 	/**
237 	 * Override preparePaintComponent in order to register the components for the current request.
238 	 *
239 	 * @param request the request being responded to
240 	 */
241 	@Override
242 	protected void preparePaintComponent(final Request request) {
243 		super.preparePaintComponent(request);
244 
245 		List<AjaxTarget> targets = getTargets();
246 		if (targets != null && !targets.isEmpty()) {
247 			WComponent triggerComponent = trigger == null ? this : trigger;
248 
249 			// The trigger maybe in a different context
250 			UIContext triggerContext = WebUtilities.getContextForComponent(triggerComponent);
251 			UIContextHolder.pushContext(triggerContext);
252 
253 			// TODO The IDs of the targets are based on the Triggers Context. Not good for targets in repeaters
254 			try {
255 				List<String> targetIds = new ArrayList<>();
256 				for (AjaxTarget target : getTargets()) {
257 					targetIds.add(target.getId());
258 				}
259 				AjaxHelper.registerComponents(targetIds, triggerComponent.getId());
260 			} finally {
261 				UIContextHolder.popContext();
262 			}
263 		}
264 	}

266 	/**
267 	 * @return a String representation of this component
268 	 */
269 	@Override
270 	public String toString() {
271 		StringBuffer buf = new StringBuffer();
272 		List<AjaxTarget> targets = getTargets();
273 
274 		buf.append("trigger=").append(trigger == null ? "null" : trigger.getClass().getSimpleName());
275 		buf.append(", targets=[");
276 
277 		for (int i = 0; i < targets.size(); i++) {
278 			if (i > 0) {
279 				buf.append(", ");
280 			}
281 
282 			AjaxTarget target = targets.get(i);
283 			buf.append(target == null ? "null" : target.getClass().getSimpleName());
284 		}
285 
286 		buf.append(']');
287 
288 		return toString(buf.toString());
289 	}

318 	 * Holds the extrinsic state information of the component.

320 	 * @author Yiannis Paschalidis

322 	public static class AjaxControlModel extends ComponentModel {
323 
324 		/**
325 		 * The components that will be repainted by the AJAX request.
326 		 */
327 		private List<AjaxTarget> targets;
328 
329 		/**
330 		 * Specifies that the WAjaxControl may only fire once per load. If false then there is no limit on the number of
331 		 * times the trigger may fire.
332 		 */
333 		private boolean loadOnce = false;
334 
335 		/**
336 		 * Delay, in milliseconds, between the control loading in the view and an automatic request. If this is set, and
337 		 * set to a value greater than 0, the WAjaxControl will fire without <em>any</em> {@link AjaxTrigger} being
338 		 * changed/invoked. This can be used to set up a trigger which polls for changes.
339 		 */
340 		private int delay;
