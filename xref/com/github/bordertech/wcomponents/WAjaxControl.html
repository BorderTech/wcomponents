8   /**
9    * <p>
10   * The WAjaxControl links an ajax trigger component with one or more ajax target components. Creating an ajax trigger
11   * will result in an ajax request being made on the 'onChange' event of the trigger element. For an ajax request the
12   * whole UI tree is serviced in the action phase but only the 'target' components related by this control will be
13   * painted in the response. Repainting only small parts of the UI aims to improve client response times.
14   * </p>
15   * <p>
16   * On the client-side, an AJAX target region must have a single html element as its parent and this element <b>must</b>
17   * have an id. In the AJAX response the whole element is replaced by the response. Components which do not implement the
18   * {@link AjaxTarget} interface can be wrapped in a {@link com.github.bordertech.wcomponents.WPanel}.
19   * </p>
20   *
21   * @author Christina Harris
22   * @since 1.0.0
23   */
24  public class WAjaxControl extends AbstractWComponent {

27  	 * The component, that onChange, will trigger the ajax request.

100 	/**
101 	 * Add target regions that should be repainted for this ajax request.
102 	 *
103 	 * @param targets The components that will be repainted for the ajax request.
104 	 */
105 	public void addTargets(final List<? extends AjaxTarget> targets) {
106 		if (targets != null) {
107 			AjaxControlModel model = getOrCreateComponentModel();
108 
109 			if (model.targets == null) {
110 				model.targets = new ArrayList<>();
111 			}
112 
113 			model.targets.addAll(targets);
114 		}
115 	}
116 
117 	/**
118 	 * Add a WComponent that will be repainted after the ajax request has been handled.
119 	 *
120 	 * @param target A WComponent to be repainted.
121 	 */
122 	public void addTarget(final AjaxTarget target) {
123 		AjaxControlModel model = getOrCreateComponentModel();
124 
125 		if (model.targets == null) {
126 			model.targets = new ArrayList<>();
127 		}
128 
129 		model.targets.add(target);
130 	}

168 	/**
169 	 * Get the target WComponents that will be repainted as a consequence of the AJAX request. An empty list is returned
170 	 * if no targets have been defined.
171 	 *
172 	 * @return The target regions that are repainted in the ajax response.

174 	public List<AjaxTarget> getTargets() {
175 		List<AjaxTarget> targets = getComponentModel().targets;
176

208 	/**
209 	 * Override preparePaintComponent in order to register the components for the current request.
210 	 *
211 	 * @param request the request being responded to.
212 	 */
213 	@Override
214 	protected void preparePaintComponent(final Request request) {
215 		super.preparePaintComponent(request);
216 
217 		List<AjaxTarget> targets = getTargets();
218 		if (targets != null && !targets.isEmpty()) {
219 			WComponent triggerComponent = trigger == null ? this : trigger;
220 
221 			UIContext triggerContext = WebUtilities.getPrimaryContext(UIContextHolder.getCurrent(),
222 					triggerComponent);
223 			UIContextHolder.pushContext(triggerContext);
224 
225 			try {
226 				List<String> targetIds = new ArrayList<>();
227 				for (AjaxTarget target : getTargets()) {
228 					targetIds.add(target.getId());
229 				}
230 				AjaxHelper.registerComponents(targetIds, request, triggerComponent.getId());
231 			} finally {
232 				UIContextHolder.popContext();
233 			}
234 
235 		}
236 	}

238 	/**
239 	 * @return a String representation of this component, for debugging purposes.
240 	 */
241 	@Override
242 	public String toString() {
243 		StringBuffer buf = new StringBuffer();
244 		List<AjaxTarget> targets = getTargets();
245 
246 		buf.append("trigger=").append(trigger == null ? "null" : trigger.getClass().getSimpleName());
247 		buf.append(", targets=[");
248 
249 		for (int i = 0; i < targets.size(); i++) {
250 			if (i > 0) {
251 				buf.append(", ");
252 			}
253 
254 			AjaxTarget target = targets.get(i);
255 			buf.append(target == null ? "null" : target.getClass().getSimpleName());
256 		}
257 
258 		buf.append(']');
259 
260 		return toString(buf.toString());
261 	}
