3   /**
4    * This interface is used to mark components which hold a name context.
5    *
6    * @author Jonathan Austin
7    * @since 1.0.0
8    */
9   public interface NamingContextable extends Container {
10  
11  	/**
12  	 * Allow components that implement {@link NamingContextable} to selectively disable prepending their ID to their
13  	 * descendent's IDs by breaking the prepending logic into a seperately callable method.
14  	 * <p>
15  	 * See {@link #getId()} for usage.
16  	 * </p>
17  	 * <p>
18  	 * By default, this method will call through to {@link #getId()} and return the result.
19  	 * </p>
20  	 *
21  	 * @return by default, return getId().
22  	 */
23  	String getNamingContextId();
24  
25  	/**
26  	 * A naming context is only considered active if an Id name has been set.
27  	 *
28  	 * @return true if active naming context.
29  	 */
30  	boolean isNamingContext();
31  }
