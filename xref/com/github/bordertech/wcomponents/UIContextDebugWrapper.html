3   import java.util.HashSet;
4   import java.util.Iterator;
5   import java.util.Map;
6   import java.util.Set;
7   import java.util.TreeMap;

9   /**
10   * This class can be used to dump the contents of a UIContext object for debugging purposes.
11   *
12   * @author Martin_Schevchenko
13   * @since 1.0.0
14   */
15  public class UIContextDebugWrapper {
16  
17  	/**
18  	 * The UIContext instance being wrapped.
19  	 */
20  	private final UIContext uic;
21  
22  	/**
23  	 * Creates a UIContextDebugWrapper.
24  	 *
25  	 * @param uic the object to log.
26  	 */
27  	public UIContextDebugWrapper(final UIContext uic) {
28  		this.uic = uic;
29  	}
30  
31  	/**
32  	 * @return the wrapped UIContext.
33  	 */
34  	protected UIContext getUIContext() {
35  		return uic;
36  	}
37  
38  	/**
39  	 * Tally up the number of WComponents storing data in the session grouped by object type. Also tally up how many are
40  	 * storing data unnecessarily. Also tally how many different top components are referenced by the components stored
41  	 * in the UIContext.
42  	 *
43  	 * @return the debugging information.
44  	 */
45  	@Override
46  	public String toString() {
47  		StringBuilder buf = new StringBuilder();
48  		UIContext cuic = getUIContext();
49  		UIContextHolder.pushContext(cuic);
50  
51  		try {
52  			Set components = cuic.getComponents();
53  
54  			// Variables that will contain the stats collected.
55  			GroupData dataOverall = new GroupData();
56  			Map<String, GroupData> tallyByClass = new TreeMap<>();
57  			Set<WComponent> topComponents = new HashSet<>();
58  
59  			for (Iterator iter = components.iterator(); iter.hasNext();) {
60  				WComponent comp = (WComponent) iter.next();
61  				String className = comp.getClass().getName();
62  
63  				GroupData dataForClass = tallyByClass.get(className);
64  
65  				if (dataForClass == null) {
66  					dataForClass = new GroupData();
67  					tallyByClass.put(className, dataForClass);
68  				}
69  
70  				dataForClass.total++;
71  				dataOverall.total++;
72  
73  				if (comp.isDefaultState()) {
74  					dataForClass.unnecessaryCount++;
75  					dataOverall.unnecessaryCount++;
76  				}
77  
78  				WComponent top = WebUtilities.getTop(comp);
79  				topComponents.add(top);
80  			}
81  
82  			buf.append("WComponent session usage overall:\n ");
83  			buf.append(dataOverall.total).append(" WComponent(s) storing data in the session.\n");
84  
85  			if (dataOverall.total > 0) {
86  				buf.append("[").append(dataOverall.unnecessaryCount).append(
87  						"] WComponent(s) in default state. WComponents in default state do not need to store data in the session.\n");
88  				buf.append(" Therefore, only ").append(
89  						dataOverall.total - dataOverall.unnecessaryCount).append(
90  								" WComponent(s) actually need(s) to store data in the session.\n");
91  
92  				buf.append("WComponent session usage by class:\n");
93  
94  				for (Map.Entry<String, GroupData> entry : tallyByClass.entrySet()) {
95  					String className = entry.getKey();
96  					GroupData dataForClass = entry.getValue();
97  					buf.append(' ').append(dataForClass.total).append(" [").append(
98  							dataForClass.unnecessaryCount).append("] ").append(className).append(
99  							'\n');
100 				}
101 
102 				buf.append("Found ").append(topComponents.size()).append(" top component(s).\n");
103 			}
104 		} finally {
105 			UIContextHolder.popContext();
106 		}
107 
108 		return buf.toString();
109 	}
110 
111 	/**
112 	 * Holds statistics for a single WComponent class.
113 	 */
114 	private static final class GroupData {
115 
116 		/**
117 		 * The count of instances of the particular WComponent class with a component model in the context/session.
118 		 */
119 		private int total = 0;
120 
121 		/**
122 		 * The count of instances of the particular WComponent class with a component model in the context/session which
123 		 * doesn't need to be there.
124 		 */
125 		private int unnecessaryCount = 0;
126 	}
127 }
