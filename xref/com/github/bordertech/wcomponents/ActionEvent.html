1   package com.github.bordertech.wcomponents;
2   
3   import java.util.EventObject;
4   
5   /**
6    * The object that indicates that a component action occured. It contains information about the event, and is passed to
7    * the Action object that will handle the event.
8    *
9    * @author Martin Shevchenko
10   * @since 1.0.0
11   */
12  public class ActionEvent extends EventObject {
13  
14  	/**
15  	 * A string that specifies a command associated with the event.
16  	 */
17  	private final String actionCommand;
18  	/**
19  	 * A data object that contains information related to the event.
20  	 */
21  	private final Object actionObject;
22  
23  	/**
24  	 * @param source the object that originated the event. (Normally a WButton)
25  	 * @param command a string that may specify a command associated with the event. (Normally the "actionCommand" from
26  	 * the source WButton)
27  	 */
28  	public ActionEvent(final Object source, final String command) {
29  		this(source, command, null);
30  	}
31  
32  	/**
33  	 *
34  	 * @param source the object that originated the event. (Normally a WButton)
35  	 * @param command a string that may specify a command associated with the event. (Normally the "actionCommand" from
36  	 * the source WButton)
37  	 * @param actionObject a data object that may contain information related to the event. (Normally the "actionObject"
38  	 * from the source WButton)
39  	 */
40  	public ActionEvent(final Object source, final String command, final Object actionObject) {
41  		super(source);
42  		this.actionCommand = command;
43  		this.actionObject = actionObject;
44  	}
45  
46  	/**
47  	 * @return The string that specifies a command associated with the event, if set
48  	 */
49  	public String getActionCommand() {
50  		return actionCommand;
51  	}
52  
53  	/**
54  	 * @return The data object that contains information related to the event, if set.
55  	 */
56  	public Object getActionObject() {
57  		return actionObject;
58  	}
59  }
