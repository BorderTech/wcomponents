24  /**

52  
53  		JPanel topPanel = new JPanel();
54  		topPanel.setLayout(new BorderLayout());
55  		topPanel.add(new JLabel("XML to validate:"), BorderLayout.NORTH);

57  				JScrollPane.HORIZONTAL_SCROLLBAR_NEVER);
58  		scrollPane.getViewport().add(in);
59  		topPanel.add(scrollPane, BorderLayout.CENTER);

61  		JButton validateButton = new JButton("Validate");
62  
63  		validateButton.addActionListener(new ActionListener() {
64  			@Override
65  			public void actionPerformed(final ActionEvent event) {
66  				validateXml();
67  			}
68  		});
69  
70  		JPanel bottomPanel = new JPanel();
71  		bottomPanel.setLayout(new BorderLayout());
72  		bottomPanel.add(validateButton, BorderLayout.NORTH);
73  		bottomPanel.add(new JScrollPane(out), BorderLayout.CENTER);

75  		getContentPane().add(topPanel, BorderLayout.CENTER);
76  		getContentPane().add(bottomPanel, BorderLayout.SOUTH);

84  	/**
85  	 * Validates the XML contained in the input text area and displays the errors (if any) in the output text area.
86  	 */
87  	private void validateXml() {
88  		String xhtml = in.getText();
89  
90  		try {
91  			assertSchemaMatch(xhtml);
92  			out.setText("Valid!");
93  			out.setCaretPosition(0);
94  		} catch (AssertionFailedError error) {
95  			out.setText(error.getMessage().replaceAll("cvc-complex-type.2.4.a:", "\n").trim());
96  			out.setCaretPosition(0);
97  		} catch (Exception e) {
98  			out.setText(e.getMessage());
99  		}
100 	}

102 	/**
103 	 * Asserts that the given xhtml matches the schema.
104 	 *
105 	 * @param xhtml the xhtmlto validate.
106 	 *
107 	 * @throws SAXException if there is a parsing error
108 	 */
109 	public void assertSchemaMatch(final String xhtml) throws SAXException {
110 		// Load the schema.
111 		Object schema = getClass().getResource("/schema/ui/v1/schema.xsd").toString();
112 
113 		// Validate the xhtml.
114 		Validator validator;
115 		StringReader reader = new StringReader(xhtml);
116 		validator = new Validator(reader);
117 		validator.useXMLSchema(true);
118 		validator.setJAXP12SchemaSource(schema);
119 
120 		validator.assertIsValid();
121 	}

123 	/**
124 	 * A namespace context that handles the "ui" namespace.
125 	 *
126 	 * @author Yiannis Paschalidis
127 	 */
128 	private static final class XmlLayoutTestNamespaceContext implements NamespaceContext {
129 
130 		/**
131 		 * The backing context, used for resolving namespaces which this context does not know about.
132 		 */
133 		private final NamespaceContext backing;
134 
135 		/**
136 		 * Creates a XmlLayoutTestNamespaceContext.
137 		 *
138 		 * @param backing the backing context, used for resolving unknown namespaces.
139 		 */
140 		XmlLayoutTestNamespaceContext(final NamespaceContext backing) {
141 			this.backing = backing;
142 		}
143 
144 		/**
145 		 * Returns the namespace uri for the given prefix.
146 		 *
147 		 * @param prefix the namespace prefix
148 		 * @return the namespace uri, or null if an unknown prefix was supplied.
149 		 */
150 		@Override
151 		public String getNamespaceURI(final String prefix) {
152 			if ("ui".equals(prefix)) {
153 				return "https://github.com/bordertech/wcomponents/namespace/ui/v1.0";
154 			} else if ("html".equals(prefix)) {
155 				return "http://www.w3.org/1999/xhtml";
156 			} else if (backing != null) {
157 				return backing.getNamespaceURI(prefix);
158 			}
159 
160 			return null;
161 		}
162 
163 		/**
164 		 * @return an iteration of the namespace prefixes in this contex.t
165 		 */
166 		@Override
167 		public Iterator<String> getPrefixes() {
168 			Set<String> prefixes = new HashSet<>(2);
169 			prefixes.add("ui");
170 			prefixes.add("html");
171 
172 			if (backing != null) {
173 				for (Iterator<?> i = backing.getPrefixes(); i.hasNext();) {
174 					prefixes.add((String) i.next());
175 				}
176 			}
177 
178 			return prefixes.iterator();
179 		}
180 	}

182 	/**
183 	 * Main entry method.
184 	 *
185 	 * @param args the program command-line arguments (ignored).
186 	 */
187 	public static void main(final String[] args) {
188 		// Need to register the "ui" prefix with XMLUnit so that we can use it in XPath expressions
189 		NamespaceContext context = XMLUnit.getXpathNamespaceContext();
190 		context = new XmlLayoutTestNamespaceContext(context);
191 		XMLUnit.setXpathNamespaceContext(context);
192 
193 		new ValidatorApp();
194 	}
