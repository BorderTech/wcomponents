3   import org.junit.runner.Description;
4   import org.junit.runner.Result;
5   import org.junit.runner.notification.RunListener;
6   
7   /**
8    * Listener to start and stop the web server before the unit tests.
9    *
10   * @author Jonathan Austin
11   */
12  public class ServerStartStopListener extends RunListener {
13  
14  	/**
15  	 * {@inheritDoc}
16  	 */
17  	@Override
18  	public void testRunStarted(final Description description) throws Exception {
19  		ServerCache.setInSuite(true);
20  		ServerCache.startServer();
21  	}
22  
23  	/**
24  	 * {@inheritDoc}
25  	 */
26  	@Override
27  	public void testRunFinished(final Result result) throws Exception {
28  		ServerCache.setInSuite(false);
29  		ServerCache.stopServer();
30  	}
31  
32  }
