30  	/**
31  	 * @return the available options for this control
32  	 */
33  	public List<WebElement> getOptions() {
34  		return findElementsImmediate(By.cssSelector(".wc-option"));
35  	}

37  	/**
38  	 * Get an option using an index.
39  	 *
40  	 * @param idx the index of the option to get
41  	 * @return the option as a WebElement
42  	 */
43  	public WebElement getOption(final int idx) {
44  		List<WebElement> options = getOptions();
45  		if (CollectionUtils.isEmpty(options)) {
46  			throw new SystemException("No options available");
47  		}
48  		return options.get(idx);
49  	}

51  	/**
52  	 * Find an option with a particular label.
53  	 *
54  	 * @param labelText the text content of the option's label
55  	 * @return the option
56  	 */
57  	public WebElement getOption(final String labelText) {
58  		List<WebElement> options = getOptions();
59  		if (CollectionUtils.isEmpty(options)) {
60  			throw new SystemException("No options available");
61  		}
62  		if (isReadOnly()) {
63  			for (WebElement o : options) {
64  				if (labelText.equalsIgnoreCase(o.getText())) {
65  					return o;
66  				}
67  			}
68  			throw new IllegalArgumentException("Could not find option identified by " + labelText);
69  		}
70  		SeleniumWComponentWebElement input = findElementImmediate(new ByLabel(labelText, false, true));
71  		return input.findElementImmediate(By.xpath(".."));
72  	}
73  
74  	/**
75  	 * Get the active control from an option.
76  	 *
77  	 * @param idx the option index
78  	 * @return the active HTML element, if any
79  	 */
80  	public WebElement getInput(final int idx) {
81  		if (isReadOnly()) {
82  			throw new SystemException("Component in a read-only state has no interactive controls.");
83  		}
84  		SeleniumWComponentWebElement option = (SeleniumWComponentWebElement) getOption(idx);
85  		return option.findElementImmediate(By.tagName(getOptionTag()));
86  	}

89  	 * Get the interactive HTML control with a given label. Only applies to components in an interactive state.
90  	 *
91  	 * @param labelText the text of the label for the option we are after
92  	 * @return the active HTML element, if any
93  	 */
94  	public WebElement getInput(final String labelText) {
95  		if (isReadOnly()) {
96  			throw new SystemException("Component in a read-only state has no interactive controls.");
97  		}
98  		return findElementImmediate(new ByLabel(labelText, false, true));
99  	}

101 	/**
102 	 * Get the input control belonging to a particular option in the group.
103 	 *
104 	 * @param option the option we are trying to interact with
105 	 * @return the input control of that option
106 	 */
107 	public WebElement getInput(final WebElement option) {
108 		if (isReadOnly()) {
109 			throw new SystemException("Components in a read-only state have no inputs.");
110 		}
111 		return SeleniumWComponentsUtil.findElementImmediateForElement(option, By.tagName(getOptionTag()));
112 	}

114 	/**
115 	 * Get the selected options. Note that a WCheckBoxSelect in a read only state contains only the selected options and
116 	 * a disabled component has no selected options.
117 	 *
118 	 * @return a list of selected options
119 	 */
120 	public List<WebElement> getSelected() {
121 		List<WebElement> options = getOptions();
122 		if (isReadOnly()) {
123 			return options;
124 		}
125 		if (!isEnabled()) {
126 			return new ArrayList<>();
127 		}
128 		List<WebElement> selected = new ArrayList<>();
129 		for (WebElement o : options) {
130 			if (getInput(o).isSelected()) {
131 				selected.add(o);
132 			}
133 		}
134 		return selected;
135 	}

137 	/**
138 	 * @param option the WebElement representing an option in the component.
139 	 * @return {@code true} if the input in option is selected or if option exists and the component is read-only.
140 	 */
141 	public boolean isSelected(final WebElement option) {
142 		if (isReadOnly()) {
143 			List<WebElement> options = getOptions();
144 			String optionText = option.getText();
145 
146 			for (WebElement o : options) {
147 				if (optionText.equals(o.getText())) {
148 					return true;
149 				}
150 			}
151 			return false;
152 		}
153 		if (!isEnabled(option)) {
154 			return false;
155 		}
156 		return getInput(option).isSelected();
157 	}
158 
159 	/**
160 	 * @param idx the option index
161 	 * @return {@code true} if the option is selected or, if the component is in a read-only state, if option merely
162 	 * exists
163 	 */
164 	public boolean isSelected(final int idx) {
165 		if (isReadOnly()) {
166 			return getOptions().size() > idx;
167 		}
168 		return isSelected(getOption(idx));
169 	}
170 
171 	/**
172 	 * @param labelText the label text for the option
173 	 * @return {@code true} if the option is selected or, if the component is in a read-only state, if the option exists
174 	 */
175 	public boolean isSelected(final String labelText) {
176 		if (isReadOnly()) {
177 			String path = ".//*[text()='" + labelText + "']";
178 			// if in a read only state the option is selected if it is in the UI as only selected options are output.
179 			try {
180 				findElementImmediate(By.xpath(path));
181 				return true;
182 			} catch (Exception e) {
183 				return false;
184 			}
185 		}
186 		return isSelected(getOption(labelText));
187 	}

195 		if (!isSelected(option)) {
196 			SeleniumCheckableGroupInputWebElement.this.clickNoWait(option);
197 		}

289 	/**
290 	 * Is a given option enabled? For most purposes this is moot as either all options are enabled or non are. Usually
291 	 * you will just need to use {@link #isEnabled()}.
292 	 *
293 	 * @param option the option to test
294 	 * @return {@code true} if the input element in option is enabled
295 	 */
296 	private boolean isEnabled(final WebElement option) {
297 		if (isReadOnly()) {
298 			return false;
299 		}
300 		return getInput(option).isEnabled();
301 	}
