13  /**
14   * Selenium WebElement class representing a WSelect.
15   *
16   * @author Joshua Barclay
17   * @author Mark Reeves
18   * @since 1.2.0
19   */
20  public class SeleniumWSelectWebElement extends SeleniumWComponentInputWebElement {
21  
22  	/**
23  	 * HTML attributes applied to the input element.
24  	 */
25  	private static final List<String> INPUT_ATTRIBUTES = Arrays.asList("disabled", "required");
26  
27  	/**
28  	 * The CSS Selector for a selected option.
29  	 */
30  	public static final String SELECTOR_SELECTED_OPTION = "option[selected]";
31  
32  	/**
33  	 * The CSS Selector for the first option.
34  	 */
35  	public static final String SELECTOR_FIRST_OPTION = "option:first-of-type";
36  
37  	/**
38  	 * The CSS Selector for the last option.
39  	 */
40  	public static final String SELECTOR_LAST_OPTION = "option:last-of-type";
41  
42  	/**
43  	 * The CSS Selector for the Nth option.
44  	 */
45  	public static final String SELECTOR_NTH_OPTION = "option:nth-child(%s)";
46  
47  	/**
48  	 * HTML tagName for select elements.
49  	 */
50  	public static final String SELECT_TAG = "select";
51  
52  	/**
53  	 * HTML tagName for multi-select components in a read-only state.
54  	 */
55  	public static final String RO_MULTI_SELECT_TAG = "ul";
56  
57  	/**
58  	 * HTML tagName for the options in a multi-select component in a read-only state.
59  	 */
60  	public static final String RO_MULTI_OPTION_TAG = "li";
61  
62  	/**
63  	 * Default constructor for this element.
64  	 *
65  	 * @param element the WebElement
66  	 * @param driver the driver.
67  	 */
68  	public SeleniumWSelectWebElement(final WebElement element, final WebDriver driver) {
69  		super(element, driver);
70  
71  		final String tagName = element.getTagName();
72  		if (!(TOP_LEVEL_TAG.equals(tagName) || RO_MULTI_SELECT_TAG.equals(tagName))) {
73  			throw new IllegalArgumentException("Element is not the expected wrapper. tag=[" + tagName + "].");
74  		}
75  	}
76  
77  	/**
78  	 * @return the value of the selected option, or null if no selected option.
79  	 */
80  	@Override
81  	public String getValue() {
82  		WebElement selected = getSelectedOption();
83  		return selected == null ? "" : selected.getText();
84  	}
85  
86  	/**
87  	 * @return the selected option for this select, or null if none selected.
88  	 */
89  	public WebElement getSelectedOption() {
90  		if (isReadOnly()) {
91  			if (isMultiSelect()) {
92  				try {
93  					return findElementImmediate(By.tagName(RO_MULTI_OPTION_TAG));
94  				} catch (final NoSuchElementException nsee) {
95  					//There is not a selected element - return null.
96  					return null;
97  				}
98  			}
99  			String text = getText();
100 			if (Util.empty(text)) {
101 				return null;
102 			}
103 			return this;
104 		}
105 
106 		SeleniumWComponentWebElement input = getInputField();
107 		Select se = new Select(input);
108 		try {
109 			return se.getFirstSelectedOption();
110 			//return getInputField().findElement(By.cssSelector(SELECTOR_SELECTED_OPTION));
111 		} catch (final NoSuchElementException nsee) {
112 			//There is not a selected element - return null.
113 			return null;
114 		}
115 	}

117 	/**
118 	 * @return the last option for this select.
119 	 */
120 	public SeleniumWComponentWebElement getFirstOption() {
121 		return getInputField().findElementImmediate(By.cssSelector(SELECTOR_FIRST_OPTION));
122 	}
123 
124 	/**
125 	 * @return the last option for this select.
126 	 */
127 	public SeleniumWComponentWebElement getLastOption() {
128 		return getInputField().findElementImmediate(By.cssSelector(SELECTOR_LAST_OPTION));
129 	}
130 
131 	/**
132 	 * @return the Nth option for this select.
133 	 * @param optionNumber - the
134 	 */
135 	public SeleniumWComponentWebElement getNthOption(final int optionNumber) {
136 		return getInputField().findElementImmediate(By.cssSelector(String.format(SELECTOR_NTH_OPTION, optionNumber)));
137 	}
138 
139 	/**
140 	 * Some attributes are applied to the wrapper, some to the input. This override sorts out which is which.
141 	 *
142 	 * @param name the name of the attribute to find
143 	 * @return the value of the attribute
144 	 */
145 	@Override
146 	public String getAttribute(final String name) {
147 		if (INPUT_ATTRIBUTES.contains(name)) {
148 			if (isReadOnly()) {
149 				return null;
150 			}
151 			return getInputField().getAttribute(name);
152 		}
153 		return super.getAttribute(name);
154 	}
155 
156 	@Override
157 	public SeleniumWComponentWebElement getInputField() {
158 		if (isReadOnly()) {
159 			return null;
160 		}
161 		return findElementImmediate(By.tagName(SELECT_TAG));
162 	}
163 
164 	/**
165 	 * Is the component a multi-select list?
166 	 *
167 	 * @return {@code true} if the component is a multi-select
168 	 */
169 	public boolean isMultiSelect() {
170 		if (isReadOnly()) {
171 			return RO_MULTI_SELECT_TAG.equals(getTagName());
172 		}
173 		return "multiple".equals(getInputField().getAttribute("multiple"));
174 	}
175 
176 	/**
177 	 *
178 	 * @return all options
179 	 */
180 	public List<WebElement> getOptions() {
181 		if (isReadOnly()) {
182 			return findElementsImmediate(By.tagName(RO_MULTI_OPTION_TAG));
183 		}
184 		WebElement input = getInputField();
185 		Select se = new Select(input);
186 		return se.getOptions();
187 	}
188 
189 	/**
190 	 *
191 	 * @return all selected options
192 	 */
193 	public List<WebElement> getSelectedOptions() {
194 		if (isReadOnly()) {
195 			return getOptions();
196 		}
197 		WebElement input = getInputField();
198 		Select se = new Select(input);
199 		return se.getAllSelectedOptions();
200 	}
201 
202 	/**
203 	 * Get an option based on its visible text.
204 	 *
205 	 * @param optionText the text of the option we are trying to find
206 	 * @return the option
207 	 */
208 	public WebElement getOption(final String optionText) {
209 		if (optionText == null) {
210 			throw new SystemException("option text must not be null");
211 		}
212 
213 		if (isReadOnly()) {
214 			if (isMultiSelect()) {
215 				for (WebElement we : getOptions()) {
216 					if (optionText.equals(we.getText())) {
217 						return we;
218 					}
219 				}
220 				throw new NoSuchElementException("No option with text provided");
221 			}
222 			if (optionText.equals(getText())) {
223 				return this;
224 			}
225 			throw new NoSuchElementException("No option with text provided");
226 		}
227 
228 		for (WebElement we : getOptions()) {
229 			if (optionText.equals(we.getText())) {
230 				return we;
231 			}
232 		}
233 		throw new NoSuchElementException("No option with text provided");
234 	}
