1   package com.github.bordertech.wcomponents.test.selenium.element;
2   
3   import com.github.bordertech.wcomponents.util.Util;
4   import org.openqa.selenium.WebDriver;
5   import org.openqa.selenium.WebElement;
6   
7   /**
8    * Group inputs include WCheckBoxSelect, WMultiTextField, WMultiDropdown, WMultiSelectPair and WRadioButtonSelect.
9    * @author Mark Reeves
10   */
11  public abstract class SeleniumGroupInputWebElement extends SeleniumWComponentWebElement {
12  
13  	/**
14  	 * All group inputs are wrapped in a HTML fieldset element when in an interactive state.
15  	 */
16  	public static final String WRAPPER_ELEMENT = "fieldset";
17  
18  	/**
19  	 * Create a SeleniumWComponentGroupInputWebElement instance.
20  	 * @param element the base WebElement
21  	 * @param driver the current Selenium driver
22  	 */
23  	public SeleniumGroupInputWebElement(final WebElement element, final WebDriver driver) {
24  		super(element, driver);
25  		String tagName = element.getTagName();
26  		if (!WRAPPER_ELEMENT.equalsIgnoreCase(tagName) && !isReadOnly()) {
27  			throw new IllegalArgumentException("Did not find expected tagName, expected " + WRAPPER_ELEMENT + " but found " + tagName);
28  		}
29  		if (!isCorrectReadOnly()) {
30  			throw new IllegalArgumentException("Did not find expected read-only component.");
31  		}
32  	}
33  
34  	/**
35  	 * A constructor helper to determine if an instance is in a readOnly state and if so to ensure that it is the correct component type.
36  	 * @return {@code true} unless the component is in a read only state and does not have the expected component name attribute.
37  	 */
38  	private boolean isCorrectReadOnly() {
39  		if (isReadOnly()) {
40  			String ro = getRoComponent();
41  			if (!getROComponentName().equals(ro)) {
42  				return false;
43  			}
44  		}
45  		return true;
46  	}
47  
48  	/**
49  	 * @return {@code true} if the component is in a read only state.
50  	 */
51  	public final boolean isReadOnly() {
52  		return !Util.empty(getAttribute("data-wc-component"));
53  	}
54  
55  	/**
56  	 * @return {@code true} if the input is mandatory
57  	 */
58  	public boolean isMandatory() {
59  		if (!isEnabled()) {
60  			return false;
61  		}
62  		String className = getAttribute("class");
63  		if (className == null) {
64  			return false;
65  		}
66  		return className.contains(SeleniumWComponentWebProperties.CLASS_REQUIRED.toString());
67  	}
68  
69  	/**
70  	 * @return the value of the read-only signifier. Used only for sub classes to make sure they have the correct element.
71  	 */
72  	private String getRoComponent() {
73  		if (!isReadOnly()) {
74  			return null;
75  		}
76  		return getAttribute(SeleniumWComponentWebProperties.ATTRIBUTE_WRAPPED_INPUT_TYPE.toString());
77  	}
78  
79  	/**
80  	 * @return the HTML tagname of the active element in an option. One of input, select or textarea.
81  	 */
82  	String getOptionTag() {
83  		return "input";
84  	}
85  
86  	/**
87  	 * @return the value of the read-only indicator attribute for a given sub-class
88  	 */
89  	abstract String getROComponentName();
90  
91  }
