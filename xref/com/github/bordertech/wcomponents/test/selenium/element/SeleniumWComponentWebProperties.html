3   /**
4    * An enum of commonly applied HTML attribute values for use in WComponents
5    * Selenium testing.
6    *
7    * @author Mark Reeves
8    * @since 1.2.3
9    */
10  public enum SeleniumWComponentWebProperties {
11  	/**
12  	 * The HTML attribute which holds the minLength property of an Input
13  	 * WComponent.
14  	 *//**
15  	 * The HTML attribute which holds the minLength property of an Input
16  	 * WComponent.
17  	 */
18  	ATTRIBUTE_MIN_LENGTH("data-wc-minlength"),
19  	/**
20  	 * HTML attribute which holds the accessibleText property of a
21  	 * WComponent.
22  	 */
23  	ATTRIBUTE_ACCESSIBLE_TEXT("aria-label"),
24  	/**
25  	 * HTML attribute which holds the ID of the element that a label is 'for'.
26  	 */
27  	ATTRIBUTE_LABEL_FOR("for"),
28  	/**
29  	 * HTML attribute which holds the id of the element that a WLabel is "for" when the labelled WComponent is not a labellable element.
30  	 */
31  	ATTRIBUTE_LABEL_FAUX_FOR("data-wc-for"),
32  	/**
33  	 * HTML attribute which holds the ID of the element that a label is 'for' when the label is for a read-only element.
34  	 */
35  	ATTRIBUTE_LABEL_FOR_READ_ONLY("data-wc-rofor"),
36  	/**
37  	 * HTML attribute which holds an Input WComponent's name if the
38  	 * component is not a native HTML form-bound element.
39  	 */
40  	ATTRIBUTE_WRAPPED_NAME("data-wc-name"),
41  	/**
42  	 * HTML attribute which holds an Input WComponent's value if the
43  	 * component is not a native HTML form-bound element.
44  	 */
45  	ATTRIBUTE_WRAPPED_VALUE("data-wc-value"),
46  	/**
47  	 * HTML attribute which holds the type of input if the component is not
48  	 * a native HTML form-bound element.
49  	 */
50  	ATTRIBUTE_WRAPPED_INPUT_TYPE("data-wc-component"),
51  	/**
52  	 * HTML attribute which holds the input's value when it is native HTML.
53  	 */
54  	ATTRIBUTE_HTML_VALUE("value"),
55  	/**
56  	 * The HTML attribute which holds the maxlength property of an Input WComponent.
57  	 */
58  	ATTRIBUTE_MAX_LENGTH("maxlength"),
59  	/**
60  	 * Common HTML class attribute value applied to an Input WComponent in a
61  	 * read-only state.
62  	 */
63  	CLASS_READ_ONLY("wc_ro"),
64  	/**
65  	 * Common HTML class attribute value applied to a checkable Input WComponent in a
66  	 * read-only state when it is selected.
67  	 */
68  	CLASS_READONLY_CHECKED("wc_ro_sel"),
69  	/**
70  	 * HTML class attribute value applied to fieldsets in which at least one field is mandatory.
71  	 */
72  	CLASS_REQUIRED("wc_req"),
73  	/**
74  	 * ID suffix used on wrapped inputs.
75  	 */
76  	ID_SUFFIX("_input");
77  
78  	/**
79  	 * The attribute itemValue for items.
80  	 */
81  	private final String itemValue;
82  
83  	/**
84  	 * Instantiate an item in the enum.
85  	 *
86  	 * @param value the attribute itemValue to apply to the item
87  	 */
88  	SeleniumWComponentWebProperties(final String value) {
89  		this.itemValue = value;
90  	}
91  
92  	/**
93  	 * @return the item value of the enum item.
94  	 */
95  	@Override
96  	public String toString() {
97  		return itemValue;
98  	}
99  }
