14  public class SeleniumWRadioButtonSelectWebElement extends SeleniumCheckableGroupInputWebElement {
15  
16  	/**
17  	 * The value of the read-only indicator for WRadioButtonSelect.
18  	 */
19  	private static final String RO_COMPONENT = "radiobuttonselect";
20  
21  	/**
22  	 * Create a SeleniumWComponentWRadioButtonSelectWebElement.
23  	 *
24  	 * @param element the backing WebElement
25  	 * @param driver the backing Selenium Driver
26  	 */
27  	public SeleniumWRadioButtonSelectWebElement(final WebElement element, final WebDriver driver) {
28  		super(element, driver);
29  	}
30  
31  	/**
32  	 * Get the radio button at a given option index. Only applies to WRadioButtonSelect in an interactive state.
33  	 *
34  	 * @param idx the option's index
35  	 * @return a HTML input element in the radio button state.
36  	 */
37  	public WebElement getRadioButton(final int idx) {
38  		if (isReadOnly()) {
39  			throw new SystemException("WRadioButtonSelect in a read-only state has no radio buttons.");
40  		}
41  		WebElement option = getOption(idx);
42  		By by = By.tagName(getOptionTag());
43  		return SeleniumWComponentsUtil.findElementImmediateForElement(option, by);
44  	}
45  
46  	@Override
47  	final String getROComponentName() {
48  		return RO_COMPONENT;
49  	}
50  }
