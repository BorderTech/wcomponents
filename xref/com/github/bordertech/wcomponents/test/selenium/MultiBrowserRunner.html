23  /**
24   * MultiBrowserRunner is a jUnit Suite which will run a single {@link WComponentSeleniumTestCase} using multiple
25   * browsers. It should only be used as follows:
26   *
27   * <pre>
28   *   &#64;RunWith(MultiBrowserRunner.class)
29   *   public class MySeleniumTest extends WComponentSeleniumTestCase
30   *   {
31   *      // your test code here
32   *   }
33   * </pre>
34   *
35   * @author Yiannis Paschalidis
36   * @since 1.0.0
37   */
38  public class MultiBrowserRunner extends Suite {
39  
40  	/**
41  	 * The runners to use for the "suite" - one for each browser.
42  	 */
43  	private final List<Runner> runners = new ArrayList<>();
44  
45  	/**
46  	 * Only called reflectively. Do not use programmatically.
47  	 *
48  	 * @param clazz the test case to run.
49  	 * @throws InitializationError if there is an error.
50  	 */
51  	public MultiBrowserRunner(final Class<?> clazz) throws InitializationError {
52  		super(clazz, Collections.<Runner>emptyList());
53  
54  		final String testClassName = getTestClass().getName();
55  
56  		String[] drivers = ConfigurationProperties.getTestSeleniumMultiBrowserDrivers(testClassName);
57  
58  		//Configuration error - no drivers defined.
59  		if (ArrayUtils.isEmpty(drivers)) {
60  			throw new SystemException("Cannot run the MultiBrowserRunner without drivers defined in default param ["
61  					+ ConfigurationProperties.TEST_SELENIUM_MULTI_BROWSER_DRIVERS + "] or test-specific param ["
62  					+ ConfigurationProperties.TEST_SELENIUM_MULTI_BROWSER_DRIVERS + "." + testClassName + "]");
63  		}
64  
65  		boolean runParallel = ConfigurationProperties.getTestSeleniumMultiBrowserDriverParallel();
66  
67  		for (String driverClassName : drivers) {
68  			try {
69  				Class<?> driverClass = Class.forName(driverClassName);
70  				if (!WebDriverType.class.isAssignableFrom(driverClass)) {
71  					throw new SystemException("parameter defined WebDriverType does not implement WebDriverType inteface. driverClass=["
72  							+ driverClass + "]");
73  				}
74  
75  				WebDriverType driverType = ((Class<WebDriverType>) driverClass).newInstance();
76  				//Reuse the driver between tests if not parallel.
77  				String driverId = runParallel ? UUID.randomUUID().toString() : null;
78  				runners.add(new TestClassRunnerForBrowser(getTestClass().getJavaClass(), driverType, driverId));
79  			} catch (ClassNotFoundException | InstantiationException | IllegalAccessException ex) {
80  				throw new SystemException("class parameter defined WebDriverType could not be instantiated. driverClassName=["
81  						+ driverClassName + "]", ex);
82  			}
83  		}
84  
85  		if (runParallel) {
86  			setScheduler(new ThreadPoolScheduler());
87  		}
88  	}
89  
90  	/**
91  	 * {@inheritDoc}
92  	 */
93  	@Override
94  	protected List<Runner> getChildren() {
95  		return runners;
96  	}

98  	/**
99  	 * This class will run all the tests for a given test class using a single browsers.
100 	 */
101 	private static final class TestClassRunnerForBrowser extends BlockJUnit4ClassRunner {
102 
103 		/**
104 		 * The unique id for this driver "Session".
105 		 */
106 		private final String driverId;
107 
108 		/**
109 		 * The WebDriverType to run the test.
110 		 */
111 		private final WebDriverType driverType;
112 
113 		/**
114 		 * Creates a TestClassRunnerForBrowser.
115 		 *
116 		 * @param type the test class to run.
117 		 * @param driverType the WebDriverType for this test.
118 		 * @param driverId the unique ID for this driver "session".
119 		 * @throws InitializationError if there is an error creating the runner.
120 		 */
121 		private TestClassRunnerForBrowser(final Class<?> type, final WebDriverType driverType, final String driverId
122 		) throws
123 				InitializationError {
124 			super(type);
125 			this.driverType = driverType;
126 			this.driverId = driverId;
127 		}
128 
129 		/**
130 		 * {@inheritDoc}
131 		 */
132 		@Override
133 		public Object createTest() throws Exception {
134 			Object test = super.createTest();
135 
136 			if (!(test instanceof WComponentSeleniumTestCase)) {
137 				throw new SystemException("MultiBrowserRunner cannot be used for test that does not extend WComponentSeleniumTestCase."
138 						+ " test class: " + test.getClass().getName());
139 			}
140 			((WComponentSeleniumTestCase) test).setDriver(driverType, driverId);
141 
142 			return test;
143 		}
144 
145 		/**
146 		 * {@inheritDoc}
147 		 */
148 		@Override
149 		protected String getName() {
150 			return super.getName() + getTestDetails();
151 		}
152 
153 		/**
154 		 * {@inheritDoc}
155 		 */
156 		@Override
157 		protected String testName(final FrameworkMethod method) {
158 			return String.format("%s[%s]", method.getName(), getTestDetails());
159 		}
160 
161 		@Override
162 		protected void validateConstructor(final List<Throwable> errors) {
163 			validateOnlyOneConstructor(errors);
164 		}
165 
166 		/**
167 		 * {@inheritDoc}
168 		 */
169 		@Override
170 		protected Statement classBlock(final RunNotifier notifier) {
171 			return childrenInvoker(notifier);
172 		}
173 
174 		/**
175 		 * @return the details of the test as a String to suffix on the name.
176 		 */
177 		private String getTestDetails() {
178 			return "." + driverType.getDriverTypeName() + ":" + driverId;
179 		}
180 	}

182 	/**
183 	 * This jUnit runner scheduler extension allows multiple threads.
184 	 */
185 	private static final class ThreadPoolScheduler implements RunnerScheduler {
186 
187 		/**
188 		 * The executor which will run the tests.
189 		 */
190 		private final ExecutorService executor;
191 
192 		/**
193 		 * Creates a ThreadPoolScheduler.
194 		 */
195 		private ThreadPoolScheduler() {
196 			executor = Executors.newCachedThreadPool();
197 		}
198 
199 		/**
200 		 * {@inheritDoc}
201 		 */
202 		@Override
203 		public void finished() {
204 			executor.shutdown();
205 
206 			try {
207 				executor.awaitTermination(10 * 60, TimeUnit.SECONDS); // 10 mins, overly generous
208 			} catch (InterruptedException exc) {
209 				throw new IllegalStateException("Test execution timed out", exc);
210 			}
211 		}
212 
213 		/**
214 		 * {@inheritDoc}
215 		 */
216 		@Override
217 		public void schedule(final Runnable childStatement) {
218 			executor.submit(childStatement);
219 		}
220 	}
