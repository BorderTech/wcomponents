1   package com.github.bordertech.wcomponents.test.selenium.driver;
2   
3   import com.github.bordertech.wcomponents.util.ConfigurationProperties;
4   import java.util.Map;
5   import java.util.Properties;
6   import org.openqa.selenium.WebDriver;
7   import org.openqa.selenium.remote.DesiredCapabilities;
8   
9   /**
10   * Superclass representing a possible WebDriver implementation to use for WComponents Selenium testing.
11   *
12   * @author Joshua Barclay
13   * @param <T> - the type of WebDriver returned.
14   * @since 1.2.0
15   */
16  public abstract class WebDriverType<T extends WebDriver> {
17  
18  	/**
19  	 * <p>
20  	 * Return a new instance of the WebDriver for this type.</p>
21  	 * <p>
22  	 * Must return a new instance each time - must not cache.</p>
23  	 *
24  	 * @return a new instance of the WebDriver.
25  	 */
26  	public abstract T getDriverImplementation();
27  
28  	/**
29  	 * @return a unique name String for this driver type.
30  	 */
31  	public abstract String getDriverTypeName();
32  
33  	/**
34  	 * @return the default DesiredCapabilities for this driver type.
35  	 */
36  	protected abstract DesiredCapabilities getDefaultDriverCapabilities();
37  
38  	/**
39  	 * @return the DesiredCapabilities configured from parameters.
40  	 */
41  	public DesiredCapabilities getCapabilities() {
42  
43  		DesiredCapabilities capabilities = getDefaultDriverCapabilities();
44  
45  		Properties props = ConfigurationProperties.getTestSeleniumDriverCapabilities(getDriverTypeName());
46  
47  		for (Map.Entry<Object, Object> property : props.entrySet()) {
48  			capabilities.setCapability((String) property.getKey(), property.getValue());
49  		}
50  
51  		return capabilities;
52  	}
53  
54  }
