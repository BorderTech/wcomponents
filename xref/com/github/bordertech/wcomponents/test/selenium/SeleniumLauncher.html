4   import com.github.bordertech.wcomponents.lde.PlainLauncher;
5   import com.github.bordertech.wcomponents.servlet.WServlet;
6   import java.util.HashMap;
7   import java.util.Map;
8   import javax.servlet.http.HttpServletRequest;
9   import javax.servlet.http.HttpServletResponse;
10  import org.apache.commons.lang.StringUtils;
11  import org.apache.commons.logging.Log;
12  import org.apache.commons.logging.LogFactory;
13  
14  /**
15   * <p>
16   * This class extends PlainLauncher to manage sessions and UIContext for Selenium.</p>
17   * <p>
18   * This launcher (or subclass) must be used to search using ByWComponent or ByWComponentPath.</p>
19   *
20   * @author Joshua Barclay
21   * @since 1.2.0
22   */
23  public class SeleniumLauncher extends PlainLauncher {
24  
25  	/**
26  	 * The logger instance for this class.
27  	 */
28  	private static final Log LOG = LogFactory.getLog(SeleniumLauncher.class);
29  
30  	/**
31  	 * List of all known session+context pairings.
32  	 */
33  	private static final Map<String, UIContext> CONTEXTS = new HashMap<>();
34  
35  	/**
36  	 * Get the UIContext for the given session ID.
37  	 *
38  	 * @param sessionId the session.
39  	 * @return the session's UIContext.
40  	 */
41  	public static UIContext getContextForSession(final String sessionId) {
42  		if (StringUtils.isBlank(sessionId)) {
43  			throw new IllegalArgumentException("sessionId must not be blank.");
44  		}
45  
46  		if (CONTEXTS.isEmpty()) {
47  			LOG.warn("No sessions exist in SeleniumLauncher. "
48  					+ "This function requires that the SeleniumLauncher (or subclass) is launched as the LDE launcher.");
49  		}
50  
51  		return CONTEXTS.get(sessionId);
52  	}
53  
54  	/**
55  	 * Remove any cached context for the given session. This method should be used when switching sessions.
56  	 *
57  	 * @param sessionId the sessionId to destroy the context.
58  	 */
59  	public static void destroyContextForSession(final String sessionId) {
60  		CONTEXTS.remove(sessionId);
61  	}
62  
63  	/**
64  	 * {@inheritDoc}.
65  	 */
66  	@Override
67  	protected WServletHelper createServletHelper(final HttpServletRequest httpServletRequest, final HttpServletResponse httpServletResponse) {
68  		return new SeleniumServletHelper(this, httpServletRequest, httpServletResponse);
69  	}
70  
71  	/**
72  	 * Selenium specific ServletHelper to ensure the UIContext is available in the JVM at all times.
73  	 */
74  	private static final class SeleniumServletHelper extends WServletHelper {
75  
76  		/**
77  		 * Construct a new helper.
78  		 *
79  		 * @param servlet the servlet.
80  		 * @param httpServletRequest the request.
81  		 * @param httpServletResponse the response.
82  		 */
83  		private SeleniumServletHelper(final WServlet servlet, final HttpServletRequest httpServletRequest,
84  				final HttpServletResponse httpServletResponse) {
85  			super(servlet, httpServletRequest, httpServletResponse);
86  		}
87  
88  		/**
89  		 * {@inheritDoc}.
90  		 */
91  		@Override
92  		protected void setUIContext(final UIContext uiContext) {
93  			super.setUIContext(uiContext);
94  
95  			String sessionId = getBackingRequest().getSession().getId();
96  			CONTEXTS.put(sessionId, uiContext);
97  		}
98  
99  	}
100 }
