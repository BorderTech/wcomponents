73  	/**
74  	 * @param httpServletRequest the request being processed
75  	 * @return an interceptor chain to use when servicing requests.
76  	 */
77  	@Override
78  	public InterceptorComponent createInterceptorChain(final Object httpServletRequest) {
79  		InterceptorComponent backing = super.createInterceptorChain(httpServletRequest);
80  		HttpServletRequest request = (HttpServletRequest) httpServletRequest;
81  
82  		// Only inject the script into normal round-trip requests.
83  		if (request.getParameter(DATA_LIST_PARAM_NAME) == null
84  				&& request.getParameter(AJAX_TRIGGER_PARAM_NAME) == null
85  				&& request.getParameter(TARGET_ID_PARAM_NAME) == null) {
86  			InterceptorComponent seleniumInterceptor = new ScriptInterceptor();
87  			seleniumInterceptor.setBackingComponent(backing);
88  
89  			return seleniumInterceptor;
90  		}
91  
92  		return backing;
93  	}

96  	 * Override in order to notify listeners when a request has been processed.
97  	 *
98  	 * @param req the {@link HttpServletRequest} object that contains the request the client made of the servlet
99  	 *
100 	 * @param res the {@link HttpServletResponse} object that contains the response the servlet returns to the client
101 	 *
102 	 * @throws ServletException a servlet exception
103 	 * @throws IOException an IO exception
104 	 */
105 	@Override
106 	public void service(final ServletRequest req, final ServletResponse res) throws ServletException, IOException {
107 		synchronized (LOCK) {
108 			super.service(req, res);
109 			incrementServiceCount();
110 		}
111 	}

155 	 * This method has been overridden to return the configured ui.
156 	 *
157 	 * @param httpServletRequest the httpServletRequest being handled.
158 	 * @return the ui to use.
159 	 */
160 	@Override
161 	public WComponent getUI(final Object httpServletRequest) {
162 		HttpServletRequest request = (HttpServletRequest) httpServletRequest;
163 		String pathInfo = request.getPathInfo();
164 
165 		if (pathInfo != null) {
166 			String testId = pathInfo.substring(pathInfo.lastIndexOf('/') + 1);
167 			UIContext uic = CONTEXT_MAP.get(testId);
168 
169 			if (uic == null) {
170 				throw new SystemException("Failed to get UIContext from map");
171 			}
172 
173 			return uic.getUI();
174 		}
175 
176 		return super.getUI(httpServletRequest);
177 	}

229 	/**
230 	 * <p>
231 	 * This servlet helper passes the request test case ID parameter around between requests, and also makes it
232 	 * available to the SeleniumTestServlet as a request attribute.</p>
233 	 *
234 	 * <p>
235 	 * The helper also ensures that we are using the test case's UIContext.</p>
236 	 */
237 	private static final class SeleniumServletHelper extends WServletHelper {
238 
239 		/**
240 		 * We need the WComponent request object created immediately so that we can place the attribute in the request.
241 		 * NOTE: The creation of this request object performs a destructive read on the servlet input stream when a file
242 		 * has been uploaded.
243 		 */
244 		private final Request request = createRequest();
245 		private final String testId;
246 
247 		/**
248 		 * Creates a SeleniumServletHelper.
249 		 *
250 		 * @param servlet the SeleniumTestServlet instance.
251 		 * @param httpServletRequest the servlet request being responded to.
252 		 * @param httpServletResponse the servlet response.
253 		 */
254 		private SeleniumServletHelper(final WServlet servlet, final HttpServletRequest httpServletRequest,
255 				final HttpServletResponse httpServletResponse) {
256 			super(servlet, httpServletRequest, httpServletResponse);
257 
258 			String pathInfo = httpServletRequest.getPathInfo();
259 
260 			if (pathInfo == null) {
261 				testId = null;
262 			} else {
263 				testId = pathInfo.substring(pathInfo.lastIndexOf('/') + 1);
264 			}
265 		}
266 
267 		/**
268 		 * We want to use the UIContext from the test case rather than creating and storing it in the HttpSession.
269 		 *
270 		 * @return the correct UIContext instance for the current test case.
271 		 */
272 		@Override
273 		protected UIContext getUIContext() {
274 			UIContext uic = super.getUIContext();
275 
276 			if (uic == null) {
277 				uic = CONTEXT_MAP.get(testId);
278 				getBackingRequest().getSession().setAttribute(getUiContextSessionKey(), uic);
279 			}
280 
281 			return uic;
282 		}
283 
284 		@Override
285 		protected Request getRequest() {
286 			return request;
287 		}
288 	}

290 	/**
291 	 * This interceptor is used to serve up the additional javascript required for Selenium tests.
292 	 */
293 	private static final class ScriptInterceptor extends InterceptorComponent {
294 		// A new interceptor chain is created for each request, so it is safe to store state here.
295 
296 		/**
297 		 * The requested script.
298 		 */
299 		private String scriptRequested = null;
300 
301 		/**
302 		 * {@inheritDoc}
303 		 */
304 		@Override
305 		public void serviceRequest(final Request request) {
306 			scriptRequested = request.getParameter("seleniumTestScript");
307 
308 			// Same basic sanity checking
309 			if (Util.empty(scriptRequested) || scriptRequested.indexOf('/') != -1 || !scriptRequested.endsWith(".js")) {
310 				scriptRequested = null;
311 			}
312 
313 			if (scriptRequested == null) {
314 				super.serviceRequest(request);
315 			}
316 		}
317 
318 		/**
319 		 * {@inheritDoc}
320 		 */
321 		@Override
322 		public void preparePaint(final Request request) {
323 			UIContext uic = UIContextHolder.getCurrent();
324 
325 			if (scriptRequested == null) {
326 				super.preparePaint(request);
327 
328 				StringBuffer path = new StringBuffer(uic.getEnvironment().getPostPath());
329 				path.append("?seleniumTestScript=seleniumBefore.js");
330 
331 				for (Map.Entry<String, String> entry : uic.getEnvironment().getHiddenParameters().entrySet()) {
332 					path.append('&');
333 					path.append(WebUtilities.escapeForUrl(entry.getKey()));
334 					path.append('=');
335 					path.append(WebUtilities.escapeForUrl(entry.getValue()));
336 				}
337 
338 				String pathStr = WebUtilities.encode(path.toString());
339 				uic.getHeaders().addHeadLine("<script type=\"text/javascript\" src=\"" + pathStr + "\"/>");
340 			} else {
341 				uic.getHeaders().setContentType("application/x-javascript");
342 			}
343 		}
344 
345 		/**
346 		 * {@inheritDoc}
347 		 */
348 		@Override
349 		public void paint(final RenderContext renderContext) {
350 			if (scriptRequested == null) {
351 				super.paint(renderContext);
352 			} else if (renderContext instanceof WebXmlRenderContext) {
353 				WebXmlRenderContext webContext = (WebXmlRenderContext) renderContext;
354 
355 				try {
356 					InputStream stream = getClass().getResourceAsStream(
357 							"/com/github/bordertech/wcomponents/test/selenium/" + scriptRequested);
358 					webContext.getWriter().write(new String(StreamUtil.getBytes(stream), "UTF-8"));
359 				} catch (Exception e) {
360 					LogFactory.getLog(getClass()).error("Failed to write selenium script", e);
361 				}
362 			} else {
363 				throw new SystemException("Unable to render to " + renderContext);
364 			}
365 		}
366 	}
