1   package com.github.bordertech.wcomponents.util;

3   /**
4    * Provides helpers to attach icons to components.
5    *
6    * @author Mark Reeves
7    * @since 1.2.1
8    */
9   public final class HtmlIconUtil {
10  
11  	/**
12  	 * Expose the HTML className to set an icon.
13  	 */
14  	public static final String CLASS_ICON = HtmlClassProperties.ICON.toString();
15  
16  	/**
17  	 * Expose the HTML className to set an icon before content.
18  	 */
19  	public static final String CLASS_ICON_BEFORE_CONTENT = HtmlClassProperties.ICON_BEFORE.toString();
20  
21  	/**
22  	 * Expose the HTML className to set an icon after content.
23  	 */
24  	public static final String CLASS_ICON_AFTER_CONTENT = HtmlClassProperties.ICON_AFTER.toString();
25  
26  	/**
27  	 * Provides access to position icons in components.
28  	 */
29  	public enum IconPosition {
30  		/**
31  		 * Put an icon before other content.
32  		 */
33  		BEFORE,
34  		/**
35  		 * Put an icon after other content.
36  		 */
37  		AFTER,
38  		/**
39  		 * Put an icon in an undefined position for use as only content.
40  		 */
41  		UNDEFINED
42  	}
43  
44  	/**
45  	 * Prevent instantiation.
46  	 */
47  	private HtmlIconUtil() {
48  	}
49  
50  	/**
51  	 * Pass in a string HTML class name and get the correct WComponents icon HTML classes.
52  	 * @param icon the icon-specific HTML className
53  	 * @return a HTML className string.
54  	 */
55  	public static String getIconClasses(final String icon) {
56  		return getIconClasses(icon, null);
57  	}
58  
59  	/**
60  	 * @param icon the HTML className representing an icon
61  	 * @param position where the icon should go in relation to the content of the component
62  	 * @return a HTML className String which will set up a component to contain an icon in the desired position.
63  	 */
64  	public static String getIconClasses(final String icon, final IconPosition position) {
65  		StringBuilder builder = new StringBuilder();
66  		if (position == null) {
67  			builder.append(CLASS_ICON);
68  		} else {
69  			switch (position) {
70  				case BEFORE:
71  					builder.append(CLASS_ICON_BEFORE_CONTENT);
72  					break;
73  				case AFTER:
74  					builder.append(CLASS_ICON_AFTER_CONTENT);
75  					break;
76  				default:
77  					builder.append(CLASS_ICON);
78  					break;
79  			}
80  		}
81  		if (!(icon == null || Util.empty(icon))) {
82  			builder.append(" ");
83  			builder.append(icon);
84  		}
85  		return builder.toString();
86  	}
87  }
