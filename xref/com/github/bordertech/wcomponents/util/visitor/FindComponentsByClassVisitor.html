3   import com.github.bordertech.wcomponents.ComponentWithContext;
4   import com.github.bordertech.wcomponents.UIContextHolder;
5   import com.github.bordertech.wcomponents.WComponent;
6   import com.github.bordertech.wcomponents.util.TreeUtil;
7   import com.github.bordertech.wcomponents.util.Util;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  /**
12   * A tree visitor implementation which finds components in the tree with a given class.
13   *
14   * @author Jonathan Austin
15   * @since 1.2.10
16   * @see TreeUtil
17   */
18  public class FindComponentsByClassVisitor extends AbstractVisitorWithResult<List<ComponentWithContext>> {
19  
20  	/**
21  	 * The root component being searched from.
22  	 */
23  	private final WComponent root;
24  
25  	/**
26  	 * The className the class name to search for.
27  	 */
28  	private final String className;
29  
30  	/**
31  	 * True if the root should be also be tested for a match, false if not.
32  	 */
33  	private final boolean includeRoot;
34  
35  	/**
36  	 * Find components of this class name including the root component being searched from.
37  	 *
38  	 * @param className the class name to search for. The package name may be omitted for convenience.
39  	 */
40  	public FindComponentsByClassVisitor(final String className) {
41  		this(null, className, true);
42  	}
43  
44  	/**
45  	 * Find components of this class name excluding the root component from the search.
46  	 *
47  	 * @param rootToIgnore the root component to ignore
48  	 * @param className the class name to search for. The package name may be omitted for convenience.
49  	 */
50  	public FindComponentsByClassVisitor(final WComponent rootToIgnore, final String className) {
51  		this(rootToIgnore, className, false);
52  	}
53  
54  	/**
55  	 * Creates a FindComponentByClassVisitor.
56  	 *
57  	 * @param root the root component being searched from.
58  	 * @param className the class name to search for. The package name may be omitted for convenience.
59  	 * @param includeRoot true if the root should be also be tested for a match, false if not.
60  	 */
61  	public FindComponentsByClassVisitor(final WComponent root, final String className,
62  			final boolean includeRoot) {
63  		this.setResult(new ArrayList<ComponentWithContext>());
64  		this.className = className;
65  		this.root = root;
66  		this.includeRoot = includeRoot;
67  	}
68  
69  	/**
70  	 * {@inheritDoc}
71  	 */
72  	@Override
73  	public VisitorResult visit(final WComponent comp) {
74  
75  		// Check if root component should be included (if ignore then just continue)
76  		if (!includeRoot && comp == root) {
77  			return VisitorResult.CONTINUE;
78  		}
79  
80  		boolean match = false;
81  		// Check inherited classes
82  		outer:
83  		for (Class<?> clazz = comp.getClass(); clazz != null; clazz = clazz.getSuperclass()) {
84  			// Check if inherited class name is a match
85  			if (classNamesMatch(clazz.getName(), className)) {
86  				match = true;
87  				break;
88  			}
89  			// Check interfaces
90  			for (Class intClazz : clazz.getInterfaces()) {
91  				if (classNamesMatch(intClazz.getName(), className)) {
92  					match = true;
93  					break outer;
94  				}
95  			}
96  		}
97  		if (match) {
98  			boolean cont = handleFoundMatch(new ComponentWithContext(comp, UIContextHolder.getCurrent()));
99  			return cont ? VisitorResult.CONTINUE : VisitorResult.ABORT;
100 		}
101 		return VisitorResult.CONTINUE;
102 	}
103 
104 	/**
105 	 * @param comp the matching component to class name
106 	 * @return true to continue processing
107 	 */
108 	protected boolean handleFoundMatch(final ComponentWithContext comp) {
109 		getResult().add(comp);
110 		return true;
111 	}
112 
113 	/**
114 	 * Tests two class names for equality, taking partical class names (no package given) into account.
115 	 *
116 	 * @param name1 the first class name to compare.
117 	 * @param name2 the second class name to compare.
118 	 * @return true if the class names match, false otherwise.
119 	 */
120 	protected boolean classNamesMatch(final String name1, final String name2) {
121 		if (name1 == null || name2 == null) {
122 			return false;
123 		}
124 		int index1 = name1.lastIndexOf('.');
125 		int index2 = name2.lastIndexOf('.');
126 		if (index1 == -1 && index2 != -1) {
127 			return Util.equals(name1, name2.substring(index2 + 1));
128 		} else if (index1 != -1 && index2 == -1) {
129 			return Util.equals(name1.substring(index1 + 1), name2);
130 		} else {
131 			return Util.equals(name1, name2);
132 		}
133 	}
134 
135 }
