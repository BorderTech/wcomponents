38  	/**
39  	 * Obtains a list of components which are visible in the given tree. Repeated components will be returned multiple
40  	 * times, one for each row which they are visible in.
41  	 *
42  	 * @param comp the root component to search from.
43  	 * @return a list of components which are visible in the given context.
44  	 */
45  	public static List<ComponentWithContext> collateVisibles(final WComponent comp) {
46  		final List<ComponentWithContext> list = new ArrayList<>();
47  
48  		WComponentTreeVisitor visitor = new WComponentTreeVisitor() {
49  			@Override
50  			public VisitorResult visit(final WComponent comp) {
51  				// In traversing the tree, special components like WInvisbleContainer, WRepeatRoot are still traversed
52  				// (so ignore them)
53  				if (comp.isVisible()) {
54  					list.add(new ComponentWithContext(comp, UIContextHolder.getCurrent()));
55  				}
56  				return VisitorResult.CONTINUE;
57  			}
58  		};
59  
60  		traverseVisible(comp, visitor);
61  
62  		return list;
63  	}

72  	public static WComponent getRoot(final UIContext uic, final WComponent comp) {
73  		UIContextHolder.pushContext(uic);
74  
75  		try {
76  			return WebUtilities.getTop(comp);
77  		} finally {
78  			UIContextHolder.popContext();
79  		}

227 			final boolean visibleOnly) {
228

231 			public VisitorResult visit(final WComponent comp) {
232 				VisitorResult result = super.visit(comp);
233 				if (result == VisitorResult.CONTINUE) {
234 					// Save closest UIC as processing tree
235 					setResult(new ComponentWithContext(comp, UIContextHolder.getCurrent()));
236 				}
237 				return result;
238 			}
239 		};
240 
241 		doTraverse(root, visibleOnly, visitor);
242 
243 		return visitor.getResult() == null ? null : visitor.getResult().getContext();

247 	 * Check if this ID is focusable.
248 	 * <p>
249 	 * Considered focusable if the component and all its ancestors are visible and not hidden.
250 	 * </p>
251 	 *
252 	 * @param root the root component to search from.
253 	 * @param id the id to search for.
254 	 * @return the component with context if it is focusable, otherwise null
255 	 */
256 	public static boolean isIdFocusable(final WComponent root, final String id) {
257 
258 		FindComponentByIdVisitor visitor = new FindComponentByIdVisitor(id) {
259 			@Override
260 			public VisitorResult visit(final WComponent comp) {
261 				VisitorResult result = super.visit(comp);
262 				// If hidden then abort branch
263 				if (result == VisitorResult.CONTINUE && comp.isHidden()) {
264 					return VisitorResult.ABORT_BRANCH;
265 				}
266 				return result;
267 			}
268 		};
269 
270 		// Only traverse visible
271 		doTraverse(root, true, visitor);
272 
273 		// Check if matching component is hidden
274 		ComponentWithContext result = visitor.getResult();
275 		return result == null ? false : !result.getComponent().isHidden();
276 	}

298 	/**
299 	 * Internal implementation of tree traversal method.
300 	 *
301 	 * @param node the node to traverse.
302 	 * @param visibleOnly if true, only visit visible components.
303 	 * @param visitor the visitor to notify as the tree is traversed.
304 	 * @return how the traversal should continue.
305 	 */
306 	private static VisitorResult doTraverse(final WComponent node, final boolean visibleOnly,
307 			final WComponentTreeVisitor visitor) {
308 		if (visibleOnly) {
309 			// Push through Invisible Containers
310 			// Certain components have their visibility altered to implement custom processing.
311 			if (node instanceof WInvisibleContainer) {
312 				WComponent parent = node.getParent();
313 				// If inside a CardManager, skip the InvisibleContainer and process the visible card.
314 				if (parent instanceof WCardManager) {
315 					WComponent visible = ((WCardManager) node.getParent()).getVisible();
316 					if (visible == null) {
317 						return VisitorResult.ABORT_BRANCH;
318 					}
319 					return doTraverse(visible, visibleOnly, visitor);
320 				} else if (parent instanceof WWindow) { // If inside a WWindow, only process if it is Active
321 					// Abort branch if WWindow is not in ACTIVE state
322 					if (((WWindow) parent).getState() != WWindow.ACTIVE_STATE) {
323 						return VisitorResult.ABORT_BRANCH;
324 					}
325 				}
326 			} else if (node instanceof WRepeatRoot) {
327 				// Let be processed.
328 			} else if (!node.isVisible()) {
329 				// For most components, we just need to see if they're marked as visible
330 				return VisitorResult.ABORT_BRANCH;
331 			}
332 		}
333 
334 		VisitorResult result = visitor.visit(node);
335 
336 		switch (result) {
337 
338 			case ABORT_BRANCH:
339 				// Continue processing, but not down this branch
340 				return VisitorResult.CONTINUE;
341 
342 			case CONTINUE:
343 				// Process repeater rows
344 				if (node instanceof WRepeater) {
345 					// Get parent repeater
346 					WRepeater repeater = (WRepeater) node;
347 					// Get row contexts
348 					List<UIContext> rowContextList = repeater.getRowContexts();
349 					WRepeatRoot repeatRoot = (WRepeatRoot) repeater.getRepeatedComponent().getParent();
350 
351 					for (UIContext rowContext : rowContextList) {
352 						UIContextHolder.pushContext(rowContext);
353 
354 						try {
355 							result = doTraverse(repeatRoot, visibleOnly, visitor);
356 						} finally {
357 							UIContextHolder.popContext();
358 						}
359 
360 						if (VisitorResult.ABORT.equals(result)) {
361 							return VisitorResult.ABORT;
362 						}
363 					}
364 				} else if (node instanceof Container) {
365 					Container container = (Container) node;
366 
367 					for (int i = 0; i < container.getChildCount(); i++) {
368 						result = doTraverse(container.getChildAt(i), visibleOnly, visitor);
369 
370 						if (VisitorResult.ABORT.equals(result)) {
371 							return VisitorResult.ABORT;
372 						}
373 					}
374 				}
375 
376 				return VisitorResult.CONTINUE;
377 
378 			default:
379 				// Abort entire traversal
380 				return VisitorResult.ABORT;
381 		}
382 	}

399 	/**
400 	 * Retrieves WComponents by their path in the WComponent tree.
401 	 * <p>
402 	 * Paths are specified using class names, starting from the furthest ancestor. To reduce the path lengths, class
403 	 * names do not need to be fully-qualified. The path does not need to explicitly state intermediate components
404 	 * between components, and may include an index suffix to select a particular instance of a component in e.g. a
405 	 * repeater or a set of fields. Some example paths are shown below.
406 	 * </p>
407 	 * Example paths.
408 	 * <dl>
409 	 * <dt><code>{ "MyComponent" }</code></dt>
410 	 * <dd>Matches the first instance of MyComponent.</dd>
411 	 * <dt><code>{ "MyComponent[0]" }</code></dt>
412 	 * <dd>Also matches the first instance of MyComponent.</dd>
413 	 * <dt><code>{ "MyComponent[1]" }</code></dt>
414 	 * <dd>Matches the second instance of MyComponent.</dd>
415 	 * <dt><code>{ "MyPanel", "MyComponent" }</code></dt>
416 	 * <dd>Matches the first instance of MyComponent which is nested anywhere under a MyPanel.</dd>
417 	 * <dt><code>{ "MyApp", "MyPanel", "MyComponent" }</code></dt>
418 	 * <dd>Matches the first instance of MyComponent, nested within a MyPanel, which is in turn nested somewhere within
419 	 * a MyApp.</dd>
420 	 * </dl>
421 	 *
422 	 * @param component the component to search from.
423 	 * @param path the path to the WComponent.
424 	 * @param visibleOnly visible only
425 	 * @return the component matching the given path, or null if not found.
426 	 */
427 	public static ComponentWithContext[] findWComponents(final WComponent component,
428 			final String[] path, final boolean visibleOnly) {
429 		UIContext uic = UIContextHolder.getCurrent();
430 		if (uic == null) {
431 			throw new IllegalStateException("No user context available.");
432 		}
433 		List<ComponentWithContext> matchAtLevel = new ArrayList<>();
434 		matchAtLevel.add(new ComponentWithContext(component, uic));
435 
436 		for (int i = 0; i < path.length; i++) {
437 			List<ComponentWithContext> matchAtLastLevel = matchAtLevel;
438 			matchAtLevel = new ArrayList<>();
439 
440 			for (ComponentWithContext comp : matchAtLastLevel) {
441 				String[] parts = path[i].trim().split("[\\[\\]]");
442 				String className = parts[0].trim();
443 				int index = parts.length == 2 ? Integer.parseInt(parts[1]) : -1;
444 
445 				FindComponentsByClassVisitor visitor = new FindComponentsByClassVisitor(comp.getComponent(),
446 						className, i == 0);
447 				UIContextHolder.pushContext(comp.getContext());
448 
449 				try {
450 					if (visibleOnly) {
451 						TreeUtil.traverseVisible(comp.getComponent(), visitor);
452 					} else {
453 						TreeUtil.traverse(comp.getComponent(), visitor);
454 					}
455 				} finally {
456 					UIContextHolder.popContext();
457 				}
458 
459 				if (index >= 0) {
460 					if (index < visitor.getResult().size()) {
461 						matchAtLevel.add(visitor.getResult().get(index));
462 					}
463 				} else {
464 					matchAtLevel.addAll(visitor.getResult());
465 				}
466 			}
467 		}
468 
469 		return matchAtLevel.toArray(new ComponentWithContext[matchAtLevel.size()]);
470 	}

504 	/**
505 	 * An implementation of WComponentTreeVisitor which can return a result.
506 	 *
507 	 * @param <T> the result type
508 	 * @deprecated Use {@link AbstractVisitorWithResult} instead
509 	 */
510 	public abstract static class AbstractTreeVisitorWithResult<T> extends AbstractVisitorWithResult<T> {
511 
512 		/**
513 		 * The current context name as processing the tree.
514 		 */
515 		private String currentContextName = null;
516 
517 		/**
518 		 * @param comp the current node in the tree
519 		 * @param id the id to locate in the tree
520 		 * @return null if continue or ABORT or ABORT_BRANCH
521 		 */
522 		protected VisitorResult checkCorrectNameContext(final WComponent comp, final String id) {
523 			// Simple check - if ids are a match, then continue
524 			String compId = comp.getId();
525 			if (compId.equals(id)) {
526 				return null;
527 			}
528 
529 			// Check if already in a context and it has changed. Dont want to start going down sibling branches to the
530 			// current context.
531 			if (currentContextName != null && !compId.startsWith(currentContextName)) {
532 				return VisitorResult.ABORT;
533 			}
534 
535 			// Check for Name Context
536 			if (WebUtilities.isActiveNamingContext(comp)) {
537 				String context = ((NamingContextable) comp).getNamingContextId();
538 				if (!Util.empty(context)) {
539 					context += WComponent.ID_CONTEXT_SEPERATOR;
540 				}
541 				// Only process branch if ID starts with the name context
542 				if (id.startsWith(context)) {
543 					currentContextName = context;
544 					return null;
545 				}
546 				// If target id doesn't start with the current component's id, they will never match
547 				return VisitorResult.ABORT_BRANCH;
548 			} else if (comp instanceof WRepeater && !id.startsWith(compId)) {
549 				// Check for a WRepeater and if we need to check the rows.
550 				// Only process the repeater rows if the ID matches its row contexts
551 				return VisitorResult.ABORT_BRANCH;
552 			}
553 			return null;
554 		}
