3   import com.github.bordertech.wcomponents.Message;
4   import com.github.bordertech.wcomponents.util.InternalMessages;
5   import com.github.bordertech.wcomponents.util.error.SystemFailureMapper;
6   import org.apache.commons.logging.Log;
7   import org.apache.commons.logging.LogFactory;
8   
9   /**
10   * A Simple {@link SystemFailureMapper} which just returns a hard coded message.
11   *
12   * @author Brian Kavanagh, 2009
13   */
14  public class DefaultSystemFailureMapper implements SystemFailureMapper {
15  
16  	/**
17  	 * Logging debugging interface object.
18  	 */
19  	private static final Log LOG = LogFactory.getLog(DefaultSystemFailureMapper.class);
20  
21  	/**
22  	 * This method converts a java Throwable into a "user friendly" error message.
23  	 *
24  	 * @param throwable the Throwable to convert
25  	 * @return A {@link Message} containing the hard coded description "The system is currently unavailable."
26  	 */
27  	@Override
28  	public Message toMessage(final Throwable throwable) {
29  		LOG.error("The system is currently unavailable", throwable);
30  		return new Message(Message.ERROR_MESSAGE, InternalMessages.DEFAULT_SYSTEM_ERROR);
31  	}
32  }
