1   package com.github.bordertech.wcomponents.util;
2   
3   import com.github.bordertech.wcomponents.WebUtilities;
4   import org.apache.commons.lang3.text.translate.AggregateTranslator;
5   import org.apache.commons.lang3.text.translate.CharSequenceTranslator;
6   import org.apache.commons.lang3.text.translate.EntityArrays;
7   import org.apache.commons.lang3.text.translate.LookupTranslator;
8   import org.apache.commons.lang3.text.translate.NumericEntityUnescaper;
9   
10  /**
11   * This utility class provides a mechanism to ensure HTML is XML valid. It does this by unescaping HTML entities but
12   * <strong>not</strong> unescaping valid XML entities: {@code &lt;}, {@code &gt;}, {@code &quot;}, {@code &apos;} and
13   * {@code &amp;}.
14   *
15   * @author Mark Reeves
16   * @since 1.2.0
17   */
18  public final class HtmlToXMLUtil {
19  
20  	/**
21  	 * The translator to unescape HTML entities and the (invalid) {@code &apos;} to valid XML. This is based on the
22  	 * apache-commons StringEscapeUtils UNESCAPE_HTML4 translator with the BASIC_UNESCAPE Entity array removed and the
23  	 * APOS_UNESCAPE Entity array added.
24  	 */
25  	public static final CharSequenceTranslator UNESCAPE_HTML_TO_XML
26  			= new AggregateTranslator(
27  					new LookupTranslator(EntityArrays.ISO8859_1_UNESCAPE()),
28  					new LookupTranslator(EntityArrays.HTML40_EXTENDED_UNESCAPE()),
29  					new LookupTranslator(EntityArrays.APOS_UNESCAPE()),
30  					new NumericEntityUnescaper());
31  
32  	/**
33  	 * Do not allow instantiation.
34  	 */
35  	private HtmlToXMLUtil() {
36  	}
37  
38  	/**
39  	 * Unescape HTML entities to safe XML.
40  	 *
41  	 * <p>
42  	 * Example</p>
43  	 * <pre>
44  	 * {@code StringEscapeHTMLToXMLUtil.unescapeToXML("&bull;&Dagger}"); // returns "•‡"}
45  	 * {@code StringEscapeHTMLToXMLUtil.unescapeToXML("&lt;p&gt;"); // returns "&lt;p&gt;" not "<p>"}
46  	 * </pre>
47  	 *
48  	 * @param input The String to unescape.
49  	 * @return the input with all HTML4 character entities unescaped except those which are also XML entities.
50  	 */
51  	public static String unescapeToXML(final String input) {
52  		if (Util.empty(input)) {
53  			return input;
54  		}
55  		// Check if input has encoded brackets
56  		String encoded = WebUtilities.doubleEncodeBrackets(input);
57  		String unescaped = UNESCAPE_HTML_TO_XML.translate(encoded);
58  		String decoded = WebUtilities.doubleDecodeBrackets(unescaped);
59  		return decoded;
60  	}
61  }
