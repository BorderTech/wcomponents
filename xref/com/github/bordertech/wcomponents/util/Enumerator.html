1   package com.github.bordertech.wcomponents.util;
2   
3   import java.util.Enumeration;
4   import java.util.Iterator;
5   import java.util.NoSuchElementException;
6   
7   /**
8    * Uitlity class to wraps an <code>Iterator</code>.
9    *
10   * @author <a href="mailto:shinsuke@yahoo.co.jp">Shinsuke Sugaya</a>
11   *
12   * @param <T> the type of object returned by the enumerator.
13   */
14  public class Enumerator<T> implements Enumeration<T> {
15  
16  	/**
17  	 * The <code>Iterator</code> over which the <code>Enumeration</code> represented by this class actually operates.
18  	 */
19  	private final Iterator<T> iterator;
20  
21  	/**
22  	 * Return an Enumeration over the values returned by the specified Iterator.
23  	 *
24  	 * @param iterator Iterator to be wrapped
25  	 */
26  	public Enumerator(final Iterator<T> iterator) {
27  		this.iterator = iterator;
28  	}
29  
30  	/**
31  	 * Tests if this enumeration contains more elements.
32  	 *
33  	 * @return <code>true</code> if and only if this enumeration object contains at least one more element to provide,
34  	 * <code>false</code> otherwise
35  	 */
36  	@Override
37  	public boolean hasMoreElements() {
38  		return iterator.hasNext();
39  	}
40  
41  	/**
42  	 * Returns the next element of this enumeration if this enumeration has at least one more element to provide.
43  	 *
44  	 * @return the next element of this enumeration
45  	 * @throws NoSuchElementException if no more elements exist
46  	 */
47  	@Override
48  	public T nextElement() {
49  		return iterator.next();
50  	}
51  }
