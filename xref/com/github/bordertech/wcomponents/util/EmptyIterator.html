3   import java.io.Serializable;
4   import java.util.Iterator;
5   import java.util.NoSuchElementException;
6   
7   /**
8    * An empty Iterator, that can be returned for an iterator on a null/empty collection.
9    *
10   * @author Yiannis Paschalidis
11   * @since 1.0.0
12   *
13   * @param <T> the type of object returned by the iterator.
14   */
15  public final class EmptyIterator<T> implements Iterator<T>, Serializable {
16  
17  	/**
18  	 * Throws an IllegalStateException, as there are no elements to remove.
19  	 */
20  	@Override
21  	public void remove() {
22  		throw new IllegalStateException("No current element");
23  	}
24  
25  	/**
26  	 * @return false as there are no elements.
27  	 */
28  	@Override
29  	public boolean hasNext() {
30  		return false;
31  	}
32  
33  	/**
34  	 * Throws a NoSuchElementException, as there are no elements.
35  	 *
36  	 * @return nothing - an exception will be thrown.
37  	 */
38  	@Override
39  	public T next() {
40  		throw new NoSuchElementException("No more elements");
41  	}
42  }
