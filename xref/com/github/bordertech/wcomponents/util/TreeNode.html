1   package com.github.bordertech.wcomponents.util;
2   
3   import java.util.Iterator;
4   
5   /**
6    * Represents a node in a tree structure. The API is loosely based on Swing's DefaultMutableTreeNode.
7    *
8    * @author Yiannis Paschalidis
9    * @since 1.0.0
10   */
11  public interface TreeNode {
12  
13  	/**
14  	 * Adds the given TreeNode to this node.
15  	 *
16  	 * @param node the node to add.
17  	 */
18  	void add(final TreeNode node);
19  
20  	/**
21  	 * @return an iteration of this TreeNode's children.
22  	 */
23  	Iterator<TreeNode> children();
24  
25  	/**
26  	 * @return a depth-first iteration of this TreeNode's children.
27  	 */
28  	Iterator<TreeNode> depthFirst();
29  
30  	/**
31  	 * @return a breadth-first iteration of this TreeNode's children.
32  	 */
33  	Iterator<TreeNode> breadthFirst();
34  
35  	/**
36  	 * Retrieves the TreeNode at the given index.
37  	 *
38  	 * @param index the child index.
39  	 * @return the child TreeNode at the given index.
40  	 */
41  	TreeNode getChildAt(final int index);
42  
43  	/**
44  	 * Return the index of the child node. Return -1 if the node is not a child of this node.
45  	 *
46  	 * @param aChild the node to return index of.
47  	 * @return the index of the child or -1 if there is no such child node.
48  	 */
49  	int getIndex(TreeNode aChild);
50  
51  	/**
52  	 * @return the number of child TreeNodes that this node contains.
53  	 */
54  	int getChildCount();
55  
56  	/**
57  	 * @return the parent TreeNode, or null if the root of the tree.
58  	 */
59  	TreeNode getParent();
60  
61  	/**
62  	 * Removes the given tree node from this node.
63  	 *
64  	 * @param child the TreeNode to remove.
65  	 */
66  	void remove(final TreeNode child);
67  
68  	/**
69  	 * Removes all children from this node.
70  	 */
71  	void removeAll();
72  
73  	/**
74  	 * Indicates whether the given node is a descendant of this node.
75  	 *
76  	 * @param otherNode the node to check
77  	 * @return true if the given node is a descendant of this node.
78  	 */
79  	boolean isDescendant(final TreeNode otherNode);
80  
81  	/**
82  	 * Indicates whether the given node is an ancestor of this node.
83  	 *
84  	 * @param otherNode the node to check
85  	 * @return true if the given node is a descendant of this node.
86  	 */
87  	boolean isAncestor(final TreeNode otherNode);
88  
89  	/**
90  	 * @return the node's level, ie. the number of parent nodes until the root.
91  	 */
92  	int getLevel();
93  
94  	/**
95  	 * True if and only if this node has no child nodes.
96  	 *
97  	 * @return <code>true</code> if the node has no child nodes.
98  	 */
99  	boolean isLeaf();
100 
101 	/**
102 	 * Returns true if and only if <code>otherNode</code> is in the same tree as this node. Returns false if
103 	 * <code>otherNode</code> is null.
104 	 *
105 	 * @param otherNode the node to check.
106 	 * @return true if <code>otherNode</code> is in the same tree as this node; false if <code>otherNode</code> is null
107 	 */
108 	boolean isRelated(final TreeNode otherNode);
109 
110 	/**
111 	 * @return the root node.
112 	 */
113 	TreeNode getRoot();
114 
115 	/**
116 	 * Returns the path from the root, to get to this node. The last element in the path is this node.
117 	 *
118 	 * @return an array of DefaultTreeNode objects giving the path, where the first element in the path is the root and
119 	 * the last element is this node.
120 	 */
121 	TreeNode[] getPath();
122 }
