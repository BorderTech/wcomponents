1   package com.github.bordertech.wcomponents.util;
2   
3   import java.io.PrintWriter;
4   import java.io.Writer;
5   
6   /**
7    * WhiteSpaceFilterPrintWriter is a writer that filters out extraneous whitespace from HTML content written to it.
8    *
9    * @author Yiannis Paschalidis
10   * @since 1.0.0
11   */
12  public class WhiteSpaceFilterPrintWriter extends PrintWriter {
13  
14  	/**
15  	 * The current state of the state machine.
16  	 */
17  	private final WhiteSpaceFilterStateMachine stateMachine;
18  
19  	/**
20  	 * Creates a new WhiteSpaceFilterOutputStream.
21  	 *
22  	 * @param writer The Writer that will receive the filtered output
23  	 */
24  	public WhiteSpaceFilterPrintWriter(final Writer writer) {
25  		super(writer);
26  		this.stateMachine = new WhiteSpaceFilterStateMachine();
27  	}
28  
29  	/**
30  	 * Writes the given byte to the underlying output stream if it passes filtering.
31  	 *
32  	 * @param c the byte to write.
33  	 */
34  	@Override
35  	public void write(final int c) {
36  		WhiteSpaceFilterStateMachine.StateChange change = stateMachine.nextState((char) c);
37  
38  		if (change.getOutputBytes() != null) {
39  			for (int i = 0; i < change.getOutputBytes().length; i++) {
40  				super.write(change.getOutputBytes()[i]);
41  			}
42  		}
43  
44  		if (!change.isSuppressCurrentChar()) {
45  			super.write(c);
46  		}
47  	}
48  
49  	/**
50  	 * Writes the given character data to the underlying output stream, filtering as necessary.
51  	 *
52  	 * @param buf the character data to write.
53  	 * @param off the data offset to start writing data from.
54  	 * @param len the number of characters to write.
55  	 */
56  	@Override
57  	public void write(final char[] buf, final int off, final int len) {
58  		for (int i = off; i < off + len; i++) {
59  			write(buf[i]);
60  		}
61  	}
62  
63  	/**
64  	 * Writes the given String to the underlying output stream, filtering as necessary.
65  	 *
66  	 * @param string the String to write.
67  	 * @param off the position in the string to start writing data from.
68  	 * @param len the number of characters to write.
69  	 */
70  	@Override
71  	public void write(final String string, final int off, final int len) {
72  		for (int i = off; i < off + len; i++) {
73  			write(string.charAt(i));
74  		}
75  	}
76  }
