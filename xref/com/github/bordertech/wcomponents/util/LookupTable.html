1   package com.github.bordertech.wcomponents.util;
2   
3   import java.util.List;
4   
5   /**
6    * The LookupTable interface describes a mechanism for providing WComponents with data sourced from an application
7    * look-up table. Applications must implement this class and provide configure the Factory by setting the following
8    * parameter in the application's {@link Config Configuration}:
9    *
10   * <code>factory.impl.com.github.bordertech.wcomponents.util.LookupTable=<i>com.something.MyLookupTable</i></code>
11   *
12   * @author Yiannis Paschalidis
13   * @since 1.0.0
14   */
15  public interface LookupTable {
16  
17  	/**
18  	 * Retrieves the contents of a given table by a table identifier. This identifier can be any object appropriate to
19  	 * your application.
20  	 *
21  	 * @param table the table to look up.
22  	 * @return a list of table objects.
23  	 */
24  	List<?> getTable(Object table);
25  
26  	/**
27  	 * <p>
28  	 * Retrieves the cache id for the given table. This is used to enable caching of data sets, including on the
29  	 * client.</p>
30  	 *
31  	 * <p>
32  	 * If your data sets are session- or user-specific, your cache id must include this information so that the cache is
33  	 * invalidated.</p>
34  	 *
35  	 * <p>
36  	 * A null return value indicates that the table is not cacheable.</p>
37  	 *
38  	 * @param table the table to look up.
39  	 * @return the cache key for the given table, or null if the table is not cacheable.
40  	 */
41  	String getCacheKeyForTable(Object table);
42  
43  	/**
44  	 * Retrieves the source table for the cache key.
45  	 *
46  	 * @param key the table's cache key, as returned by {@link #getCacheKeyForTable(Object)}.
47  	 * @return the table to lookup.
48  	 */
49  	Object getTableForCacheKey(String key);
50  
51  	/**
52  	 * Retrieves the code for a given table entry, or null if the type of object is not supported.
53  	 *
54  	 * @param table the source table
55  	 * @param entry the table entry.
56  	 * @return the code for the given entry.
57  	 */
58  	String getCode(Object table, Object entry);
59  
60  	/**
61  	 * Retrieves the description for a given table entry, or null if the type of object is not supported.
62  	 *
63  	 * @param table the source table.
64  	 * @param entry the table entry.
65  	 * @return the description for the given entry.
66  	 */
67  	String getDescription(Object table, Object entry);
68  }
