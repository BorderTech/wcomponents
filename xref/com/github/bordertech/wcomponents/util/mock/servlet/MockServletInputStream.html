3   import java.io.IOException;
4   import java.io.InputStream;
5   import javax.servlet.ServletInputStream;
6   
7   /**
8    * MockServletInputStream - mock servlet input stream for unit testing.
9    *
10   * @author Yiannis Paschalidis
11   * @since 1.0.0
12   */
13  public class MockServletInputStream extends ServletInputStream {
14  
15  	/**
16  	 * The backing input stream to read from.
17  	 */
18  	private final InputStream backing;
19  
20  	/**
21  	 * Creates a MockServletInputStream with the specified backing.
22  	 *
23  	 * @param backing the backing stream to read from.
24  	 */
25  	public MockServletInputStream(final InputStream backing) {
26  		this.backing = backing;
27  	}
28  
29  	/**
30  	 * Reads the next byte of data from the backing stream. The value byte is returned as an <code>int</code> in the
31  	 * range <code>0</code> to <code>255</code>. If no byte is available because the end of the stream has been reached,
32  	 * the value <code>-1</code> is returned. This method blocks until input data is available, the end of the stream is
33  	 * detected, or an exception is thrown.
34  	 *
35  	 * @return the next byte of data, or <code>-1</code> if the end of the stream is reached.
36  	 * @throws IOException if an I/O error occurs.
37  	 */
38  	@Override
39  	public int read() throws IOException {
40  		return backing.read();
41  	}
42  }
