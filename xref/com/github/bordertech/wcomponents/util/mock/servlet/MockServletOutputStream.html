1   package com.github.bordertech.wcomponents.util.mock.servlet;
2   
3   import java.io.ByteArrayOutputStream;
4   import java.io.IOException;
5   import javax.servlet.ServletOutputStream;
6   
7   /**
8    * MockServletOutputStream - mock Servlet output stream for unit testing.
9    *
10   * @author Yiannis Paschalidis
11   * @since 1.0.0
12   */
13  public class MockServletOutputStream extends ServletOutputStream {
14  
15  	/**
16  	 * The backing output stream to write content to, so that it may later be retrieved using {@link #getOutput}.
17  	 */
18  	private final ByteArrayOutputStream backing = new ByteArrayOutputStream();
19  
20  	/**
21  	 * Writes the specified byte to this output stream. The byte to be written is the eight low-order bits of the
22  	 * argument <code>b</code>. The 24 high-order bits of <code>b</code> are ignored.
23  	 *
24  	 * @param byt the <code>byte</code> to be written.
25  	 * @throws IOException if an I/O error occurs.
26  	 */
27  	@Override
28  	public void write(final int byt) throws IOException {
29  		backing.write(byt);
30  	}
31  
32  	/**
33  	 * Returns the data that has been written to this output stream.
34  	 *
35  	 * @return the byte data written to the stream. May be empty, but will not be null.
36  	 */
37  	public byte[] getOutput() {
38  		return backing.toByteArray();
39  	}
40  }
