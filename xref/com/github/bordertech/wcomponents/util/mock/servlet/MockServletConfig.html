3   import com.github.bordertech.wcomponents.util.Enumerator;
4   import java.util.Enumeration;
5   import java.util.HashMap;
6   import java.util.Map;
7   import javax.servlet.ServletConfig;
8   import javax.servlet.ServletContext;
9   
10  /**
11   * MockServletConfig - mock servlet config for unit testing.
12   *
13   * @author Yiannis Paschalidis
14   * @since 1.0.0
15   */
16  public class MockServletConfig implements ServletConfig {
17  
18  	private String servletName;
19  	private final Map initParameters = new HashMap();
20  
21  	/**
22  	 * Sets an init parameter value.
23  	 *
24  	 * @param name the name of the parameter.
25  	 * @param value the value of the parameter.
26  	 */
27  	public void setInitParameter(final String name, final String value) {
28  		initParameters.put(name, value);
29  	}
30  
31  	/**
32  	 * Returns the value of the init parameter with the specified name.
33  	 *
34  	 * @param name the name of the parameter to read.
35  	 * @return the init parameter value, or null if not set.
36  	 */
37  	@Override
38  	public String getInitParameter(final String name) {
39  		return (String) initParameters.get(name);
40  	}
41  
42  	/**
43  	 * @return an enumeration of the parameter names
44  	 */
45  	@Override
46  	public Enumeration getInitParameterNames() {
47  		return new Enumerator(initParameters.keySet().iterator());
48  	}
49  
50  	/**
51  	 * @return null, as this is not implemented
52  	 */
53  	@Override
54  	public ServletContext getServletContext() {
55  		return null;
56  	}
57  
58  	/**
59  	 * @return the servlet name
60  	 */
61  	@Override
62  	public String getServletName() {
63  		return servletName;
64  	}
65  
66  	/**
67  	 * Sets the servlet name.
68  	 *
69  	 * @param servletName the servlet name to set.
70  	 */
71  	public void setServletName(final String servletName) {
72  		this.servletName = servletName;
73  	}
74  }
