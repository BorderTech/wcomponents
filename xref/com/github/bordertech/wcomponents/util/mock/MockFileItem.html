3   import java.io.ByteArrayInputStream;
4   import java.io.File;
5   import java.io.IOException;
6   import java.io.InputStream;
7   import java.io.OutputStream;
8   import java.io.UnsupportedEncodingException;
9   import org.apache.commons.fileupload.FileItem;
10  import org.apache.commons.fileupload.FileItemHeaders;
11  import org.apache.commons.io.FileUtils;
12  
13  /**
14   * Mock implementation of {@link FileItem}.
15   *
16   * @author Yiannis Paschalidis
17   * @author Anthony O'Connor - extracted from WFileWidget_Test.
18   * @since 1.0.0
19   */
20  public final class MockFileItem implements FileItem {
21  
22  	/**
23  	 * The field name used to reference this file item.
24  	 */
25  	private String fieldName;
26  
27  	/**
28  	 * <code>True</code> if the instance represents a simple form field, or <code>false</code> if it represents an
29  	 * uploaded file.
30  	 */
31  	private boolean formField = true;
32  
33  	/**
34  	 * The original filename in the client's filesystem.
35  	 */
36  	private String fileName;
37  
38  	/**
39  	 * The file's mime-type.
40  	 */
41  	private String contentType;
42  
43  	/**
44  	 * Flags whether the {@link #delete()} method has been called.
45  	 */
46  	private boolean deleted = false;
47  
48  	/**
49  	 * The binary data contained in the file.
50  	 */
51  	private byte[] contents;
52  
53  	private FileItemHeaders headers;
54  
55  	/**
56  	 * {@inheritDoc}
57  	 */
58  	@Override
59  	public void delete() {
60  		deleted = true;
61  	}
62  
63  	/**
64  	 * {@inheritDoc}
65  	 */
66  	@Override
67  	public byte[] get() {
68  		if (deleted) {
69  			throw new IllegalStateException("delete() called");
70  		}
71  
72  		if (contents != null) {
73  			byte[] data = new byte[contents.length];
74  			System.arraycopy(contents, 0, data, 0, contents.length);
75  			return data;
76  		}
77  		return null;
78  	}
79  
80  	/**
81  	 * Sets the binary data for this MockFileItem.
82  	 *
83  	 * @param binaryContents the binary content.
84  	 */
85  	public void set(final byte[] binaryContents) {
86  		formField = false;
87  		this.contents = binaryContents;
88  	}
89  
90  	/**
91  	 * Sets the content MIME-type for thos MockFileItem.
92  	 *
93  	 * @param contentType the contentType.
94  	 */
95  	public void setContentType(final String contentType) {
96  		this.contentType = contentType;
97  	}
98  
99  	/**
100 	 * {@inheritDoc}
101 	 */
102 	@Override
103 	public String getContentType() {
104 		return contentType;
105 	}
106 
107 	/**
108 	 * {@inheritDoc}
109 	 */
110 	@Override
111 	public String getFieldName() {
112 		return fieldName;
113 	}
114 
115 	/**
116 	 * {@inheritDoc}
117 	 */
118 	@Override
119 	public InputStream getInputStream() throws IOException {
120 		return new ByteArrayInputStream(get());
121 	}
122 
123 	/**
124 	 * Returns the file item headers.
125 	 *
126 	 * @return The file items headers.
127 	 */
128 	@Override
129 	public FileItemHeaders getHeaders() {
130 		return headers;
131 	}
132 
133 	/**
134 	 * Set the file item headers.
135 	 */
136 	@Override
137 	public void setHeaders(final FileItemHeaders headers) {
138 		this.headers = headers;
139 	}
140 
141 	/**
142 	 * Sets the name of the uploaded file for this MockFileItem.
143 	 *
144 	 * @param name the name of the uploaded file.
145 	 */
146 	public void setName(final String name) {
147 		this.fileName = name;
148 	}
149 
150 	/**
151 	 * {@inheritDoc}
152 	 */
153 	@Override
154 	public String getName() {
155 		return fileName;
156 	}
157 
158 	/**
159 	 * {@inheritDoc}
160 	 */
161 	@Override
162 	public OutputStream getOutputStream() throws IOException {
163 		return null;
164 	}
165 
166 	/**
167 	 * {@inheritDoc}
168 	 */
169 	@Override
170 	public long getSize() {
171 		return contents == null ? 0 : contents.length;
172 	}
173 
174 	/**
175 	 * {@inheritDoc}
176 	 */
177 	@Override
178 	public String getString() {
179 		return new String(get());
180 	}
181 
182 	/**
183 	 * {@inheritDoc}
184 	 */
185 	@Override
186 	public String getString(final String encoding) throws UnsupportedEncodingException {
187 		return new String(get(), encoding);
188 	}
189 
190 	/**
191 	 * {@inheritDoc}
192 	 */
193 	@Override
194 	public boolean isFormField() {
195 		return formField;
196 	}
197 
198 	/**
199 	 * {@inheritDoc}
200 	 */
201 	@Override
202 	public boolean isInMemory() {
203 		return true;
204 	}
205 
206 	/**
207 	 * {@inheritDoc}
208 	 */
209 	@Override
210 	public void setFieldName(final String name) {
211 		this.fieldName = name;
212 	}
213 
214 	/**
215 	 * {@inheritDoc}
216 	 */
217 	@Override
218 	public void setFormField(final boolean state) {
219 		this.formField = state;
220 	}
221 
222 	/**
223 	 * {@inheritDoc}
224 	 */
225 	@Override
226 	public void write(final File file) throws IOException {
227 		FileUtils.writeByteArrayToFile(file, get());
228 	}
229 }
