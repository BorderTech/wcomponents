32  	/**
33  	 * @param jsonString the string of JSON to convert to a custom tree of nodes
34  	 * @return the custom tree structure of item ids
35  	 */
36  	public static TreeItemIdNode convertJsonToTree(final String jsonString) {
37  
38  		JsonParser parser = new JsonParser();
39  		JsonObject json = parser.parse(jsonString).getAsJsonObject();
40  
41  		TreeItemIdNode root = new TreeItemIdNode(null);
42  
43  		JsonArray children = json.getAsJsonArray("root");
44  		for (int i = 0; i < children.size(); i++) {
45  			JsonObject child = children.get(i).getAsJsonObject();
46  			processJsonToTree(root, child);
47  		}
48  
49  		return root;
50  	}

52  	/**
53  	 * Iterate over the JSON objects to create the tree structure.
54  	 *
55  	 * @param parentNode the parent node
56  	 * @param json the current JSON object
57  	 */
58  	private static void processJsonToTree(final TreeItemIdNode parentNode, final JsonObject json) {
59  
60  		String id = json.getAsJsonPrimitive("id").getAsString();
61  
62  		TreeItemIdNode node = new TreeItemIdNode(id);
63  		parentNode.addChild(node);
64  
65  		JsonArray children = json.getAsJsonArray("items");
66  		for (int i = 0; i < children.size(); i++) {
67  			JsonObject child = children.get(i).getAsJsonObject();
68  			processJsonToTree(node, child);
69  		}
70  	}

73  	 *
74  	 * @param tree1 the first tree to compare
75  	 * @param tree2 the second tree to compare
76  	 * @return true if the trees match
77  	 */
78  	public static boolean isTreeSame(final TreeItemIdNode tree1, final TreeItemIdNode tree2) {
79  
80  		// Check IDs match
81  		if (!Util.equals(tree1.getItemId(), tree2.getItemId())) {
82  			return false;
83  		}
84  
85  		// Check have same number of children
86  		if (tree1.getChildren().size() != tree2.getChildren().size()) {
87  			return false;
88  		}
89  
90  		// Check child IDs match
91  		for (int i = 0; i < tree1.getChildren().size(); i++) {
92  			if (!isTreeSame(tree1.getChildren().get(i), tree2.getChildren().get(i))) {
93  				return false;
94  			}
95  		}
96  		return true;
97  	}

102 	 */
103 	public static TreeItemIdNode copyTreeNode(final TreeItemIdNode node) {
104 		TreeItemIdNode copy = new TreeItemIdNode(node.getItemId());
105 		copy.setHasChildren(node.hasChildren());
106 
107 		for (TreeItemIdNode childItem : node.getChildren()) {
108 			TreeItemIdNode childCopy = copyTreeNode(childItem);
109 			copy.addChild(childCopy);
110 		}
111 		return copy;
112 	}

115 	 * Convert the data to a set (if necessary).
116 	 *
117 	 * @param data the data to convert to a set
118 	 * @return the data converted to a set
119 	 */
120 	public static Set<String> convertDataToSet(final Object data) {
121 		if (data == null) {
122 			return null;
123 		} else if (data instanceof Set) {
124 			return (Set<String>) data;
125 		} else if (data instanceof Collection) {
126 			Collection items = (Collection) data;
127 			Set<String> set = new HashSet<>();
128 			for (Object item : items) {
129 				if (item != null) {
130 					set.add(item.toString());
131 				}
132 			}
133 			return set;
134 		} else if (data instanceof Object[]) {
135 			Object[] items = (Object[]) data;
136 			Set<String> set = new HashSet<>();
137 			for (Object item : items) {
138 				if (item != null) {
139 					set.add(item.toString());
140 				}
141 			}
142 			return set;
143 		} else {
144 			Set<String> set = new HashSet<>();
145 			set.add(data.toString());
146 			return set;
147 		}
148 	}

165 	 *
166 	 * @param map the map of custom items and their node
167 	 * @param node the current node being processed
168 	 */
169 	private static void processCustomIdMapping(final Map<String, TreeItemIdNode> map, final TreeItemIdNode node) {
170 		String itemId = node.getItemId();
171 		if (!Util.empty(itemId)) {
172 			map.put(itemId, node);
173 		}
174 
175 		for (TreeItemIdNode childItem : node.getChildren()) {
176 			processCustomIdMapping(map, childItem);
177 		}

181 	 * @param tree the tree component to create a map of item ids
182 	 * @return the map of item ids to their row index.
183 	 */
184 	public static Map<String, List<Integer>> createItemIdIndexMap(final WTree tree) {
185 		Map<String, List<Integer>> map = new HashMap<>();
186 		TreeItemModel treeModel = tree.getTreeModel();
187 		int rows = treeModel.getRowCount();
188 		Set<String> expanded = tree.getExpandedRows();
189 		WTree.ExpandMode mode = tree.getExpandMode();
190 
191 		for (int i = 0; i < rows; i++) {
192 			List<Integer> index = new ArrayList<>();
193 			index.add(i);
194 			processItemIdIndexMapping(map, index, treeModel, mode, expanded);
195 		}
196 		return Collections.unmodifiableMap(map);
197 	}

199 	/**
200 	 * Iterate through the table model to add the item ids and their row index.
201 	 *
202 	 * @param map the map of item ids
203 	 * @param rowIndex the current row index
204 	 * @param treeModel the tree model
205 	 * @param mode the expand mode
206 	 * @param expandedRows the set of expanded rows
207 	 */
208 	private static void processItemIdIndexMapping(final Map<String, List<Integer>> map, final List<Integer> rowIndex, final TreeItemModel treeModel, final WTree.ExpandMode mode, final Set<String> expandedRows) {
209 
210 		// Add current item
211 		String id = treeModel.getItemId(rowIndex);
212 		if (id == null) {
213 			return;
214 		}
215 		map.put(id, rowIndex);
216 
217 		// Check row is expandable
218 		if (!treeModel.isExpandable(rowIndex)) {
219 			return;
220 		}
221 
222 		// Check has children
223 		if (!treeModel.hasChildren(rowIndex)) {
224 			return;
225 		}
226 
227 		// Always add children if CLIENT mode or row is expanded
228 		boolean addChildren = (mode == WTree.ExpandMode.CLIENT) || (expandedRows != null && expandedRows.contains(
229 				id));
230 		if (!addChildren) {
231 			return;
232 		}
233 
234 		// Get actual child count
235 		int children = treeModel.getChildCount(rowIndex);
236 		if (children == 0) {
237 			// Could be there are no children even though hasChildren returned true
238 			return;
239 		}
240 
241 		// Add children by processing each child row
242 		for (int i = 0; i < children; i++) {
243 			// Add next level
244 			List<Integer> nextRow = new ArrayList<>(rowIndex);
245 			nextRow.add(i);
246 			processItemIdIndexMapping(map, nextRow, treeModel, mode, expandedRows);
247 		}
248 	}

260 	/**
261 	 * Update the custom tree node to make sure any nodes that dont have any children in the custom layout have their
262 	 * children loaded from the tree model (if they have any).
263 	 *
264 	 * @param node the current node.
265 	 * @param treeModel the tree model
266 	 * @param mode the expand mode
267 	 * @param expandedRows the set of expanded rows
268 	 * @param mapItemIds the map of item ids to row index
269 	 */
270 	private static void processCustomTreeNodes(final TreeItemIdNode node, final TreeItemModel treeModel, final WTree.ExpandMode mode, final Set<String> expandedRows, final Map<String, List<Integer>> mapItemIds) {
271 
272 		// Node has no children so check if they need to be loaded from the tree model
273 		if (node.getChildren().isEmpty()) {
274 			List<Integer> rowIndex = mapItemIds.get(node.getItemId());
275 			loadCustomNodesFromModel(node, rowIndex, treeModel, mode, expandedRows);
276 		} else {
277 			// Check children
278 			for (TreeItemIdNode child : node.getChildren()) {
279 				processCustomTreeNodes(child, treeModel, mode, expandedRows, mapItemIds);
280 			}
281 		}
282 	}

284 	/**
285 	 * Load the child items from the tree model onto the custom node.
286 	 *
287 	 * @param node the node to update
288 	 * @param rowIndex the current row index
289 	 * @param treeModel the tree model
290 	 * @param mode the expand mode
291 	 * @param expandedRows the set of expanded rows
292 	 */
293 	private static void loadCustomNodesFromModel(final TreeItemIdNode node, final List<Integer> rowIndex, final TreeItemModel treeModel, final WTree.ExpandMode mode, final Set<String> expandedRows) {
294 
295 		// Defualt to no children
296 		node.setHasChildren(false);
297 
298 		// Row is not expandable
299 		if (!treeModel.isExpandable(rowIndex)) {
300 			return;
301 		}
302 
303 		// Row has no children
304 		if (!treeModel.hasChildren(rowIndex)) {
305 			return;
306 		}
307 
308 		// OK. So has children
309 		node.setHasChildren(true);
310 
311 		// Always add children if CLIENT mode or row is expanded
312 		boolean addChildren = (mode == WTree.ExpandMode.CLIENT) || (expandedRows != null && expandedRows.contains(node.getItemId()));
313 		if (!addChildren) {
314 			return;
315 		}
316 
317 		// Get actual child count
318 		int children = treeModel.getChildCount(rowIndex);
319 		if (children == 0) {
320 			// Could be there are no children even though hasChildren returned true
321 			node.setHasChildren(false);
322 			return;
323 		}
324 
325 		// Add children by processing each child row
326 		for (int i = 0; i < children; i++) {
327 			// Calc next level index
328 			List<Integer> nextRow = new ArrayList<>(rowIndex);
329 			nextRow.add(i);
330 			// Get child item id
331 			String nextId = treeModel.getItemId(nextRow);
332 			// Add child node
333 			TreeItemIdNode childNode = new TreeItemIdNode(nextId);
334 			node.addChild(childNode);
335 			// Process this child node
336 			loadCustomNodesFromModel(childNode, nextRow, treeModel, mode, expandedRows);
337 		}
338 
339 	}

341 	/**
342 	 * Convert the row index to its string representation.
343 	 *
344 	 * @param row the row index
345 	 * @return the string representation of the row index
346 	 */
347 	public static String rowIndexListToString(final List<Integer> row) {
348 		if (row == null || row.isEmpty()) {
349 			return null;
350 		}
351 
352 		StringBuilder index = new StringBuilder();
353 		boolean addDelimiter = false;
354 
355 		for (Integer lvl : row) {
356 			if (addDelimiter) {
357 				index.append('-');
358 			}
359 			index.append(lvl);
360 			addDelimiter = true;
361 		}
362 
363 		return index.toString();
364 	}
