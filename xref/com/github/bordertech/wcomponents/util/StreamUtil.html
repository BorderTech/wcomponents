1   package com.github.bordertech.wcomponents.util;

3   import java.io.ByteArrayOutputStream;
4   import java.io.Closeable;
5   import java.io.IOException;
6   import java.io.InputStream;
7   import java.io.OutputStream;
8   import org.apache.commons.logging.Log;
9   import org.apache.commons.logging.LogFactory;
10  
11  /**
12   * Static utility methods related to working with Streams.
13   *
14   * @author Martin Shevchenko
15   */
16  public final class StreamUtil {
17  
18  	/**
19  	 * The logger instance for this class.
20  	 */
21  	private static final Log LOG = LogFactory.getLog(StreamUtil.class);
22  
23  	/**
24  	 * No instance methods here.
25  	 */
26  	private StreamUtil() {
27  	}
28  
29  	/**
30  	 * Default value is 2048.
31  	 */
32  	public static final int DEFAULT_BUFFER_SIZE = 2048;
33  
34  	/**
35  	 * Copies information from the input stream to the output stream, using the default copy buffer size.
36  	 *
37  	 * @param in the source stream.
38  	 * @param out the destination stream.
39  	 * @throws IOException if there is an error reading or writing to the streams.
40  	 */
41  	public static void copy(final InputStream in, final OutputStream out)
42  			throws IOException {
43  		copy(in, out, DEFAULT_BUFFER_SIZE);
44  	}
45  
46  	/**
47  	 * Copies information from the input stream to the output stream using a specified buffer size.
48  	 *
49  	 * @param in the source stream.
50  	 * @param out the destination stream.
51  	 * @param bufferSize the buffer size.
52  	 * @throws IOException if there is an error reading or writing to the streams.
53  	 */
54  	public static void copy(final InputStream in, final OutputStream out, final int bufferSize)
55  			throws IOException {
56  		final byte[] buf = new byte[bufferSize];
57  		int bytesRead = in.read(buf);
58  
59  		while (bytesRead != -1) {
60  			out.write(buf, 0, bytesRead);
61  			bytesRead = in.read(buf);
62  		}
63  
64  		out.flush();
65  	}
66  
67  	/**
68  	 * Returns a byte array containing all the information contained in the given input stream.
69  	 *
70  	 * @param in the stream to read from.
71  	 * @return the stream contents as a byte array.
72  	 * @throws IOException if there is an error reading from the stream.
73  	 */
74  	public static byte[] getBytes(final InputStream in) throws IOException {
75  		ByteArrayOutputStream result = new ByteArrayOutputStream();
76  		copy(in, result);
77  		result.close();
78  		return result.toByteArray();
79  	}
80  
81  	/**
82  	 * Closes an OutputStream, logging any exceptions.
83  	 *
84  	 * @param stream the stream to close.
85  	 */
86  	public static void safeClose(final Closeable stream) {
87  		if (stream != null) {
88  			try {
89  				stream.close();
90  			} catch (IOException e) {
91  				LOG.error("Failed to close resource stream", e);
92  			}
93  		}
94  	}
95  
96  	/**
97  	 * This method returns the contents of a {@link InputStream} as a byte[].
98  	 *
99  	 * @param is The {@link InputStream} to be read into a byte[]
100 	 * @return A byte[] representing the contents of the given <em>is</em>.
101 	 * @throws IOException If there is an error reading from the <em>is</em>.
102 	 */
103 	public static byte[] streamToByteArray(final InputStream is) throws IOException {
104 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
105 		copy(is, baos);
106 		byte[] bytes = baos.toByteArray();
107 		return bytes;
108 	}
109 }
