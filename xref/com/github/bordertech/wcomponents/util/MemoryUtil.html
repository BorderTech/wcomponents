1   package com.github.bordertech.wcomponents.util;
2   
3   import org.apache.commons.logging.Log;
4   import org.apache.commons.logging.LogFactory;
5   
6   /**
7    * This class is intended to help detect and trace possible memory leaks.
8    *
9    * @author Rick Brown
10   */
11  public final class MemoryUtil {
12  
13  	/**
14  	 * For convenience this can be used as a "rule of thumb" when testing the size of a List of components in the UI.
15  	 */
16  	public static final int WARN_THRESHOLD;
17  
18  	static {
19  		WARN_THRESHOLD = ConfigurationProperties.getMemoryUtilWarnThreshold();
20  	}
21  
22  	/**
23  	 * The logger instance for this class.
24  	 */
25  	private static final Log LOG = LogFactory.getLog(MemoryUtil.class);
26  
27  	/**
28  	 * Utility class needs no constructor.
29  	 */
30  	private MemoryUtil() {
31  
32  	}
33  
34  	/**
35  	 * Performs default check on the size of a "registry" of UI widgets (or similar).
36  	 * In practice "size" is likely to be the size of a java Collection but it could be anything you want.
37  	 *
38  	 * This is a convenience which should be adequate for the majority of cases.
39  	 *
40  	 * @param size The current size of a component registry (or similar).
41  	 * @param className The name of the class which holds the registry.
42  	 */
43  	public static void checkSize(final int size, final String className) {
44  		if (MemoryUtil.WARN_THRESHOLD < size) {
45  			MemoryUtil.log(className + " may be leaking memory, it contains a large number of items: " + size);
46  		}
47  	}
48  
49  	/**
50  	 * Log a possible memory leak as necessary.
51  	 * The benefit of using this to log possible memory leaks is that there is one global logger to enable/disable
52  	 * as debugging dictates.
53  	 *
54  	 * This method also creates a stack trace so that possible memory leaks are easier to track.
55  	 *
56  	 * @param message A meaningful message that provides information about the possible memory leak.
57  	 * The message should ideally provide information about the source of the leak.
58  	 */
59  	public static void log(final String message) {
60  		if (LOG.isWarnEnabled()) {
61  			try {
62  				throw new PossibleMemLeakException(message);
63  			} catch (PossibleMemLeakException ex) {
64  				LOG.warn(message, ex);
65  			}
66  		}
67  	}
68  
69  	/**
70  	 * An exception that is used to generate a stack trace to aid tracing the original cause of the possible memory leak.
71  	 */
72  	private static final class PossibleMemLeakException extends RuntimeException {
73  		/**
74  		 * Creates an exception that indicates there is possibly (probably) a memory leak.
75  		 * @param message A meaningful message that provides information about the possible leak.
76  		 */
77  		private PossibleMemLeakException(final String message) {
78  			super(message);
79  		}
80  	}
81  }
