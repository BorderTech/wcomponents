1   package com.github.bordertech.wcomponents.util;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   
5   /**
6    * A visitor interface used when traversing WComponent trees.
7    *
8    * @author Yiannis Paschalidis
9    * @since 1.0.0
10   */
11  public interface WComponentTreeVisitor {
12  
13  	/**
14  	 * An enumeration used to short-circuit tree traversal for efficiency.
15  	 */
16  	enum VisitorResult {
17  		/**
18  		 * Continue tree traversal.
19  		 */
20  		CONTINUE,
21  		/**
22  		 * Continue tree traversal, but not in this branch.
23  		 */
24  		ABORT_BRANCH,
25  		/**
26  		 * Stop tree traversal altogether.
27  		 */
28  		ABORT
29  	}
30  
31  	/**
32  	 * Called for each component in the WComponent hierarchy.
33  	 *
34  	 * @param comp the component in the tree being observed
35  	 *
36  	 * @return how the traversal should proceed.
37  	 */
38  	VisitorResult visit(WComponent comp);
39  }
