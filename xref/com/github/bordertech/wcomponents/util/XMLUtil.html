7    * Utility class for rendering file element.

12  public final class XMLUtil {
13  
14  	/**
15  	 * Theme URI.
16  	 */
17  	public static final String THEME_URI = "https://github.com/bordertech/wcomponents/namespace/ui/v1.0";
18  
19  	/**
20  	 * XHtml URI.
21  	 */
22  	public static final String XHTML_URI = "http://www.w3.org/1999/xhtml";
23  
24  	/**
25  	 * XML Declaration.
26  	 */
27  	public static final String XML_DECLARATION = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>";
28  
29  	/**
30  	 * UI Namespace attribute.
31  	 */
32  	public static final String UI_NAMESPACE = " xmlns:ui=\"" + THEME_URI + "\"";
33  
34  	/**
35  	 * Standard Namespace attributes.
36  	 */
37  	public static final String STANDARD_NAMESPACES = " xmlns=\"" + XHTML_URI + "\" xmlns:html=\"" + XHTML_URI + "\""
38  			+ UI_NAMESPACE;
39  
40  	/**
41  	 * XML Declaration.
42  	 * @deprecated 1.4.0 never used internally. Must never be used!
43  	 */
44  	@Deprecated
45  	public static final String XML_DECLERATION = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>";
46  
47  	/**
48  	 * Prevent instantiation of utility class.
49  	 */
50  	private XMLUtil() {
51  		// Do nothing
52  	}
53  
54  	/**
55  	 * @param uic the current user context
56  	 * @return the xml declaration with the theme processing instruction
57  	 */
58  	public static String getXMLDeclarationWithThemeXslt(final UIContext uic) {
59  		String theme = WebUtilities.encode(ThemeUtil.getThemeXslt(uic));
60  		String dec = XML_DECLARATION + "\n<?xml-stylesheet type=\"text/xsl\" href=\"" + theme + "\"?>";
61  		return dec;
62  	}
63  }
