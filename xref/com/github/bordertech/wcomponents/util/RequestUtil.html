3   import java.util.Map;
4   import org.apache.commons.fileupload.FileItem;
5   
6   /**
7    * This class provides helpers for processing Requests.
8    *
9    * @author Rick Brown
10   * @since 1.0.0
11   */
12  public final class RequestUtil {
13  
14  	/**
15  	 * Hide the constructor as there are no instance methods.
16  	 */
17  	private RequestUtil() {
18  	}
19  
20  	/**
21  	 * Add the file data to the files collection. If a file already exists with the given name then the value for this
22  	 * name will be an array of all registered files.
23  	 *
24  	 * @param files the map in which to store file data.
25  	 * @param name the name of the file, i.e. the key to store the file against.
26  	 * @param item the file data.
27  	 */
28  	public static void addFileItem(final Map<String, FileItem[]> files, final String name, final FileItem item) {
29  		if (files.containsKey(name)) {
30  			// This field contains multiple values, append the new value to the existing values.
31  			FileItem[] oldValues = files.get(name);
32  			FileItem[] newValues = new FileItem[oldValues.length + 1];
33  			System.arraycopy(oldValues, 0, newValues, 0, oldValues.length);
34  			newValues[newValues.length - 1] = item;
35  			files.put(name, newValues);
36  		} else {
37  			files.put(name, new FileItem[]{item});
38  		}
39  	}
40  
41  	/**
42  	 * Add the request parameter to the parameters collection. If a parameter already exists with the given name then
43  	 * the Map will contain an array of all registered values.
44  	 *
45  	 * @param parameters the map to store non-file request parameters in.
46  	 * @param name the name of the parameter.
47  	 * @param value the parameter value.
48  	 */
49  	public static void addParameter(final Map<String, String[]> parameters, final String name, final String value) {
50  		if (parameters.containsKey(name)) {
51  			// This field contains multiple values, append the new value to the existing values.
52  			String[] oldValues = parameters.get(name);
53  			String[] newValues = new String[oldValues.length + 1];
54  			System.arraycopy(oldValues, 0, newValues, 0, oldValues.length);
55  			newValues[newValues.length - 1] = value;
56  			parameters.put(name, newValues);
57  		} else {
58  			parameters.put(name, new String[]{value});
59  		}
60  	}
61  
62  }
