3   import javax.servlet.ServletContextEvent;

7   /**
8    * This context listener is used to stop the reaper {@link Thread} that is started by the jakarta commons file upload
9    * API.
10   *
11   * @author Christina Harris
12   * @since 1.0.0
13   */
14  public class WFileWidgetCleanup implements ServletContextListener {
15  
16  	/**
17  	 * @param sce the servlet context event.
18  	 */
19  	@Override
20  	public void contextDestroyed(final ServletContextEvent sce) {
21  		FileCleaner.exitWhenFinished();
22  	}
23  
24  	/**
25  	 * Does nothing.
26  	 *
27  	 * @param sce the servlet context event.
28  	 */
29  	@Override
30  	public void contextInitialized(final ServletContextEvent sce) {
31  		// does nothing.
32  	}
33  }
