1   package com.github.bordertech.wcomponents;
2   
3   import java.io.Serializable;
4   import java.util.Map;
5   
6   /**
7    * Methods for finding out about the environment in which wcomponents are hosted.
8    *
9    * @author James Gifford
10   * @since 1.0.0
11   */
12  public interface Environment extends Serializable {
13  
14  	/**
15  	 * The URL/Post variable that will store the session token. The session token is used to prevent Cross-site request
16  	 * forgery (CSRF) attacks.
17  	 */
18  	String SESSION_TOKEN_VARIABLE = "wc_t";
19  
20  	/**
21  	 * The URL/Post variable that will store the step number. The step number is used to identify the request is from
22  	 * the most currently rendered view.
23  	 */
24  	String STEP_VARIABLE = "wc_s";
25  
26  	/**
27  	 * The URL/Post variable that will store the action step number. The action step number is used to identify the
28  	 * request is still related to the current action step.
29  	 *
30  	 * @deprecated portal specific
31  	 */
32  	@Deprecated
33  	String STEP_ACTION_VARIABLE = "wc_a";
34  
35  	/**
36  	 * The key used for caching content.
37  	 */
38  	String CONTENT_CACHE_KEY = "contentCacheKey";
39  
40  	/**
41  	 * The URL/Post variable that will identify a specific wcomponent to process the request. A request that includes
42  	 * this variable is called a targeted request. Targeted requests are used to return document content such as image
43  	 * data and PDFs. It is also used to support AJAX regions. Targeted requests are handled by the
44  	 * WContentHelperServlet.
45  	 */
46  	String TARGET_ID = "wc_target";
47  
48  	/**
49  	 * The name of the theme path that indicates a theme resource has been requested.
50  	 */
51  	String THEME_RESOURCE_PATH_NAME = "wc_th";
52  
53  	/**
54  	 * The URL query string parameter variable name that is used to ensure that the URL appears unique to the browser.
55  	 * This is used to prevent caching of PDFs, TR5s etc by the client.
56  	 */
57  	String UNIQUE_RANDOM_PARAM = "no-cache";
58  
59  	/**
60  	 * Gets the relative URL path to which this WComponent application should post its requests.
61  	 * <p>
62  	 * Examples of what this method may return:
63  	 * </p>
64  	 * <dl>
65  	 * <dt>Servlet</dt><dd>/app</dd>
66  	 * </dl>
67  	 *
68  	 * @return the relative URL path to which this WComponent application should post its requests.
69  	 */
70  	String getPostPath();
71  
72  	/**
73  	 * Sets the relative URL path to which this WComponent application should post its requests.
74  	 *
75  	 * @param postPath the post path.
76  	 */
77  	void setPostPath(final String postPath);
78  
79  	/**
80  	 * Gets the URL path for the WComponent support servlets, which handle serving content for Targetable WComponents,
81  	 * AJAX and data lists. There needs to be a servlet (WServlet) deployed to serve Targetable content. This method
82  	 * returns the URL to that servlet.
83  	 * <p>
84  	 * Examples of what this method may return:
85  	 * </p>
86  	 * <dl>
87  	 * <dt>Servlet</dt><dd>/app</dd>
88  	 * </dl>
89  	 *
90  	 * @return the URL path suitable for serving the content of Targetable wcomponents.
91  	 */
92  	String getWServletPath();
93  
94  	/**
95  	 * Gets the base URL under which web applications are hosted. This AppHostPath can be used to create url links to
96  	 * applications that have been given fixed relative names.
97  	 *
98  	 * @return the base URL under which web applications are hosted.
99  	 */
100 	String getAppHostPath();
101 
102 	/**
103 	 * Gets the base URL under which the web content for the Theme is hosted. Web content is static content such as css,
104 	 * js, html, and image files.
105 	 *
106 	 * @return the base URL under which the web content for the Theme is hosted.
107 	 */
108 	String getThemePath();
109 
110 	/**
111 	 * Get the base url at which this web application is hosted.
112 	 * <p>
113 	 * Implementations ensure that this method returns a URL WITHOUT a trailing slash, as in above example.
114 	 * </p>
115 	 *
116 	 * @return the base url at which this web application is hosted.
117 	 */
118 	String getBaseUrl();
119 
120 	/**
121 	 * Get the "host-free" part of the base url for this web application. Eg, if the base url is
122 	 * http://localhost:8080/visas, then the "host-free" part is "/visas".
123 	 *
124 	 * @return Get the "host-free" part of the base url for this web application.
125 	 */
126 	String getHostFreeBaseUrl();
127 
128 	/**
129 	 * Get a unique "app" id. This is intended to help with portals where multiple applications are present in the one
130 	 * document. The app id will be a machine generated unique identifier, but must not contain the character
131 	 * {@link WComponent#ID_CONTEXT_SEPERATOR}.
132 	 *
133 	 * @return the unique application id for this application.
134 	 * @deprecated no longer used. Use {@link WApplication#getIdName()} instead.
135 	 */
136 	@Deprecated
137 	String getAppId();
138 
139 	/**
140 	 * Sets the unique "app" id of the portlet.
141 	 *
142 	 * @param appId the unique application id.
143 	 * @deprecated no longer used. Use {@link WApplication#setIdName(String)} instead.
144 	 */
145 	@Deprecated
146 	void setAppId(String appId);
147 
148 	/**
149 	 * Gets the session token. This is intended to make sure the request is for the correct session.
150 	 *
151 	 * @return the session token.
152 	 */
153 	String getSessionToken();
154 
155 	/**
156 	 * Sets the session token.
157 	 *
158 	 * @param sessionToken the session token.
159 	 */
160 	void setSessionToken(String sessionToken);
161 
162 	/**
163 	 * Gets the page step number. This is intended to make sure the request being processed is on the correct step.
164 	 *
165 	 * @return the page step number.
166 	 */
167 	int getStep();
168 
169 	/**
170 	 * Sets the step page number.
171 	 *
172 	 * @param step the step number.
173 	 */
174 	void setStep(int step);
175 
176 	/**
177 	 * Gets the page action step number. This is intended to make sure the request being processed is on the correct
178 	 * action step.
179 	 *
180 	 * @return the page action step number.
181 	 *
182 	 * @deprecated portal specific
183 	 */
184 	@Deprecated
185 	int getActionStep();
186 
187 	/**
188 	 * Sets the page action step number.
189 	 *
190 	 * @param actionStep the step action number.
191 	 *
192 	 * @deprecated portal specific
193 	 */
194 	@Deprecated
195 	void setActionStep(int actionStep);
196 
197 	/**
198 	 * Return a Map containing the hidden parameters that are needed for the WServlet to work correctly. Amongst these
199 	 * are the step number and the window id. This map may be mutated (ie, a new Map instance is created and returned).
200 	 * The map may have additional variables added, and it may be turned into a path using {@link WebUtilities#getPath}
201 	 *
202 	 * @return a Map containing the hidden parameters required by WServlet.
203 	 */
204 	Map<String, String> getHiddenParameters();
205 
206 	/**
207 	 * Sets the form encoding type, e.g. "multipart/form-data" when there are files being uploaded.
208 	 *
209 	 * @param enctype the encoding type.
210 	 */
211 	void setFormEncType(String enctype);
212 
213 	/**
214 	 * @return the form encoding type.
215 	 */
216 	String getFormEncType();
217 
218 	/**
219 	 * @return Details about the client browser.
220 	 */
221 	UserAgentInfo getUserAgentInfo();
222 }
