3   import java.io.IOException;
4   import javax.servlet.ServletException;
5   import javax.servlet.http.HttpServlet;
6   import javax.servlet.http.HttpServletRequest;

9   /**
10   * <p>
11   * This servlet is designed to serve up theme files from within the WComponent project or jar file. This saves
12   * developers from having to configure where the webdocs directory is, or extract the theme into their WAR file.
13   * </p>
14   * <p>
15   * The ThemeServlet expects all static resources to be available in the classpath under <code>/theme/XXXXXX/</code>.
16   * Where "XXXXX" is the value of the configuration parameter "<code>theme.name</code>".
17   * </p>
18   *
19   * @author Yiannis Paschalidis
20   * @since 1.0.0
21   * @deprecated no longer required. Theme is now available via WServlet.The bordertech.wcomponents.theme.content.path
22   * parameter im wcomponents-app.properties must not be set and there must be servlet-mappings to the
23   * url-patterns "/{my-mapping}"  and "/{my-mapping}/*" where you specify the value of {my-mapping}.
24   */
25  @Deprecated
26  public class ThemeServlet extends HttpServlet {
27  
28  	/**
29  	 * {@inheritDoc}
30  	 */
31  	@Override
32  	protected void doPost(final HttpServletRequest req, final HttpServletResponse resp) throws
33  			ServletException,
34  			IOException {
35  		doGet(req, resp);
36  	}
37  
38  	/**
39  	 * Serves up a file from the theme.
40  	 *
41  	 * @param req the request with the file name in parameter "f", or following the servlet path.
42  	 * @param resp the response to write to.
43  	 * @throws ServletException on error.
44  	 * @throws IOException if there is an error reading the file / writing the response.
45  	 */
46  	@Override
47  	protected void doGet(final HttpServletRequest req, final HttpServletResponse resp) throws
48  			ServletException,
49  			IOException {
50  		ServletUtil.handleThemeResourceRequest(req, resp);
51  	}
52  }
