129 	 * Check if the request is for a resource (eg static, theme...).
130 	 *
131 	 * @param request the http servlet request.
132 	 * @param response the http servlet response.
133 	 * @return true to continue processing
134 	 * @throws ServletException a servlet exception
135 	 * @throws IOException an IO Exception
136 	 */
137 	public static boolean checkResourceRequest(final HttpServletRequest request,
138 			final HttpServletResponse response)
139 			throws ServletException, IOException {
140 		// Static resource
141 		if (isStaticResourceRequest(request)) {
142 			handleStaticResourceRequest(request, response);
143 			return false;
144 		} else if (isThemeResourceRequest(request)) {  // Theme resource
145 			handleThemeResourceRequest(request, response);
146 			return false;
147 		}
148 
149 		String method = request.getMethod();
150 
151 		if ("HEAD".equals(method)) {
152 			response.setContentType(WebUtilities.CONTENT_TYPE_XML);
153 			return false;
154 		} else if (!"POST".equals(method) && !"GET".equals(method)) {
155 			response.setStatus(HttpServletResponse.SC_NOT_IMPLEMENTED);
156 			return false;
157 		}
158 
159 		return true;
160 	}

162 	/**
163 	 * This method does the real work in servicing the http request. It integrates wcomponents into a servlet
164 	 * environment via a servlet specific helper class.
165 	 *
166 	 * @param helper the servlet helper
167 	 * @param ui the application ui
168 	 * @param interceptorChain the chain of interceptors
169 	 * @throws ServletException a servlet exception
170 	 * @throws IOException an IO Exception
171 	 */
172 	public static void processRequest(final HttpServletHelper helper, final WComponent ui,
173 			final InterceptorComponent interceptorChain) throws ServletException, IOException {
174 
175 		try {
176 			// Tell the support container about the top most web component
177 			// that will service the request/response.
178 			if (interceptorChain == null) {
179 				helper.setWebComponent(ui);
180 			} else {
181 				interceptorChain.attachUI(ui);
182 				helper.setWebComponent(interceptorChain);
183 			}
184 
185 			// Prepare user context
186 			UIContext uic = helper.prepareUserContext();
187 
188 			synchronized (uic) {
189 				// Process the action phase.
190 				helper.processAction();
191 
192 				// Process the render phase.
193 				helper.render();
194 			}
195 		} finally {
196 			// We need to ensure that the AJAX operation is cleared
197 			// The interceptors can not guarantee this
198 			// TODO: Investigate changing to not use a thread-local
199 			AjaxHelper.clearCurrentOperationDetails();
200 		}
201 	}

211 	/**
212 	 * Handles a request for static resources.
213 	 *
214 	 * @param request the http request.
215 	 * @param response the http response.
216 	 */
217 	public static void handleStaticResourceRequest(final HttpServletRequest request,
218 			final HttpServletResponse response) {
219 		String staticRequest = request.getParameter(WServlet.STATIC_RESOURCE_PARAM_NAME);
220 
221 		try {
222 			InternalResource staticResource = InternalResourceMap.getResource(staticRequest);
223 			boolean headersOnly = "HEAD".equals(request.getMethod());
224 
225 			if (staticResource == null) {
226 				LOG.warn("Static resource [" + staticRequest + "] not found.");
227 				response.setStatus(HttpServletResponse.SC_NOT_FOUND);
228 				return;
229 			}
230 
231 			InputStream resourceStream = staticResource.getStream();
232 			if (resourceStream == null) {
233 				LOG.warn(
234 						"Static resource [" + staticRequest + "] not found. Stream for content is null.");
235 				response.setStatus(HttpServletResponse.SC_NOT_FOUND);
236 				return;
237 			}
238 
239 			int size = resourceStream.available();
240 			String fileName = WebUtilities.encodeForContentDispositionHeader(staticRequest.
241 					substring(staticRequest
242 							.lastIndexOf('/') + 1));
243 
244 			if (size > 0) {
245 				response.setContentLength(size);
246 			}
247 
248 			response.setContentType(WebUtilities.getContentType(staticRequest));
249 			response.setHeader("Cache-Control", CacheType.CONTENT_CACHE.getSettings());
250 
251 			String param = request.getParameter(WContent.URL_CONTENT_MODE_PARAMETER_KEY);
252 			if ("inline".equals(param)) {
253 				response.setHeader("Content-Disposition", "inline; filename=" + fileName);
254 			} else if ("attach".equals(param)) {
255 				response.setHeader("Content-Disposition", "attachment; filename=" + fileName);
256 			} else {
257 				// added "filename=" to comply with https://tools.ietf.org/html/rfc6266
258 				response.setHeader("Content-Disposition", "filename=" + fileName);
259 			}
260 
261 			if (!headersOnly) {
262 				StreamUtil.copy(resourceStream, response.getOutputStream());
263 			}
264 		} catch (IOException e) {
265 			LOG.warn("Could not process static resource [" + staticRequest + "]. ", e);
266 			response.reset();
267 			response.setStatus(HttpServletResponse.SC_NOT_FOUND);
268 		}
269 	}

280 	/**
281 	 * Serves up a file from the theme. In practice it is generally a bad idea to use this servlet to serve up static
282 	 * resources. Instead it would make more sense to move CSS, JS, HTML resources to a CDN or similar.
283 	 *
284 	 *
285 	 * @param req the request with the file name in parameter "f", or following the servlet path.
286 	 * @param resp the response to write to.
287 	 * @throws ServletException on error.
288 	 * @throws IOException if there is an error reading the file / writing the response.
289 	 */
290 	public static void handleThemeResourceRequest(final HttpServletRequest req,
291 			final HttpServletResponse resp)
292 			throws ServletException, IOException {
293 
294 		if (req.getHeader("If-Modified-Since") != null) {
295 			resp.reset();
296 			resp.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
297 			return;
298 		}
299 
300 		String fileName = req.getParameter("f");
301 
302 		String path = req.getPathInfo();
303 		if (fileName == null && !Util.empty(path)) {
304 			int offset = path.startsWith(THEME_RESOURCE_PATH_PARAM) ? THEME_RESOURCE_PATH_PARAM.
305 					length() : 1;
306 			fileName = path.substring(offset);
307 		}
308 
309 		if (fileName == null || !checkThemeFile(fileName)) {
310 			resp.setStatus(HttpServletResponse.SC_NOT_FOUND);
311 			return;
312 		}
313 
314 		InputStream resourceStream = null;
315 
316 		try {
317 			URL url = null;
318 
319 			// Check for project translation file
320 			if (fileName.startsWith(THEME_TRANSLATION_RESOURCE_PREFIX)) {
321 				String resourceFileName = fileName.substring(THEME_TRANSLATION_RESOURCE_PREFIX.length());
322 				url = ServletUtil.class.getResource(THEME_PROJECT_TRANSLATION_RESOURCE_PATH + resourceFileName);
323 			}
324 
325 			// Load from the theme path
326 			if (url == null) {
327 				String resourceName = ThemeUtil.getThemeBase() + fileName;
328 				url = ServletUtil.class.getResource(resourceName);
329 			}
330 
331 			if (url == null) {
332 				resp.setStatus(HttpServletResponse.SC_NOT_FOUND);
333 			} else {
334 				URLConnection connection = url.openConnection();
335 				resourceStream = connection.getInputStream();
336 				int size = resourceStream.available();
337 				if (size > 0) {
338 					resp.setContentLength(size);
339 				}
340 
341 				/*
342 				I have commented out the setting of the Content-Disposition on static theme resources because, well why is it there?
343 				If this needs to be reinstated please provide a thorough justification comment here so the reasons are clear.
344 
345 				Note that setting this header breaks Polymer 1.0 when it is present on HTML imports.
346 
347 				String encodedName = WebUtilities.encodeForContentDispositionHeader(fileName.
348 						substring(fileName
349 								.lastIndexOf('/') + 1));
350 				resp.setHeader("Content-Disposition", "filename=" + encodedName);  // "filename=" to comply with https://tools.ietf.org/html/rfc6266
351 				 */
352 				resp.setContentType(WebUtilities.getContentType(fileName));
353 				resp.setHeader("Cache-Control", CacheType.THEME_CACHE.getSettings());
354 
355 				resp.setHeader("Expires", "31536000");
356 				resp.setHeader("ETag", "\"" + WebUtilities.getProjectVersion() + "\"");
357 				// resp.setHeader("Last-Modified", "Mon, 02 Jan 2015 01:00:00 GMT");
358 				long modified = connection.getLastModified();
359 				resp.setDateHeader("Last-Modified", modified);
360 				StreamUtil.copy(resourceStream, resp.getOutputStream());
361 			}
362 		} finally {
363 			StreamUtil.safeClose(resourceStream);
364 		}
365 	}

369 	 *
370 	 * @param name the file name
371 	 * @return true if the requested file name is ok, false if not.
372 	 */
373 	private static boolean checkThemeFile(final String name) {
374 		return !(Util.empty(name) // name must exist
375 				|| name.contains("..") // prevent directory traversal
376 				|| name.charAt(0) == '/' // all theme references should be relative
377 				|| name.indexOf(':') != -1 // forbid use of protocols such as jar:, http: etc.
378 				);
379 	}

382 	 * Creates a new interceptor chain to handle the given request.
383 	 *
384 	 * @param request the request to handle
385 	 * @return a new interceptor chain for the request.
386 	 */
387 	public static InterceptorComponent createInterceptorChain(final HttpServletRequest request) {
388 
389 		// Allow for multi part parameters
390 		Map<String, String[]> parameters = getRequestParameters(request);
391 
392 		InterceptorComponent[] chain;
393 
394 		if (parameters.get(WServlet.DATA_LIST_PARAM_NAME) != null) { // Datalist
395 			chain = new InterceptorComponent[]{
396 				new TransformXMLInterceptor(),
397 				new DataListInterceptor()};
398 
399 		} else if (parameters.get(WServlet.AJAX_TRIGGER_PARAM_NAME) != null) { // AJAX
400 			chain = new InterceptorComponent[]{
401 				new TemplateRenderInterceptor(),
402 				new TransformXMLInterceptor(),
403 				new ValidateXMLInterceptor(),
404 				new AjaxErrorInterceptor(),
405 				new SessionTokenAjaxInterceptor(),
406 				new ResponseCacheInterceptor(CacheType.NO_CACHE),
407 				new UIContextDumpInterceptor(),
408 				new AjaxSetupInterceptor(),
409 				new WWindowInterceptor(true),
410 				new WrongStepAjaxInterceptor(),
411 				new ContextCleanupInterceptor(),
412 				new WhitespaceFilterInterceptor(),
413 				new SubordinateControlInterceptor(),
414 				new AjaxPageShellInterceptor(),
415 				new AjaxDebugStructureInterceptor(),
416 				new AjaxInterceptor()};
417 
418 		} else if (parameters.get(WServlet.TARGET_ID_PARAM_NAME) != null) { // Targetted Content
419 			chain = new InterceptorComponent[]{
420 				new TargetableErrorInterceptor(),
421 				new SessionTokenContentInterceptor(),
422 				new UIContextDumpInterceptor(),
423 				new TargetableInterceptor(),
424 				new WWindowInterceptor(false),
425 				new WrongStepContentInterceptor()};
426 
427 		} else {
428 			chain = new InterceptorComponent[]{ // Page submit
429 				new TemplateRenderInterceptor(),
430 				new TransformXMLInterceptor(),
431 				new ValidateXMLInterceptor(),
432 				new SessionTokenInterceptor(),
433 				new ResponseCacheInterceptor(CacheType.NO_CACHE),
434 				new UIContextDumpInterceptor(),
435 				new WWindowInterceptor(true),
436 				new WrongStepServerInterceptor(),
437 				new AjaxCleanupInterceptor(),
438 				new ContextCleanupInterceptor(),
439 				new WhitespaceFilterInterceptor(),
440 				new SubordinateControlInterceptor(),
441 				new PageShellInterceptor(),
442 				new FormInterceptor(),
443 				new DebugStructureInterceptor()};
444 		}
445 
446 		// Link the interceptors together in a chain.
447 		for (int i = 0; i < chain.length - 1; i++) {
448 			chain[i].setBackingComponent(chain[i + 1]);
449 		}
450 
451 		// Return the top of the chain.
452 		return chain[0];
453 	}

455 	/**
456 	 * Called if a Throwable is caught by the top-level service method. By default we display an error and terminate the
457 	 * session.
458 	 *
459 	 * @param helper the current servlet helper
460 	 * @param throwable the throwable
461 	 * @throws ServletException a servlet exception
462 	 * @throws IOException an IO Exception
463 	 */
464 	public static void handleError(final HttpServletHelper helper, final Throwable throwable) throws
465 			ServletException,
466 			IOException {
467 		HttpServletRequest httpServletRequest = helper.getBackingRequest();
468 		HttpServletResponse httpServletResponse = helper.getBackingResponse();
469 
470 		// Allow for multi part requests
471 		Map<String, String[]> parameters = getRequestParameters(httpServletRequest);
472 
473 		// Set error code for AJAX, Content or data requests
474 		boolean dataRequest = parameters.get(WServlet.DATA_LIST_PARAM_NAME) != null;
475 		Object target = parameters.get(WServlet.AJAX_TRIGGER_PARAM_NAME);
476 		if (target == null) {
477 			target = parameters.get(WServlet.TARGET_ID_PARAM_NAME);
478 		}
479 		if (target != null || dataRequest) {
480 			httpServletResponse.sendError(500, "Internal Error");
481 			return;
482 		}
483 
484 		// Decide whether we should use the ErrorPageFactory.
485 		boolean handleErrorWithFatalErrorPageFactory = ConfigurationProperties.getHandleErrorWithFatalErrorPageFactory();
486 
487 		// use the new technique and delegate to the ErrorPageFactory.
488 		if (handleErrorWithFatalErrorPageFactory) {
489 			helper.handleError(throwable);
490 			helper.dispose();
491 		} else { // use the old technique and just display a raw message.
492 			// First, decide whether we are in friendly mode or not.
493 			boolean friendly = ConfigurationProperties.getDeveloperErrorHandling();
494 
495 			String message = InternalMessages.DEFAULT_SYSTEM_ERROR;
496 
497 			// If we are unfriendly, terminate the session
498 			if (!friendly) {
499 				HttpSession session = httpServletRequest.getSession(true);
500 				session.invalidate();
501 
502 				message = InternalMessages.DEFAULT_SYSTEM_ERROR_SEVERE;
503 			}
504 
505 			// Display an error to the user.
506 			UIContext uic = helper.getUIContext();
507 			Locale locale = uic == null ? null : uic.getLocale();
508 			message = I18nUtilities.format(locale, message);
509 			httpServletResponse.getWriter().println(message);
510 		}
511 	}

619 	 * Extract the parameters and file items allowing for multi part form fields.
620 	 *
621 	 * @param request the request being processed
622 	 * @param parameters the map to store non-file request parameters in.
623 	 * @param files the map to store the uploaded file parameters in.
624 	 */
625 	public static void extractParameterMap(final HttpServletRequest request, final Map<String, String[]> parameters,
626 			final Map<String, FileItem[]> files) {
627 
628 		if (isMultipart(request)) {
629 			ServletFileUpload upload = new ServletFileUpload();
630 			upload.setFileItemFactory(new DiskFileItemFactory());
631 			try {
632 				List fileItems = upload.parseRequest(request);
633 
634 				uploadFileItems(fileItems, parameters, files);
635 			} catch (FileUploadException ex) {
636 				throw new SystemException(ex);
637 			}
638 			// Include Query String Parameters (only if parameters were not included in the form fields)
639 			for (Object entry : request.getParameterMap().entrySet()) {
640 				Map.Entry<String, String[]> param = (Map.Entry<String, String[]>) entry;
641 				if (!parameters.containsKey(param.getKey())) {
642 					parameters.put(param.getKey(), param.getValue());
643 				}
644 			}
645 		} else {
646 			parameters.putAll(request.getParameterMap());
647 		}
648 	}
649 
650 	/**
651 	 * <p>
652 	 * {@link FileItem} classes (if attachements) will be kept as part of the request. The default behaviour of the file
653 	 * item is to store the upload in memory until it reaches a certain size, after which the content is streamed to a
654 	 * temp file.</p>
655 	 *
656 	 * <p>
657 	 * If, in the future, performance of uploads becomes a focus we can instead look into using the Jakarta Commons
658 	 * Streaming API. In this case, the content of the upload isn't stored anywhere. It will be up to the user to
659 	 * read/store the content of the stream.</p>
660 	 *
661 	 * @param fileItems a list of {@link FileItem}s corresponding to POSTed form data.
662 	 * @param parameters the map to store non-file request parameters in.
663 	 * @param files the map to store the uploaded file parameters in.
664 	 */
665 	public static void uploadFileItems(final List<FileItem> fileItems, final Map<String, String[]> parameters,
666 			final Map<String, FileItem[]> files) {
667 
668 		for (FileItem item : fileItems) {
669 			String name = item.getFieldName();
670 			boolean formField = item.isFormField();
671 
672 			if (LOG.isDebugEnabled()) {
673 				LOG.debug(
674 						"Uploading form " + (formField ? "field" : "attachment") + " \"" + name + "\"");
675 			}
676 
677 			if (formField) {
678 				String value;
679 				try {
680 					// Without specifying UTF-8, apache commons DiskFileItem defaults to ISO-8859-1.
681 					value = item.getString("UTF-8");
682 				} catch (UnsupportedEncodingException e) {
683 					throw new SystemException("Encoding error on formField item", e);
684 				}
685 				RequestUtil.addParameter(parameters, name, value);
686 			} else {
687 				// Form attachment
688 				RequestUtil.addFileItem(files, name, item);
689 				String value = item.getName();
690 				RequestUtil.addParameter(parameters, name, value);
691 			}
692 		}
693 
694 	}

696 	/**
697 	 * Determine the user's device type from the {@link HttpServletRequest}.
698 	 *
699 	 * @param request the request being processed
700 	 * @return the device type
701 	 */
702 	public static DeviceType getDevice(final HttpServletRequest request) {
703 		// User agent
704 		String userAgent = ((HttpServletRequest) request).getHeader("User-Agent");
705 		if (Util.empty(userAgent)) {
706 			LOG.warn("No User-Agent details in the request headers. Will assume normal device.");
707 			return DeviceType.NORMAL;
708 		}
709 
710 		// Check for device type
711 		UAgentInfo agentInfo = new UAgentInfo(userAgent, null);
712 		if (agentInfo.detectMobileQuick()) {
713 			return DeviceType.MOBILE;
714 		} else if (agentInfo.detectTierTablet()) {
715 			return DeviceType.TABLET;
716 		}
717 		return DeviceType.NORMAL;
718 	}

720 	/**
721 	 * Helper method that uses the current WComponents {@link Request} interface to determine the user's device type.
722 	 *
723 	 * @param request the request being processed
724 	 * @return the device type
725 	 */
726 	public static DeviceType getDevice(final Request request) {
727 		if (request instanceof ServletRequest) {
728 			return getDevice(((ServletRequest) request).getBackingRequest());
729 		}
730 		return DeviceType.NORMAL;
