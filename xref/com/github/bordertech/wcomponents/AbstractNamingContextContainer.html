1   package com.github.bordertech.wcomponents;

3   /**
4    * This abstract container implements {@link NamingContextable}.
5    *
6    * @author Jonathan Austin
7    * @since 1.0.0
8    */
9   public abstract class AbstractNamingContextContainer extends AbstractContainer implements NamingContextable {
10  	/**
11  	 * A naming context is only considered active if it has been set active via {@link #setNamingContext(boolean)} and
12  	 * also has an id name set via {@link #setIdName(String)}.
13  	 *
14  	 * @param context set true if this is a naming context.
15  	 */
16  	public void setNamingContext(final boolean context) {
17  		setFlag(ComponentModel.NAMING_CONTEXT_FLAG, context);
18  	}
19  
20  	/**
21  	 * {@inheritDoc}
22  	 */
23  	@Override
24  	public boolean isNamingContext() {
25  		return isFlagSet(ComponentModel.NAMING_CONTEXT_FLAG);
26  	}
27  
28  	/**
29  	 * {@inheritDoc}
30  	 */
31  	@Override
32  	public String getNamingContextId() {
33  		return getId();
34  	}
35  }
