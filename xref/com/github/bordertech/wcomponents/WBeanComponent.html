13  /**
14   * WBeanComponent provides a default implementation of a bean-aware component, and is the basis for most bean-aware
15   * WComponents. It can be used as a starting point for custom application bean-aware components.
16   * <p>
17   * A fix has been made to the logic for a bean container, that has a bean property set, to pass the correct bean value
18   * to its child bean components. This fix is an "opt in". To enable the correct bean container logic set the parameter
19   * "bordertech.wcomponents.bean.logic.correct=true".
20   * </p>
21   *
22   * @author Yiannis Paschalidis
23   * @author Jonathan Austin
24   * @since 1.0.0
25   */
26  public class WBeanComponent extends AbstractWComponent implements DataBound, BeanBound,
27  		BeanProviderBound {

44  	/**
45  	 * <p>
46  	 * Retrieves the bean. The following are searched in order:</p>
47  	 * <ul>
48  	 * <li>a bean set explicitly using <code>setBean</code>;</li>
49  	 * <li>a bean cached in the scratch map;</li>
50  	 * <li>a bean provided by a BeanProvider (and subsequently cached in the scratch map);</li>
51  	 * <li>a bean provided by a bean-aware parent component (and subsequently cached in the scratch map).</li>
52  	 * </ul>
53  	 *
54  	 * @return this component's bean for the given context
55  	 */
56  	@Override
57  	public Object getBean() {
58  
59  		// Check if bean is on the user model
60  		BeanAndProviderBoundComponentModel model = getComponentModel();
61  		Object bean = model.getBean();
62  		if (bean != null) {
63  			return bean;
64  		}
65  
66  		// Check if using a bean provider
67  		BeanProvider beanProvider = getBeanProvider();
68  		if (beanProvider != null) {
69  			Object beanId = getBeanId();
70  
71  			// Check if bean is in the scratch map
72  			if (isBeanInScratchMap(beanId)) {
73  				return getBeanFromScratchMap();
74  			}
75  
76  			// Get the bean from the provider
77  			bean = beanProvider.getBean(this);
78  
79  			// Cache the value as it may have been the result of an expensive operation.
80  			addBeanToScratchMap(beanId, bean);
81  
82  			return bean;
83  		}
84  
85  		// Check if search ancestors for bean
86  		if (!isSearchAncestors()) {
87  			return null;
88  		}
89  
90  		// Search for a bean in a bean aware parent.
91  		// We explicitly do not cache the bean, as we can't tell if the parent's
92  		// value changes. In any case, it will have been cached by the parent.
93  		BeanAware parent = WebUtilities.getAncestorOfClass(BeanAware.class, this);
94  		if (parent != null) {
95  			String parentBeanProperty = parent.getBeanProperty();
96  			// Correct
97  			if (ConfigurationProperties.getCorrectBeanLogic()) {
98  				if (parentBeanProperty == null || ".".equals(parentBeanProperty)) {
99  					bean = parent.getBean();
100 				} else {
101 					bean = parent.getBeanValue();
102 				}
103 			} else { // Legacy
104 				if (parentBeanProperty != null && !".".equals(parentBeanProperty) && parentBeanProperty.
105 						contains(".")) {
106 					LOG.warn("Possible bean property logic error with bean property ["
107 							+ parentBeanProperty + "]. Check runtime parameter " + ConfigurationProperties.getCorrectBeanLogic() + ".");
108 				}
109 				bean = parent.getBean();
110 			}
111 		}
112 
113 		return bean;
114 	}

203 	 *
204 	 * @param beanProvider the bean provider to associate
205 	 */
206 	@Override

208 		if (isLocked() && beanProvider != null && !(beanProvider instanceof Serializable)) {
209 			throw new SystemException(
210 					"Unable to store bean provider in user's session as it is not serializable: " + beanProvider.
211 					getClass());
212 		}
213 
214 		getOrCreateComponentModel().setBeanProvider(beanProvider);
215 		String beanProperty = getBeanProperty();
216 
217 		if (beanProperty == null) {
218 			setBeanProperty(".");
219 		}
220 
221 		// Remove values in scratch map
222 		removeBeanFromScratchMap();
223 	}

232 	/**
233 	 * Returns the data for this component. The following are searched in order:
234 	 * <ul>
235 	 * <li>a value set explicitly in the ui context using {@link #setData(Object)};</li>
236 	 * <li>if a bean is available; the bean's value; or</li>
237 	 * <li>the value set on the shared model.</li>
238 	 * </ul>
239 	 *
240 	 * @return the current value of this component for the given context
241 	 */
242 	@Override
243 	public Object getData() {
244 		BeanAndProviderBoundComponentModel model = getComponentModel();
245 		Object data = model.getData();
246 		String beanProperty = getBeanProperty();
247 
248 		if (beanProperty != null) {
249 			Object sharedData = ((BeanAndProviderBoundComponentModel) getDefaultModel()).getData();
250 
251 			if (!isFlagSet(ComponentModel.USER_DATA_SET) && Util.equals(data, sharedData)) {
252 				Object bean = getBean();
253 
254 				if (bean != null) {
255 					data = getBeanValue();
256 				}
257 			}
258 		}
259 
260 		return data;
261 	}

263 	/**
264 	 * Sets the data that this component displays/edits. For bean aware components, this should only be called from
265 	 * handleRequest to set user-entered data.
266 	 *
267 	 * @param data the data to set
268 	 */
269 	@Override
270 	public void setData(final Object data) {
271 		UIContext uic = UIContextHolder.getCurrent();
272 
273 		if (uic == null) {
274 			getComponentModel().setData(data);
275 		} else {
276 			Object sharedValue = ((BeanAndProviderBoundComponentModel) getDefaultModel()).getData();
277 
278 			if (getBeanProperty() != null) {
279 				sharedValue = getBeanValue();
280 			}
281 
282 			getOrCreateComponentModel().setData(data);
283 			setFlag(ComponentModel.USER_DATA_SET, !Util.equals(data, sharedValue));
284 		}
285 	}

287 	/**
288 	 * Retrieves the bean value. The value is (temporarily) cached in the scratch map to speed up subsequent accesses.
289 	 *
290 	 * @return the bean value, or null if no bean value is available
291 	 */
292 	@Override
293 	public Object getBeanValue() {
294 		Object bean = getBean();
295 		Object beanValue = null;
296 
297 		if (bean != null) {
298 			String beanProperty = getBeanProperty();
299 
300 			if (beanProperty == null || ".".equals(beanProperty)) {
301 				beanValue = bean;
302 			} else {
303 				try {
304 					beanValue = PropertyUtils.getProperty(bean, beanProperty);
305 				} catch (Exception e) {
306 					LOG.error("Failed to read bean property " + beanProperty + " from " + bean, e);
307 				}
308 			}
309 		}
310 
311 		return beanValue;
312 	}

322 	/**
323 	 * Updates the bean value with the new value.
324 	 *
325 	 * @param value the new value with which to update the bean
326 	 */
327 	protected void doUpdateBeanValue(final Object value) {
328 		String beanProperty = getBeanProperty();
329 		if (beanProperty != null && beanProperty.length() > 0 && !".".equals(beanProperty)) {
330 			Object bean = getBean();
331 			if (bean != null) {
332 				try {
333 					Object beanValue = getBeanValue();
334 					if (!Util.equals(beanValue, value)) {
335 						PropertyUtils.setProperty(bean, beanProperty, value);
336 					}
337 				} catch (Exception e) {
338 					LOG.error("Failed to set bean property " + beanProperty + " on " + bean);
339 				}
340 			}
341 		}
342 	}

353 	 * Remove the bean from the scratch maps.
354 	 */
355 	protected void removeBeanFromScratchMap() {
356 		Map<Object, Object> scratchMap = getBeanScratchMap();
357 		if (scratchMap == null) {
358 			return;
359 		}
360 		scratchMap.remove(SCRATCHMAP_BEAN_ID_KEY);
361 		scratchMap.remove(SCRATCHMAP_BEAN_OBJECT_KEY);
362 	}

364 	/**
365 	 * @param beanId check if this bean id is in the scratch map
366 	 * @return true if current bean is in the scratch map
367 	 */
368 	protected boolean isBeanInScratchMap(final Object beanId) {
369 		Map<Object, Object> scratchMap = getBeanScratchMap();
370 		if (scratchMap == null) {
371 			return false;
372 		}
373 		if (scratchMap.containsKey(SCRATCHMAP_BEAN_OBJECT_KEY) && scratchMap.containsKey(SCRATCHMAP_BEAN_ID_KEY)) {
374 			Object scratchId = scratchMap.get(SCRATCHMAP_BEAN_ID_KEY);
375 			return Objects.equals(scratchId, beanId);
376 		}
377 		return false;
378 	}
379 
380 	/**
381 	 * @return the bean from the scratch map
382 	 */
383 	protected Object getBeanFromScratchMap() {
384 		Map<Object, Object> scratchMap = getBeanScratchMap();
385 		if (scratchMap == null) {
386 			return null;
387 		}
388 		Object bean = scratchMap.get(SCRATCHMAP_BEAN_OBJECT_KEY);
389 		return bean;
390 	}
391 
392 	/**
393 	 *
394 	 * @param beanId the bean id to put in the scratch map
395 	 * @param bean the bean to put in the scratch map
396 	 */
397 	protected void addBeanToScratchMap(final Object beanId, final Object bean) {
398 		Map<Object, Object> scratchMap = getBeanScratchMap();
399 		if (scratchMap != null) {
400 			scratchMap.put(SCRATCHMAP_BEAN_ID_KEY, beanId);
401 			scratchMap.put(SCRATCHMAP_BEAN_OBJECT_KEY, bean);
402 		}
403 	}
404 
405 	/**
406 	 * @return the scratch map with the correct scope.
407 	 */
408 	protected Map<Object, Object> getBeanScratchMap() {
409 		UIContext uic = UIContextHolder.getCurrent();
410 		if (uic == null) {
411 			return null;
412 		}
413 		if (isUseRequestScopeScratchMap()) {
414 			return uic.getRequestScratchMap(this);
415 		} else {
416 			return uic.getScratchMap(this);
417 		}
418 	}

421 	 * Indicates whether this component's data has changed from the default value.
422 	 *
423 	 * TODO: This needs to be added to the databound interface after the bulk of the components have been converted.
424 	 *
425 	 * @return true if this component's current value differs from the default value for the given context.
426 	 */
427 	public boolean isChanged() {
428 		Object currentValue = getData();
429 		Object sharedValue = ((BeanAndProviderBoundComponentModel) getDefaultModel()).getData();
430 
431 		if (getBeanProperty() != null) {
432 			sharedValue = getBeanValue();
433 		}
434 
435 		return !Util.equals(currentValue, sharedValue);
436 	}
