8   /**
9    * <p>
10   * This component is used to provide either a dropmenu (when added directly to a {@link WMenu} or nestable sub menus
11   * (when added to another <code>WSubMenu</code> instance).</p>
12   *
13   * <p>
14   * Sub Menus may contain the following components:
15   * </p>
16   * <ul>
17   * <li>Other sub menus.</li>
18   * <li>Menu items ({@link WMenuItem})</li>
19   * <li>Menu separators (see {@link #addSeparator()})</li>
20   * </ul>
21   *
22   * <p>
23   * Actions on sub-menus are only supported for {@link WMenu.MenuType#COLUMN} menus.</p>
24   *
25   * @author Adam Millard
26   * @author Yiannis Paschalidis - re-written to not extend WButton.
27   */
28  public class WSubMenu extends AbstractNamingContextContainer implements Disableable {

31  	 * The available types of operation.
32  	 *
33  	 * @author Yiannis Paschalidis

35  	public enum MenuMode {
36  		/**
37  		 * Indicates that a round-trip should be made whenever the menu is opened.
38  		 *
39  		 * @deprecated Use MenuMode DYNAMIC instead as a like-for-like replacement or any other mode if it is more
40  		 * appropriate to the individual use case.
41  		 */
42  		SERVER,
43  		/**
44  		 * Indicates that an ajax request should be made the first time the menu is opened.
45  		 */
46  		LAZY,
47  		/**
48  		 * Indicates that the sub-menu content is always sent to the client.
49  		 */
50  		CLIENT,
51  		/**
52  		 * Indicates that an ajax request should be made whenever the menu is opened.
53  		 */
54  		DYNAMIC,
55  		/**
56  		 * Indicates that an ajax request should be made immediately after the page is loaded.
57  		 */
58  		EAGER
59  	};

364 	/**
365 	 * Override handleRequest in order to perform processing for this component. This implementation checks for submenu
366 	 * selection and executes the associated action if it has been set.
367 	 *
368 	 * @param request the request being responded to.
369 	 */
370 	@Override
371 	public void handleRequest(final Request request) {
372 		if (isDisabled()) {
373 			// Protect against client-side tampering of disabled/read-only fields.
374 			return;
375 		}
376 
377 		if (isMenuPresent(request)) {
378 			// If current ajax trigger, process menu for current selections
379 			if (AjaxHelper.isCurrentAjaxTrigger(this)) {
380 				WMenu menu = WebUtilities.getAncestorOfClass(WMenu.class, this);
381 				menu.handleRequest(request);
382 			}
383 
384 			String requestValue = request.getParameter(getId());
385 
386 			if (requestValue != null) {
387 				// Execute associated action, if set
388 				final Action action = getAction();
389 
390 				if (action != null) {
391 					final ActionEvent event = new ActionEvent(this, this.getActionCommand(),
392 							this.getActionObject());
393 
394 					Runnable later = new Runnable() {
395 						@Override
396 						public void run() {
397 							action.execute(event);
398 						}
399 					};
400 
401 					invokeLater(later);
402 				}
403 			}
404 
405 			boolean openState = "true".equals(request.getParameter(getId() + ".open"));
406 			setOpen(openState);
407 		}
408 	}

421 	/**
422 	 * Override preparePaintComponent in order to correct the visibility of the sub-menu's children before they are
423 	 * rendered.
424 	 *
425 	 * @param request the request being responded to.
426 	 */
427 	@Override
428 	protected void preparePaintComponent(final Request request) {
429 		super.preparePaintComponent(request);
430 		List<WComponent> children = new ArrayList<>(getChildCount());
431 
432 		for (int i = 0; i < getChildCount(); i++) {
433 			WComponent child = getChildAt(i);
434 
435 			if (child != label) {
436 				children.add(child);
437 			}
438 		}
439 
440 		// String of children Ids
441 		List<String> childrenIds = new ArrayList<>();
442 		for (WComponent child : children) {
443 			childrenIds.add(child.getId());
444 		}
445 
446 		switch (getComponentModel().mode) {
447 			case LAZY: {
448 				for (WComponent child : children) {
449 					child.setVisible(isOpen());
450 				}
451 
452 				if (!isOpen()) {
453 					AjaxHelper.
454 							registerContainer(getId(), getId() + "-content", childrenIds, request);
455 				}
456 
457 				break;
458 			}
459 			case DYNAMIC: {
460 				AjaxHelper.registerContainer(getId(), getId() + "-content", childrenIds, request);
461 
462 				for (WComponent child : children) {
463 					child.setVisible(isOpen());
464 				}
465 
466 				break;
467 			}
468 			case EAGER: {
469 				AjaxHelper.registerContainer(getId(), getId() + "-content", childrenIds, request);
470 				// Will always be visible
471 				break;
472 			}
473 			case SERVER: {
474 				for (WComponent child : children) {
475 					child.setVisible(isOpen());
476 				}
477 
478 				break;
479 			}
480 			case CLIENT: {
481 				// Will always be visible
482 				break;
483 			}
484 		}
485 	}
