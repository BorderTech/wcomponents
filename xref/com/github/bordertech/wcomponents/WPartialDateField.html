14  /**
15   * <p>
16   * This component is a date entry component. Individual themes may add features such as automatic text formatting and
17   * date picker widgets. Convenience methods exist to set and get the value as a java date object.
18   * </p>
19   * <p>
20   * "Partial dates" can be specified without day or month components, e.g. "JAN 2001", or just "2001".
21   * </p>
22   * <p>
23   * When the component is bound to a bean, the bean property must be a string containing the date in the format
24   * "yyyyMMdd". Parts of the date that do not have a value are padded with space characters by default. The padding
25   * character can be changed using the {@link #setPaddingChar(char)} method. An example of a partial date bean value for
26   * "Mar 2012" would be "201203". Note, trailing spaces are removed. If a padding character of '@' was specified, then
27   * the bean value would be "201203@@".
28   * </p>
29   *
30   * @author Ming Gao
31   * @author Jonathan Austin
32   */
33  public class WPartialDateField extends AbstractInput implements AjaxTrigger, AjaxTarget,

297 	@Override
298 	protected boolean doHandleRequest(final Request request) {
299 		// Valid date entered by the user
300 		String dateValue = getRequestValue(request);
301 		// Text entered by the user (An empty string is treated as null)
302 		String value = request.getParameter(getId());
303 		String text = (Util.empty(value)) ? null : value;
304 
305 		// Current date value
306 		String currentDate = getValue();
307 
308 		boolean changed = false;
309 
310 		// If a "valid" date value has not been entered, then check if the "user text" has changed
311 		if (dateValue == null) {
312 			// User entered text
313 			changed = !Util.equals(text, getText()) || currentDate != null;
314 		} else {
315 			// Valid Date
316 			changed = !Util.equals(dateValue, currentDate);
317 		}
318 
319 		if (changed) {
320 			boolean valid = dateValue != null || text == null;
321 			handleRequestValue(dateValue, valid, text);
322 		}
323 
324 		return changed;
325 	}

342 	/**
343 	 * {@inheritDoc}
344 	 */
345 	@Override
346 	public String getRequestValue(final Request request) {
347 		if (isPresent(request)) {
348 			// User entered a valid date
349 			String dateParam = request.getParameter(getId() + "-date");
350 			if (dateParam == null) {
351 				return null;
352 			}
353 
354 			// Validate Transfer Date Format - YYYY-MM-DD
355 			if (dateParam.length() != INTERNAL_DATE_TOTAL_CHARS || dateParam.charAt(
356 					INTERNAL_DASH1_POS) != '-'
357 					|| dateParam.charAt(INTERNAL_DASH2_POS) != '-') {
358 				LOG.warn("Date parameter is not in the format yyyy-MM-dd (" + dateParam
359 						+ ") and will be treated as null.");
360 				return null;
361 			}
362 
363 			// Transform YYYY-MM-DD to YYYYMMDD
364 			StringBuffer buf = new StringBuffer(DATE_TOTAL_CHARS);
365 			buf.append(dateParam.substring(INTERNAL_YEAR_START, INTERNAL_YEAR_END));
366 			buf.append(dateParam.substring(INTERNAL_MONTH_START, INTERNAL_MONTH_END));
367 			buf.append(dateParam.substring(INTERNAL_DAY_START, INTERNAL_DAY_END));
368 
369 			String dateFormat = buf.toString();
370 
371 			// Validate the date
372 			if (!isValidPartialDateStringFormat(dateFormat, THEME_PADDING_CHAR)) {
373 				LOG.warn("Date parameter ("
374 						+ dateParam
375 						+ ") could not be transformed from YYYY-MM-DD to the format YYYYMMDD and will be treated as null.");
376 				return null;
377 			}
378 
379 			// Convert from Theme padding character to the correct padding character
380 			return dateFormat.replace(THEME_PADDING_CHAR, getPaddingChar());
381 		} else {
382 			return getValue();
383 		}
384 	}

581 	 * Set the WPartialDateField with the given day, month and year. Each of the day, month and year parameters that
582 	 * make up the partial date are optional.
583 	 *
584 	 * @param day a number from 1 to 31 or null if unknown.
585 	 * @param month a number from 1 to 12, or null if unknown.
586 	 * @param year a number, or null if unknown.
587 	 * @param padding the padding character for the partial date
588 	 * @return the formatted partial date, or null if invalid
589 	 */
590 	private String formatPartialDateToString(final Integer day, final Integer month,
591 			final Integer year,
592 			final char padding) {
593 		if (year == null && month == null && day == null) {
594 			return null;
595 		}
596 
597 		if (!isValidYear(year) || !isValidMonth(month) || !isValidDay(day)) {
598 			return null;
599 		}
600 
601 		StringBuffer dateString = new StringBuffer(DATE_TOTAL_CHARS);
602 
603 		append(dateString, year, YEAR_DIGITS, padding);
604 		append(dateString, month, MONTH_DIGITS, padding);
605 		append(dateString, day, DAY_DIGITS, padding);
606 
607 		// TRIM trailing spaces (will only "trim" if the padding character is a space)
608 		String trimmed = Util.rightTrim(dateString.toString());
609 		return trimmed;
610 	}

642 	/**
643 	 * Parses a component of a partial date.
644 	 *
645 	 * @param component the date component.
646 	 * @param padding the padding character.
647 	 * @return the parsed value, may be null.
648 	 */
649 	private boolean isValidPartialDateStringFormat(final String component, final char padding) {
650 		// Empty is not valid
651 		if (Util.empty(component)) {
652 			return false;
653 		}
654 
655 		int length = component.length();
656 
657 		// Check length YYYY, or YYYYMM, or YYYYMMDD
658 		if (length != YEAR_END && length != MONTH_END && length != DAY_END) {
659 			return false;
660 		}
661 
662 		// Year - ???? or YYYY
663 		String year = component.substring(YEAR_START, YEAR_END);
664 		if (!isValidCharacters(year, padding) || !isValidYear(parseDateComponent(year, padding))) {
665 			return false;
666 		}
667 
668 		// Month - ?? or MM
669 		if (component.length() >= MONTH_END) {
670 			String month = component.substring(MONTH_START, MONTH_END);
671 			if (!isValidCharacters(month, padding) || !isValidMonth(parseDateComponent(month,
672 					padding))) {
673 				return false;
674 			}
675 		}
676 
677 		// Day - ?? or DD
678 		if (component.length() == DAY_END) {
679 			String day = component.substring(DAY_START, DAY_END);
680 			if (!isValidCharacters(day, padding) || !isValidDay(parseDateComponent(day, padding))) {
681 				return false;
682 			}
683 		}
684 
685 		return true;
686 	}

688 	/**
689 	 * Check the component is either all padding chars or all digit chars.
690 	 *
691 	 * @param component the date component.
692 	 * @param padding the padding character.
693 	 * @return true if the component is valid, otherwise false
694 	 */
695 	private boolean isValidCharacters(final String component, final char padding) {
696 		// Check the component is either all padding chars or all digit chars
697 		boolean paddingChars = false;
698 		boolean digitChars = false;
699 		for (int i = 0; i < component.length(); i++) {
700 			char chr = component.charAt(i);
701 			// Padding
702 			if (chr == padding) {
703 				if (digitChars) {
704 					return false;
705 				}
706 				paddingChars = true;
707 			} else if (chr >= '0' && chr <= '9') {  // Digit
708 				if (paddingChars) {
709 					return false;
710 				}
711 				digitChars = true;
712 			} else {
713 				return false;
714 			}
715 		}
716 
717 		return true;
718 	}
719 
720 	/**
721 	 * Parses a component of a partial date.
722 	 *
723 	 * @param component the date component.
724 	 * @param padding the padding character.
725 	 * @return the parsed value, may be null.
726 	 */
727 	private Integer parseDateComponent(final String component, final char padding) {
728 		if (component != null && component.indexOf(padding) == -1) {
729 			try {
730 				return Integer.valueOf(component);
731 			} catch (NumberFormatException e) {
732 				return null;
733 			}
734 		}
735 		return null;
736 	}
737 
738 	/**
739 	 * Appends a single date component to the given StringBuffer. Nulls are replaced with the padding char, and numbers
740 	 * are padded with zeros.
741 	 *
742 	 * @param buf the buffer to append to.
743 	 * @param num the number to append, may be null.
744 	 * @param digits the minimum number of digits to append.
745 	 * @param padding the padding character for null values
746 	 */
747 	private void append(final StringBuffer buf, final Integer num, final int digits,
748 			final char padding) {
749 		if (num == null) {
750 			for (int i = 0; i < digits; i++) {
751 				buf.append(padding);
752 			}
753 		} else {
754 			for (int digit = 1, test = 10; digit < digits; digit++, test *= 10) {
755 				if (num < test) {
756 					buf.append('0');
757 				}
758 			}
759 
760 			buf.append(num);
761 		}
762 	}
