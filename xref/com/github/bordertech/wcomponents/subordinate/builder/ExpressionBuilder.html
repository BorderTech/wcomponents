158 	// Group Expressions - OR, AND, NOT
159 	//
160 	/**
161 	 * Appends an OR expression to the RHS of the expression. The current RHS of the expression must be an Operand.
162 	 *
163 	 * @return this ExpressionBuilder.
164 	 */
165 	public ExpressionBuilder or() {
166 		GroupExpression lastGroupExpression = stack.isEmpty() ? null : stack.peek();
167 
168 		if (lhsExpression == null) {
169 			throw new SyntaxException("Syntax exception: OR missing LHS operand");
170 		} else if (lastGroupExpression == null) {
171 			GroupExpression or = new GroupExpression(GroupExpression.Type.OR);
172 			or.add(lhsExpression);
173 			stack.push(or);
174 			expression.setExpression(or);
175 			lhsExpression = null;
176 		} else if (lastGroupExpression.getType().equals(GroupExpression.Type.OR)) {
177 			// Keep using the existing OR
178 			lhsExpression = null;
179 			return this;
180 		} else if (lastGroupExpression.getType().equals(GroupExpression.Type.AND)) {
181 			// AND takes precedence over OR, so we wrap the AND
182 			// by removing it from any parent expressions and inserting the OR in its place
183 
184 			GroupExpression and = stack.pop();
185 			GroupExpression or = new GroupExpression(GroupExpression.Type.OR);
186 
187 			if (stack.isEmpty()) {
188 				expression.setExpression(or);
189 				or.add(and);
190 			} else {
191 				// need to get at the parent
192 				GroupExpression parent = stack.pop();
193 				parent.remove(and);
194 				parent.add(or);
195 				or.add(and);
196 			}
197 
198 			stack.push(and);
199 			stack.push(or);
200 			lhsExpression = null;
201 		}
202 
203 		return this;
204 	}

206 	/**
207 	 * Appends an AND expression to the RHS of the expression. The current RHS of the expression must be an Operand.
208 	 *
209 	 * @return this ExpressionBuilder.
210 	 */
211 	public ExpressionBuilder and() {
212 		GroupExpression lastGroupExpression = stack.isEmpty() ? null : stack.peek();
213 
214 		if (lhsExpression == null) {
215 			throw new SyntaxException("Syntax exception: AND missing LHS operand");
216 		} else if (lastGroupExpression == null) {
217 			GroupExpression and = new GroupExpression(GroupExpression.Type.AND);
218 			and.add(lhsExpression);
219 			stack.push(and);
220 			expression.setExpression(and);
221 			lhsExpression = null;
222 		} else if (lastGroupExpression.getType().equals(GroupExpression.Type.AND)) {
223 			// Keep using the existing AND
224 			lhsExpression = null;
225 			return this;
226 		} else if (lastGroupExpression.getType().equals(GroupExpression.Type.OR)) {
227 			// AND takes precedence over OR, so we steal the OR's RHS
228 			GroupExpression or = lastGroupExpression;
229 			GroupExpression and = new GroupExpression(GroupExpression.Type.AND);
230 
231 			or.remove(lhsExpression);
232 			and.add(lhsExpression);
233 			or.add(and);
234 			stack.push(and);
235 			lhsExpression = null;
236 		}
237 
238 		return this;
239 	}

264 	 * Appends a NOT expression to this expression.
265 	 *
266 	 * @param exp the builder containing the expression to NOT with the current expression.
267 	 * @return this ExpressionBuilder.
268 	 */
269 	public ExpressionBuilder not(final ExpressionBuilder exp) {
270 		// Note - NOT Expressions are not added to the stack (like the other group expressions AND and OR) and are
271 		// treated the same as a "compare" expression.
272 		// It is expected a NOT Expression is used by itself or with an AND or NOT.
273 		GroupExpression not = new GroupExpression(Type.NOT);
274 		not.add(exp.expression.getExpression());
275 
276 		appendExpression(not);
277 
278 		return this;
279 	}

281 	/**
282 	 * Appends the given expression to this expression.
283 	 *
284 	 * @param newExpression the expression to append.
285 	 */
286 	private void appendExpression(final BooleanExpression newExpression) {
287 		if (lhsExpression != null) {
288 			throw new SyntaxException("Syntax exception: use AND or OR to join expressions");
289 		}
290 
291 		lhsExpression = newExpression;
292 
293 		GroupExpression currentExpression = stack.isEmpty() ? null : stack.peek();
294 
295 		if (currentExpression == null) {
296 			this.expression.setExpression(newExpression);
297 		} else {
298 			currentExpression.add(newExpression);
299 		}
300 	}

302 	/**
303 	 * Returns the built expression if it passes basic validation.
304 	 *
305 	 * @return the built expression, or null if the expression is invalid.
306 	 */
307 	protected BooleanExpression build() {
308 		if (validate()) {
309 			return expression.getExpression();
310 		}
311 
312 		return null;
313 	}
314 
315 	/**
316 	 * Determines whether the current expression is syntactically correct.
317 	 *
318 	 * @return true if the current expression is valid, false if not.
319 	 */
320 	public boolean validate() {
321 		try {
322 			BooleanExpression built = expression.getExpression();
323 
324 			if (built == null) {
325 				// nothing to evaluate.
326 				return false;
327 			}
328 
329 			// First check the nesting (an expression must not contain itself)
330 			checkNesting(built, new ArrayList<BooleanExpression>());
331 
332 			// If the expression evaluates correctly, the syntax is correct.
333 			built.evaluate();
334 		} catch (Exception e) {
335 			LogFactory.getLog(getClass()).warn("Invalid expression: " + e.getMessage());
336 			return false;
337 		}
338 
339 		return true;
340 	}

342 	/**
343 	 * Checks nesting of expressions to ensure we don't end up in an infinite recursive loop during evaluation.
344 	 *
345 	 * @param expression the expression to check
346 	 * @param visitedExpressions a list of expressions which have been visited so far.
347 	 */
348 	private static void checkNesting(final BooleanExpression expression,
349 			final List<BooleanExpression> visitedExpressions) {
350 		if (visitedExpressions.contains(expression)) {
351 			// Unfortunately, we can't give much more information - even calling toString() will overflow the stack.
352 			throw new SyntaxException("An expression can not contain itself.");
353 		}
354 
355 		visitedExpressions.add(expression);
356 
357 		if (expression instanceof GroupExpression) {
358 			GroupExpression group = (GroupExpression) expression;
359 			for (BooleanExpression operand : group.getOperands()) {
360 				checkNesting(operand, visitedExpressions);
361 			}
362 		}
363 	}
