9   import com.github.bordertech.wcomponents.subordinate.EnableInGroup;

17  /**
18   * Describes an action to execute. This class is used by the {@link SubordinateBuilder} class to define what needs to be
19   * done when a condition is met.
20   *
21   * @author Yiannis Paschalidis
22   * @author Jonathan Austin
23   * @since 1.0.0
24   */
25  public final class Action {
26  
27  	/**
28  	 * The type of action to perform.
29  	 */
30  	private final com.github.bordertech.wcomponents.subordinate.Action.ActionType type;
31  
32  	/**
33  	 * The object to perform the action on.
34  	 */
35  	private final SubordinateTarget target;
36  
37  	/**
38  	 * The group of targets that may be used in the action.
39  	 */
40  	private final WComponentGroup<? extends SubordinateTarget> group;
41  
42  	/**
43  	 * Creates an action.
44  	 *
45  	 * @param type the type of action to execute.
46  	 * @param target the target to execute the action on.
47  	 */
48  	public Action(final com.github.bordertech.wcomponents.subordinate.Action.ActionType type,
49  			final SubordinateTarget target) {
50  		this(type, target, null);
51  	}
52  
53  	/**
54  	 * Creates an action.
55  	 *
56  	 * @param type the type of action to execute.
57  	 * @param target the target to execute the action on.
58  	 * @param group the group of targets to use on the action.
59  	 */
60  	public Action(final com.github.bordertech.wcomponents.subordinate.Action.ActionType type,
61  			final SubordinateTarget target,
62  			final WComponentGroup<? extends SubordinateTarget> group) {
63  		if (type == null) {
64  			throw new IllegalArgumentException("Action type can not be null");
65  		}
66  
67  		if (target == null) {
68  			throw new IllegalArgumentException("Action target can not be null");
69  		}
70  
71  		if (group == null
72  				&& (com.github.bordertech.wcomponents.subordinate.Action.ActionType.HIDEIN.equals(
73  						type)
74  				|| com.github.bordertech.wcomponents.subordinate.Action.ActionType.SHOWIN.equals(
75  						type)
76  				|| com.github.bordertech.wcomponents.subordinate.Action.ActionType.ENABLEIN.equals(
77  						type) || com.github.bordertech.wcomponents.subordinate.Action.ActionType.DISABLEIN.
78  				equals(type))) {
79  			throw new IllegalArgumentException("Group can not be null");
80  		}
81  
82  		this.type = type;
83  		this.target = target;
84  		this.group = group;
85  	}
86  
87  	/**
88  	 * @return the action type.
89  	 */
90  	public com.github.bordertech.wcomponents.subordinate.Action.ActionType getType() {
91  		return type;
92  	}
93  
94  	/**
95  	 * @return the target.
96  	 */
97  	public SubordinateTarget getTarget() {
98  		return target;
99  	}
100 
101 	/**
102 	 * @return the group.
103 	 */
104 	public WComponentGroup<? extends SubordinateTarget> getGroup() {
105 		return group;
106 	}
107 
108 	/**
109 	 * @return an instance of the subordinate action for this builder action
110 	 */
111 	public com.github.bordertech.wcomponents.subordinate.Action build() {
112 		switch (type) {
113 			case DISABLE:
114 				return new Disable(target);
115 
116 			case ENABLE:
117 				return new Enable(target);
118 
119 			case HIDE:
120 				return new Hide(target);
121 
122 			case SHOW:
123 				return new Show(target);
124 
125 			case MANDATORY:
126 				return new Mandatory(target);
127 
128 			case OPTIONAL:
129 				return new Optional(target);
130 
131 			case SHOWIN:
132 				return new ShowInGroup(target, group);
133 
134 			case HIDEIN:
135 				return new HideInGroup(target, group);
136 
137 			case ENABLEIN:
138 				return new EnableInGroup(target, group);
139 
140 			case DISABLEIN:
141 				return new DisableInGroup(target, group);
142 
143 			default:
144 				throw new IllegalArgumentException("Unknown action type: " + type);
145 		}
146 
147 	}
148 
149 	/**
150 	 * {@inheritDoc}
151 	 */
152 	@Override
153 	public String toString() {
154 		String targetName = target.getClass().getSimpleName();
155 
156 		WLabel label = target.getLabel();
157 		if (label != null) {
158 			targetName = label.getText();
159 		}
160 
161 		switch (type) {
162 			case DISABLE:
163 				return "disable " + targetName;
164 
165 			case ENABLE:
166 				return "enable " + targetName;
167 
168 			case HIDE:
169 				return "hide " + targetName;
170 
171 			case MANDATORY:
172 				return "set " + targetName + " mandatory";
173 
174 			case OPTIONAL:
175 				return "set " + targetName + " optional";
176 
177 			case SHOW:
178 				return "show " + targetName;
179 
180 			case SHOWIN:
181 				return "show " + targetName + " in " + group;
182 
183 			case HIDEIN:
184 				return "hide " + targetName + " in " + group;
185 
186 			case ENABLEIN:
187 				return "enable " + targetName + " in " + group;
188 
189 			case DISABLEIN:
190 				return "disable " + targetName + " in " + group;
191 
192 			default:
193 				throw new IllegalArgumentException("Unknown type: " + type);
194 		}
195 	}
196 }
