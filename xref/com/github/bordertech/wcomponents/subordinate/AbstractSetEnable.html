1   package com.github.bordertech.wcomponents.subordinate;

3   import com.github.bordertech.wcomponents.Container;
4   import com.github.bordertech.wcomponents.Disableable;
5   import com.github.bordertech.wcomponents.SubordinateTarget;
6   import com.github.bordertech.wcomponents.WComponent;
7   
8   /**
9    * An "action function" that sets the enable/disable attribute on a wcomponent or group of wcomponents.
10   *
11   * @author Martin Shevchenko
12   * @author Jonathan Austin
13   * @since 1.0.0
14   */
15  public abstract class AbstractSetEnable extends AbstractAction {
16  
17  	/**
18  	 * Creates an action with the given target.
19  	 *
20  	 * @param target the component to enable/disable.
21  	 * @param value the value to enable/disable.
22  	 */
23  	public AbstractSetEnable(final SubordinateTarget target, final Boolean value) {
24  		super(target, value);
25  	}
26  
27  	/**
28  	 * Apply the action against the target.
29  	 *
30  	 * @param target the target of this action
31  	 * @param value is the evaluated value
32  	 */
33  	@Override
34  	protected void applyAction(final SubordinateTarget target, final Object value) {
35  		// Should always be Boolean, but check anyway
36  		if (value instanceof Boolean) {
37  			applyEnableAction(target, ((Boolean) value));
38  		}
39  	}
40  
41  	/**
42  	 * Apply the enable action against the target and its children.
43  	 *
44  	 * @param target the target of this action
45  	 * @param enabled is the evaluated value
46  	 */
47  	private void applyEnableAction(final WComponent target, final boolean enabled) {
48  		// Found Disableable component
49  		if (target instanceof Disableable) {
50  			target.setValidate(enabled);
51  			((Disableable) target).setDisabled(!enabled);
52  		} else if (target instanceof Container) {  // Apply to any Disableable children
53  			Container cont = (Container) target;
54  			final int size = cont.getChildCount();
55  
56  			for (int i = 0; i < size; i++) {
57  				WComponent child = cont.getChildAt(i);
58  				applyEnableAction(child, enabled);
59  			}
60  		}
61  	}
62  
63  }
