46  	/**
47  	 * Create a Compare condition with a trigger and compare value.
48  	 *
49  	 * @param trigger the trigger input field.
50  	 * @param value the value to use in the compare.
51  	 */
52  	public AbstractCompare(final SubordinateTrigger trigger, final Object value) {
53  		if (trigger == null) {
54  			throw new IllegalArgumentException("Trigger cannot be null.");
55  		}
56  
57  		// Check Date and Number field trigger values
58  		if (value != null && !CompareType.MATCH.equals(getCompareType())) {
59  			// WNumberField trigger value must be a BigDecimal
60  			if (trigger instanceof WNumberField && !(value instanceof BigDecimal)) {
61  				throw new IllegalArgumentException(
62  						"The value for a WNumberField trigger must be null or a BigDecimal.");
63  			}
64  
65  			// WDateField trigger value must be a Date
66  			if (trigger instanceof WDateField && !(value instanceof Date)) {
67  				throw new IllegalArgumentException(
68  						"The value for a WDateField trigger must be null or a Date.");
69  			}
70  		}
71  
72  		this.trigger = trigger;
73  		this.value = value;
74  	}

83  	/**
84  	 * Compare the trigger and compare value.
85  	 *
86  	 * @return true if the trigger input's value compares to the compare value, otherwise false
87  	 */
88  	@Override
89  	protected boolean execute() {
90  		// Disabled triggers are always false
91  		if ((trigger instanceof Disableable) && ((Disableable) trigger).isDisabled()
92  				&& !(trigger instanceof Input && ((Input) trigger).isReadOnly())) {
93  			return false;
94  		}
95  
96  		final Object triggerValue = getTriggerValue(null);
97  		final Object compareValue = getCompareValue();
98  
99  		return executeCompare(triggerValue, compareValue);
100 	}

102 	/**
103 	 * Compare the trigger value from the request and compare value.
104 	 *
105 	 * @param request the request being processed.
106 	 * @return true if the trigger input's value compares to the compare value, otherwise false
107 	 */
108 	@Override
109 	protected boolean execute(final Request request) {
110 		// Disabled triggers are always false
111 		if ((trigger instanceof Disableable) && ((Disableable) trigger).isDisabled()
112 				&& !(trigger instanceof Input && ((Input) trigger).isReadOnly())) {
113 			return false;
114 		}
115 
116 		final Object triggerValue = getTriggerValue(request);
117 		final Object compareValue = getCompareValue();
118 
119 		return executeCompare(triggerValue, compareValue);
120 	}

122 	/**
123 	 * @param triggerValue the trigger value
124 	 * @param compareValue the compare value
125 	 * @return true if the values compare, otherwise false
126 	 */
127 	private boolean executeCompare(final Object triggerValue, final Object compareValue) {
128 		// If the trigger value is a list, check if any option in the list compares
129 		if (triggerValue instanceof List<?>) {
130 			final List<?> selected = (List<?>) triggerValue;
131 			for (Object option : selected) {
132 				if (doCompare(option, compareValue)) {
133 					return true;
134 				}
135 			}
136 			return false;
137 		} else {
138 			return doCompare(triggerValue, compareValue);
139 		}
140 	}

143 	 * Return true if the two values compare.

145 	 * @param aVal the trigger value
146 	 * @param bVal the compare value

149 	protected abstract boolean doCompare(final Object aVal, final Object bVal);

151 	/**
152 	 * Get the value for the trigger.
153 	 * <p>
154 	 * If no request is passed in, the current value of the trigger is used.
155 	 * </p>
156 	 * <p>
157 	 * It will return the same "value" the client would have used in its subordinate logic.
158 	 * </p>
159 	 * <p>
160 	 * The trigger value will either be (i) a date formatted String for WDateFields, (ii) a BigDecimal for WNumberFields
161 	 * or (iii) a List of String values for MultiSelect components or (iv) a String value.
162 	 * </p>
163 	 *
164 	 * @param request the request being processed, can be null
165 	 * @return the value to be used for the trigger
166 	 */
167 	protected Object getTriggerValue(final Request request) {
168 		// Date Compare (Use Date Formatted String - YYYY-MM-DD)
169 		if (trigger instanceof WDateField) {
170 			final WDateField input = (WDateField) trigger;
171 			Date date;
172 			if (request == null) {
173 				date = input.getValue();
174 			} else {
175 				date = input.getRequestValue(request);
176 			}
177 			return date == null ? null : new SimpleDateFormat(INTERNAL_DATE_FORMAT).format(date);
178 		} else if (trigger instanceof WNumberField) { // Number Compare (Use Number Object)
179 			final WNumberField input = (WNumberField) trigger;
180 			if (request == null) {
181 				return input.getValue();
182 			} else {
183 				return input.getRequestValue(request);
184 			}
185 		} else if (trigger instanceof AbstractWSingleSelectList) { // String Compare for Single Select Lists (Use the Option's Code)
186 			final AbstractWSingleSelectList list = (AbstractWSingleSelectList) trigger;
187 			final Object selected;
188 			if (request == null) {
189 				selected = list.getValue();
190 			} else {
191 				selected = list.getRequestValue(request);
192 			}
193 			// Convert selected option to its "code" (Should always have a value)
194 			String code = list.optionToCode(selected);
195 			return code;
196 		} else if (trigger instanceof AbstractWMultiSelectList) { // String Compare for Multi Select Lists (Use the Option's Code)
197 			final AbstractWMultiSelectList list = (AbstractWMultiSelectList) trigger;
198 			final List<?> selected;
199 			if (request == null) {
200 				selected = list.getValue();
201 			} else {
202 				selected = list.getRequestValue(request);
203 			}
204 			// Empty is treated the same as null
205 			if (selected == null || selected.isEmpty()) {
206 				return null;
207 			}
208 
209 			// Convert selected options to their "code" (Should always have a value)
210 			List<String> codes = new ArrayList<>(selected.size());
211 			for (Object select : selected) {
212 				String code = list.optionToCode(select);
213 				codes.add(code);
214 			}
215 			return codes;
216 		} else if (trigger instanceof Input) { // String Compare - Use the String Value of the Input
217 			final Input input = (Input) trigger;
218 			final Object inputValue;
219 			if (request == null) {
220 				inputValue = input.getValue();
221 			} else {
222 				inputValue = input.getRequestValue(request);
223 			}
224 			// Treat empty the same as null
225 			return (inputValue == null || Util.empty(inputValue.toString())) ? null : inputValue.
226 					toString();
227 		} else {
228 			throw new SystemException("Trigger is not a valid type.");
229 		}
230 	}

232 	/**
233 	 * Get the value to use in the compare.
234 	 * <p>
235 	 * It will return the same "value" the client would have used in its subordinate logic.
236 	 * </p>
237 	 * <p>
238 	 * The compare value will either be (i) a date formatted String for WDateFields, (ii) a BigDecimal for WNumberFields
239 	 * or (iii) a String value.
240 	 * </p>
241 	 *
242 	 * @return the value to be used for the compare.
243 	 */
244 	protected Object getCompareValue() {
245 		// Date Compare (Use Date Formatted String - YYYY-MM-DD)
246 		if (trigger instanceof WDateField) {
247 			return value == null ? null : new SimpleDateFormat(INTERNAL_DATE_FORMAT).format(value);
248 		} else if (trigger instanceof WNumberField) { // Number Compare (Use Number Object)
249 			return value;
250 		} else if (trigger instanceof AbstractWSelectList) {  // String Compare - List (Use the Option's Code)
251 			final AbstractWSelectList listTrigger = (AbstractWSelectList) trigger;
252 			final List<?> options = listTrigger.getOptions();
253 
254 			// No options, just return the compare value (so that a test against null works correctly)
255 			if (options == null || options.isEmpty()) {
256 				return value == null ? null : value.toString();
257 			}
258 
259 			// Check if the value is a valid option allowing for "Legacy" matching
260 			if (SelectListUtil.containsOptionWithMatching(options, value)) {
261 				Object option = SelectListUtil.getOptionWithMatching(options, value);
262 				String code = listTrigger.optionToCode(option);
263 				return code;
264 			}
265 
266 			// Return the value as a String - Treat empty the same as null
267 			return (value == null || Util.empty(value.toString())) ? null : value.toString();
268 		} else if (trigger instanceof RadioButtonGroup && value instanceof WRadioButton) {
269 			// String Compare for RadioButtonGroup and value is WRadioButton (Use the button value)
270 			// Note - This is only for backward compatibility where projects have used a radio button
271 			// in the trigger. Projects should use the value expected, not the radio button.
272 			// If the radio button passed into the compare is used in a repeater, then this compare will not work.
273 
274 			String data = ((WRadioButton) value).getValue();
275 			// Treat empty the same as null
276 			return Util.empty(data) ? null : data;
277 		} else { // String Compare
278 			// Treat empty the same as null
279 			return (value == null || Util.empty(value.toString())) ? null : value.toString();
280 		}
281 	}
