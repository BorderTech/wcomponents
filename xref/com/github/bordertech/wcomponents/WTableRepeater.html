9    */
10  @Deprecated
11  public class WTableRepeater extends WRepeater {
12  
13  	/**
14  	 * Parent table.
15  	 */
16  	private final WDataTable table;
17  
18  	/**
19  	 * @param table the parent table.
20  	 */
21  	public WTableRepeater(final WDataTable table) {

23  		setNamingContext(true);
24  	}
25  
26  	/**

29  	 * @param renderContext the RenderContext to send output to.
30  	 */

32  	protected void paintComponent(final RenderContext renderContext) {
33  		// Do nothing
34  	}
35  
36  	/**
37  	 * {@inheritDoc}
38  	 */
39  	@Override
40  	public String getId() {
41  		return table.getId() + ID_CONTEXT_SEPERATOR + getIdName();
42  	}
43  
44  	/**
45  	 * {@inheritDoc}
46  	 */
47  	@Override
48  	public String getIdName() {
49  		return "row";
50  	}
