4   import java.util.Arrays;
5   import java.util.List;

7   /**
8    * <p>
9    * The WMultiDropdown component allows multiple dropdown elements to be generated dynamically on the client, without
10   * requiring the page to be reloaded. This component takes its appearance and attributes from a regular drop down but
11   * allows for one or more items of text to be entered via the add link adjacent to the right of the input.
12   * </p>
13   * <p>
14   * This component is useful in instances where the user needs to select one or more items into the interfaces for a
15   * particular field. For instance, the user may have one or more aliases that need to be entered into the system.
16   * </p>
17   * <p>
18   * The following attributes can be set on WMultiDropdown:
19   * </p>
20   * <ul>
21   * <li>maxInputs: The maximum number of text inputs the user can add to the component. Client-side functionality will
22   * stop users adding more than the allowable number of inputs via the UI. This class chops off any excess inputs if an
23   * attempt is made to add them programmatically.</li>
24   * </ul>
25   *
26   * @author Yiannis Paschalidis
27   * @author Jonathan Austin
28   * @since 1.0.0
29   */
30  public class WMultiDropdown extends AbstractWMultiSelectList implements AjaxTrigger, AjaxTarget, SubordinateTarget, MultiInputComponent {
31  
32  	/**
33  	 * Creates an empty WMultiDropdown.
34  	 */
35  	public WMultiDropdown() {
36  		this((List) null);
37  	}
38  
39  	/**
40  	 * Creates a WMultiDropdown with the specified list of options.
41  	 *
42  	 * @param aList the list of available options.
43  	 */
44  	public WMultiDropdown(final List aList) {
45  		super(aList, false);
46  	}
47  
48  	/**
49  	 * Creates a WMultiDropdown with the specified list of options.
50  	 *
51  	 * @param options the list of available options.
52  	 */
53  	public WMultiDropdown(final Object[] options) {
54  		super(Arrays.asList(options), false);
55  	}
56  
57  	/**
58  	 * Creates a WMultiDropdown with the options provided by the given table.
59  	 *
60  	 * @param table the table to obtain the list's options from.
61  	 */
62  	public WMultiDropdown(final Object table) {
63  		super(table, false);
64  	}
65  
66  	/**
67  	 * Sets the maximum allowable number of inputs.
68  	 *
69  	 * @param maxInputs the maximum allowable number of inputs, or &lt;=0 for unlimited inputs.
70  	 * @deprecated Use {{@link #setMaxSelect(int)} instead
71  	 */
72  	@Deprecated
73  	public void setMaxInputs(final int maxInputs) {
74  		setMaxSelect(maxInputs);
75  	}
76  
77  	/**
78  	 * @return the maximum allowable number of inputs
79  	 * @deprecated Use {{@link #getMaxSelect()} instead
80  	 */
81  	@Deprecated
82  	public int getMaxInputs() {
83  		return getMaxSelect();
84  	}
85  
86  	/**
87  	 * Override getNewSelections to ensure that the max number of inputs is honoured and that there are no duplicates.
88  	 * This should not normally occur, as the client side js should prevent it.
89  	 *
90  	 * @param request the current request
91  	 * @return a list of selections that have been added in the given request.
92  	 */
93  	@Override
94  	protected List getNewSelections(final Request request) {
95  		List selections = super.getNewSelections(request);
96  
97  		if (selections != null) {
98  			// Ensure that there are no duplicates
99  			for (int i = 0; i < selections.size(); i++) {
100 				Object selection = selections.get(i);
101 
102 				for (int j = i + 1; j < selections.size(); j++) {
103 					if (Util.equals(selection, selections.get(j))) {
104 						selections.remove(j--);
105 					}
106 				}
107 			}
108 		}
109 
110 		return selections;
111 	}

113 }
