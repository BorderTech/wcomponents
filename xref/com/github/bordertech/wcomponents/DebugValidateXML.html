17  /**
18   * A class used to make sure the HTML/XML generated by each {@link WComponent} is well formed. {@link DebugUtil} is used
19   * to determine if XML Validation is enabled. As each WComponent paints its output, and if XML validation is enabled,
20   * WComponent calls this class to verify the HTML/XML. If a WComponent has badly formed HTML/XML, then an error message,
21   * along with the component details are stored in a framework attribute to be retrieved later by the
22   * {@link ValidateXMLInterceptor} so they can be reported in the response HTML.
23   *
24   * @author Jonathan Austin
25   * @since 1.0.0
26   */
27  public final class DebugValidateXML {
28  
29  	/**
30  	 * The logger instance for this class.
31  	 */
32  	private static final Log LOG = LogFactory.getLog(DebugValidateXML.class);
33  
34  	/**
35  	 * Hide the constructor as there are no instance methods.
36  	 */
37  	private DebugValidateXML() {
38  	}
39  
40  	/**
41  	 * Check if Validate XML is enabled.
42  	 *
43  	 * @return true or false
44  	 */
45  	public static boolean isEnabled() {
46  		return DebugUtil.isValidateXMLEnabled();
47  	}
48  
49  	/**
50  	 * Validate the component to make sure the generated XML is schema compliant.
51  	 *
52  	 * @param xml the xml to validate
53  	 * @return Any errors found, or null if the XML is valid.
54  	 */
55  	public static String validateXMLAgainstSchema(final String xml) {
56  		// Validate XML against schema
57  		if (xml != null && !xml.equals("")) {
58  			// Wrap XML with a root element (if required)
59  			String testXML = wrapXMLInRootElement(xml);
60  			try {
61  				// Create SAX Parser Factory
62  				SAXParserFactory spf = SAXParserFactory.newInstance();
63  				spf.setNamespaceAware(true);
64  				spf.setValidating(true);
65  
66  				// Create SAX Parser
67  				SAXParser parser = spf.newSAXParser();
68  				parser.setProperty("http://java.sun.com/xml/jaxp/properties/schemaLanguage",
69  						XMLConstants.W3C_XML_SCHEMA_NS_URI);
70  
71  				// Set schema location
72  				Object schema = DebugValidateXML.class.getResource(getSchemaPath()).toString();
73  				parser.setProperty("http://java.sun.com/xml/jaxp/properties/schemaSource", schema);
74  
75  				// Setup the handler to throw an exception when an error occurs
76  				DefaultHandler handler = new DefaultHandler() {
77  					@Override
78  					public void warning(final SAXParseException e) throws SAXException {
79  						LOG.warn("XML Schema warning: " + e.getMessage(), e);
80  						super.warning(e);
81  					}
82  
83  					@Override
84  					public void fatalError(final SAXParseException e) throws SAXException {
85  						throw e;
86  					}
87  
88  					@Override
89  					public void error(final SAXParseException e) throws SAXException {
90  						throw e;
91  					}
92  				};
93  
94  				// Validate the XML
95  				InputSource xmlSource = new InputSource(new StringReader(testXML));
96  				parser.parse(xmlSource, handler);
97  
98  			} catch (SAXParseException e) {
99  				return "At line " + e.getLineNumber() + ", column: " + e.getColumnNumber()
100 						+ " ==> " + e.getMessage();
101 			} catch (Exception e) {
102 				return e.getMessage();
103 			}
104 		}
105 
106 		return null;
107 	}
108 
109 	/**
110 	 * Wrap the XML in a root element before validating.
111 	 *
112 	 * @param xml the XML to wrap
113 	 * @return the XML wrapped in a root element
114 	 */
115 	public static String wrapXMLInRootElement(final String xml) {
116 		// The XML may not need to be wrapped.
117 		if (xml.startsWith("<?xml") || xml.startsWith("<!DOCTYPE")) {
118 			return xml;
119 		} else {
120 			// ENTITY definition required for NBSP.
121 			// ui namepsace required for xml theme.
122 			return XMLUtil.XML_DECLARATION + "<ui:root" + XMLUtil.STANDARD_NAMESPACES + ">" + xml + "</ui:root>";
123 		}
124 	}
125 
126 	/**
127 	 * Return the path to the schema to test against. This can be overridden if client applications wish to validate
128 	 * their own components.
129 	 *
130 	 * @return the schema path for the current theme in use.
131 	 */
132 	protected static String getSchemaPath() {
133 		return "/schema/ui/v1/schema.xsd";
134 	}
135 }
