5   /**
6    * <p>
7    * TableDataModel provides the data for tables. In a MVC sense, the TableDataModel is the Model, the {@link WDataTable}
8    * is the controller and the view is comprised of the WDataTable layout and column renderers.</p>
9    *
10   * <p>
11   * Note that Data may be stored locally or sourced remotely, depending on the particular TableDataModel
12   * implementation.</p>
13   *
14   * <p>
15   * Row and column indices for all methods are zero-based, and TableDataModels are not expected to perform
16   * bounds-checking.</p>
17   *
18   * @author Yiannis Paschalidis
19   * @since 1.0.0
20   *
21   * @deprecated Use {@link WTable} and {@link WTable.TableModel} instead.
22   */
23  @Deprecated
24  public interface TableDataModel {
25  
26  	/**
27  	 * Retrieves the value at the given row and column.
28  	 *
29  	 * @param row - the row index.
30  	 * @param col - the column index.
31  	 *
32  	 * @return the value at the given row and column.
33  	 */
34  	Object getValueAt(int row, int col);
35  
36  	/**
37  	 * Indicates whether the given cell is editable.
38  	 *
39  	 * @param row - the row index.
40  	 * @param col - the column index.
41  	 *
42  	 * @return true if the given cell is editable, false otherwise.
43  	 */
44  	boolean isCellEditable(int row, int col);
45  
46  	/**
47  	 * Sets the value at the given row and column.
48  	 *
49  	 * @param value the value to set.
50  	 * @param row - the row index.
51  	 * @param col - the column index.
52  	 */
53  	void setValueAt(Object value, int row, int col);
54  
55  	/**
56  	 * Indicates whether the model supports sorting by the given column.
57  	 *
58  	 * @param col the column index.
59  	 * @return true if the model is sortable by the given column, false otherwise.
60  	 */
61  	boolean isSortable(int col);
62  
63  	/**
64  	 * <p>
65  	 * Sorts the data by the given column. Any previous sorting should be disregarded.</p>
66  	 *
67  	 * <p>
68  	 * Data models must implement sorting in one of two ways.</p>
69  	 * <ol>
70  	 * <li>
71  	 * <p>
72  	 * If the data is accessible locally by the data model (ie. a sort won't result in a service call to obtain sorted
73  	 * data), then this method should not sort the actual data, but return a row-index mapping which the table will use
74  	 * to access the data. Row selection and expansion will be updated to use the new row indices.</p>
75  	 *
76  	 * <p>
77  	 * For example, if the data for the column is {"a", "b", "d", "c"}, then an ascending sort should return {0, 1, 3,
78  	 * 2}, and a descending sort {2, 3, 1, 0}.</p>
79  	 * </li>
80  	 * <li>
81  	 * <p>
82  	 * If the data is not accessible locally by the data model, or the model is otherwise unable to perform a mapping
83  	 * between old and new row indices, then the model should sort the actual data, and return null. In this case, the
84  	 * table will reset any row selection or expansion.</p>
85  	 * </li>
86  	 * </ol>
87  	 *
88  	 * @param col the column to sort on
89  	 * @param ascending true for an ascending sort, false for descending.
90  	 *
91  	 * @return the row indices in sort order, or null if row mappings can not be determined.
92  	 */
93  	int[] sort(int col, boolean ascending);
94  
95  	/**
96  	 * Indicates whether the given row is disabled.
97  	 *
98  	 * @param row the row index
99  	 * @return true if the row is disabled, false otherwise.
100 	 */
101 	boolean isDisabled(int row);
102 
103 	/**
104 	 * Indicates whether the given row is selectable.
105 	 *
106 	 * @param row the row index
107 	 * @return true if the row is disabled, false otherwise.
108 	 */
109 	boolean isSelectable(int row);
110 
111 	/**
112 	 * Retrieves the filter values for this row.
113 	 *
114 	 * @param row the row index
115 	 * @return the filter values for this row.
116 	 */
117 	List<String> getFilterValues(int row);
118 
119 	/**
120 	 * Retrieves the number of rows in the model.
121 	 *
122 	 * @return the number of rows in the model.
123 	 */
124 	int getRowCount();
125 
126 	/**
127 	 * Retrieves the row heading text for the given row.
128 	 *
129 	 * @param row the row index
130 	 * @return the row heading text for the given row.
131 	 */
132 	String getRowHeader(int row);
133 }
