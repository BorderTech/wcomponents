3   import com.github.bordertech.wcomponents.Input;
4   import com.github.bordertech.wcomponents.WLabel;
5   import com.github.bordertech.wcomponents.validation.Diagnostic;
6   import com.github.bordertech.wcomponents.validation.DiagnosticImpl;
7   import java.io.Serializable;
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  /**
12   * This class provides the common functionality required by a {@link FieldValidator} implementation.
13   *
14   * @author Adam Millard
15   */
16  public abstract class AbstractFieldValidator implements FieldValidator, Serializable {
17  
18  	/**
19  	 * The input field to validate.
20  	 */
21  	private Input input;
22  
23  	/**
24  	 * The error message to display on validation failure.
25  	 */
26  	private String errorMessage;
27  
28  	/**
29  	 * Creates an AbstractFieldValidator with no message.
30  	 */
31  	public AbstractFieldValidator() {
32  	}
33  
34  	/**
35  	 * Creates an AbstractFieldValidator with the given error message.
36  	 *
37  	 * The format of the error message String must conform to the pattern required by <code>MessageFormat</code>.
38  	 *
39  	 * @see java.text.MessageFormat
40  	 *
41  	 * @param errorMessage the error message.
42  	 */
43  	public AbstractFieldValidator(final String errorMessage) {
44  		this.errorMessage = errorMessage;
45  	}
46  
47  	/**
48  	 * Sets the input field.
49  	 *
50  	 * @param inputField the input field to set.
51  	 */
52  	@Override
53  	public void setInputField(final Input inputField) {
54  		this.input = inputField;
55  	}
56  
57  	/**
58  	 * @return the input field.
59  	 */
60  	@Override
61  	public Input getInputField() {
62  		return input;
63  	}
64  
65  	/**
66  	 * Sets the error message to display on validation failure.
67  	 *
68  	 * This is the format of the error message for a validator. It must conform to the pattern required by
69  	 * <code>MessageFormat</code>.
70  	 *
71  	 * @see java.text.MessageFormat
72  	 *
73  	 * @param errorMessage the error message.
74  	 */
75  	@Override
76  	public void setErrorMessage(final String errorMessage) {
77  		this.errorMessage = errorMessage;
78  	}
79  
80  	/**
81  	 * @return the error message to display on validation failure.
82  	 */
83  	@Override
84  	public String getErrorMessage() {
85  		return errorMessage;
86  	}
87  
88  	/**
89  	 * Retrieves the input field's value in the given context.
90  	 *
91  	 * @return the input field's value, as a string.
92  	 */
93  	public String getInputAsString() {
94  		return getInputField().getValueAsString();
95  	}
96  
97  	/**
98  	 * Sub classes should remember to call super.getMessageArguments() to ensure that the field label is added to the
99  	 * list of arguments.
100 	 *
101 	 * @return The list of arguments to be applied to the validators error message.
102 	 */
103 	protected List<Serializable> getMessageArguments() {
104 		List<Serializable> args = new ArrayList<>(1);
105 		WLabel label = input.getLabel();
106 
107 		if (label == null) {
108 			args.add("");
109 		} else {
110 			args.add(label.getText());
111 		}
112 
113 		return args;
114 	}
115 
116 	/**
117 	 * Validates the input field.
118 	 *
119 	 * @param diags the list of Diagnostics to add validation errors to.
120 	 *
121 	 * @return the updated list of validation errors.
122 	 */
123 	@Override
124 	public List<Diagnostic> validate(final List<Diagnostic> diags) {
125 		if (!isValid()) {
126 			List<Serializable> argList = getMessageArguments();
127 			Serializable[] args = argList.toArray(new Serializable[argList.size()]);
128 
129 			diags.add(new DiagnosticImpl(Diagnostic.ERROR, input, getErrorMessage(), args));
130 		}
131 
132 		return diags;
133 	}
134 
135 	/**
136 	 * Subclasses should implement this method with the actual validation logic.
137 	 *
138 	 * @return true if the component is valid.
139 	 */
140 	protected abstract boolean isValid();
141 }
