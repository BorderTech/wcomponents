1   package com.github.bordertech.wcomponents.validator;
2   
3   import com.github.bordertech.wcomponents.util.InternalMessages;
4   import java.util.regex.Pattern;
5   
6   /**
7    * <p>
8    * A <code>FieldValidator</code> implementation used to match an input fields value with the supplied regular
9    * expression. An error is added if the value does not match the supplied pattern.
10   * </p>
11   * <p>
12   * NOTE: When using this validator it is probably a good idea to always provide an error message.
13   * </p>
14   *
15   * @author Adam Millard
16   * @since 1.0.0
17   */
18  public class RegExFieldValidator extends AbstractFieldValidator {
19  
20  	/**
21  	 * The regular expression to validate against.
22  	 */
23  	private final Pattern pattern;
24  
25  	/**
26  	 * Creates a RegExFieldValidator.
27  	 *
28  	 * @param pattern the regular expression to validate against.
29  	 */
30  	public RegExFieldValidator(final String pattern) {
31  		this(pattern, InternalMessages.DEFAULT_VALIDATION_ERROR_INVALID);
32  	}
33  
34  	/**
35  	 * Creates a RegExFieldValidator.
36  	 *
37  	 * @param pattern the regular expression to validate against.
38  	 * @param errorMessage the error message to display when validation fails.
39  	 */
40  	public RegExFieldValidator(final String pattern, final String errorMessage) {
41  		super(errorMessage);
42  		this.pattern = Pattern.compile(pattern);
43  	}
44  
45  	/**
46  	 * {@inheritDoc}
47  	 */
48  	@Override
49  	protected boolean isValid() {
50  		String value = getInputAsString();
51  		return (value == null) || (value.length() == 0) || (pattern.matcher(value).matches());
52  	}
53  }
