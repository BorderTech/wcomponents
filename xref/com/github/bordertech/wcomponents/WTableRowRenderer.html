62  	/**
63  	 * <p>
64  	 * The preparePaintComponent method has been overridden to ensure that expanded row renderers have been correctly
65  	 * initialised.
66  	 * </p>
67  	 * <p>
68  	 * Expanded row renderers are lazily instantiated and added to the shared structure as needed. This means for the
69  	 * first use of a renderer, it will not have been part of the WComponent tree, and would not have had its
70  	 * preparePaintComponent called. We therefore add the renderer to the tree here, and manually call its preparePaint.
71  	 * </p>
72  	 *
73  	 * @param request the Request being responded to.
74  	 */
75  	@Override
76  	protected void preparePaintComponent(final Request request) {
77  		super.preparePaintComponent(request);
78  
79  		Class<? extends WComponent> rowRendererClass = getRowRendererClass();
80  
81  		if (rowRendererClass != null && !expandedRenderers.containsKey(rowRendererClass)) {
82  			getExpandedTreeNodeRenderer(rowRendererClass).preparePaint(request);
83  		}
84  	}
85  
86  	/**
87  	 * <p>
88  	 * This is called to lazily add expanded renderers as necessary. To save memory, only one instance of a renderer
89  	 * class is ever added to the row renderer instance. The RendererWrapper ensures that data binding occurs at the
90  	 * right time.
91  	 * </p>
92  	 *
93  	 * @param rendererClass the renderer class.
94  	 * @return the expanded renderer for the given row.
95  	 */
96  	public WComponent getExpandedTreeNodeRenderer(final Class<? extends WComponent> rendererClass) {
97  		if (rendererClass == null) {
98  			return null;
99  		}
100 
101 		// If we already have an additional renderer for this class, return it.
102 		// This is synchronized, as it updates the shared model using locking/unlocking, which is a bit dodgy.
103 		synchronized (expandedRenderers) {
104 			WComponent renderer = expandedRenderers.get(rendererClass);
105 
106 			if (renderer != null) {
107 				return renderer;
108 			}
109 
110 			// Not found, create a new instance of the given class
111 			renderer = new RendererWrapper(this, rendererClass, -1);
112 			expandedRenderers.put(rendererClass, renderer);
113 			setLocked(false);
114 			add(renderer);
115 			setLocked(true);
116 
117 			return renderer;
118 		}
119 	}

121 	/**
122 	 * Retrieves the renderer class for a given row.
123 	 *
124 	 * @return the renderer class for a given row (if it's an expanded row), otherwise null.
125 	 */
126 	private Class<? extends WComponent> getRowRendererClass() {
127 		RowIdWrapper wrapper = getCurrentRowIdWrapper();
128 		if (wrapper != null) {
129 			return table.getTableModel().getRendererClass(wrapper.getRowIndex());
130 		}
131 		return null;
132 	}

134 	/**
135 	 * Retrieve the current {@link RowIdWrapper}.
136 	 * <p>
137 	 * Intended for internal use only.
138 	 * </p>
139 	 *
140 	 * @return the current row id wrapper, or null if not found
141 	 */
142 	public RowIdWrapper getCurrentRowIdWrapper() {
143 		UIContext uic = UIContextHolder.getCurrent();
144 		if (uic instanceof SubUIContext) {
145 			int index = ((SubUIContext) uic).getRowIndex();
146 			RowIdWrapper wrapper = table.getRepeater().getBeanList().get(index);
147 			return wrapper;
148 		}
149 		return null;
150 	}

152 	/**
153 	 * The renderer wrapper is responsible for ensuring that the renderer is only used when needed (ie. it is only
154 	 * involved in processing of certain rows), and ensuring that data is passed to / from the renderer when required.
155 	 */
156 	private static final class RendererWrapper extends WBeanContainer implements BeanProvider {
157 
158 		/**
159 		 * The table that this wrapper belongs to.
160 		 */
161 		private final WTableRowRenderer rowRenderer;
162 
163 		/**
164 		 * The actual renderer used to render the column/expanded content.
165 		 */
166 		private final WComponent renderer;
167 
168 		/**
169 		 * The index of the column which this wrapper renders, or -1 if this is an expanded renderer.
170 		 */
171 		private final int columnIndex;
172 
173 		/**
174 		 * Creates a RendererWrapper.
175 		 *
176 		 * @param rowRenderer the row renderer.
177 		 * @param rendererClass the column/expanded content renderer class.
178 		 * @param columnIndex the index of the column renderer, or -1 if the renderer is for expanded content.
179 		 */
180 		private RendererWrapper(final WTableRowRenderer rowRenderer,
181 				final Class<? extends WComponent> rendererClass,
182 				final int columnIndex) {
183 			this.rowRenderer = rowRenderer;
184 			this.columnIndex = columnIndex;
185 			WComponent rendererComponent = null;
186 
187 			// Not found, create a new instance of the given class
188 			try {
189 				rendererComponent = rendererClass.newInstance();
190 
191 				if (rendererComponent instanceof BeanProviderBound) {
192 					((BeanProviderBound) rendererComponent).setBeanProvider(this);
193 				}
194 
195 				add(rendererComponent);
196 			} catch (Exception e) {
197 				LOG.error("Failed to instantiate renderer: " + rendererClass.getName(), e);
198 				setVisible(false);
199 			}
200 
201 			renderer = rendererComponent;
202 		}
203 
204 		/**
205 		 * Creates a RendererWrapper.
206 		 *
207 		 * @param rowRenderer the row renderer.
208 		 * @param renderer the column/expanded content renderer component.
209 		 * @param columnIndex the index of the column renderer, or -1 if the renderer is for expanded content.
210 		 */
211 		private RendererWrapper(final WTableRowRenderer rowRenderer, final WComponent renderer,
212 				final int columnIndex) {
213 			this.rowRenderer = rowRenderer;
214 			this.columnIndex = columnIndex;
215 			this.renderer = renderer;
216 
217 			if (renderer instanceof BeanProviderBound) {
218 				((BeanProviderBound) renderer).setBeanProvider(this);
219 			}
220 
221 			add(renderer);
222 		}
223 
224 		/**
225 		 * Column Renderers are invisible for expanded rows with an expanded renderer set.
226 		 *
227 		 * @return true if this component is visible, false if invisible.
228 		 */
229 		@Override
230 		public boolean isVisible() {
231 			if (!super.isVisible()) {
232 				return false;
233 			}
234 
235 			Class<? extends WComponent> rendererClass = rowRenderer.getRowRendererClass();
236 
237 			if (columnIndex >= 0) {
238 				// column renderer
239 				return rendererClass == null;
240 			} else {
241 				// expanded content renderer
242 				return renderer != null && Util.equals(renderer.getClass(), rendererClass);
243 			}
244 		}
245 
246 		/**
247 		 * Provides data to a component rendering a column.
248 		 *
249 		 * @param beanProviderBound the component rendering the column.
250 		 * @return a bean value for component that is rendering the specified row/column
251 		 */
252 		@Override
253 		public Object getBean(final BeanProviderBound beanProviderBound) {
254 			UIContext uic = UIContextHolder.getCurrent();
255 
256 			// Make sure we have the correct uic
257 			while (uic instanceof SubUIContext && !((SubUIContext) uic).isInContext(
258 					(WComponent) beanProviderBound)) {
259 				uic = ((SubUIContext) uic).getParentContext();
260 			}
261 
262 			if (!(uic instanceof SubUIContext)) {
263 				LOG.error("Unable to handle UIContext type: " + uic.getClass().getName());
264 				return null;
265 			}
266 
267 			RowIdWrapper wrapper = rowRenderer.getCurrentRowIdWrapper();
268 			List<Integer> rowIndex = wrapper.getRowIndex();
269 
270 			// ColunIndex -1 means we have a row renderer
271 			return rowRenderer.table.getTableModel().getValueAt(rowIndex, columnIndex);
272 
273 		}
274 
275 		/**
276 		 * Some renderers may not be bean provider bound, or not bean-aware. We need to make sure that the data is set
277 		 * correctly for these columns.
278 		 *
279 		 * @param request the request being responded to.
280 		 */
281 		@Override
282 		protected void preparePaintComponent(final Request request) {
283 			super.preparePaintComponent(request);
284 
285 			TableModel model = rowRenderer.table.getTableModel();
286 			RowIdWrapper wrapper = rowRenderer.getCurrentRowIdWrapper();
287 			List<Integer> rowIndex = wrapper.getRowIndex();
288 
289 			// Update input read-only status
290 			if (renderer instanceof Input) {
291 				boolean readOnly = !rowRenderer.table.isEditable() || !model.
292 						isCellEditable(rowIndex, columnIndex);
293 				Input input = (Input) renderer;
294 				if (input.isReadOnly() != readOnly) {
295 					input.setReadOnly(readOnly);
296 				}
297 			}
298 
299 			if (!(renderer instanceof BeanProviderBound)) {
300 				// Column index -1 means row renderer
301 				Object bean = model.getValueAt(rowIndex, columnIndex);
302 
303 				if (renderer instanceof BeanBound) {
304 					((BeanBound) renderer).setBean(bean);
305 				} else if (renderer instanceof DataBound) {
306 					((DataBound) renderer).setData(bean);
307 				}
308 			}
309 		}
310 	}

312 	/**
313 	 * Adds a column to the renderer. This method is called by {@link WTable} to keep the renderer's and table's columns
314 	 * in sync.
315 	 *
316 	 * @param column the column to add.
317 	 * @param columnIndex the index of the column. Zero based.
318 	 */
319 	void addColumn(final WTableColumn column, final int columnIndex) {
320 		WComponent renderer = column.getRenderer();
321 
322 		if (renderer == null) {
323 			Class<? extends WComponent> rendererClass = column.getRendererClass();
324 
325 			if (!(BeanProviderBound.class.isAssignableFrom(rendererClass))
326 					&& !(BeanBound.class.isAssignableFrom(rendererClass))
327 					&& !(DataBound.class.isAssignableFrom(rendererClass))) {
328 				throw new IllegalArgumentException(
329 						"Column renderers must be BeanProvider-, Bean- or Data-Bound");
330 			}
331 
332 			add(new RendererWrapper(this, rendererClass, columnIndex));
333 		} else {
334 			if (!(renderer instanceof BeanProviderBound) && !(renderer instanceof BeanBound)
335 					&& !(renderer instanceof DataBound)) {
336 				throw new IllegalArgumentException(
337 						"Column renderers must be BeanProvider-, Bean- or Data-Bound");
338 			}
339 
340 			add(new RendererWrapper(this, renderer, columnIndex));
341 		}
342 	}
