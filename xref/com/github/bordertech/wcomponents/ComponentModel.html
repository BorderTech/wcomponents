28  /**
29   * Holds the extrinsic state information of a WComponent. Subclasses can extend this class and add extra model
30   * attributes.
31   *
32   * <p>
33   * Subclasses must adhere to the following rules:
34   * </p>
35   * <ul>
36   * <li>They must define a public no-args constructor (requirement of {@link Externalizable}).</li>
37   * <li>They must not contain any final fields (needed for initialising from shared model).</li>
38   * <li>They must not contain any non-serializable data.</li>
39   * </ul>

41   * @author Yiannis Paschalidis
42   * @since 1.0.0

254 	/**
255 	 * Indicates whether this ComponentModel is equal to the given object.
256 	 *
257 	 * Subclasses should not need to override this method, as reflection is used to read in all fields.
258 	 *
259 	 * @param obj the object to check for equality.
260 	 * @return true if the other object is a ComponentModel and is equal to this model.
261 	 */
262 	@Override
263 	public boolean equals(final Object obj) {
264 		if (obj == null) {
265 			return false;
266 		} else if (obj == this) {
267 			// Trivial case
268 			return true;
269 		} else if (!getClass().equals(obj.getClass())) {
270 			// Can't be equal if we're not the same model class
271 			return false;
272 		} else {
273 			// Check all fields - this saves subclasses having to override
274 			// equals and remembering to call super.equals.
275 			final Field[] fields = getFields(this);
276 
277 			try {
278 				for (int i = fields.length - 1; i >= 0; i--) {
279 					Object value = fields[i].get(this);
280 					Object otherValue = fields[i].get(obj);
281 
282 					if (!Util.equals(value, otherValue)) {
283 						return false;
284 					}
285 				}
286 			} catch (IllegalAccessException e) {
287 				LOG.error("Failed to read field", e);
288 				return false;
289 			}
290 
291 			return true;
292 		}
293 	}

313 	 * Sets the shared component model. If there are any unset fields, they are set from the values in the shared model.
314 	 *
315 	 * @param sharedModel the shared ComponentModel
316 	 */
317 	protected final void setSharedModel(final ComponentModel sharedModel) {
318 		this.sharedModel = sharedModel;
319 
320 		if (unsetFields != null) {
321 			// Copy the data from the shared model
322 			for (Field field : unsetFields) {
323 				try {
324 					Object sharedValue = field.get(sharedModel);
325 					Object sessionValue = copyData(sharedValue);
326 					field.set(this, sessionValue);
327 				} catch (IllegalAccessException | IllegalArgumentException e) {
328 					LOG.error("Failed to set field " + field.getName() + " on " + getClass().
329 							getName(), e);
330 				}
331 			}
332 
333 			unsetFields = null;
334 		}
335 	}

337 	/**
338 	 * Creates a copy of mutable data, to ensure updates to one component model do not update the other.
339 	 *
340 	 * @param data the data to copy
341 	 * @return a copy of the data, if it is mutable, otherwise the original data.
342 	 */
343 	protected Object copyData(final Object data) {
344 		// For mutable objects, we perform a shallow copy
345 		// so that we don't change the state of the shared value
346 		if (data instanceof Stack) {
347 			Stack copy = new Stack();
348 			copy.addAll((Stack) data);
349 			return copy;
350 		} else if (data instanceof List) {
351 			return new ArrayList((List) data);
352 		} else if (data instanceof Hashtable) {
353 			return new Hashtable((Hashtable) data);
354 		} else if (data instanceof Map) {
355 			return new HashMap((Map) data);
356 		} else if (data instanceof Set) {
357 			return new HashSet((Set) data);
358 		} else if (data instanceof String[]) {
359 			String[] array = (String[]) data;
360 			String[] copy = new String[array.length];
361 			System.arraycopy(array, 0, copy, 0, copy.length);
362 			return copy;
363 		} else if (data instanceof int[]) {
364 			int[] array = (int[]) data;
365 			int[] copy = new int[array.length];
366 			System.arraycopy(array, 0, copy, 0, copy.length);
367 			return copy;
368 		} else if (data instanceof long[]) {
369 			long[] array = (long[]) data;
370 			long[] copy = new long[array.length];
371 			System.arraycopy(array, 0, copy, 0, copy.length);
372 			return copy;
373 		} else if (data instanceof float[]) {
374 			float[] array = (float[]) data;
375 			float[] copy = new float[array.length];
376 			System.arraycopy(array, 0, copy, 0, copy.length);
377 			return copy;
378 		} else if (data instanceof double[]) {
379 			double[] array = (double[]) data;
380 			double[] copy = new double[array.length];
381 			System.arraycopy(array, 0, copy, 0, copy.length);
382 			return copy;
383 		} else if (data instanceof boolean[]) {
384 			boolean[] array = (boolean[]) data;
385 			boolean[] copy = new boolean[array.length];
386 			System.arraycopy(array, 0, copy, 0, copy.length);
387 			return copy;
388 		} else if (data instanceof char[]) {
389 			char[] array = (char[]) data;
390 			char[] copy = new char[array.length];
391 			System.arraycopy(array, 0, copy, 0, copy.length);
392 			return copy;
393 		} else if (data instanceof Object[]) {
394 			Class<?> componentClass = data.getClass().getComponentType();
395 			Object[] copy = (Object[]) Array.newInstance(componentClass, ((Object[]) data).length);
396 			System.arraycopy(data, 0, copy, 0, copy.length);
397 			return copy;
398 		}
399 
400 		return data;
401 	}

546 	 */
547 	protected Serializable getHtmlClass() {
548 		if (this.htmlClasses == null || this.htmlClasses.isEmpty()) {
549 			return null;
550 		}
551 		int i = 0;
552 		StringBuilder result = new StringBuilder();
553 
554 		for (String c : this.htmlClasses) {
555 			if (i++ > 0) {
556 				result.append(" ");
557 			}
558 			result.append(c);
559 		}
560 		return result.toString();
561 	}

685 	/**
686 	 * <p>
687 	 * Implementation of the Externalizable interface to ensure that we don't serialize redundant data (anything
688 	 * contained by the {@link #sharedModel}.</p>
689 	 *
690 	 * <p>
691 	 * Data is obtained using reflection so that subclasses don't need concern themselves with the ComponentModel
692 	 * serialization mechanism.</p>
693 	 *
694 	 * <p>
695 	 * After the ComponentModel data is read in, some fields may be in a "unset" state. The static model needs to be
696 	 * supplied using {@link #setSharedModel(ComponentModel)}
697 	 *
698 	 * @param in the ObjectInput to read from.
699 	 *
700 	 * @throws IOException if there is an error reading from the ObjectInput
701 	 * @throws ClassNotFoundException If the class of a serialized object cannot be found.
702 	 */
703 	@Override
704 	public void readExternal(final ObjectInput in) throws IOException, ClassNotFoundException {
705 		Field[] fields = getFields(this);
706 		unsetFields = new ArrayList<>();
707 
708 		for (Field field : fields) {
709 			try {
710 				Object value = in.readObject();
711 
712 				if (value instanceof NoOverride) {
713 					// No override, so remember for later
714 					unsetFields.add(field);
715 				} else {
716 					field.set(this, value);
717 				}
718 			} catch (IllegalAccessException e) {
719 				LOG.error("Failed to read field " + field.getName(), e);
720 			}
721 		}
722 	}

724 	/**
725 	 * <p>
726 	 * Implementation of the Externalizable interface to ensure that we don't serialize redundant data (anything
727 	 * contained by the {@link #sharedModel}.</p>
728 	 *
729 	 * <p>
730 	 * Data is obtained using reflection so that subclasses don't need concern themselves with the ComponentModel
731 	 * serialization mechanism.</p>
732 	 *
733 	 * @param out the ObjectOutput to write to.
734 	 *
735 	 * @throws IOException if there is an error writing to the ObjectOutput
736 	 */
737 	@Override
738 	public void writeExternal(final ObjectOutput out) throws IOException {
739 		Field[] fields = getFields(this);
740 
741 		for (Field field : fields) {
742 			try {
743 				if (sharedModel == null) {
744 					// Support serialization of the static model,
745 					// even though this should not occur.
746 					Object value = field.get(this);
747 					out.writeObject(value);
748 				} else if (unsetFields != null && unsetFields.contains(field)) {
749 					// Support the unlikely case being deserialized/serialized
750 					// in short succession without the shared model being set.
751 					out.writeObject(NoOverride.INSTANCE);
752 				} else {
753 					Object sharedValue = field.get(sharedModel);
754 					Object value = field.get(this);
755 
756 					if (Util.equals(value, sharedValue)) {
757 						out.writeObject(NoOverride.INSTANCE);
758 					} else {
759 						out.writeObject(value);
760 					}
761 				}
762 			} catch (IllegalAccessException e) {
763 				LOG.error("Failed to write field " + field.getName(), e);
764 			}
765 		}
766 	}
767 
768 	/**
769 	 * Returns the list of fields for the given ComponentModel.
770 	 *
771 	 * @param model the ComponentModel to read fields for.
772 	 * @return an array of fields for the given model.
773 	 */
774 	private static Field[] getFields(final ComponentModel model) {
775 		Class modelClass = model.getClass();
776 		Field[] fields;
777 
778 		synchronized (FIELDS_BY_CLASS) {
779 			fields = FIELDS_BY_CLASS.get(modelClass);
780 
781 			if (fields == null) {
782 				List fieldList = ReflectionUtil.getAllFields(model, true, true);
783 				Collections.sort(fieldList, new FieldComparator());
784 
785 				fields = (Field[]) fieldList.toArray(new Field[fieldList.size()]);
786 				FIELDS_BY_CLASS.put(modelClass, fields);
787 			}
788 		}
789 
790 		return fields;
791 	}

793 	/**
794 	 * When serializing the model, we need to be distinguish a null override of the static model's value vs a field that
795 	 * hasn't been written because it hasn't been overridden. This object is used as a marker.
796 	 *
797 	 * @author Yiannis Paschalidis
798 	 */
799 	private static final class NoOverride implements Serializable {
800 
801 		/**
802 		 * Singleton instance.
803 		 */
804 		public static final NoOverride INSTANCE = new NoOverride();
805 
806 		/**
807 		 * Don't allow instantiation from outside of this class.
808 		 */
809 		private NoOverride() {
810 		}
811 	}

813 	/**
814 	 * The reflection API doesn't guarantee that fields will be returned in any particular order. To ensure that fields
815 	 * are deserialized in the same order that they were serialized, we need to sort the field list.
816 	 *
817 	 * @author Yiannis Paschalidis
818 	 */
819 	private static final class FieldComparator implements Comparator<Field>, Serializable {
820 
821 		/**
822 		 * Compares two objects for order. Returns a negative integer, zero, or a positive integer as the first object
823 		 * is less than, equal to, or greater than the second.<p>
824 		 *
825 		 * @param field1 the first field to compare.
826 		 * @param field2 the second field to compare.
827 		 * @return a negative integer, zero, or a positive integer as the first object is less than, equal to, or
828 		 * greater than the second.
829 		 */
830 		@Override
831 		public int compare(final Field field1, final Field field2) {
832 			String field1Str = field1.getDeclaringClass().getName() + '.' + field1.getName();
833 			String field2Str = field2.getDeclaringClass().getName() + '.' + field2.getName();
834 
835 			return field1Str.compareTo(field2Str);
836 		}
837 	}
