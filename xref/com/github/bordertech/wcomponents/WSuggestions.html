10  /**
11   * WSuggestions represents a device for providing suggested input for a text-like input field. The suggestions may be a
12   * static list, derived from a data url or acquired on the fly (via AJAX) based on user input into an associated input.
13   * <p>
14   * WSuggestions has no effect unless it is associated with a text-like input control such as WTextField. If it is
15   * associated with a constrained input (such as WEmailField) then it is expected (but not enforced) that the suggestions
16   * would be in line with the associated field's constraints.
17   * </p>
18   * <p>
19   * It allows for client caching of frequently used lists via a data key or lists that can be produced via AJAX depending
20   * on the text entered in the related TextField.
21   * </p>
22   * <p>
23   * Suggestions provided via a lookup table are cached on the client and filtered on the client.
24   * </p>
25   * <p>
26   * To have a suggestion list dynamically updated via AJAX, do not use a lookup table, but manually set the options and
27   * set a refresh action via {@link #setRefreshAction(Action)}. The text entered by the user that triggered the refresh
28   * is provided by {@link #getAjaxFilter()}.
29   * </p>
30   *
31   * @author Jonathan Austin
32   * @since 1.0.0
33   */
34  public class WSuggestions extends AbstractWComponent implements AjaxInternalTrigger, AjaxTarget {

46  	/**
47  	 * The way in which the suggestion is provided to and selected by the user. Defaults to BOTH and should remain as
48  	 * this for combobox implementations. LIST should be used for implementations which enforce selection of an existing
49  	 * value. We may want to consider INLINE in the future but I see no reason for NONE since then it would not be a
50  	 * combo.
51  	 */
52  	public enum Autocomplete {
53  		/**
54  		 * Indicates that autocomplete may be from the textbox or from the suggestion list.
55  		 */
56  		BOTH,
57  		/**
58  		 * Indicates the autocomplete must only be from the suggestion list.
59  		 */
60  		LIST
61  	};

90  	 */
91  	@Override
92  	public void handleRequest(final Request request) {
93  		// Check if this suggestion list is the current AJAX trigger
94  		if (AjaxHelper.isCurrentAjaxTrigger(this)) {
95  			String filter = request.getParameter(getId());
96  			setAjaxFilter(filter);
97  			doHandleAjaxRefresh();
98  		}
99  	}

101 	/**
102 	 * Handle the AJAX refresh request.
103 	 */
104 	protected void doHandleAjaxRefresh() {
105 		final Action action = getRefreshAction();
106 		if (action == null) {
107 			return;
108 		}
109 
110 		final ActionEvent event = new ActionEvent(this, AJAX_REFRESH_ACTION_COMMAND, getAjaxFilter());
111 		Runnable later = new Runnable() {
112 			@Override
113 			public void run() {
114 				action.execute(event);
115 			}
116 		};
117 
118 		invokeLater(later);
119 	}

121 	/**
122 	 * Returns the complete list of suggestions available for selection for this user's session.
123 	 *
124 	 * @return the list of suggestions available for the given user's session.
125 	 */
126 	public List<String> getSuggestions() {
127 		// Lookup table
128 		Object table = getLookupTable();
129 
130 		if (table == null) {
131 			SuggestionsModel model = getComponentModel();
132 			List<String> suggestions = model.getSuggestions();
133 			return suggestions == null ? Collections.EMPTY_LIST : suggestions;
134 		} else {
135 			List<?> lookupSuggestions = APPLICATION_LOOKUP_TABLE.getTable(table);
136 			if (lookupSuggestions == null || lookupSuggestions.isEmpty()) {
137 				return Collections.EMPTY_LIST;
138 			}
139 			// Build list of String suggestions
140 			List<String> suggestions = new ArrayList<>(lookupSuggestions.size());
141 			for (Object suggestion : lookupSuggestions) {
142 				String sugg = APPLICATION_LOOKUP_TABLE.getDescription(table, suggestion);
143 				if (sugg != null) {
144 					suggestions.add(sugg);
145 				}
146 			}
147 			return Collections.unmodifiableList(suggestions);
148 		}
149 	}

151 	/**
152 	 * Retrieves the data list cache key for this component.
153 	 *
154 	 * @return the cache key if client-side caching is enabled, null otherwise.
155 	 */
156 	public String getListCacheKey() {
157 		Object table = getLookupTable();
158 
159 		if (table != null && ConfigurationProperties.getDatalistCaching()) {
160 			String key = APPLICATION_LOOKUP_TABLE.getCacheKeyForTable(table);
161 			return key;
162 		}
163 
164 		return null;
165 	}

338 					unmodifiableList(suggestions);

343 		 * @param lookupTable the lookup table name to set.

348 		}
