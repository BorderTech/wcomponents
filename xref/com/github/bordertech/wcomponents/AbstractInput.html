31  	@Override
32  	public void addValidator(final FieldValidator validator) {
33  		InputModel model = getOrCreateComponentModel();
34  
35  		if (model.validators == null) {
36  			model.validators = new ArrayList<>();
37  		}
38  
39  		validator.setInputField(this);
40  		model.validators.add(validator);
41  	}

44  	 * {@inheritDoc}
45  	 */
46  	@Override
47  	public Iterator<FieldValidator> getValidators() {
48  		List<FieldValidator> validators = getComponentModel().validators;
49  
50  		if (validators != null) {
51  			return validators.iterator();
52  		}
53  
54  		return new EmptyIterator<>();
55  	}

57  	/**
58  	 * Override WComponent's validatorComponent in order to use the validators which have been added to this input
59  	 * field. Subclasses may still override this method, but it is suggested to call super.validateComponent to ensure
60  	 * that the validators are still used.
61  	 *
62  	 * @param diags the list into which any validation diagnostics are added.
63  	 */
64  	@Override
65  	protected void validateComponent(final List<Diagnostic> diags) {
66  		// Mandatory validation
67  		if (isMandatory() && isEmpty()) {
68  			diags.add(createMandatoryDiagnostic());
69  		}
70  
71  		// Other validations
72  		List<FieldValidator> validators = getComponentModel().validators;
73  
74  		if (validators != null) {
75  			for (FieldValidator validator : validators) {
76  				validator.validate(diags);
77  			}
78  		}
79  	}

81  	/**
82  	 * <p>
83  	 * This method is called by validateComponent to create the mandatory diagnostic error message if the mandatory
84  	 * validation check does not pass.
85  	 * </p>
86  	 * <p>
87  	 * Subclasses may override this method to customise the message, however in most cases it is easier to supply a
88  	 * custom error message pattern to the setMandatory method.
89  	 * </p>
90  	 *
91  	 * @return a new diagnostic for when mandatory validation fails.
92  	 */
93  	protected Diagnostic createMandatoryDiagnostic() {
94  		String errorMessage = getComponentModel().errorMessage;
95  		String msg = errorMessage == null ? InternalMessages.DEFAULT_VALIDATION_ERROR_MANDATORY : errorMessage;
96  		return createErrorDiagnostic(msg, this);
97  	}

227 	// Handle Request
228 	/**
229 	 * Override handleRequest in order to perform consistent processing for input components.
230 	 * <p>
231 	 * To protect against client-side tampering of the request, disabled/readonly input fields will not have their
232 	 * handle request processing performed.
233 	 * </p>
234 	 * <p>
235 	 * This method will call {@link #doHandleRequest(Request)} for input components to process the request. If the input
236 	 * component has changed, then this method will call {@link #doHandleChanged()} and set the
237 	 * {@link #isChangedInLastRequest()} flag.
238 	 * </p>
239 	 *
240 	 * @param request the request being responded to.
241 	 */
242 	@Override
243 	public final void handleRequest(final Request request) {
244 		if (!beforeHandleRequest(request)) {
245 			return;
246 		}
247 
248 		setChangedInLastRequest(false);
249 
250 		if (isDisabled() || isReadOnly()) {
251 			// Protect against client-side tampering of disabled/read-only fields.
252 			return;
253 		}
254 
255 		if (isPresent(request)) {
256 
257 			// Only process on a POST
258 			if (!"POST".equals(request.getMethod())) {
259 				LOG.warn("Input component on a request that is not a POST. Will be ignored.");
260 				return;
261 			}
262 
263 			boolean changed = doHandleRequest(request);
264 
265 			if (changed) {
266 				setChangedInLastRequest(true);
267 				doHandleChanged();
268 			}
269 		}
270 	}
271 
272 	/**
273 	 * Handle before handle request processing.
274 	 *
275 	 * @param request the request being responded to.
276 	 * @return true to continue
277 	 */
278 	protected boolean beforeHandleRequest(final Request request) {
279 		return true;
280 	}
281 
282 	/**
283 	 * Specific handle request processing for an input component is provided here.
284 	 * <p>
285 	 * Input components are required to determine if the component has changed in the request, set the component data to
286 	 * the new value (if changed) and return the changed flag.
287 	 * </p>
288 	 *
289 	 * @param request the request being responded to.
290 	 * @return true if the input component has changed, otherwise return false
291 	 */
292 	protected abstract boolean doHandleRequest(final Request request);
293 
294 	/**
295 	 * Perform change logic for this component.
296 	 */
297 	protected void doHandleChanged() {
298 		// If there is an associated action, execute it
299 		if (getActionOnChange() != null) {
300 			final ActionEvent event = new ActionEvent(this, getActionCommand(), getActionObject());
301 			Runnable later = new Runnable() {
302 				@Override
303 				public void run() {
304 					getActionOnChange().execute(event);
305 
306 					if (isSubmitOnChange() && UIContextHolder.getCurrent().getFocussed() == null) {
307 						setFocussed();
308 					}
309 				}
310 			};
311 
312 			invokeLater(later);
313 		} else if (isSubmitOnChange() && UIContextHolder.getCurrent().getFocussed() == null) {
314 			setFocussed();
315 		}
316 	}
