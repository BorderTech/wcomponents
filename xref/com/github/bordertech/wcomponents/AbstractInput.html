29  	/**
30  	 * {@inheritDoc}
31  	 */
32  	@Override
33  	public void addValidator(final FieldValidator validator) {
34  		InputModel model = getOrCreateComponentModel();
35  
36  		if (model.validators == null) {
37  			model.validators = new ArrayList<>();
38  		}
39  
40  		validator.setInputField(this);
41  		model.validators.add(validator);
42  		// possible source of memory leaks
43  		MemoryUtil.checkSize(model.validators.size(), this.getClass().getSimpleName());
44  	}

47  	 * {@inheritDoc}
48  	 */
49  	@Override
50  	public Iterator<FieldValidator> getValidators() {
51  		List<FieldValidator> validators = getComponentModel().validators;
52  
53  		if (validators != null) {
54  			return validators.iterator();
55  		}
56  
57  		return new EmptyIterator<>();
58  	}

60  	/**
61  	 * Override WComponent's validatorComponent in order to use the validators which have been added to this input
62  	 * field. Subclasses may still override this method, but it is suggested to call super.validateComponent to ensure
63  	 * that the validators are still used.
64  	 *
65  	 * @param diags the list into which any validation diagnostics are added.
66  	 */
67  	@Override
68  	protected void validateComponent(final List<Diagnostic> diags) {
69  		// Mandatory validation
70  		if (isMandatory() && isEmpty()) {
71  			diags.add(createMandatoryDiagnostic());
72  		}
73  
74  		// Other validations
75  		List<FieldValidator> validators = getComponentModel().validators;
76  
77  		if (validators != null) {
78  			for (FieldValidator validator : validators) {
79  				validator.validate(diags);
80  			}
81  		}
82  	}

84  	/**
85  	 * <p>
86  	 * This method is called by validateComponent to create the mandatory diagnostic error message if the mandatory
87  	 * validation check does not pass.
88  	 * </p>
89  	 * <p>
90  	 * Subclasses may override this method to customise the message, however in most cases it is easier to supply a
91  	 * custom error message pattern to the setMandatory method.
92  	 * </p>
93  	 *
94  	 * @return a new diagnostic for when mandatory validation fails.
95  	 */
96  	protected Diagnostic createMandatoryDiagnostic() {
97  		String errorMessage = getComponentModel().errorMessage;
98  		String msg = errorMessage == null ? InternalMessages.DEFAULT_VALIDATION_ERROR_MANDATORY : errorMessage;
99  		return createErrorDiagnostic(msg, this);
100 	}

230 	// Handle Request
231 	/**
232 	 * Override handleRequest in order to perform consistent processing for input components.
233 	 * <p>
234 	 * To protect against client-side tampering of the request, disabled/readonly input fields will not have their
235 	 * handle request processing performed.
236 	 * </p>
237 	 * <p>
238 	 * This method will call {@link #doHandleRequest(Request)} for input components to process the request. If the input
239 	 * component has changed, then this method will call {@link #doHandleChanged()} and set the
240 	 * {@link #isChangedInLastRequest()} flag.
241 	 * </p>
242 	 *
243 	 * @param request the request being responded to.
244 	 */
245 	@Override
246 	public final void handleRequest(final Request request) {
247 		if (!beforeHandleRequest(request)) {
248 			return;
249 		}
250 
251 		setChangedInLastRequest(false);
252 
253 		if (isDisabled() || isReadOnly()) {
254 			// Protect against client-side tampering of disabled/read-only fields.
255 			return;
256 		}
257 
258 		if (isPresent(request)) {
259 
260 			// Only process on a POST
261 			if (!"POST".equals(request.getMethod())) {
262 				LOG.warn("Input component on a request that is not a POST. Will be ignored.");
263 				return;
264 			}
265 
266 			boolean changed = doHandleRequest(request);
267 
268 			if (changed) {
269 				setChangedInLastRequest(true);
270 				doHandleChanged();
271 			}
272 		}
273 	}
274 
275 	/**
276 	 * Handle before handle request processing.
277 	 *
278 	 * @param request the request being responded to.
279 	 * @return true to continue
280 	 */
281 	protected boolean beforeHandleRequest(final Request request) {
282 		return true;
283 	}
284 
285 	/**
286 	 * Specific handle request processing for an input component is provided here.
287 	 * <p>
288 	 * Input components are required to determine if the component has changed in the request, set the component data to
289 	 * the new value (if changed) and return the changed flag.
290 	 * </p>
291 	 *
292 	 * @param request the request being responded to.
293 	 * @return true if the input component has changed, otherwise return false
294 	 */
295 	protected abstract boolean doHandleRequest(final Request request);
296 
297 	/**
298 	 * Perform change logic for this component.
299 	 * <p>Reset focus ONLY if the current Request is an Ajax request. See https://github.com/BorderTech/wcomponents/issues/501.</p>
300 	 */
301 	protected void doHandleChanged() {
302 		// If there is an associated action, execute it
303 		if (getActionOnChange() != null) {
304 			final ActionEvent event = new ActionEvent(this, getActionCommand(), getActionObject());
305 			final boolean isCAT = isCurrentAjaxTrigger();
306 			Runnable later = new Runnable() {
307 				@Override
308 				public void run() {
309 					getActionOnChange().execute(event);
310 					if (isCAT && UIContextHolder.getCurrent().getFocussed() == null) {
311 						setFocussed();
312 					}
313 				}
314 			};
315 
316 			invokeLater(later);
317 		} else if (AjaxHelper.isCurrentAjaxTrigger(this) && UIContextHolder.getCurrent().getFocussed() == null) {
318 			setFocussed();
319 		}
320 	}
