13  /**
14   * Note that WLink is different to WButton rendered as a link because WLink will not post the form, and does not support
15   * Actions. By default it opens up a new browser window and shows the given url. If the new window attributes needs to
16   * be specified then a builder can be used <code>
17   * WLink wLink = new WLink.Builder("WLink using builder and with attrs", "http://bordertech.github.io/").
18   * window("myWcomponentsWindow").width(200).height(200).scrollbars(true).build();
19   * </code> Caution must be taken when specifying attributes so that window name is not same for two links on the same
20   * page. This can occur when window name is not provided with attributes causing same default window name.
21   *
22   * @author Martin Shevchenko
23   * @since 1.0.0
24   */
25  public class WLink extends WBeanComponent implements Container, Disableable, AjaxInternalTrigger, AjaxTarget,

371 	/**
372 	 * Override handleRequest in order to perform processing for this component. This implementation checks whether the
373 	 * link has been pressed via the current ajax operation.
374 	 *
375 	 * @param request the request being responded to.
376 	 */
377 	@Override
378 	public void handleRequest(final Request request) {
379 		// Check if this link was the AJAX Trigger
380 		AjaxOperation operation = AjaxHelper.getCurrentOperation();
381 		boolean pressed = (operation != null && getId().equals(operation.getTriggerId()));
382 
383 		// Protect against client-side tampering of disabled/read-only fields.
384 		if (isDisabled() && pressed) {
385 			LOG.warn("A disabled link has been triggered. " + getText() + ". " + getId());
386 			return;
387 		}
388 
389 		// If an action has been supplied then execute it, but only after
390 		// handle request has been performed on the entire component tree.
391 		final Action action = getAction();
392 
393 		if (pressed && action != null) {
394 			final ActionEvent event = new ActionEvent(this, getActionCommand(), getActionObject());
395 
396 			Runnable later = new Runnable() {
397 				@Override
398 				public void run() {
399 					action.execute(event);
400 				}
401 			};
402 
403 			invokeLater(later);
404 		}
405 	}

407 	/**
408 	 * Override preparePaintComponent to register an AJAX operation if this link has an action.
409 	 *
410 	 * @param request the request being responded to.
411 	 */
412 	@Override
413 	protected void preparePaintComponent(final Request request) {
414 		super.preparePaintComponent(request);
415 		UIContext uic = UIContextHolder.getCurrent();
416 
417 		// If the link has an action, register it for AJAX
418 		final Action action = getAction();
419 		final AjaxTarget[] actionTargets = getActionTargets();
420 
421 		if (action != null && uic.getUI() != null) {
422 			// Register AJAX operation if AJAX targets are set, otherwise rely on InternalAjax request
423 			if (actionTargets != null && actionTargets.length > 0) {
424 				List<String> targetIds = new ArrayList<>();
425 				for (AjaxTarget target : actionTargets) {
426 					targetIds.add(target.getId());
427 				}
428 				// Register the action targets
429 				AjaxHelper.registerComponents(targetIds, getId());
430 			}
431 		}
432 	}
