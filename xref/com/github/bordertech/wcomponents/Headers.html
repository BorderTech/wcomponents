1   package com.github.bordertech.wcomponents;
2   
3   import java.util.List;
4   
5   /**
6    * WComponents can use this interface to communicate "global" or "shared" items to the server. Examples are lines in the
7    * html/head element. It also allows for extra attributes on the html element itself.
8    * <p>
9    * It is important to note that "headlines" must be added in the preparePaint phase. Headlines adding in serviceRequest
10   * will be lost because the headers are reset at the start of preparePaint. Adding headlines in the paint phase is too
11   * late as the headers have already been written. Note also that it would be possible to make the adding of headlines
12   * work for the paint phase, but at the expense of requiring the entire body of the page to be cached.
13   *
14   * @author James Gifford
15   * @author Martin Shevchenko
16   * @since 1.0.0
17   */
18  public interface Headers {
19  
20  	/**
21  	 * Untyped headline.
22  	 */
23  	String UNTYPED_HEADLINE = "un-typed";
24  	/**
25  	 * Javascript headline.
26  	 */
27  	String JAVASCRIPT_HEADLINE = "javascript";
28  	/**
29  	 * CSS headline.
30  	 */
31  	String CSS_HEADLINE = "css";
32  
33  	/**
34  	 * Called by the servlet between service requests.
35  	 */
36  	void reset();
37  
38  	/**
39  	 * Records a "shared" line for inclusion in the output.
40  	 *
41  	 * @param aLine the line to add.
42  	 */
43  	void addHeadLine(String aLine);
44  
45  	/**
46  	 * Records a "shared" line (of a specified type) for inclusion in the output.
47  	 *
48  	 * @param type the type of line to add.
49  	 * @param aLine the line to add.
50  	 */
51  	void addHeadLine(String type, String aLine);
52  
53  	/**
54  	 * Records a line for inclusion in the html/head, if it has not already been included.
55  	 *
56  	 * @param aLine the line to add.
57  	 */
58  	void addUniqueHeadLine(String aLine);
59  
60  	/**
61  	 * Records a "shared" line (of a specified type) for inclusion in the output, if it has not already been included.
62  	 * <p>
63  	 * An example of where this is useful is for adding shared JavaScript code.
64  	 *
65  	 * @param type the type of line to add.
66  	 * @param aLine the line to add.
67  	 */
68  	void addUniqueHeadLine(String type, String aLine);
69  
70  	/**
71  	 * @return the "un-typed" head lines.
72  	 */
73  	List getHeadLines();
74  
75  	/**
76  	 * Get the head lines, of a specified type.
77  	 *
78  	 * @param type the type of headlines to return.
79  	 * @return the head lines, of a specified type
80  	 */
81  	List getHeadLines(String type);
82  
83  	/**
84  	 * Get the content type (usually text/html).
85  	 *
86  	 * @return the content mime type.
87  	 */
88  	String getContentType();
89  
90  	/**
91  	 * Set the content type (usually text/html).
92  	 *
93  	 * @param type The content type to set on the response.
94  	 */
95  	void setContentType(String type);
96  }
