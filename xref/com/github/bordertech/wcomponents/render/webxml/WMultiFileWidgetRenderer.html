13  /**
14   * The Renderer for the {@link WMultiFileWidget} component.
15   *
16   * @author Yiannis Paschalidis
17   * @since 1.0.0
18   */
19  final class WMultiFileWidgetRenderer extends AbstractWebXmlRenderer {
20  
21  	/**
22  	 * Paints the given WMultiFileWidget.
23  	 *
24  	 * @param component the WMultiFileWidget to paint.
25  	 * @param renderContext the RenderContext to paint to.
26  	 */
27  	@Override
28  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
29  		WMultiFileWidget widget = (WMultiFileWidget) component;
30  		XmlStringBuilder xml = renderContext.getWriter();
31  
32  		// Check if rendering a file upload response
33  		String uploadId = widget.getFileUploadRequestId();
34  		if (uploadId != null) {
35  			handleFileUploadRequest(widget, xml, uploadId);
36  			return;
37  		}
38  		boolean readOnly = widget.isReadOnly();
39  
40  
41  		xml.appendTagOpen("ui:multifileupload");
42  		xml.appendAttribute("id", component.getId());
43  		xml.appendOptionalAttribute("class", component.getHtmlClass());
44  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
45  		xml.appendOptionalAttribute("hidden", widget.isHidden(), "true");
46  		if (readOnly) {
47  			xml.appendAttribute("readOnly", "true");
48  		} else {
49  			long maxFileSize = widget.getMaxFileSize();
50  			int maxFiles = widget.getMaxFiles();
51  			WComponent dropzone = widget.getDropzone();
52  			WImageEditor editor = widget.getEditor();
53  
54  			xml.appendOptionalAttribute("disabled", widget.isDisabled(), "true");
55  			xml.appendOptionalAttribute("required", widget.isMandatory(), "true");
56  			xml.appendOptionalAttribute("tabIndex", widget.hasTabIndex(), widget.getTabIndex());
57  			xml.appendOptionalAttribute("toolTip", widget.getToolTip());
58  			xml.appendOptionalAttribute("accessibleText", widget.getAccessibleText());
59  			xml.appendOptionalAttribute("acceptedMimeTypes", typesToString(widget.getFileTypes()));
60  			xml.appendOptionalAttribute("maxFileSize", maxFileSize > 0, maxFileSize);
61  			xml.appendOptionalAttribute("maxFiles", maxFiles > 0, maxFiles);
62  			if (dropzone != null) {
63  				xml.appendAttribute("dropzone", dropzone.getId());
64  			}
65  			if (editor != null) {
66  				xml.appendAttribute("editor", editor.getId());
67  				if (editor.getUseCamera()) {
68  					xml.appendAttribute("camera", true);
69  				}
70  			}
71  		}
72  		if (widget.getColumns() != null) {
73  			xml.appendAttribute("cols", widget.getColumns());
74  		}
75  		if (widget.getFileAjaxAction() != null) {
76  			xml.appendAttribute("ajax", "true");
77  		}
78  
79  		if (widget.getFiles().isEmpty()) {
80  			xml.appendEnd();
81  		} else {
82  			xml.appendClose();
83  			// Render files
84  			int i = 0;
85  			for (FileWidgetUpload file : widget.getFiles()) {
86  				FileWidgetRendererUtil.renderFileElement(widget, xml, file, i++);
87  			}
88  			xml.appendEndTag("ui:multifileupload");
89  		}
90  	}
91  
92  	/**
93  	 * Paint the response for the file uploaded in this request.
94  	 *
95  	 * @param widget the file widget to render
96  	 * @param xml the XML string builder
97  	 * @param uploadId the file id uploaded in this request
98  	 */
99  	protected void handleFileUploadRequest(final WMultiFileWidget widget, final XmlStringBuilder xml, final String uploadId) {
100 
101 		FileWidgetUpload file = widget.getFile(uploadId);
102 		if (file == null) {
103 			throw new SystemException("Invalid file id [" + uploadId + "] to render uploaded response.");
104 		}
105 		int idx = widget.getFiles().indexOf(file);
106 		FileWidgetRendererUtil.renderFileElement(widget, xml, file, idx);
107 	}
108 
109 	/**
110 	 * Flattens the list of accepted mime types into a format suitable for rendering.
111 	 *
112 	 * @param types the list of accepted mime types
113 	 * @return the list of accepted mime types in a format suitable for rendering
114 	 */
115 	private String typesToString(final List<String> types) {
116 		if (types == null || types.isEmpty()) {
117 			return null;
118 		}
119 
120 		StringBuffer typesString = new StringBuffer();
121 
122 		for (Iterator<String> iter = types.iterator(); iter.hasNext();) {
123 			typesString.append(iter.next());
124 
125 			if (iter.hasNext()) {
126 				typesString.append(',');
127 			}
128 		}
129 
130 		return typesString.toString();
131 	}
