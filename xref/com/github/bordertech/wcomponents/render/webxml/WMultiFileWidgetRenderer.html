12  /**
13   * The Renderer for the {@link WMultiFileWidget} component.
14   *
15   * @author Yiannis Paschalidis
16   * @since 1.0.0
17   */
18  final class WMultiFileWidgetRenderer extends AbstractWebXmlRenderer {
19  
20  	/**
21  	 * Paints the given WMultiFileWidget.
22  	 *
23  	 * @param component the WMultiFileWidget to paint.
24  	 * @param renderContext the RenderContext to paint to.
25  	 */
26  	@Override
27  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
28  		WMultiFileWidget widget = (WMultiFileWidget) component;
29  		XmlStringBuilder xml = renderContext.getWriter();
30  		long maxFileSize = widget.getMaxFileSize();
31  		int maxFiles = widget.getMaxFiles();
32  		WComponent dropzone = widget.getDropzone();
33  		WImageEditor editor = widget.getEditor();
34  
35  		xml.appendTagOpen("ui:fileupload");
36  		xml.appendAttribute("id", component.getId());
37  		xml.appendOptionalAttribute("class", component.getHtmlClass());
38  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
39  		xml.appendOptionalAttribute("disabled", widget.isDisabled(), "true");
40  		xml.appendOptionalAttribute("hidden", widget.isHidden(), "true");
41  		xml.appendOptionalAttribute("required", widget.isMandatory(), "true");
42  		xml.appendOptionalAttribute("readOnly", widget.isReadOnly(), "true");
43  		xml.appendOptionalAttribute("tabIndex", widget.hasTabIndex(), widget.getTabIndex());
44  		xml.appendOptionalAttribute("toolTip", widget.getToolTip());
45  		xml.appendOptionalAttribute("accessibleText", widget.getAccessibleText());
46  		xml.appendOptionalAttribute("acceptedMimeTypes", typesToString(widget.getFileTypes()));
47  		xml.appendOptionalAttribute("maxFileSize", maxFileSize > 0, maxFileSize);
48  		xml.appendOptionalAttribute("maxFiles", maxFiles > 0, maxFiles);
49  		if (widget.getColumns() != null) {
50  			xml.appendAttribute("cols", widget.getColumns());
51  		}
52  		if (dropzone != null) {
53  			xml.appendAttribute("dropzone", dropzone.getId());
54  		}
55  		if (editor != null) {
56  			xml.appendAttribute("editor", editor.getId());
57  			if (editor.getUseCamera()) {
58  				xml.appendAttribute("camera", true);
59  			}
60  		}
61  		if (widget.getFileAjaxAction() != null) {
62  			xml.appendAttribute("ajax", "true");
63  		}
64  
65  		if (widget.getFiles().isEmpty()) {
66  			xml.appendEnd();
67  		} else {
68  			xml.appendClose();
69  			// Render files
70  			int i = 0;
71  			for (FileWidgetUpload file : widget.getFiles()) {
72  				FileWidgetRendererUtil.renderFileElement(widget, xml, file, i++);
73  			}
74  			xml.appendEndTag("ui:fileupload");
75  		}
76  	}
77  
78  	/**
79  	 * Flattens the list of accepted mime types into a format suitable for rendering.
80  	 *
81  	 * @param types the list of accepted mime types
82  	 * @return the list of accepted mime types in a format suitable for rendering
83  	 */
84  	private String typesToString(final List<String> types) {
85  		if (types == null || types.isEmpty()) {
86  			return null;
87  		}
88  
89  		StringBuffer typesString = new StringBuffer();
90  
91  		for (Iterator<String> iter = types.iterator(); iter.hasNext();) {
92  			typesString.append(iter.next());
93  
94  			if (iter.hasNext()) {
95  				typesString.append(',');
96  			}
97  		}
98  
99  		return typesString.toString();
100 	}
