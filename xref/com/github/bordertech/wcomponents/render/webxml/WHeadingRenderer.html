3   import com.github.bordertech.wcomponents.Renderer;
4   import com.github.bordertech.wcomponents.WComponent;
5   import com.github.bordertech.wcomponents.WHeading;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   
9   /**
10   * {@link Renderer} for the {@link WHeading} component.
11   *
12   * @author Yiannis Paschalidis
13   * @since 1.0.0
14   */
15  final class WHeadingRenderer extends AbstractWebXmlRenderer {
16  
17  	/**
18  	 * Paints the given WHeading.
19  	 *
20  	 * @param component the WHeading to paint.
21  	 * @param renderContext the RenderContext to paint to.
22  	 */
23  	@Override
24  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
25  		WHeading heading = (WHeading) component;
26  		XmlStringBuilder xml = renderContext.getWriter();
27  
28  		xml.appendTagOpen("ui:heading");
29  		xml.appendAttribute("id", component.getId());
30  		xml.appendOptionalAttribute("class", component.getHtmlClass());
31  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
32  		xml.appendAttribute("level", heading.getHeadingLevel().getLevel());
33  		xml.appendOptionalAttribute("accessibleText", heading.getAccessibleText());
34  		xml.appendClose();
35  
36  		// Render margin
37  		MarginRendererUtil.renderMargin(heading, renderContext);
38  
39  		if (heading.getDecoratedLabel() == null) {
40  			// Constructed with a String
41  			xml.append(heading.getText(), heading.isEncodeText());
42  		} else {
43  			heading.getDecoratedLabel().paint(renderContext);
44  		}
45  
46  		xml.appendEndTag("ui:heading");
47  	}
48  }
