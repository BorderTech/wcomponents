23  /**
24   * {@link Renderer} for the {@link WDataTableRowRenderer} component.
25   *
26   * @author Yiannis Paschalidis
27   * @since 1.0.0
28   */
29  final class WDataTableRowRendererRenderer extends AbstractWebXmlRenderer {
30  
31  	/**
32  	 * Paints the given WDataTableRowRenderer.
33  	 *
34  	 * @param component the WDataTableRowRenderer to paint.
35  	 * @param renderContext the RenderContext to paint to.
36  	 */
37  	@Override
38  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
39  		WDataTableRowRenderer renderer = (WDataTableRowRenderer) component;
40  		XmlStringBuilder xml = renderContext.getWriter();
41  		WDataTable table = renderer.getTable();
42  		TableDataModel dataModel = table.getDataModel();
43  		UIContext uic = UIContextHolder.getCurrent();
44  
45  		final int numCols = table.getColumnCount();
46  		int[] columnOrder = table.getColumnOrder();
47  		int rowIndex = getRowIndex(table, (SubUIContext) uic);
48  		boolean unselectable = table.getSelectMode() != SelectMode.NONE
49  				&& !dataModel.isSelectable(rowIndex);
50  
51  		xml.appendTagOpen("ui:tr");
52  		xml.appendAttribute("rowIndex", rowIndex);
53  		xml.appendOptionalAttribute("unselectable", unselectable, "true");
54  		xml.appendOptionalAttribute("selected", table.getSelectedRows().contains(rowIndex), "true");
55  		xml.appendOptionalAttribute("filterValues", getFilterValues(dataModel, rowIndex));
56  
57  		if (table.getExpandMode() != WDataTable.ExpandMode.NONE && dataModel instanceof TreeTableDataModel) {
58  			TableTreeNode node = ((TreeTableDataModel) dataModel).getNodeAtLine(rowIndex);
59  			boolean expandable = !node.isLeaf() && !node.isExpanded();
60  			xml.appendOptionalAttribute("expandable", expandable, "true");
61  		}
62  
63  		xml.appendClose();
64  
65  		if (table.isShowRowHeaders()) {
66  			xml.appendTag("ui:th");
67  			renderer.getRowHeader().paint(renderContext);
68  			xml.appendEndTag("ui:th");
69  		}
70  
71  		for (int i = 0; i < numCols; i++) {
72  			int colIndex = columnOrder == null ? i : columnOrder[i];
73  			WTableColumn col = table.getColumn(colIndex);
74  
75  			if (col.isVisible()) {
76  				xml.appendTag("ui:td");
77  				renderer.getRenderer(colIndex).paint(renderContext);
78  				xml.appendEndTag("ui:td");
79  			}
80  		}
81  
82  		if (table.getExpandMode() != WDataTable.ExpandMode.NONE && dataModel instanceof TreeTableDataModel) {
83  			TreeTableDataModel treeModel = (TreeTableDataModel) dataModel;
84  			TableTreeNode node = treeModel.getNodeAtLine(rowIndex);
85  
86  			if (!node.isLeaf()) {
87  				xml.appendTagOpen("ui:subtr");
88  				xml.appendOptionalAttribute("open", node.isExpanded(), "true");
89  				xml.appendClose();
90  
91  				if (node.isExpanded() || table.getExpandMode() == ExpandMode.CLIENT) {
92  					WRepeater repeater = table.getRepeater();
93  
94  					// If there a renderer specified by any child, we only paint content that has a specified renderer
95  					boolean rendererPresent = false;
96  
97  					for (Iterator<TreeNode> i = node.children(); !rendererPresent && i.hasNext();) {
98  						TableTreeNode child = (TableTreeNode) i.next();
99  
100 						if (child.getRendererClass() != null) {
101 							rendererPresent = true;
102 						}
103 					}
104 
105 					// Paint immediate children only.
106 					if (rendererPresent) {
107 						xml.appendTagOpen("ui:content");
108 						xml.appendOptionalAttribute("spanAllCols", node.isRendererSpansAllCols(),
109 								"true");
110 						xml.appendClose();
111 
112 						for (Iterator<TreeNode> i = node.children(); i.hasNext();) {
113 							TableTreeNode child = (TableTreeNode) i.next();
114 							Integer rowId = child.getRowIndex() - 1;
115 							UIContext nodeContext = repeater.getRowContext(rowId);
116 
117 							WComponent expandedRenderer = renderer.getExpandedTreeNodeRenderer(
118 									child.getRendererClass());
119 
120 							if (expandedRenderer != null) {
121 								UIContextHolder.pushContext(nodeContext);
122 
123 								try {
124 									expandedRenderer.paint(renderContext);
125 								} finally {
126 									UIContextHolder.popContext();
127 								}
128 							}
129 						}
130 
131 						xml.appendEndTag("ui:content");
132 					} else {
133 						for (Iterator<TreeNode> i = node.children(); i.hasNext();) {
134 							TableTreeNode child = (TableTreeNode) i.next();
135 							Integer rowId = child.getRowIndex() - 1;
136 							UIContext nodeContext = repeater.getRowContext(rowId);
137 
138 							UIContextHolder.pushContext(nodeContext);
139 
140 							try {
141 								render(component, renderContext);
142 							} finally {
143 								UIContextHolder.popContext();
144 							}
145 						}
146 					}
147 				}
148 
149 				xml.appendEndTag("ui:subtr");
150 			}
151 		}
152 
153 		xml.appendEndTag("ui:tr");
154 	}

156 	/**
157 	 * Retrieves the filter values for the given row in the data model, as a comma-separated string.
158 	 *
159 	 * @param dataModel the data model.
160 	 * @param rowIndex the row index.
161 	 * @return the filter values string.
162 	 */
163 	private String getFilterValues(final TableDataModel dataModel, final int rowIndex) {
164 		List<String> filterValues = dataModel.getFilterValues(rowIndex);
165 
166 		if (filterValues == null || filterValues.isEmpty()) {
167 			return null;
168 		}
169 
170 		StringBuffer buf = new StringBuffer(filterValues.get(0));
171 
172 		for (int i = 1; i < filterValues.size(); i++) {
173 			buf.append(", ");
174 			buf.append(filterValues.get(i));
175 		}
176 
177 		return buf.toString();
178 	}
179 
180 	/**
181 	 * @param table the table that the row renderer belongs to
182 	 * @param uic the SubUIContext to determine the row index for
183 	 * @return the row index for the specified SubUIContext.
184 	 */
185 	private int getRowIndex(final WDataTable table, final SubUIContext uic) {
186 		// SubUIContext does not cater for pagination
187 		return (Integer) table.getRepeater().getBeanList().get(uic.getRowIndex());
188 	}
