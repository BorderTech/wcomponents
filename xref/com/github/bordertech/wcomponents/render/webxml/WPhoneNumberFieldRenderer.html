1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WPhoneNumberField;
5   import com.github.bordertech.wcomponents.WSuggestions;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   import com.github.bordertech.wcomponents.util.Util;
9   
10  /**
11   * The Renderer for {@link WPhoneNumberField}.
12   *
13   * @author Yiannis Paschalidis
14   * @since 1.0.0
15   */
16  class WPhoneNumberFieldRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WPhoneNumberField.
20  	 *
21  	 * @param component the WPhoneNumberField to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WPhoneNumberField field = (WPhoneNumberField) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  
29  		boolean readOnly = field.isReadOnly();
30  
31  		xml.appendTagOpen("ui:phonenumberfield");
32  		xml.appendAttribute("id", component.getId());
33  		xml.appendOptionalAttribute("class", component.getHtmlClass());
34  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
35  		xml.appendOptionalAttribute("hidden", component.isHidden(), "true");
36  
37  		if (readOnly) {
38  			xml.appendAttribute("readOnly", "true");
39  		} else {
40  			int cols = field.getColumns();
41  			int minLength = field.getMinLength();
42  			int maxLength = field.getMaxLength();
43  			String pattern = field.getPattern();
44  
45  			WSuggestions suggestions = field.getSuggestions();
46  			String suggestionsId = suggestions == null ? null : suggestions.getId();
47  
48  			WComponent submitControl = field.getDefaultSubmitButton();
49  			String submitControlId = submitControl == null ? null : submitControl.getId();
50  
51  			xml.appendOptionalAttribute("disabled", field.isDisabled(), "true");
52  			xml.appendOptionalAttribute("required", field.isMandatory(), "true");
53  			xml.appendOptionalAttribute("minLength", minLength > 0, minLength);
54  			xml.appendOptionalAttribute("maxLength", maxLength > 0, maxLength);
55  			xml.appendOptionalAttribute("tabIndex", field.hasTabIndex(), field.getTabIndex());
56  			xml.appendOptionalAttribute("toolTip", field.getToolTip());
57  			xml.appendOptionalAttribute("accessibleText", field.getAccessibleText());
58  			xml.appendOptionalAttribute("size", cols > 0, cols);
59  			xml.appendOptionalAttribute("buttonId", submitControlId);
60  			xml.appendOptionalAttribute("pattern", !Util.empty(pattern), pattern);
61  			xml.appendOptionalAttribute("list", suggestionsId);
62  			xml.appendOptionalAttribute("placeholder", field.getPlaceholder());
63  		}
64  		xml.appendClose();
65  
66  		xml.appendEscaped(field.getText());
67  
68  		xml.appendEndTag("ui:phonenumberfield");
69  	}
70  }
