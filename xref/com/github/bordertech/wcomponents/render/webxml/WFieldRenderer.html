1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WField;
5   import com.github.bordertech.wcomponents.WLabel;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   
9   /**
10   * The Renderer for {@link WField}.
11   *
12   * @author Jonathan Austin
13   * @since 1.0.0
14   */
15  final class WFieldRenderer extends AbstractWebXmlRenderer {
16  
17  	/**
18  	 * Paints the given WField.
19  	 *
20  	 * @param component the WField to paint.
21  	 * @param renderContext the RenderContext to paint to.
22  	 */
23  	@Override
24  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
25  		WField field = (WField) component;
26  		XmlStringBuilder xml = renderContext.getWriter();
27  		int inputWidth = field.getInputWidth();
28  
29  		xml.appendTagOpen("ui:field");
30  		xml.appendAttribute("id", component.getId());
31  		xml.appendOptionalAttribute("class", component.getHtmlClass());
32  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
33  		xml.appendOptionalAttribute("hidden", field.isHidden(), "true");
34  		xml.appendOptionalAttribute("inputWidth", inputWidth > 0, inputWidth);
35  		xml.appendClose();
36  
37  		// Label
38  		WLabel label = field.getLabel();
39  
40  		if (label != null) {
41  			label.paint(renderContext);
42  		}
43  
44  		// Field
45  		if (field.getField() != null) {
46  			xml.appendTag("ui:input");
47  			field.getField().paint(renderContext);
48  
49  			if (field.getErrorIndicator() != null) {
50  				field.getErrorIndicator().paint(renderContext);
51  			}
52  
53  			if (field.getWarningIndicator() != null) {
54  				field.getWarningIndicator().paint(renderContext);
55  			}
56  
57  			xml.appendEndTag("ui:input");
58  		}
59  
60  		xml.appendEndTag("ui:field");
61  	}
62  }
