12  /**
13   * The {@link Renderer} for {@link WDropdown}.
14   *
15   * @author Yiannis Paschalidis
16   * @since 1.0.0
17   */
18  final class WDropdownRenderer extends AbstractWebXmlRenderer {
19  
20  	/**
21  	 * Paints the given WDropdown.
22  	 *
23  	 * @param component the WDropdown to paint.
24  	 * @param renderContext the RenderContext to paint to.
25  	 */
26  	@Override
27  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
28  		WDropdown dropdown = (WDropdown) component;
29  		XmlStringBuilder xml = renderContext.getWriter();
30  		boolean readOnly = dropdown.isReadOnly();
31  		String dataKey = dropdown.getListCacheKey();
32  
33  		// Start tag
34  		xml.appendTagOpen("ui:dropdown");
35  		xml.appendAttribute("id", component.getId());
36  		xml.appendOptionalAttribute("class", component.getHtmlClass());
37  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
38  		xml.appendOptionalAttribute("hidden", dropdown.isHidden(), "true");
39  		if (readOnly) {
40  			xml.appendAttribute("readOnly", "true");
41  		} else {
42  			xml.appendOptionalAttribute("data", dataKey != null, dataKey);
43  			xml.appendOptionalAttribute("disabled", dropdown.isDisabled(), "true");
44  			xml.appendOptionalAttribute("required", dropdown.isMandatory(), "true");
45  			xml.appendOptionalAttribute("submitOnChange", dropdown.isSubmitOnChange(), "true");
46  			xml.appendOptionalAttribute("tabindex", component.hasTabIndex(), component.getTabIndex());
47  			xml.appendOptionalAttribute("toolTip", dropdown.getToolTip());
48  			xml.appendOptionalAttribute("accessibleText", dropdown.getAccessibleText());
49  			int optionWidth = dropdown.getOptionWidth();
50  			xml.appendOptionalAttribute("optionWidth", optionWidth > 0, optionWidth);
51  			xml.appendOptionalAttribute("type", getTypeAsString(dropdown.getType()));
52  		}
53  		xml.appendClose();
54  
55  		// Options
56  		List<?> options = dropdown.getOptions();
57  		Object selectedOption = dropdown.getSelected();
58  
59  		// For an editable dropdown (combo box), the selected option may not be in the list.
60  		if (dropdown.getType() == WDropdown.DropdownType.COMBO
61  				&& selectedOption != null
62  				&& (options == null || !options.contains(selectedOption))) {
63  			xml.appendTagOpen("ui:option");
64  			xml.appendAttribute("value", selectedOption.toString());
65  			xml.appendAttribute("selected", "true");
66  			xml.appendClose();
67  			xml.appendEscaped(selectedOption.toString());
68  			xml.appendEndTag("ui:option");
69  		}
70  
71  		if (options != null) {
72  			int optionIndex = 0;
73  			boolean renderSelectionsOnly = readOnly || dataKey != null;
74  
75  			for (Object option : options) {
76  				if (option instanceof OptionGroup) {
77  					xml.appendTagOpen("ui:optgroup");
78  					xml.appendAttribute("label", ((OptionGroup) option).getDesc());
79  					xml.appendClose();
80  
81  					for (Object nestedOption : ((OptionGroup) option).getOptions()) {
82  						renderOption(dropdown, nestedOption, optionIndex++, xml, selectedOption,
83  								renderSelectionsOnly);
84  					}
85  
86  					xml.appendEndTag("ui:optgroup");
87  				} else {
88  					renderOption(dropdown, option, optionIndex++, xml, selectedOption,
89  							renderSelectionsOnly);
90  				}
91  			}
92  		}
93  
94  		// End tag
95  		xml.appendEndTag("ui:dropdown");
96  	}
97  
98  	/**
99  	 * Renders a single option within the dropdown.
100 	 *
101 	 * @param dropdown the dropdown being rendered.
102 	 * @param option the option to render.
103 	 * @param optionIndex the index of the option. OptionGroups are not counted.
104 	 * @param html the XmlStringBuilder to paint to.
105 	 * @param selectedOption the dropdown's selected option.
106 	 * @param renderSelectionsOnly true to only render selected options, false to render all options.
107 	 */
108 	private void renderOption(final WDropdown dropdown, final Object option,
109 			final int optionIndex, final XmlStringBuilder html, final Object selectedOption,
110 			final boolean renderSelectionsOnly) {
111 		boolean selected = Util.equals(option, selectedOption);
112 
113 		if (selected || !renderSelectionsOnly) {
114 			// Get Code and Desc
115 			String code = dropdown.getCode(option, optionIndex);
116 			String desc = dropdown.getDesc(option, optionIndex);
117 
118 			// Check for null option (ie null or empty). Match isEmpty() logic.
119 			boolean isNull = option == null ? true : (option.toString().length() == 0);
120 
121 			// Render option
122 			html.appendTagOpen("ui:option");
123 			html.appendAttribute("value", code);
124 			html.appendOptionalAttribute("selected", selected, "true");
125 			html.appendOptionalAttribute("isNull", isNull, "true");
126 			html.appendClose();
127 			html.appendEscaped(desc);
128 			html.appendEndTag("ui:option");
129 		}
130 	}
131 
132 	/**
133 	 * Determines the theme tabset type name for the given TabSetType.
134 	 *
135 	 * @param type the TabSetType
136 	 * @return the theme tabset type name for the given TabSetType.
137 	 */
138 	public static String getTypeAsString(final WDropdown.DropdownType type) {
139 		if (type == null) {
140 			return null;
141 		}
142 
143 		switch (type) {
144 			case NATIVE:
145 				return null;
146 			case COMBO:
147 				return "combo";
148 			default:
149 				throw new IllegalStateException("Invalid dropdown type: " + type);
150 		}
151 	}
152 }
