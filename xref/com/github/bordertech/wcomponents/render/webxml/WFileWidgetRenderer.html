3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WFileWidget;
5   import com.github.bordertech.wcomponents.XmlStringBuilder;
6   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
7   import java.util.Iterator;
8   import java.util.List;
9   
10  /**
11   * The Renderer for {@link WFileWidget}.
12   *
13   * @author Yiannis Paschalidis
14   * @since 1.0.0
15   */
16  final class WFileWidgetRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WFileWidget.
20  	 *
21  	 * @param component the WFileWidget to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WFileWidget fileWidget = (WFileWidget) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  		boolean readOnly = fileWidget.isReadOnly();
29  
30  		xml.appendTagOpen("ui:fileupload");
31  		xml.appendAttribute("id", component.getId());
32  		xml.appendOptionalAttribute("class", component.getHtmlClass());
33  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
34  		xml.appendOptionalAttribute("hidden", component.isHidden(), "true");
35  		if (readOnly) {
36  			xml.appendAttribute("readOnly", "true");
37  		} else {
38  			xml.appendOptionalAttribute("disabled", fileWidget.isDisabled(), "true");
39  			xml.appendOptionalAttribute("required", fileWidget.isMandatory(), "true");
40  			xml.appendOptionalAttribute("tabIndex", fileWidget.hasTabIndex(), fileWidget.getTabIndex());
41  			xml.appendOptionalAttribute("toolTip", fileWidget.getToolTip());
42  			xml.appendOptionalAttribute("accessibleText", fileWidget.getAccessibleText());
43  			xml.appendOptionalAttribute("acceptedMimeTypes", typesToString(fileWidget.getFileTypes()));
44  			long maxFileSize = fileWidget.getMaxFileSize();
45  			xml.appendOptionalAttribute("maxFileSize", maxFileSize > 0, maxFileSize);
46  		}
47  		xml.appendEnd();
48  	}
49  
50  	/**
51  	 * Flattens the list of accepted mime types into a format suitable for rendering.
52  	 *
53  	 * @param types the list of accepted mime types
54  	 * @return the list of accepted mime types in a format suitable for rendering
55  	 */
56  	private String typesToString(final List<String> types) {
57  		if (types == null || types.isEmpty()) {
58  			return null;
59  		}
60  
61  		StringBuffer typesString = new StringBuffer();
62  
63  		for (Iterator<String> iter = types.iterator(); iter.hasNext();) {
64  			typesString.append(iter.next());
65  
66  			if (iter.hasNext()) {
67  				typesString.append(',');
68  			}
69  		}
70  
71  		return typesString.toString();
72  	}
73  }
