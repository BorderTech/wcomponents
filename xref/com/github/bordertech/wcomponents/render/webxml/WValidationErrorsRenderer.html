13  /**
14   * The Renderer for the {@link WValidationErrors} component.
15   *
16   * @author Jonathan Austin
17   * @author Mark Reeves
18   * @since 1.0.0
19   */
20  final class WValidationErrorsRenderer extends AbstractWebXmlRenderer {
21  
22  	/**
23  	 * Paints the given {@link WValidationErrors} component.
24  	 *
25  	 * @param component The {@link WValidationErrors} component to paint.
26  	 * @param renderContext the RenderContext to paint to.
27  	 */
28  	@Override
29  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
30  		WValidationErrors errors = (WValidationErrors) component;
31  		XmlStringBuilder xml = renderContext.getWriter();
32  
33  		if (errors.hasErrors()) {
34  			xml.appendTagOpen("ui:validationerrors");
35  			xml.appendAttribute("id", component.getId());
36  		xml.appendOptionalAttribute("class", component.getHtmlClass());
37  			xml.appendOptionalAttribute("track", component.isTracking(), "true");
38  			xml.appendOptionalAttribute("title", errors.getTitleText());
39  			xml.appendClose();
40  
41  			for (GroupedDiagnositcs nextGroup : errors.getGroupedErrors()) {
42  				// Render each diagnostic message in this group.
43  				for (Diagnostic nextMessage : nextGroup.getDiagnostics()) {
44  					xml.appendTagOpen("ui:error");
45  					WComponent forComponent = nextMessage.getComponent();
46  
47  					if (forComponent != null) {
48  						UIContextHolder.pushContext(nextMessage.getContext());
49  
50  						try {
51  							xml.appendAttribute("for", forComponent.getId());
52  						} finally {
53  							UIContextHolder.popContext();
54  						}
55  					}
56  
57  					xml.appendClose();
58  
59  					// DiagnosticImpl has been extended to support rendering
60  					// of a WComponent as the message.
61  					if (nextMessage instanceof DiagnosticImpl) {
62  						WComponent messageComponent = ((DiagnosticImpl) nextMessage)
63  								.createDiagnosticErrorComponent();
64  
65  						// We add the component to a throw-away container so that it renders with the correct ID.
66  						WContainer container = new WContainer() {
67  							@Override
68  							public String getId() {
69  								return component.getId();
70  							}
71  						};
72  
73  						container.add(messageComponent);
74  						messageComponent.paint(renderContext);
75  						container.remove(messageComponent);
76  						container.reset();
77  					} else {
78  						xml.append(nextMessage.getDescription());
79  					}
80  
81  					xml.appendEndTag("ui:error");
82  				}
83  			}
84  
85  			xml.appendEndTag("ui:validationerrors");
86  		}
87  	}
