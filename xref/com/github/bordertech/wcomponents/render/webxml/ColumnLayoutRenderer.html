11  /**
12   * This {@link Renderer} renders the children of a {@link WPanel} which have been arranged using a {@link ColumnLayout}.
13   *
14   * @author Yiannis Paschalidis
15   * @author Mark Reeves
16   * @since 1.0.0
17   */
18  final class ColumnLayoutRenderer extends AbstractWebXmlRenderer {
19  
20  	/**
21  	 * Paints the given WPanel's children.
22  	 *
23  	 * @param component the container to paint.
24  	 * @param renderContext the RenderContext to paint to.
25  	 */
26  	@Override
27  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
28  		WPanel panel = (WPanel) component;
29  		XmlStringBuilder xml = renderContext.getWriter();
30  		ColumnLayout layout = (ColumnLayout) panel.getLayout();
31  		int childCount = panel.getChildCount();
32  		Size hgap = layout.getHorizontalGap();
33  		String hgapString = hgap == null ? null : hgap.toString();
34  		Size vgap = layout.getVerticalGap();
35  		String vgapString = vgap == null ? null : vgap.toString();
36  		int cols = layout.getColumnCount();
37  
38  		xml.appendTagOpen("ui:columnlayout");
39  		xml.appendOptionalAttribute("hgap", hgapString);
40  		xml.appendOptionalAttribute("vgap", vgapString);
41  		xml.appendClose();
42  
43  		// Column Definitions
44  		for (int col = 0; col < cols; col++) {
45  			xml.appendTagOpen("ui:column");
46  			int width = layout.getColumnWidth(col);
47  			xml.appendOptionalAttribute("width", width > 0, width);
48  
49  			switch (layout.getColumnAlignment(col)) {
50  				case LEFT:
51  					// left is assumed if omitted
52  					break;
53  
54  				case RIGHT:
55  					xml.appendAttribute("align", "right");
56  					break;
57  
58  				case CENTER:
59  					xml.appendAttribute("align", "center");
60  					break;
61  
62  				default:
63  					throw new IllegalArgumentException("Invalid alignment: " + layout.
64  							getColumnAlignment(col));
65  			}
66  
67  			xml.appendEnd();
68  		}
69  
70  		for (int i = 0; i < childCount; i++) {
71  			xml.appendTag("ui:cell");
72  			WComponent child = panel.getChildAt(i);
73  			child.paint(renderContext);
74  			xml.appendEndTag("ui:cell");
75  		}
76  
77  		xml.appendEndTag("ui:columnlayout");
78  	}
79  }
