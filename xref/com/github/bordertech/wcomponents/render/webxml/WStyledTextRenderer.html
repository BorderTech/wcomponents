12  /**
13   * {@link Renderer} for the {@link WStyledText} component.
14   *
15   * @author Yiannis Paschalidis
16   * @since 1.0.0
17   */
18  final class WStyledTextRenderer extends AbstractWebXmlRenderer {
19  
20  	/**
21  	 * Paints the given WStyledText.
22  	 *
23  	 * @param component the WStyledText to paint.
24  	 * @param renderContext the RenderContext to paint to.
25  	 */
26  	@Override
27  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
28  		WStyledText text = (WStyledText) component;
29  		XmlStringBuilder xml = renderContext.getWriter();
30  		String textString = text.getText();
31  
32  		if (textString != null && textString.length() > 0) {
33  			xml.appendTagOpen("ui:text");
34  
35  			xml.appendOptionalAttribute("class", component.getHtmlClass());
36  			switch (text.getType()) {
37  				case EMPHASISED:
38  					xml.appendAttribute("type", "emphasised");
39  					break;
40  
41  				case HIGH_PRIORITY:
42  					xml.appendAttribute("type", "highPriority");
43  					break;
44  
45  				case LOW_PRIORITY:
46  					xml.appendAttribute("type", "lowPriority");
47  					break;
48  
49  				case MEDIUM_PRIORITY:
50  					xml.appendAttribute("type", "mediumPriority");
51  					break;
52  
53  				case ACTIVE_INDICATOR:
54  					xml.appendAttribute("type", "activeIndicator");
55  					break;
56  
57  				case MATCH_INDICATOR:
58  					xml.appendAttribute("type", "matchIndicator");
59  					break;
60  
61  				case INSERT:
62  					xml.appendAttribute("type", "insert");
63  					break;
64  
65  				case DELETE:
66  					xml.appendAttribute("type", "delete");
67  					break;
68  
69  				case MANDATORY_INDICATOR:
70  					xml.appendAttribute("type", "mandatoryIndicator");
71  					break;
72  
73  				case PLAIN:
74  				default:
75  					xml.appendAttribute("type", "plain");
76  					break;
77  			}
78  
79  			switch (text.getWhitespaceMode()) {
80  				case PARAGRAPHS:
81  					xml.appendAttribute("space", "paragraphs");
82  					break;
83  
84  				case PRESERVE:
85  					xml.appendAttribute("space", "preserve");
86  					break;
87  
88  				case DEFAULT:
89  					break;
90  
91  				default:
92  					throw new IllegalArgumentException("Unknown white space mode: " + text.
93  							getWhitespaceMode());
94  			}
95  
96  			xml.appendClose();
97  
98  			if (WStyledText.WhitespaceMode.PARAGRAPHS.equals(text.getWhitespaceMode())) {
99  				textString = text.isEncodeText() ? WebUtilities.encode(textString) : HtmlToXMLUtil.unescapeToXML(textString);
100 				writeParagraphs(textString, xml);
101 			} else {
102 				xml.append(textString, text.isEncodeText());
103 			}
104 
105 			xml.appendEndTag("ui:text");
106 		}
107 	}
108 
109 	/**
110 	 * Writes out paragraph delimited content.
111 	 *
112 	 * @param text the String content to output.
113 	 * @param xml the XmlStringBuilder to paint to.
114 	 */
115 	private static void writeParagraphs(final String text, final XmlStringBuilder xml) {
116 		if (!Util.empty(text)) {
117 			int start = 0;
118 			int end = text.length() - 1;
119 
120 			// Set the start index to the first non-linebreak, so we don't emit leading ui:nl tags
121 			for (; start < end; start++) {
122 				char c = text.charAt(start);
123 
124 				if (c != '\n' && c != '\r') {
125 					break;
126 				}
127 			}
128 
129 			// Set the end index to the last non-linebreak, so we don't emit trailing ui:nl tags
130 			for (; start < end; end--) {
131 				char c = text.charAt(end);
132 
133 				if (c != '\n' && c != '\r') {
134 					break;
135 				}
136 			}
137 
138 			char lastChar = 0;
139 
140 			for (int i = start; i <= end; i++) {
141 				char c = text.charAt(i);
142 
143 				if (c == '\n' || c == '\r') {
144 					if (lastChar != 0) {
145 						xml.write("<ui:nl/>");
146 					}
147 
148 					lastChar = 0;
149 				} else {
150 					xml.write(c);
151 					lastChar = c;
152 				}
153 			}
154 		}
155 	}
