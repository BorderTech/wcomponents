12  import java.util.ArrayList;
13  import java.util.List;

15  /**
16   * This {@link Renderer} renders the children of a {@link WPanel} which have been arranged using a {@link BorderLayout}.
17   *
18   * @author Yiannis Paschalidis
19   * @author Mark Reeves
20   * @since 1.0.0
21   */
22  final class BorderLayoutRenderer extends AbstractWebXmlRenderer {
23  
24  	/**
25  	 * Paints the given WPanel's children.
26  	 *
27  	 * @param component the container to paint.
28  	 * @param renderContext the RenderContext to paint to.
29  	 */
30  	@Override
31  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
32  		WPanel panel = (WPanel) component;
33  		XmlStringBuilder xml = renderContext.getWriter();
34  		BorderLayout layout = (BorderLayout) panel.getLayout();
35  		Size hgap = layout.getHorizontalGap();
36  		String hgapString = hgap == null ? null : hgap.toString();
37  		Size vgap = layout.getVerticalGap();
38  		String vgapString = vgap == null ? null : vgap.toString();
39  
40  		xml.appendTagOpen("ui:borderlayout");
41  
42  		xml.appendOptionalAttribute("hgap", hgapString);
43  		xml.appendOptionalAttribute("vgap", vgapString);
44  
45  		xml.appendClose();
46  
47  		// Fetch the children and their constraints.
48  		final int childCount = panel.getChildCount();
49  		List<Duplet<WComponent, BorderLayoutConstraint>> children = new ArrayList<>(childCount);
50  
51  		for (int i = 0; i < childCount; i++) {
52  			WComponent child = panel.getChildAt(i);
53  			children.add(new Duplet<>(child, getConstraints(panel, child)));
54  		}
55  
56  		// Now paint them
57  		paintChildrenWithConstraint(children, renderContext, BorderLayout.NORTH);
58  		paintChildrenWithConstraint(children, renderContext, BorderLayout.EAST);
59  		paintChildrenWithConstraint(children, renderContext, BorderLayout.SOUTH);
60  		paintChildrenWithConstraint(children, renderContext, BorderLayout.WEST);
61  		paintChildrenWithConstraint(children, renderContext, BorderLayout.CENTER);
62  
63  		xml.appendEndTag("ui:borderlayout");
64  	}
65  
66  	/**
67  	 * Retrieves the name of the element that will contain components with the given constraint.
68  	 *
69  	 * @param constraint the constraint to retrieve the tag for.
70  	 * @return the tag for the given constraint.
71  	 */
72  	private String getTag(final BorderLayoutConstraint constraint) {
73  		switch (constraint) {
74  			case EAST:
75  				return "ui:east";
76  			case NORTH:
77  				return "ui:north";
78  			case SOUTH:
79  				return "ui:south";
80  			case WEST:
81  				return "ui:west";
82  			case CENTER:
83  
84  			default:
85  				return "ui:center";
86  		}
87  	}
88  
89  	/**
90  	 * Paints all the child components with the given constraint.
91  	 *
92  	 * @param children the list of potential children to paint.
93  	 * @param renderContext the RenderContext to paint to.
94  	 * @param constraint the target constraint.
95  	 */
96  	private void paintChildrenWithConstraint(
97  			final List<Duplet<WComponent, BorderLayoutConstraint>> children,
98  			final WebXmlRenderContext renderContext, final BorderLayoutConstraint constraint) {
99  		String containingTag = null;
100 		XmlStringBuilder xml = renderContext.getWriter();
101 
102 		final int size = children.size();
103 
104 		for (int i = 0; i < size; i++) {
105 			Duplet<WComponent, BorderLayoutConstraint> child = children.get(i);
106 
107 			if (constraint.equals(child.getSecond())) {
108 				if (containingTag == null) {
109 					containingTag = getTag(constraint);
110 					xml.appendTag(containingTag);
111 				}
112 
113 				child.getFirst().paint(renderContext);
114 			}
115 		}
116 
117 		if (containingTag != null) {
118 			xml.appendEndTag(containingTag);
119 		}
120 	}
121 
122 	/**
123 	 * Retrieves the layout constraint for the given component. If the constraint is incorrectly configured, an
124 	 * {@link IllegalStateException} will be thrown.
125 	 *
126 	 * @param panel the panel which contains the child
127 	 * @param child the component to retrieve the constraint for.
128 	 *
129 	 * @return the layout constraint for the given component.
130 	 */
131 	private static BorderLayoutConstraint getConstraints(final WPanel panel, final WComponent child) {
132 		Object constraints = panel.getLayoutConstraints(child);
133 
134 		if (constraints instanceof BorderLayoutConstraint) {
135 			return (BorderLayoutConstraint) constraints;
136 		} else if (constraints == null) {
137 			return BorderLayout.CENTER;
138 		} else {
139 			throw new IllegalStateException("Constraint must be a BorderLayoutConstraint");
140 		}
141 	}
142 }
