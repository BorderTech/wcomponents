1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WSelectToggle;
5   import com.github.bordertech.wcomponents.WSelectToggle.State;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   
9   /**
10   * The Renderer for {@link WSelectToggle}.
11   *
12   * @author Yiannis Paschalidis
13   * @since 1.0.0
14   */
15  final class WSelectToggleRenderer extends AbstractWebXmlRenderer {
16  
17  	/**
18  	 * Paints the given WSelectToggle.
19  	 *
20  	 * @param component the WSelectToggle to paint.
21  	 * @param renderContext the RenderContext to paint to.
22  	 */
23  	@Override
24  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
25  		WSelectToggle toggle = (WSelectToggle) component;
26  		XmlStringBuilder xml = renderContext.getWriter();
27  
28  		xml.appendTagOpen("ui:selecttoggle");
29  		xml.appendAttribute("id", component.getId());
30  		xml.appendOptionalAttribute("class", component.getHtmlClass());
31  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
32  
33  		State state = toggle.getState();
34  
35  		if (State.ALL.equals(state)) {
36  			xml.appendAttribute("selected", "all");
37  		} else if (State.NONE.equals(state)) {
38  			xml.appendAttribute("selected", "none");
39  		} else {
40  			xml.appendAttribute("selected", "some");
41  		}
42  
43  		xml.appendOptionalAttribute("disabled", toggle.isDisabled(), "true");
44  		xml.appendAttribute("target", toggle.getTarget().getId());
45  		xml.appendAttribute("renderAs", toggle.isRenderAsText() ? "text" : "control");
46  		xml.appendOptionalAttribute("roundTrip", !toggle.isClientSide(), "true");
47  		xml.appendEnd();
48  	}
49  }
