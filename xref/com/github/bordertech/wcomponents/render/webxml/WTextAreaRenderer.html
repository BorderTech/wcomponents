1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WTextArea;
5   import com.github.bordertech.wcomponents.XmlStringBuilder;
6   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
7   import com.github.bordertech.wcomponents.util.HtmlToXMLUtil;
8   
9   /**
10   * The Renderer for WTextArea.
11   *
12   * @author Yiannis Paschalidis
13   * @author Rick Brown
14   * @since 1.0.0
15   */
16  class WTextAreaRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WTextArea.
20  	 *
21  	 * @param component the WTextArea to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WTextArea textArea = (WTextArea) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  		boolean readOnly = textArea.isReadOnly();
29  		xml.appendTagOpen("ui:textarea");
30  		xml.appendAttribute("id", component.getId());
31  		xml.appendOptionalAttribute("class", component.getHtmlClass());
32  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
33  		xml.appendOptionalAttribute("hidden", textArea.isHidden(), "true");
34  
35  		if (readOnly) {
36  			xml.appendAttribute("readOnly", "true");
37  		} else {
38  			int cols = textArea.getColumns();
39  			int rows = textArea.getRows();
40  			int minLength = textArea.getMinLength();
41  			int maxLength = textArea.getMaxLength();
42  			WComponent submitControl = textArea.getDefaultSubmitButton();
43  			String submitControlId = submitControl == null ? null : submitControl.getId();
44  
45  			xml.appendOptionalAttribute("disabled", textArea.isDisabled(), "true");
46  			xml.appendOptionalAttribute("required", textArea.isMandatory(), "true");
47  			xml.appendOptionalAttribute("minLength", minLength > 0, minLength);
48  			xml.appendOptionalAttribute("maxLength", maxLength > 0, maxLength);
49  			xml.appendOptionalAttribute("tabIndex", textArea.hasTabIndex(), textArea.getTabIndex());
50  			xml.appendOptionalAttribute("toolTip", textArea.getToolTip());
51  			xml.appendOptionalAttribute("accessibleText", textArea.getAccessibleText());
52  			xml.appendOptionalAttribute("rows", rows > 0, rows);
53  			xml.appendOptionalAttribute("cols", cols > 0, cols);
54  			xml.appendOptionalAttribute("buttonId", submitControlId);
55  			xml.appendOptionalAttribute("placeholder", textArea.getPlaceholder());
56  		}
57  		xml.appendClose();
58  
59  		if (textArea.isRichTextArea()) {
60  			/*
61  			 * This is a nested element instead of an attribute to cater for future enhancements
62  			 * such as turning rich text features on or off, or specifying JSON config either as
63  			 * a URL attribute or a nested CDATA section.
64  			 */
65  			xml.append("<ui:rtf />");
66  		}
67  
68  		String textString = textArea.getText();
69  		if (textString != null) {
70  			if (textArea.isReadOnly() && textArea.isRichTextArea()) {
71  				// read only we want to output unescaped, but it must still be XML valid.
72  				xml.write(HtmlToXMLUtil.unescapeToXML(textString));
73  			} else {
74  				xml.appendEscaped(textString);
75  			}
76  		}
77  
78  		xml.appendEndTag("ui:textarea");
79  	}
80  }
