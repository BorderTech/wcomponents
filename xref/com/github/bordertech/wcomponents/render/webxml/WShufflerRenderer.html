1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.Renderer;
4   import com.github.bordertech.wcomponents.WComponent;
5   import com.github.bordertech.wcomponents.WShuffler;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   import java.util.List;
9   
10  /**
11   * The {@link Renderer} for {@link WShuffler}.
12   *
13   * @author Jonathan Austin
14   * @since 1.0.0
15   */
16  final class WShufflerRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WShuffler.
20  	 *
21  	 * @param component the WShuffler to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WShuffler shuffler = (WShuffler) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  		boolean readOnly = shuffler.isReadOnly();
29  
30  		// Start tag
31  		xml.appendTagOpen("ui:shuffler");
32  		xml.appendAttribute("id", component.getId());
33  		xml.appendOptionalAttribute("class", component.getHtmlClass());
34  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
35  		xml.appendOptionalAttribute("hidden", shuffler.isHidden(), "true");
36  		if (readOnly) {
37  			xml.appendAttribute("readOnly", "true");
38  		} else {
39  			xml.appendOptionalAttribute("disabled", shuffler.isDisabled(), "true");
40  			xml.appendOptionalAttribute("tabindex", component.hasTabIndex(), component.getTabIndex());
41  			xml.appendOptionalAttribute("toolTip", shuffler.getToolTip());
42  			xml.appendOptionalAttribute("accessibleText", shuffler.getAccessibleText());
43  			int rows = shuffler.getRows();
44  			xml.appendOptionalAttribute("rows", rows > 0, rows);
45  		}
46  		xml.appendClose();
47  
48  		// Options
49  		List<?> options = shuffler.getOptions();
50  
51  		if (options != null && !options.isEmpty()) {
52  			for (int i = 0; i < options.size(); i++) {
53  				String stringOption = String.valueOf(options.get(i));
54  				xml.appendTagOpen("ui:option");
55  				xml.appendAttribute("value", stringOption);
56  				xml.appendClose();
57  				xml.appendEscaped(stringOption);
58  				xml.appendEndTag("ui:option");
59  			}
60  		}
61  
62  		// End tag
63  		xml.appendEndTag("ui:shuffler");
64  	}
65  }
