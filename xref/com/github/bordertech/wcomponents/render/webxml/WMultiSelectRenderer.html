3   import com.github.bordertech.wcomponents.OptionGroup;
4   import com.github.bordertech.wcomponents.WComponent;
5   import com.github.bordertech.wcomponents.WMultiSelect;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   import java.util.List;
9   
10  /**
11   * The Renderer for {@link WMultiSelect}.
12   *
13   * @author Jonathan Austin
14   * @since 1.0.0
15   */
16  final class WMultiSelectRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WMultiSelect.
20  	 *
21  	 * @param component the WMultiSelect to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WMultiSelect listBox = (WMultiSelect) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  		String dataKey = listBox.getListCacheKey();
29  		boolean readOnly = listBox.isReadOnly();
30  		int rows = listBox.getRows();
31  		int min = listBox.getMinSelect();
32  		int max = listBox.getMaxSelect();
33  
34  		xml.appendTagOpen("ui:listbox");
35  		xml.appendAttribute("id", component.getId());
36  		xml.appendOptionalAttribute("class", component.getHtmlClass());
37  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
38  		xml.appendOptionalAttribute("hidden", listBox.isHidden(), "true");
39  		if (readOnly) {
40  			xml.appendAttribute("readOnly", "true");
41  		} else {
42  			xml.appendOptionalAttribute("data", dataKey != null && !readOnly, dataKey);
43  			xml.appendOptionalAttribute("disabled", listBox.isDisabled(), "true");
44  			xml.appendOptionalAttribute("required", listBox.isMandatory(), "true");
45  			xml.appendOptionalAttribute("submitOnChange", listBox.isSubmitOnChange(), "true");
46  			xml.appendOptionalAttribute("tabIndex", component.hasTabIndex(), listBox.getTabIndex());
47  			xml.appendOptionalAttribute("toolTip", component.getToolTip());
48  			xml.appendOptionalAttribute("accessibleText", component.getAccessibleText());
49  			xml.appendOptionalAttribute("rows", rows >= 2, rows);
50  			xml.appendOptionalAttribute("min", min > 0, min);
51  			xml.appendOptionalAttribute("max", max > 0, max);
52  		}
53  		xml.appendClose();
54  
55  		// Options
56  		List<?> options = listBox.getOptions();
57  		boolean renderSelectionsOnly = readOnly || dataKey != null;
58  
59  		if (options != null) {
60  			int optionIndex = 0;
61  			List<?> selections = listBox.getSelected();
62  			for (Object option : options) {
63  				if (option instanceof OptionGroup) {
64  					xml.appendTagOpen("ui:optgroup");
65  					xml.appendAttribute("label", ((OptionGroup) option).getDesc());
66  					xml.appendClose();
67  
68  					for (Object nestedOption : ((OptionGroup) option).getOptions()) {
69  						renderOption(listBox, nestedOption, optionIndex++, xml, selections,
70  								renderSelectionsOnly);
71  					}
72  
73  					xml.appendEndTag("ui:optgroup");
74  				} else {
75  					renderOption(listBox, option, optionIndex++, xml, selections,
76  							renderSelectionsOnly);
77  				}
78  			}
79  		}
80  
81  		xml.appendEndTag("ui:listbox");
82  	}
83  
84  	/**
85  	 * Renders a single option within the list box.
86  	 *
87  	 * @param listBox the list box being rendered.
88  	 * @param option the option to render.
89  	 * @param optionIndex the index of the option. OptionGroups are not counted.
90  	 * @param html the XmlStringBuilder to paint to.
91  	 * @param selections the list of selected options.
92  	 * @param renderSelectionsOnly true to only render selected options, false to render all options.
93  	 */
94  	private void renderOption(final WMultiSelect listBox, final Object option,
95  			final int optionIndex, final XmlStringBuilder html,
96  			final List<?> selections, final boolean renderSelectionsOnly) {
97  		boolean selected = selections.contains(option);
98  
99  		if (selected || !renderSelectionsOnly) {
100 			// Get Code and Desc
101 			String code = listBox.getCode(option, optionIndex);
102 			String desc = listBox.getDesc(option, optionIndex);
103 			// Render Option
104 			html.appendTagOpen("ui:option");
105 			html.appendAttribute("value", code);
106 			html.appendOptionalAttribute("selected", selected, "true");
107 			html.appendClose();
108 			html.appendEscaped(desc);
109 			html.appendEndTag("ui:option");
110 		}
111 	}
112 }
