3   import com.github.bordertech.wcomponents.Renderer;
4   import com.github.bordertech.wcomponents.Track;
5   import com.github.bordertech.wcomponents.Video;
6   import com.github.bordertech.wcomponents.WComponent;
7   import com.github.bordertech.wcomponents.WVideo;
8   import com.github.bordertech.wcomponents.XmlStringBuilder;
9   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
10  import com.github.bordertech.wcomponents.util.I18nUtilities;
11  import org.apache.commons.logging.Log;
12  import org.apache.commons.logging.LogFactory;
13  
14  /**
15   * {@link Renderer} for the {@link WVideo} component.
16   *
17   * @author Yiannis Paschalidis
18   * @since 1.0.0
19   */
20  final class WVideoRenderer extends AbstractWebXmlRenderer {
21  
22  	/**
23  	 * The logger instance for this class.
24  	 */
25  	private static final Log LOG = LogFactory.getLog(WVideoRenderer.class);
26  
27  	/**
28  	 * Paints the given WVideo.
29  	 *
30  	 * @param component the WVideo to paint.
31  	 * @param renderContext the RenderContext to paint to.
32  	 */
33  	@Override
34  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
35  		WVideo videoComponent = (WVideo) component;
36  		XmlStringBuilder xml = renderContext.getWriter();
37  		Video[] video = videoComponent.getVideo();
38  
39  		if (video == null || video.length == 0) {
40  			return;
41  		}
42  
43  		Track[] tracks = videoComponent.getTracks();
44  		WVideo.Controls controls = videoComponent.getControls();
45  		int width = videoComponent.getWidth();
46  		int height = videoComponent.getHeight();
47  		int duration = video[0].getDuration();
48  
49  		// Check for alternative text
50  		String alternativeText = videoComponent.getAltText();
51  
52  		if (alternativeText == null) {
53  			LOG.warn("Video should have a description.");
54  			alternativeText = null;
55  		} else {
56  			alternativeText = I18nUtilities.format(null, alternativeText);
57  		}
58  
59  		xml.appendTagOpen("ui:video");
60  		xml.appendAttribute("id", component.getId());
61  		xml.appendOptionalAttribute("class", component.getHtmlClass());
62  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
63  		xml.appendOptionalUrlAttribute("poster", videoComponent.getPosterUrl());
64  		xml.appendOptionalAttribute("alt", alternativeText);
65  		xml.appendOptionalAttribute("autoplay", videoComponent.isAutoplay(), "true");
66  		xml.appendOptionalAttribute("mediagroup", videoComponent.getMediaGroup());
67  		xml.appendOptionalAttribute("loop", videoComponent.isLoop(), "true");
68  		xml.appendOptionalAttribute("muted", videoComponent.isMuted(), "true");
69  		xml.appendOptionalAttribute("hidden", videoComponent.isHidden(), "true");
70  		xml.appendOptionalAttribute("disabled", videoComponent.isDisabled(), "true");
71  		xml.appendOptionalAttribute("toolTip", videoComponent.getToolTip());
72  		xml.appendOptionalAttribute("width", width > 0, width);
73  		xml.appendOptionalAttribute("height", height > 0, height);
74  		xml.appendOptionalAttribute("duration", duration > 0, duration);
75  
76  		switch (videoComponent.getPreload()) {
77  			case NONE:
78  				xml.appendAttribute("preload", "none");
79  				break;
80  
81  			case META_DATA:
82  				xml.appendAttribute("preload", "metadata");
83  				break;
84  
85  			case AUTO:
86  			default:
87  				break;
88  		}
89  
90  		if (controls != null && !WVideo.Controls.NATIVE.equals(controls)) {
91  			switch (controls) {
92  				case NONE:
93  					xml.appendAttribute("controls", "none");
94  					break;
95  
96  				case ALL:
97  					xml.appendAttribute("controls", "all");
98  					break;
99  
100 				case PLAY_PAUSE:
101 					xml.appendAttribute("controls", "play");
102 					break;
103 
104 				case DEFAULT:
105 					xml.appendAttribute("controls", "default");
106 					break;
107 
108 				default:
109 					LOG.error("Unknown control type: " + controls);
110 			}
111 		}
112 
113 		xml.appendClose();
114 
115 		String[] urls = videoComponent.getVideoUrls();
116 
117 		for (int i = 0; i < urls.length; i++) {
118 			xml.appendTagOpen("ui:src");
119 			xml.appendUrlAttribute("uri", urls[i]);
120 			xml.appendOptionalAttribute("type", video[i].getMimeType());
121 
122 			if (video[i].getSize() != null) {
123 				xml.appendOptionalAttribute("width", video[i].getSize().width > 0, video[i].
124 						getSize().width);
125 				xml.appendOptionalAttribute("height", video[i].getSize().height > 0, video[i].
126 						getSize().height);
127 			}
128 
129 			xml.appendEnd();
130 		}
131 
132 		if (tracks != null && tracks.length > 0) {
133 			String[] trackUrls = videoComponent.getTrackUrls();
134 
135 			for (int i = 0; i < tracks.length; i++) {
136 				xml.appendTagOpen("ui:track");
137 				xml.appendUrlAttribute("src", trackUrls[i]);
138 				xml.appendOptionalAttribute("lang", tracks[i].getLanguage());
139 				xml.appendOptionalAttribute("desc", tracks[i].getDescription());
140 				xml.appendOptionalAttribute("kind", trackKindToString(tracks[i].getKind()));
141 				xml.appendEnd();
142 			}
143 		}
144 
145 		xml.appendEndTag("ui:video");
146 	}
147 
148 	/**
149 	 * Converts a Track kind to the client track kind identifier.
150 	 *
151 	 * @param kind the Track Kind to convert.
152 	 * @return a client track kind identifier, or null if it could not be converted.
153 	 */
154 	private String trackKindToString(final Track.Kind kind) {
155 		if (kind == null) {
156 			return null;
157 		}
158 
159 		switch (kind) {
160 			case SUBTITLES:
161 				return "subtitles";
162 
163 			case CAPTIONS:
164 				return "captions";
165 
166 			case DESCRIPTIONS:
167 				return "descriptions";
168 
169 			case CHAPTERS:
170 				return "chapters";
171 
172 			case METADATA:
173 				return "metadata";
174 
175 			default:
176 				LOG.error("Unknown track kind " + kind);
177 				return null;
178 		}
179 	}
180 }
