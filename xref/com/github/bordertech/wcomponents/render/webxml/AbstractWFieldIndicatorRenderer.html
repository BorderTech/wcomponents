7   import com.github.bordertech.wcomponents.util.SystemException;
8   import com.github.bordertech.wcomponents.validation.AbstractWFieldIndicator;
9   import com.github.bordertech.wcomponents.validation.Diagnostic;
10  import java.util.List;
11  
12  /**
13   * {@link Renderer} for the {@link AbstractWFieldIndicator} component.
14   *
15   * @author Jonathan Austin
16   * @since 1.0.0
17   */
18  abstract class AbstractWFieldIndicatorRenderer extends AbstractWebXmlRenderer {
19  
20  	/**
21  	 * Paints the given AbstractWFieldIndicator.
22  	 *
23  	 * @param component the WFieldErrorIndicator to paint.
24  	 * @param renderContext the RenderContext to paint to.
25  	 */
26  	@Override
27  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
28  		AbstractWFieldIndicator fieldIndicator = (AbstractWFieldIndicator) component;
29  		XmlStringBuilder xml = renderContext.getWriter();
30  
31  		List<Diagnostic> diags = fieldIndicator.getDiagnostics();
32  
33  		if (diags != null && !diags.isEmpty()) {
34  			xml.appendTagOpen("ui:fieldindicator");
35  			xml.appendAttribute("id", component.getId());
36  			xml.appendOptionalAttribute("track", component.isTracking(), "true");
37  
38  			switch (fieldIndicator.getFieldIndicatorType()) {
39  				case INFO:
40  					xml.appendAttribute("type", "info");
41  					break;
42  
43  				case WARN:
44  					xml.appendAttribute("type", "warn");
45  					break;
46  
47  				case ERROR:
48  					xml.appendAttribute("type", "error");
49  					break;
50  
51  				default:
52  					throw new SystemException(
53  							"Cannot paint field indicator due to an invalid field indicator type: " + fieldIndicator.
54  							getFieldIndicatorType());
55  			}
56  
57  			xml.appendAttribute("for", fieldIndicator.getRelatedFieldId());
58  			xml.appendClose();
59  
60  			for (Diagnostic diag : diags) {
61  				xml.appendTag("ui:message");
62  				xml.appendEscaped(diag.getDescription());
63  				xml.appendEndTag("ui:message");
64  			}
65  
66  			xml.appendEndTag("ui:fieldindicator");
67  		}
68  	}
69  }
