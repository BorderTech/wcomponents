14  /**
15   * {@link Renderer} for the {@link WTemplate} component.
16   *
17   * @author Jonathan Austin
18   * @since 1.0.3
19   */
20  final class WTemplateRenderer extends AbstractWebXmlRenderer {
21  
22  	/**
23  	 * Paints the given WTemplate.
24  	 *
25  	 * @param component the WTemplate to paint.
26  	 * @param renderContext the RenderContext to paint to.
27  	 */
28  	@Override
29  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
30  		WTemplate template = (WTemplate) component;
31  
32  		// Setup the context
33  		Map<String, Object> context = new HashMap<>();
34  		// Make the component available under the "wc" key.
35  		context.put("wc", template);
36  		// Load the parameters
37  		context.putAll(template.getParameters());
38  
39  		// Get template renderer for the engine
40  		String engine = template.getEngineName();
41  		if (Util.empty(engine)) {
42  			engine = ConfigurationProperties.getDefaultRenderingEngine();
43  		}
44  		TemplateRenderer templateRenderer = TemplateRendererFactory.newInstance(engine);
45  
46  		// Render
47  		if (!Util.empty(template.getTemplateName())) {
48  			// Render the template
49  			templateRenderer.renderTemplate(template.getTemplateName(), context, template.getTaggedComponents(), renderContext.getWriter(),
50  					template.getEngineOptions());
51  		} else if (!Util.empty(template.getInlineTemplate())) {
52  			// Render inline
53  			templateRenderer.renderInline(template.getInlineTemplate(), context, template.getTaggedComponents(), renderContext.getWriter(),
54  					template.getEngineOptions());
55  		}
56  	}
57  
58  }
