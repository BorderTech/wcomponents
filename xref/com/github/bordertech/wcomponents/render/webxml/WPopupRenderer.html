3   import com.github.bordertech.wcomponents.Renderer;
4   import com.github.bordertech.wcomponents.WComponent;
5   import com.github.bordertech.wcomponents.WPopup;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   import com.github.bordertech.wcomponents.util.Util;
9   
10  /**
11   * The {@link Renderer} for {@link WPopup}.
12   *
13   * @author Jonathan Austin
14   * @since 1.0.0
15   */
16  final class WPopupRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WPopup.
20  	 *
21  	 * @param component the WPopup to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WPopup popup = (WPopup) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  		int width = popup.getWidth();
29  		int height = popup.getHeight();
30  		String targetWindow = popup.getTargetWindow();
31  
32  		xml.appendTagOpen("ui:popup");
33  		xml.appendUrlAttribute("url", popup.getUrl());
34  		xml.appendOptionalAttribute("width", width > 0, width);
35  		xml.appendOptionalAttribute("height", height > 0, height);
36  		xml.appendOptionalAttribute("resizable", popup.isResizable(), "true");
37  		xml.appendOptionalAttribute("showScrollbars", popup.isScrollable(), "true");
38  		xml.appendOptionalAttribute("targetWindow", !Util.empty(targetWindow), targetWindow);
39  		xml.appendClose();
40  		xml.appendEndTag("ui:popup");
41  	}
42  }
