1   package com.github.bordertech.wcomponents.render.webxml;

3   import com.github.bordertech.wcomponents.Container;
4   import com.github.bordertech.wcomponents.RenderContext;
5   import com.github.bordertech.wcomponents.Renderer;
6   import com.github.bordertech.wcomponents.WComponent;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   import com.github.bordertech.wcomponents.util.SystemException;
9   
10  /**
11   * An abstract web xml renderer.
12   *
13   * All web xml renderers should extend this class, and must be thread-safe, as only a single instance of the renderer is
14   * created by the UIManager. It is recommended that renderers do not keep any state information, to avoid the
15   * performance penality of having to synchronize methods.
16   *
17   * @author Yiannis Paschalidis
18   * @since 1.0.0
19   */
20  public abstract class AbstractWebXmlRenderer implements Renderer {
21  
22  	/**
23  	 * Renders the component.
24  	 *
25  	 * @param component the component to paint
26  	 * @param renderContext the context for rendering.
27  	 */
28  	@Override
29  	public void render(final WComponent component, final RenderContext renderContext) {
30  		if (renderContext instanceof WebXmlRenderContext) {
31  			doRender(component, (WebXmlRenderContext) renderContext);
32  		} else {
33  			throw new SystemException("Unable to render web xml output to " + renderContext);
34  		}
35  	}
36  
37  	/**
38  	 * Paints the children of the given component.
39  	 *
40  	 * @param container the component whose children will be painted.
41  	 * @param renderContext the context for rendering.
42  	 */
43  	protected final void paintChildren(final Container container,
44  			final WebXmlRenderContext renderContext) {
45  		final int size = container.getChildCount();
46  
47  		for (int i = 0; i < size; i++) {
48  			WComponent child = container.getChildAt(i);
49  			child.paint(renderContext);
50  		}
51  	}
52  
53  	/**
54  	 * Paints the component and its children using the manager's layout algorithm.
55  	 *
56  	 * @param component the component to paint
57  	 * @param renderContext the context for rendering.
58  	 */
59  	public abstract void doRender(WComponent component, WebXmlRenderContext renderContext);
60  }
