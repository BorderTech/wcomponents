1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WMessageBox;
5   import com.github.bordertech.wcomponents.XmlStringBuilder;
6   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
7   
8   /**
9    * The Renderer for the {@link WMessageBox} component.
10   *
11   * @author Yiannis Paschalidis
12   * @author Jonathan Austin
13   * @author Mark Reeves
14   * @since 1.0.0
15   */
16  final class WMessageBoxRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WMessageBox.
20  	 *
21  	 * @param component the WMessageBox to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WMessageBox messageBox = (WMessageBox) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  
29  		if (messageBox.hasMessages()) {
30  			xml.appendTagOpen("ui:messagebox");
31  			xml.appendAttribute("id", component.getId());
32  			xml.appendOptionalAttribute("class", component.getHtmlClass());
33  			xml.appendOptionalAttribute("track", component.isTracking(), "true");
34  
35  			switch (messageBox.getType()) {
36  				case SUCCESS:
37  					xml.appendOptionalAttribute("type", "success");
38  					break;
39  
40  				case INFO:
41  					xml.appendOptionalAttribute("type", "info");
42  					break;
43  
44  				case WARN:
45  					xml.appendOptionalAttribute("type", "warn");
46  					break;
47  
48  				case ERROR:
49  				default:
50  					xml.appendOptionalAttribute("type", "error");
51  					break;
52  			}
53  
54  			xml.appendOptionalAttribute("title", messageBox.getTitleText());
55  
56  			xml.appendClose();
57  
58  			for (String message : messageBox.getMessages()) {
59  				xml.appendTag("ui:message");
60  				xml.print(message);
61  				xml.appendEndTag("ui:message");
62  			}
63  
64  			xml.appendEndTag("ui:messagebox");
65  		}
66  	}
67  }
