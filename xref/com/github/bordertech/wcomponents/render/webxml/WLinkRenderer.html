14  /**
15   * The {@link Renderer} for the {@link WLink} component.
16   *
17   * @author Yiannis Paschalidis
18   * @since 1.0.0
19   */
20  final class WLinkRenderer extends AbstractWebXmlRenderer {
21  
22  	/**
23  	 * Paints the given {@link WLink}.
24  	 *
25  	 * @param component the WLink to paint.
26  	 * @param renderContext the RenderContext to paint to.
27  	 */
28  	@Override
29  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
30  		WLink link = (WLink) component;
31  		XmlStringBuilder xml = renderContext.getWriter();
32  		String text = link.getText();
33  		String targetWindowName = link.getOpenNewWindow() ? link.getTargetWindowName() : null;
34  		String imageUrl = link.getImageUrl();
35  
36  		xml.appendTagOpen("ui:link");
37  		xml.appendAttribute("id", component.getId());
38  		xml.appendOptionalAttribute("class", component.getHtmlClass());
39  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
40  		xml.appendOptionalAttribute("disabled", link.isDisabled(), "true");
41  		xml.appendOptionalAttribute("hidden", component.isHidden(), "true");
42  		xml.appendOptionalAttribute("tabIndex", link.hasTabIndex(), link.getTabIndex());
43  		xml.appendOptionalAttribute("toolTip", link.getToolTip());
44  		xml.appendOptionalAttribute("accessibleText", link.getAccessibleText());
45  		xml.appendUrlAttribute("url", link.getUrl());
46  		xml.appendOptionalAttribute("rel", link.getRel());
47  		xml.appendOptionalAttribute("accessKey", Util.upperCase(link.getAccessKeyAsString()));
48  
49  		if (imageUrl != null) {
50  			xml.appendUrlAttribute("imageUrl", imageUrl);
51  			ImagePosition imagePosition = link.getImagePosition();
52  
53  			if (imagePosition != null) {
54  				switch (imagePosition) {
55  					case NORTH:
56  						xml.appendAttribute("imagePosition", "n");
57  						break;
58  					case EAST:
59  						xml.appendAttribute("imagePosition", "e");
60  						break;
61  					case SOUTH:
62  						xml.appendAttribute("imagePosition", "s");
63  						break;
64  					case WEST:
65  						xml.appendAttribute("imagePosition", "w");
66  						break;
67  					default:
68  						throw new SystemException("Unknown image position: " + imagePosition);
69  				}
70  			}
71  			// we have an image. We must have a text equivalent
72  			if (Util.empty(text) && Util.empty(link.getToolTip()) && Util.empty(link.getAccessibleText())) {
73  				// If the link has an umageUrl but no text equivalent get the text equivalent off of the image
74  				WImage linkImage = link.getImageHolder();
75  				if (null != linkImage) {
76  					xml.appendOptionalAttribute("toolTip", linkImage.getAlternativeText());
77  				}
78  			}
79  		}
80  
81  		if (link.isRenderAsButton()) {
82  			xml.appendAttribute("type", "button");
83  		}
84  
85  		xml.appendClose();
86  
87  		if (targetWindowName != null) {
88  			xml.appendTagOpen("ui:windowAttributes");
89  			xml.appendAttribute("name", targetWindowName);
90  
91  			WLink.WindowAttributes attributes = link.getWindowAttrs();
92  
93  			if (attributes != null) {
94  				xml.appendOptionalAttribute("top", attributes.getTop() >= 0, attributes.getTop());
95  				xml.appendOptionalAttribute("left", attributes.getLeft() >= 0, attributes.getLeft());
96  				xml.appendOptionalAttribute("width", attributes.getWidth() > 0, attributes.
97  						getWidth());
98  				xml.appendOptionalAttribute("height", attributes.getHeight() > 0, attributes.
99  						getHeight());
100 				xml.appendOptionalAttribute("resizable", attributes.isResizable(), "true");
101 				xml.appendOptionalAttribute("showMenubar", attributes.isMenubar(), "true");
102 				xml.appendOptionalAttribute("showToolbar", attributes.isToolbars(), "true");
103 				xml.appendOptionalAttribute("showLocation", attributes.isLocation(), "true");
104 				xml.appendOptionalAttribute("showStatus", attributes.isStatus(), "true");
105 				xml.appendOptionalAttribute("showScrollbars", attributes.isScrollbars(), "true");
106 				xml.appendOptionalAttribute("showDirectories", attributes.isDirectories(), "true");
107 			}
108 
109 			xml.appendEnd();
110 		}
111 
112 		if (text != null) {
113 			xml.appendEscaped(text);
114 		}
115 
116 		xml.appendEndTag("ui:link");
117 
118 		// Paint the AJAX trigger if the link has an action
119 		if (link.getAction() != null) {
120 			paintAjaxTrigger(link, xml);
121 		}
122 
123 	}
124 
125 	/**
126 	 * Paint the AJAX trigger if the link has an action.
127 	 *
128 	 * @param link the link component being rendered
129 	 * @param xml the XmlStringBuilder to paint to.
130 	 */
131 	private void paintAjaxTrigger(final WLink link, final XmlStringBuilder xml) {
132 		AjaxTarget[] actionTargets = link.getActionTargets();
133 
134 		// Start tag
135 		xml.appendTagOpen("ui:ajaxtrigger");
136 		xml.appendAttribute("triggerId", link.getId());
137 		xml.appendClose();
138 
139 		if (actionTargets != null && actionTargets.length > 0) {
140 			// Targets
141 			for (AjaxTarget target : actionTargets) {
142 				xml.appendTagOpen("ui:ajaxtargetid");
143 				xml.appendAttribute("targetId", target.getId());
144 				xml.appendEnd();
145 			}
146 		} else {
147 			// Target itself
148 			xml.appendTagOpen("ui:ajaxtargetid");
149 			xml.appendAttribute("targetId", link.getId());
150 			xml.appendEnd();
151 		}
152 
153 		// End tag
154 		xml.appendEndTag("ui:ajaxtrigger");
155 	}
156 }
