1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.Renderer;
4   import com.github.bordertech.wcomponents.WComponent;
5   import com.github.bordertech.wcomponents.WImageEditor;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   import java.awt.Dimension;
9   
10  /**
11   * {@link Renderer} for the {@link WImageEditor} component.
12   *
13   * @author Rick Brown
14   * @since 1.0.3
15   */
16  final class WImageEditorRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given {@link WImageEditor}.
20  	 *
21  	 * @param component the WImageEditor to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WImageEditor editorComponent = (WImageEditor) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  
29  		xml.appendTagOpen("wc-imageedit");
30  		xml.appendAttribute("id", editorComponent.getId());
31  		xml.appendOptionalAttribute("class", editorComponent.getHtmlClass());
32  		xml.appendOptionalUrlAttribute("overlay", editorComponent.getOverlayUrl());
33  		xml.appendOptionalAttribute("camera", editorComponent.getUseCamera());
34  		xml.appendOptionalAttribute("face", editorComponent.getIsFace());
35  
36  		// Check for size information on the image
37  		Dimension size = editorComponent.getSize();
38  		if (size != null) {
39  			if (size.getHeight() >= 0) {
40  				xml.appendAttribute("height", size.height);
41  			}
42  
43  			if (size.getWidth() >= 0) {
44  				xml.appendAttribute("width", size.width);
45  			}
46  		}
47  
48  		xml.appendEnd();
49  	}
50  }
