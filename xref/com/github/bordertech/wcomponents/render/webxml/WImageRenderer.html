11  /**
12   * {@link Renderer} for the {@link WImage} component.
13   *
14   * @author Yiannis Paschalidis
15   * @since 1.0.0
16   */
17  final class WImageRenderer extends AbstractWebXmlRenderer {
18  	/**
19  	 * Builds the "open tag" part of the XML, that is the tagname and attributes.
20  	 *
21  	 * E.g. &lt;ui:image src="example.png" alt="some alt txt"
22  	 *
23  	 * The caller may then append any additional attributes and then close the XML tag.
24  	 *
25  	 * @param imageComponent The WImage to render.
26  	 * @param xml The buffer to render the XML into.
27  	 */
28  	protected static void renderTagOpen(final WImage imageComponent, final XmlStringBuilder xml) {
29  
30  		// Check for alternative text on the image
31  		String alternativeText = imageComponent.getAlternativeText();
32  		if (alternativeText == null) {
33  			alternativeText = "";
34  		} else {
35  			alternativeText = I18nUtilities.format(null, alternativeText);
36  		}
37  
38  		xml.appendTagOpen("img");
39  		xml.appendAttribute("id", imageComponent.getId());
40  		xml.appendOptionalAttribute("class", imageComponent.getHtmlClass());
41  		xml.appendOptionalAttribute("track", imageComponent.isTracking(), "true");
42  		xml.appendUrlAttribute("src", imageComponent.getTargetUrl());
43  		xml.appendAttribute("alt", alternativeText);
44  		xml.appendOptionalAttribute("hidden", imageComponent.isHidden(), "hidden");
45  
46  		// Check for size information on the image
47  		Dimension size = imageComponent.getSize();
48  		if (size != null) {
49  			if (size.getHeight() >= 0) {
50  				xml.appendAttribute("height", size.height);
51  			}
52  
53  			if (size.getWidth() >= 0) {
54  				xml.appendAttribute("width", size.width);
55  			}
56  		}
57  	}
58  
59  	/**
60  	 * Paints the given {@link WImage}.
61  	 *
62  	 * @param component the WImage to paint.
63  	 * @param renderContext the RenderContext to paint to.
64  	 */
65  	@Override
66  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
67  		WImage imageComponent = (WImage) component;
68  		XmlStringBuilder xml = renderContext.getWriter();
69  
70  		// No image set
71  		if (imageComponent.getImage() == null && imageComponent.getImageUrl() == null) {
72  			return;
73  		}
74  
75  		renderTagOpen(imageComponent, xml);
76  		xml.appendEnd();
77  	}
78  }
