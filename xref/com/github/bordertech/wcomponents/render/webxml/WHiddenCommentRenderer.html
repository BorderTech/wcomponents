10  /**
11   * {@link Renderer} for the {@link WHiddenComment} component.
12   *
13   * @author Darian Bridge
14   * @since 1.0.0
15   */
16  final class WHiddenCommentRenderer extends AbstractWebXmlRenderer {
17  
18  	/**
19  	 * Paints the given WHiddenComment.
20  	 *
21  	 * @param component the WHiddenComment to paint.
22  	 * @param renderContext the RenderContext to paint to.
23  	 */
24  	@Override
25  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
26  		WHiddenComment hiddenComponent = (WHiddenComment) component;
27  		XmlStringBuilder xml = renderContext.getWriter();
28  
29  		String hiddenText = hiddenComponent.getText();
30  
31  		if (!Util.empty(hiddenText)) {
32  			xml.appendTag("ui:comment");
33  			xml.appendEscaped(hiddenText);
34  			xml.appendEndTag("ui:comment");
35  		}
36  	}
37  }
