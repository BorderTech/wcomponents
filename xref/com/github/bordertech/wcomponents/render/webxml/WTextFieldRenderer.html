1   package com.github.bordertech.wcomponents.render.webxml;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WSuggestions;
5   import com.github.bordertech.wcomponents.WTextField;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   import com.github.bordertech.wcomponents.util.Util;
9   
10  /**
11   * The Renderer for WTextField.
12   *
13   * @author Yiannis Paschalidis
14   * @author Jonathan Austin
15   * @since 1.0.0
16   */
17  class WTextFieldRenderer extends AbstractWebXmlRenderer {
18  
19  	/**
20  	 * Paints the given WTextField.
21  	 *
22  	 * @param component the WTextField to paint.
23  	 * @param renderContext the RenderContext to paint to.
24  	 */
25  	@Override
26  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
27  		WTextField textField = (WTextField) component;
28  		XmlStringBuilder xml = renderContext.getWriter();
29  		boolean readOnly = textField.isReadOnly();
30  
31  		xml.appendTagOpen("ui:textfield");
32  		xml.appendAttribute("id", component.getId());
33  		xml.appendOptionalAttribute("class", component.getHtmlClass());
34  		xml.appendOptionalAttribute("track", component.isTracking(), "true");
35  		xml.appendOptionalAttribute("hidden", component.isHidden(), "true");
36  		if (readOnly) {
37  			xml.appendAttribute("readOnly", "true");
38  		} else {
39  			int cols = textField.getColumns();
40  			int minLength = textField.getMinLength();
41  			int maxLength = textField.getMaxLength();
42  			String pattern = textField.getPattern();
43  
44  			WSuggestions suggestions = textField.getSuggestions();
45  			String suggestionsId = suggestions == null ? null : suggestions.getId();
46  
47  			WComponent submitControl = textField.getDefaultSubmitButton();
48  			String submitControlId = submitControl == null ? null : submitControl.getId();
49  
50  			xml.appendOptionalAttribute("disabled", textField.isDisabled(), "true");
51  			xml.appendOptionalAttribute("required", textField.isMandatory(), "true");
52  			xml.appendOptionalAttribute("minLength", minLength > 0, minLength);
53  			xml.appendOptionalAttribute("maxLength", maxLength > 0, maxLength);
54  			xml.appendOptionalAttribute("tabIndex", textField.hasTabIndex(), textField.getTabIndex());
55  			xml.appendOptionalAttribute("toolTip", textField.getToolTip());
56  			xml.appendOptionalAttribute("accessibleText", textField.getAccessibleText());
57  			xml.appendOptionalAttribute("size", cols > 0, cols);
58  			xml.appendOptionalAttribute("buttonId", submitControlId);
59  			xml.appendOptionalAttribute("pattern", !Util.empty(pattern), pattern);
60  			xml.appendOptionalAttribute("list", suggestionsId);
61  			xml.appendOptionalAttribute("placeholder", textField.getPlaceholder());
62  		}
63  
64  		xml.appendClose();
65  
66  		xml.appendEscaped(textField.getText());
67  
68  		xml.appendEndTag("ui:textfield");
69  	}
70  }
