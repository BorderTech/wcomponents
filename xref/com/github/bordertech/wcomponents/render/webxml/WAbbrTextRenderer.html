3   import com.github.bordertech.wcomponents.Renderer;
4   import com.github.bordertech.wcomponents.WAbbrText;
5   import com.github.bordertech.wcomponents.WComponent;
6   import com.github.bordertech.wcomponents.XmlStringBuilder;
7   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
8   
9   /**
10   * {@link Renderer} for the {@link WAbbrText} component.
11   *
12   * @author Yiannis Paschalidis
13   * @since 1.0.0
14   */
15  final class WAbbrTextRenderer extends AbstractWebXmlRenderer {
16  
17  	/**
18  	 * Paints the given WAbbrText.
19  	 *
20  	 * @param component the WAbbrText to paint.
21  	 * @param renderContext the RenderContext to paint to.
22  	 */
23  	@Override
24  	public void doRender(final WComponent component, final WebXmlRenderContext renderContext) {
25  		WAbbrText abbrText = (WAbbrText) component;
26  		XmlStringBuilder xml = renderContext.getWriter();
27  
28  		xml.appendTagOpen("abbr");
29  		xml.appendOptionalAttribute("title", abbrText.getToolTip());
30  		xml.appendOptionalAttribute("class", abbrText.getHtmlClass());
31  		xml.appendClose();
32  		xml.append(abbrText.getText(), abbrText.isEncodeText());
33  		xml.appendEndTag("abbr");
34  	}
35  }
