9   /**
10   * The layout factory for the nil renderer package. This factory produces a renderer which doesn't do anything, and is
11   * useful for testing.
12   *
13   * @author Yiannis Paschalidis
14   * @since 1.0.0
15   */
16  public final class RendererFactoryImpl implements RendererFactory {
17  
18  	/**
19  	 * The singleton NullRenderer instance.
20  	 */
21  	private static final Renderer NULL_RENDERER = new NullRenderer();
22  
23  	/**
24  	 * {@inheritDoc}
25  	 */
26  	@Override
27  	public Renderer getRenderer(final Class<?> clazz) {
28  		return NULL_RENDERER;
29  	}
30  
31  	/**
32  	 * {@inheritDoc}
33  	 *
34  	 * @deprecated Use {@link WTemplate} instead
35  	 */
36  	@Deprecated
37  	@Override
38  	public Renderer getTemplateRenderer() {
39  		return NULL_RENDERER;
40  	}
41  
42  	/**
43  	 * A no-op renderer.
44  	 */
45  	private static final class NullRenderer implements Renderer {
46  
47  		/**
48  		 * Doesn't do anything.
49  		 *
50  		 * @param component ignored.
51  		 * @param renderContext ignored.
52  		 */
53  		@Override
54  		public void render(final WComponent component, final RenderContext renderContext) {
55  			// NO-OP
56  		}
57  	};
58  }
