275 		MenuContainer container = WebUtilities.getAncestorOfClass(MenuContainer.class, this);

283 	/**
284 	 * @return {@code true} if a WMenuItem is allowed to have a selection state. This is determined by its nearest ancestor
285 	 * {@link MenuSelectContainer}'s selectionMode.
286 	 */
287 	@Override
288 	public boolean isSelectAllowed() {
289 		Boolean selectability = getSelectability();
290 		if (selectability != null) {
291 			return selectability;
292 		}
293 		MenuSelectContainer container = WebUtilities.getAncestorOfClass(MenuSelectContainer.class, this);
294 		if (container == null) {
295 			return false;
296 		}
297 		SelectionMode mode = container.getSelectionMode();
298 		return !SelectionMode.NONE.equals(mode);
299 	}

422 	/**
423 	 * Override handleRequest in order to perform processing for this component. This implementation checks for
424 	 * selection of the menu item, and executes the associated action if it has been set.
425 	 *
426 	 * @param request the request being responded to.
427 	 */
428 	@Override
429 	public void handleRequest(final Request request) {
430 		if (isDisabled()) {
431 			// Protect against client-side tampering of disabled/read-only fields.
432 			return;
433 		}
434 
435 		if (isMenuPresent(request)) {
436 			String requestValue = request.getParameter(getId());
437 
438 			if (requestValue != null) {
439 				// Only process on a POST
440 				if (!"POST".equals(request.getMethod())) {
441 					LOG.warn("Menu item on a request that is not a POST. Will be ignored.");
442 					return;
443 				}
444 
445 				// Execute associated action, if set
446 				final Action action = getAction();
447 
448 				if (action != null) {
449 					final ActionEvent event = new ActionEvent(this, this.getActionCommand(), this.
450 							getActionObject());
451 
452 					Runnable later = new Runnable() {
453 						@Override
454 						public void run() {
455 							action.execute(event);
456 						}
457 					};
458 
459 					invokeLater(later);
460 				}
461 			}
462 		}
463 	}

465 	/**
466 	 * Determine if this WMenuItem's parent WMenu is on the Request.
467 	 *
468 	 * @param request the request being responded to.
469 	 * @return true if this WMenuItem's WMenu is on the Request, otherwise return false.
470 	 */
471 	protected boolean isMenuPresent(final Request request) {
472 		WMenu menu = WebUtilities.getAncestorOfClass(WMenu.class, this);
473 
474 		if (menu != null) {
475 			return menu.isPresent(request);
476 		}
477 
478 		return false;
479 	}
