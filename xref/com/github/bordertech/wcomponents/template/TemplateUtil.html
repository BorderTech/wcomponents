7   /**
8    * Utility class for template renderers.
9    *
10   * @author Jonathan Austin
11   * @since 1.0.3
12   */
13  public final class TemplateUtil {
14  
15  	/**
16  	 * Prevent instantiation.
17  	 */
18  	private TemplateUtil() {
19  	}
20  
21  	/**
22  	 * Replace each component tag with the key so it can be used in the replace writer.
23  	 *
24  	 * @param context the context to modify.
25  	 * @param taggedComponents the tagged components
26  	 *
27  	 * @return the keyed components
28  	 */
29  	public static Map<String, WComponent> mapTaggedComponents(final Map<String, Object> context, final Map<String, WComponent> taggedComponents) {
30  
31  		Map<String, WComponent> componentsByKey = new HashMap<>();
32  
33  		// Replace each component tag with the key so it can be used in the replace writer
34  		for (Map.Entry<String, WComponent> tagged : taggedComponents.entrySet()) {
35  			String tag = tagged.getKey();
36  			WComponent comp = tagged.getValue();
37  
38  			// The key needs to be something which would never be output by a Template.
39  			String key = "[WC-TemplateLayout-" + tag + "]";
40  			componentsByKey.put(key, comp);
41  
42  			// Map the tag to the key in the context
43  			context.put(tag, key);
44  		}
45  
46  		return componentsByKey;
47  	}
48  }
