1   package com.github.bordertech.wcomponents.template;

3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.WTemplate;
5   import java.io.Writer;
6   import java.util.Map;
7   
8   /**
9    * Template renderer.
10   * <p>
11   * A template can be passed inline or loaded from a resource file.
12   * </p>
13   * <p>
14   * Variables are passed to the template via the context which is a map of String keys and data objects.
15   * </p>
16   * <p>
17   * The template engines used to render the template can have configuration passed in via the engine options map.
18   * </p>
19   *
20   * @see TemplateRendererFactory
21   * @see WTemplate
22   *
23   * @author Jonathan Austin
24   * @since 1.0.3
25   */
26  public interface TemplateRenderer {
27  
28  	/**
29  	 * Render a template loaded from a resource.
30  	 * <p>
31  	 * Depending on how the Template Engine loads the template resource (ie by ClassLoader or Class) will determine the
32  	 * format of the template resource.
33  	 * </p>
34  	 * <p>
35  	 * ClassLoader.getResource() and Class.getResource() work differently.
36  	 * </p>
37  	 * <p>
38  	 * The methods in ClassLoader use the given String as the name of the resource without applying any
39  	 * absolute/relative transformation. The name should not have a leading “/”.
40  	 * </p>
41  	 * <p>
42  	 * The methods in Class can use a relative (“path/resource.xml”) or absolute path (“/path/resource.xml”). Relative
43  	 * means, relative to the location in the classpath, where the method was called. The path will be appended if
44  	 * needed (ie my/packaage/path/resopurce.xml). Absolute will be used as is, only first / will be removed before the
45  	 * search (ie path/resource.xml).
46  	 * </p>
47  	 *
48  	 * @param templateName the template name and its path
49  	 * @param context the context for the template
50  	 * @param taggedComponents the tagged components
51  	 * @param writer the writer
52  	 * @param options the engine options
53  	 */
54  	void renderTemplate(final String templateName, final Map<String, Object> context, final Map<String, WComponent> taggedComponents,
55  			final Writer writer, final Map<String, Object> options);
56  
57  	/**
58  	 * Render an inline template.
59  	 *
60  	 * @param templateInline the inline template
61  	 * @param context the context for the template
62  	 * @param taggedComponents the tagged components
63  	 * @param writer the writer
64  	 * @param options the engine options
65  	 */
66  	void renderInline(final String templateInline, final Map<String, Object> context, final Map<String, WComponent> taggedComponents,
67  			final Writer writer, final Map<String, Object> options);
68  
69  }
