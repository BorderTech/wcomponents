3   /**
4    * Indicates a menu container can have its menu items selected.
5    *
6    * @author Jonathan Austin
7    * @since 1.0.2
8    */
9   public interface MenuSelectContainer extends MenuContainer {
10  
11  	/**
12  	 * The available types of selection mode for the items in a menu.
13  	 */
14  	enum SelectionMode {
15  		/**
16  		 * No items can be selected.
17  		 */
18  		NONE,
19  		/**
20  		 * A single item can be selected.
21  		 */
22  		SINGLE,
23  		/**
24  		 * Multiple items can be selected.
25  		 */
26  		MULTIPLE
27  	};
28  
29  	/**
30  	 * @return the selection mode of the container
31  	 */
32  	SelectionMode getSelectionMode();
33  
34  	/**
35  	 * @param selectionMode the selection mode for the items in this menu container.
36  	 */
37  	void setSelectionMode(final SelectionMode selectionMode);
38  
39  }
