3   /**
4    * This component enables the creation of links within a page.
5    *
6    * <p>
7    * Current limitations of this component are:</p>
8    * <ul>
9    * <li>It will only link to components that render out their id</li>
10   * <li>It will not link to components nested within WRepeater of WTable components</li>
11   * <li>The text for this link is the same for all users.
12   * </ul>
13   *
14   * @author Martin Shevchenko
15   * @since 1.0.0
16   */
17  public class WInternalLink extends AbstractWComponent implements AjaxTarget {
18  
19  	/**
20  	 * Creates a WInternalLink with no text or component.
21  	 */
22  	public WInternalLink() {
23  	}
24  
25  	/**
26  	 * Creates a WInternalLink with the specified text and component to link to.
27  	 *
28  	 * @param text the text to display on the link.
29  	 * @param reference the component to link to.
30  	 */
31  	public WInternalLink(final String text, final WComponent reference) {
32  		InternalLinkModel model = getComponentModel();
33  		model.text = text;
34  		model.reference = reference;
35  	}
36  
37  	/**
38  	 * @return the component to which this component is linked.
39  	 */
40  	public WComponent getReference() {
41  		return getComponentModel().reference;
42  	}
43  
44  	/**
45  	 * Sets the component to which this component is linked.
46  	 *
47  	 * @param reference the component.
48  	 */
49  	public void setReference(final WComponent reference) {
50  		getOrCreateComponentModel().reference = reference;
51  	}
52  
53  	/**
54  	 * @return the text to be displayed for this link.
55  	 */
56  	public String getText() {
57  		return getComponentModel().text;
58  	}
59  
60  	/**
61  	 * Sets the text to be displayed for this link.
62  	 *
63  	 * @param text the text to be displayed.
64  	 */
65  	public void setText(final String text) {
66  		getOrCreateComponentModel().text = text;
67  	}
68  
69  	/**
70  	 * @return a String representation of this component, for debugging purposes.
71  	 */
72  	@Override
73  	public String toString() {
74  		String text = getText();
75  		text = text == null ? "null" : ('"' + text + '"');
76  		return toString(text);
77  	}
78  
79  	/**
80  	 * Creates a new InternalLinkModel.
81  	 *
82  	 * @return a new InternalLinkModel
83  	 */
84  	@Override
85  	protected InternalLinkModel newComponentModel() {
86  		return new InternalLinkModel();
87  	}
88  
89  	/**
90  	 * {@inheritDoc}
91  	 */
92  	@Override // For type safety only
93  	protected InternalLinkModel getComponentModel() {
94  		return (InternalLinkModel) super.getComponentModel();
95  	}
96  
97  	/**
98  	 * {@inheritDoc}
99  	 */
100 	@Override // For type safety only
101 	protected InternalLinkModel getOrCreateComponentModel() {
102 		return (InternalLinkModel) super.getOrCreateComponentModel();
103 	}
104 
105 	/**
106 	 * InternalLinkModel holds Extrinsic state management of the component.
107 	 */
108 	public static class InternalLinkModel extends ComponentModel {
109 
110 		/**
111 		 * The Wcomponent to which this component is linked.
112 		 */
113 		private WComponent reference;
114 
115 		/**
116 		 * The text displayed for this link.
117 		 */
118 		private String text;
119 	}
120 }
