30  /**
31   * <p>
32   * AbstractWComponent is the parent class of all standard WComponents.
33   * </p>
34   * <p>
35   * WComponent trees (UIs) are intended to be shared between sessions in order to reduce their memory footprint. To
36   * archive this a class called UIContext has been introduced to store WComponent information specific to an individual
37   * session. Each session has its own UIContext instance which is passed to the component tree whenever it needs to
38   * handle events and paint.
39   * </p>
40   * <p>
41   * The attributes of a WComponent have an initial shared value that can be overridden on a per session basis. We call
42   * this a private attribute value. The methods that manipulate a components attributes will normally have two method
43   * signatures. One will manipulate the shared value and does not require a UIContext to be passed. The other will
44   * manipulate the private session based value and will include a UIContext as the first parameter. The methods that
45   * access a components attributes only require one method signature. Accessor methods will include a UIContext as their
46   * first parameter. They will return the private attribute value if one exists else the shared value.
47   * </p>
48   * <p>
49   * The shared/private concept makes WComponents very flexible but has the dangerous ramification that it is easy to
50   * accidentally dynamically share attribute values and even whole chunks of UI with everyone. To reduce this risk, it is
51   * possible to lock a component and all its children. When the lock is set, it becomes impossible to update a shared
52   * value. Trying to update a shared value will result in a runtime exception. The intention is that shared component
53   * trees will be held in a registry. Adding a component tree to a registry would be a sensible time to lock it.
54   * </p>
55   *
56   * @author James Gifford, Martin Shevchenko
57   * @since 1.0.0
58   */
59  public abstract class AbstractWComponent implements WComponent {

88  	public String getInternalId() {
89  		String iid = null;
90  
91  		// As determining the internal id involves a fair bit of tree traversal, it is cached in the scratch map.
92  		// Try to retrieve the cached id first.
93  		Map scratchMap = getScratchMap();
94  
95  		if (scratchMap != null) {
96  			iid = (String) scratchMap.get("iid");
97  		}
98  
99  		if (iid == null) {
100 			// An id is a function of a component's position in the component tree.
101 			// The id will remain constant so long as the component tree does not
102 			// change.
103 			Container parent = getParent();
104 
105 			if (parent == null) {
106 				return DEFAULT_INTERNAL_ID;
107 			}
108 
109 			iid = parent.getInternalId();
110 			final int nameLen = iid.length();
111 			StringBuffer nameBuf = new StringBuffer(nameLen + 3);
112 			nameBuf.append(iid);
113 			nameBuf.append(getIndexOfChild(parent, this));
114 
115 			if (iid.charAt(nameLen - 1) <= '9') {
116 				// last char was a number, change current to letters
117 				for (int i = nameLen; i < nameBuf.length(); i++) {
118 					nameBuf.setCharAt(i, (char) (nameBuf.charAt(i) + ('a' - '0')));
119 				}
120 			}
121 
122 			iid = nameBuf.toString();
123 
124 			if (scratchMap != null) {
125 				scratchMap.put("iid", iid);
126 			}
127 		}
128 
129 		return iid;
130 	}

179 	public String getId() {
180 		// As determining the name involves a fair bit of tree traversal, it is cached in the scratch map.
181 		// Try to retrieve the cached name first.
182 		Map scratchMap = getScratchMap();
183 		if (scratchMap != null) {
184 			String name = (String) scratchMap.get("name");
185 			if (name != null) {
186 				return name;
187 			}
188 		}
189 
190 		// Get ID name
191 		String idName = getIdName();
192 
193 		String name;
194 		// No ID name, so generate an ID
195 		if (idName == null) {
196 			name = generateId();
197 		} else { // Has ID name, so derive the full context name
198 			name = deriveId(idName);
199 		}
200 
201 		if (scratchMap != null) {
202 			scratchMap.put("name", name);
203 		}
204 
205 		// Log warning if an Active Naming Context has no id name
206 		if (this instanceof NamingContextable && ((NamingContextable) this).isNamingContext() && idName == null) {
207 			LOG.warn(
208 					"NamingContext [ID:" + name + "] does not have an id name set. Will be ignored.");
209 		}
210 
211 		return name;
212 	}

214 	/**
215 	 * @return the generated unique id
216 	 */
217 	private String generateId() {
218 		// Direct parent
219 		Container parent = getParent();
220 
221 		// No Parent
222 		if (parent == null) {
223 			return DEFAULT_NO_ID;
224 		}
225 
226 		// Check if parent is an active naming context
227 		boolean parentIsNamingContext = WebUtilities.isActiveNamingContext(parent);
228 
229 		// Get ID prefix
230 		String prefix;
231 		if (parentIsNamingContext) {
232 			prefix = ((NamingContextable) parent).getNamingContextId();
233 		} else {
234 			prefix = parent.getId();
235 		}
236 
237 		// Setup name buffer
238 		int prefixLen = prefix.length();
239 		StringBuffer nameBuf = new StringBuffer(prefixLen + 3);
240 
241 		// Append prefix (if required)
242 		if (prefixLen != 0) {
243 			nameBuf.append(prefix);
244 			// Parent is a NamingContext, so include context separator
245 			if (parentIsNamingContext) {
246 				nameBuf.append(ID_CONTEXT_SEPERATOR);
247 			}
248 		}
249 
250 		// Use the component's position in the tree (ie index) to build a unique ID
251 		int idx = getIndexOfChild(parent, this);
252 
253 		// Parent has a generated ID (ie no id name)
254 		// (NamingContexts must have an id name, so never null)
255 		if (parent.getIdName() == null) {
256 			// Generate ID - Append the index
257 			nameBuf.append(idx);
258 			if (prefix.charAt(prefixLen - 1) <= '9') {
259 				// last char was a number, change current to letters
260 				for (int i = prefixLen; i < nameBuf.length(); i++) {
261 					nameBuf.setCharAt(i, (char) (nameBuf.charAt(i) + ('a' - '0')));
262 				}
263 			}
264 		} else { // Parent has an id name assigned
265 			// Generate ID (with separator)
266 			nameBuf.append(ID_FRAMEWORK_ASSIGNED_SEPERATOR);
267 			nameBuf.append(idx);
268 		}
269 
270 		return nameBuf.toString();
271 	}

274 	 * Derive the full id from its naming context.
275 	 *
276 	 * @param idName the component id name
277 	 * @return the derived id in its context
278 	 */
279 	private String deriveId(final String idName) {
280 		// Find parent naming context
281 		NamingContextable parent = WebUtilities.getParentNamingContext(this);
282 
283 		// No Parent
284 		if (parent == null) {
285 			return idName;
286 		}
287 
288 		// Get ID prefix
289 		String prefix = parent.getNamingContextId();
290 
291 		// No Prefix, just use id name
292 		if (prefix.length() == 0) {
293 			return idName;
294 		}
295 
296 		// Add Prefix
297 		StringBuffer nameBuf = new StringBuffer(prefix.length() + idName.length() + 1);
298 		nameBuf.append(prefix);
299 		nameBuf.append(ID_CONTEXT_SEPERATOR);
300 		nameBuf.append(idName);
301 
302 		return nameBuf.toString();
303 	}

305 	/**
306 	 * Register this component's ID in its naming context.
307 	 */
308 	void registerInContext() {
309 		if (!ConfigurationProperties.getCheckDuplicateIds()) {
310 			return;
311 		}
312 
313 		// Register Component if it has an ID name set
314 		if (getIdName() != null) {
315 			// Find parent context
316 			NamingContextable context = WebUtilities.getParentNamingContext(this);
317 
318 			if (context == null) {
319 				// If this is the top context, then register itself
320 				if (WebUtilities.isActiveNamingContext(this)) {
321 					this.registerId(this);
322 				} else {
323 					LOG.warn("Component with id name [" + getIdName()
324 							+ "] is not in a naming context and cannot be verified for duplicate id.");
325 				}
326 				return;
327 			}
328 			// Assume context is AbstractWComponent
329 			((AbstractWComponent) context).registerId(this);
330 		}
331 	}

334 	 * Helper method to be used by a {@link NamingContextable} to register a component.
335 	 *
336 	 * @param component the component to register
337 	 */
338 	void registerId(final WComponent component) {
339 		if (!WebUtilities.isActiveNamingContext(this)) {
340 			throw new SystemException("Can only register a component on an active NamingContext");
341 		}
342 
343 		// Get registered id names
344 		ComponentModel model = getOrCreateComponentModel();
345 		Map<String, WComponent> ids = model.getContextIds();
346 		if (ids == null) {
347 			ids = new HashMap<>();
348 			model.setContextIds(ids);
349 		}
350 
351 		// Get id name
352 		String idName = component.getIdName();
353 
354 		// Check if already used
355 		WComponent mapped = ids.get(idName);
356 
357 		// Not in map
358 		if (mapped == null) {
359 			// Save ID / Component
360 			ids.put(idName, component);
361 		} else if (mapped != component) { // Check is same component
362 			String contextName = getId();
363 			throw new SystemException("Duplicate ID. ID \"" + idName + "\" for " + component.
364 					getClass().getName()
365 					+ " is already in use by " + mapped.getClass().getName() + " in context \"" + contextName + "\".");
366 		}
367 
368 	}

402 	public final void serviceRequest(final Request request) {
403 		ArrayList<WComponent> visibles = new ArrayList<>();
404 		collateVisible(this, visibles);
405 
406 		if (LOG.isDebugEnabled()) {
407 			LOG.debug("Calling handleRequest for " + visibles.size() + " components");
408 		}
409 
410 		for (int i = 0; i < visibles.size(); i++) {
411 			WComponent next = visibles.get(i);
412 
413 			if (LOG.isDebugEnabled()) {
414 				LOG.debug("Calling handleRequest on " + next.getId() + " " + next.getClass());
415 			}
416 
417 			next.handleRequest(request);
418 		}
419 
420 		invokeLaters();
421 	}

424 	 * Collates all the visible components in this branch of the WComponent tree. WComponents are added to the
425 	 * <code>list</code> in depth-first order, as this list is traversed in order during the request handling phase.
426 	 *
427 	 * @param component the current branch to collate visible items in.
428 	 * @param list the list to add the visible components to.
429 	 */
430 	private static void collateVisible(final WComponent component, final List<WComponent> list) {
431 		if (component.isVisible()) {
432 
433 			if (component instanceof Container) {
434 
435 				final int size = ((Container) component).getChildCount();
436 
437 				for (int i = 0; i < size; i++) {
438 					collateVisible(((Container) component).getChildAt(i), list);
439 				}
440 			}
441 
442 			list.add(component);
443 		}
444 	}

558 	 * Prepares this component and all child componenents for painting (e.g. rendering to XML). This implementation
559 	 * calls {@link #preparePaintComponent(Request)}, then calls {@link #preparePaint(Request)} on all its children.
560 	 * Note that the this component's {@link #preparePaintComponent(Request)} is called before the childrens'
561 	 * {@link #preparePaintComponent(Request)} is called.
562 	 *
563 	 * @param request the request being responded to.
564 	 */
565 	@Override
566 	public final void preparePaint(final Request request) {
567 		// Don't prepare if it's invisible.
568 		if (!isVisible()) {
569 			return;
570 		}
571 
572 		// Prepare this component.
573 		preparePaintComponent(request);
574 
575 		// Prepare its children.
576 		final ArrayList<WComponent> children = (ArrayList<WComponent>) getComponentModel().
577 				getChildren();
578 
579 		if (children != null) {
580 			final int size = children.size();
581 
582 			for (int i = 0; i < size; i++) {
583 				children.get(i).preparePaint(request);
584 			}
585 
586 			// Chances are that the WComponent tree is fairly stable now, so take
587 			// the opportunity to trim the child list to size if we have one.
588 			// This is just a memory optimization - it won't prevent adding more
589 			// children later, of course.
590 			children.trimToSize();
591 		}
592 	}

662 	 * This is where most of the painting work is normally done. If a layout has been supplied either directly or by
663 	 * supplying a velocity template, then painting is delegated to the layout manager. If there is no layout, the
664 	 * default behaviour is to paint the child components in sequence.
665 	 *
666 	 * @param renderContext the context to render to.
667 	 */
668 	protected void paintComponent(final RenderContext renderContext) {
669 		Renderer renderer = UIManager.getRenderer(this, renderContext);
670 
671 		if (getTemplate() != null || getTemplateMarkUp() != null) {
672 			Renderer templateRenderer = UIManager.getTemplateRenderer(renderContext);
673 			templateRenderer.render(this, renderContext);
674 		} else if (renderer == null) {
675 			// Default is juxtaposition
676 			List<WComponent> children = getComponentModel().getChildren();
677 
678 			if (children != null) {
679 				final int size = children.size();
680 
681 				for (int i = 0; i < size; i++) {
682 					children.get(i).paint(renderContext);
683 				}
684 			}
685 		} else {
686 			renderer.render(this, renderContext);
687 		}
688 	}

693 	 * {@inheritDoc}
694 	 */
695 	@Override
696 	public void validate(final List<Diagnostic> diags) {
697 		// Don't validate if it's invisible.
698 		if (!isVisible()) {
699 			return;
700 		}
701 
702 		// Also check the validation override flag.
703 		if (!isValidate()) {
704 			return;
705 		}
706

1160 	/**
1161 	 * <p>
1162 	 * Resets this component and its children to their initial state for the given user context / session.
1163 	 * </p>
1164 	 * <p>
1165 	 * <b>NOTE:</b> The exception to this rule is if this component has been dynamically added to the UI, then the
1166 	 * parent component will differ from the initial state.
1167 	 * </p>
1168 	 */
1169 	@Override
1170 	public void reset() {
1171 		UIContext uic = UIContextHolder.getCurrent();
1172 
1173 		if (uic != null) {
1174 			// Note: The parent component is stored as a field within AbstractWComponent's
1175 			// component model, which previously caused problems with code like the following:
1176 			//
1177 			// ...
1178 			// WContainer parent = new WContainer();
1179 			// WComponent child = new WTextField();
1180 			// ...
1181 			// parent.add(uic, child);
1182 			// ...
1183 			// child.reset(uic);
1184 			// ...
1185 			//
1186 			// In this scenario, the parent still kept a reference to the child component,
1187 			// but the child component has lost its reference to the parent, which caused
1188 			// ancestor lookups to fail, and duplicate IDs to be generated.
1189 			//
1190 			// This implementation therefore does a top-down reset, clearing this
1191 			// component's model before any child models. This allows child components
1192 			// to correctly check for dynamic parenting in their reset methods and
1193 			// restore the reference to the parent component if necessary.
1194 
1195 			final int childCount = getChildCount();
1196 			List<WComponent> children = null;
1197 
1198 			if (childCount > 0) {
1199 				children = new ArrayList<>(getComponentModel().getChildren());
1200 			}
1201 
1202 			// Keep a reference to the current (possibly dynamic) parent before it is potentially lost
1203 			Container dynamicParent = getParent();
1204 
1205 			// Reset this component's data first.
1206 			this.removeComponentModel();
1207 			uic.clearScratchMap(this);
1208 			uic.clearRequestScratchMap(this);
1209 
1210 			// Now reset all descendant components
1211 			if (children != null) {
1212 				for (WComponent child : children) {
1213 					child.reset();
1214 				}
1215 			}
1216 
1217 			// At this point, everything in the sub-tree where the reset
1218 			// method was initially called has now been reset.
1219 			// If the component was dynamically added
1220 			// and the dynamic parent still has this component as a child
1221 			if (dynamicParent != null && getParent() != dynamicParent && getIndexOfChild(
1222 					dynamicParent, this) != -1) {
1223 				// then re-instate the reference to the dynamic parent.
1224 				getOrCreateComponentModel().setParent(dynamicParent);
1225 			}
1226 		}
1227 	}

1415 			model.setChildren(copyChildren(getComponentModel().getChildren()));
1416 		}
1417 
1418 		if (model.getChildren().remove(aChild)) {
1419 			// Deallocate children list if possible, to reduce session size.
1420 			if (model.getChildren().isEmpty()) {

1747 	 * @param childEndIndex the end index of children to include in the output
1748 	 * @return a String representation of this component
1749 	 */
1750 	final String toString(final String details, final int childStartIndex, final int childEndIndex) {
1751 		// The simple class name will be empty for anonymous subclasses,
1752 		// which is not particularly useful, so we recurse until we find a suitable one.
1753 		String className = null;
1754 
1755 		for (Class<?> clazz = getClass(); Util.empty(className) && clazz != null; clazz = clazz.
1756 				getSuperclass()) {
1757 			className = clazz.getSimpleName();
1758 		}
1759 
1760 		StringBuffer buf = new StringBuffer(className == null ? "?" : className);
1761 
1762 		if (!isDefaultState()) {
1763 			buf.append("<user model>");
1764 		}

1767 		buf.append(!isHidden() ? "" : "<hidden>");
1768 
1769 		if (this instanceof Disableable) {
1770 			buf.append(!((Disableable) this).isDisabled() ? "" : "<disabled>");
1771 		}
1772 
1773 		if (details != null) {
1774 			buf.append('(').append(details).append(')');
1775 		}
1776 
1777 		if (this instanceof Container && childStartIndex >= 0 && childEndIndex < getChildCount()
1778 				&& childStartIndex <= childEndIndex) {
1779 			WComponent[] children = new WComponent[childEndIndex - childStartIndex + 1];
1780 
1781 			for (int i = childStartIndex; i <= childEndIndex; i++) {
1782 				children[i - childStartIndex] = getChildAt(i);

1819 	/**
1820 	 * <p>
1821 	 * This class is used to hold a reference to the shared singleton instance of a wcomponent for the purpose of
1822 	 * serialisation. Serialization of WComponent session information is tricky because of the separation of the session
1823 	 * state data (the UIContext) and the shared application definition (the wcompontent stored in the UIRegistry). When
1824 	 * serializing a UIContext, we don't want to serialize the entire shared application definition in each users
1825 	 * session. The problem is that the data stored in the UIContext is keyed using the shared WComponent instances, so
1826 	 * we must convert them all to references in order to prevent the entire application from being serialized.
1827 	 * </p>
1828 	 * <p>
1829 	 * This extends WComponent in order to fulfil the readResolve contract, however it overrides writeObject/readObject
1830 	 * to only write out the data needed to find the shared instance.
1831 	 * </p>
1832 	 *
1833 	 * @author Yiannis Paschalidis
1834 	 */
1835 	public static final class WComponentRef extends AbstractWComponent {
1836 
1837 		/**
1838 		 * The UIRegistry key under which the UI root component is registered. This is assumed to be the fully qualified
1839 		 * class name of the root component.
1840 		 */
1841 		private String repositoryKey;
1842 
1843 		/**
1844 		 * The location path of the component in the UI tree, specified as child indices.
1845 		 */
1846 		private int[] nodeLocation;
1847 
1848 		/**
1849 		 * Creates a WComponentRef.
1850 		 *
1851 		 * @param repositoryKey the UIRegistry key under which the UI root component is registered.
1852 		 * @param nodeLocation the location of the component in the UI tree.
1853 		 */
1854 		public WComponentRef(final String repositoryKey, final int[] nodeLocation) {
1855 			this.repositoryKey = repositoryKey;
1856 			this.nodeLocation = nodeLocation;
1857 		}
1858 
1859 		/**
1860 		 * Implement writeObject to only write out the repositoryKey and nodeLocation fields.
1861 		 *
1862 		 * @param out the ObjectOutputStream to write to.
1863 		 * @throws IOException if there is an error writing to the stream.
1864 		 * @see java.io.Serializable
1865 		 */
1866 		private void writeObject(final ObjectOutputStream out) throws IOException {
1867 			out.writeObject(repositoryKey);
1868 			out.writeInt(nodeLocation.length);
1869 
1870 			for (int i = 0; i < nodeLocation.length; i++) {
1871 				out.writeInt(nodeLocation[i]);
1872 			}
1873 		}
1874 
1875 		/**
1876 		 * Implement readObject to only read in the repositoryKey and nodeLocation fields.
1877 		 *
1878 		 * @param in the ObjectInputStream to read from.
1879 		 * @throws IOException if there is an error reading from the stream.
1880 		 * @throws ClassNotFoundException if the class can't be found.
1881 		 * @see java.io.Serializable
1882 		 */
1883 		private void readObject(final ObjectInputStream in) throws IOException,
1884 				ClassNotFoundException {
1885 			repositoryKey = (String) in.readObject();
1886 			int len = in.readInt();
1887 			nodeLocation = new int[len];
1888 
1889 			for (int i = 0; i < nodeLocation.length; i++) {
1890 				nodeLocation[i] = in.readInt();
1891 			}
1892 		}
1893 
1894 		/**
1895 		 * Implement readResolve so that on deserialization, the WComponent that is referred to by this WComponentRef is
1896 		 * returned. The WComponent
1897 		 *
1898 		 * @return the WComponent instance that is registered with the registry.
1899 		 * @throws ObjectStreamException never, but Serializable requires this method signature to declare it.
1900 		 * @see java.io.Serializable
1901 		 */
1902 		private Object readResolve() throws ObjectStreamException {
1903 			if (repositoryKey == null || nodeLocation == null) {
1904 				// Should not happen
1905 				throw new IllegalStateException(
1906 						"Unable to resolve component in repository '" + repositoryKey + '\'');
1907 			}
1908 
1909 			WComponent comp = UIRegistry.getInstance().getUI(repositoryKey);
1910 
1911 			for (int i = 0; comp != null && i < nodeLocation.length; i++) {
1912 				comp = ((Container) comp).getChildAt(nodeLocation[i]);
1913 			}
1914 
1915 			// Component not found - BAD!
1916 			if (comp == null) {
1917 				throw new IllegalStateException("Unable to resolve component: " + toString());
1918 			}
1919 
1920 			return comp;
1921 		}

1925 		 */

1928 			StringBuffer buf = new StringBuffer();
1929 			buf.append(repositoryKey);
1930 
1931 			for (int i = 0; i < nodeLocation.length; i++) {
1932 				buf.append(',').append(nodeLocation[i]);

1939 	/**
1940 	 * Implement writeReplace so that on serialization, WComponents that are registered in the UIRegistry write a
1941 	 * reference to the registered component rather than the component itself. This ensures that, on deserialization,
1942 	 * only one copy of the registered component will be present in the VM.
1943 	 *
1944 	 * @return the WComponent instance that is registered with the registry.
1945 	 * @throws ObjectStreamException never, but Serializable requires this method signature to declare it.
1946 	 * @see java.io.Serializable
1947 	 */
1948 	protected Object writeReplace() throws ObjectStreamException {
1949 		WComponent top = WebUtilities.getTop(this);
1950 		String repositoryKey;
1951 		if (top instanceof WApplication) {
1952 			repositoryKey = ((WApplication) top).getUiVersionKey();
1953 		} else {
1954 			repositoryKey = top.getClass().getName();
1955 		}
1956 
1957 		if (UIRegistry.getInstance().isRegistered(repositoryKey)
1958 				&& top == UIRegistry.getInstance().getUI(repositoryKey)) {
1959 			// Calculate the node location.
1960 			// The node location is a list of "shared" child indexes of each
1961 			// ancestor going right back to the top node.
1962 			ArrayList<Integer> reversedIndexList = new ArrayList<>();
1963 			WComponent node = this;
1964 			Container parent = node.getParent();
1965 
1966 			try {
1967 				while (parent != null) {
1968 					int index = getIndexOfChild(parent, node);
1969 					reversedIndexList.add(index);
1970 					node = parent;
1971 					parent = node.getParent();
1972 				}
1973 			} catch (Exception ex) {
1974 				LOG.error("Unable to determine component index relative to top.", ex);
1975 			}
1976 
1977 			final int depth = reversedIndexList.size();
1978 			int[] nodeLocation = new int[depth];
1979 
1980 			for (int i = 0; i < depth; i++) {
1981 				Integer index = reversedIndexList.get(depth - i - 1);
1982 				nodeLocation[i] = index.intValue();
1983 			}
1984 
1985 			WComponentRef ref = new WComponentRef(repositoryKey, nodeLocation);
1986 
1987 			if (LOG.isDebugEnabled()) {
1988 				LOG.debug(
1989 						"WComponent converted to reference. Ref = " + ref + ". Component = " + getClass().
1990 						getName());
1991 			}
1992 
1993 			return ref;
1994 		} else {
1995 			LOG.debug(
1996 					"WComponent not accessible via the repository, so it will be serialised. Component = "
1997 					+ getClass().getName());
1998 
1999 			return this;
2000 		}
2001 	}
