20  	/**
21  	 * The available types of client-side menus.
22  	 *
23  	 * @author Yiannis Paschalidis
24  	 */
25  	public enum MenuType {
26  		/**
27  		 * The menu is displayed as a horizontal bar, similar to menu bars in desktop applications.
28  		 */
29  		BAR,
30  		/**
31  		 * The menu is displayed as a series of buttons. This is most commonly used to display a single submenu as a
32  		 * button.
33  		 */
34  		FLYOUT,
35  		/**
36  		 * The menu is displayed as a tree structure.
37  		 */
38  		TREE,
39  		/**
40  		 * The menu is displayed columns from left to right. This menu type supports multiple selections of menu items
41  		 * and sub-menus.
42  		 */
43  		COLUMN
44  	};

431 	/**
432 	 * Override handleRequest in order to perform processing specific to WMenu.
433 	 *
434 	 * @param request the request being handled.
435 	 */
436 	@Override
437 	public void handleRequest(final Request request) {
438 		if (isDisabled()) {
439 			// Protect against client-side tampering of disabled/read-only fields.
440 			return;
441 		}
442 
443 		if (isPresent(request)) {
444 			List<MenuItemSelectable> selectedItems = new ArrayList<>();
445 			// Unfortunately, we need to recurse through all the menu/sub-menus
446 			findSelections(request, this, selectedItems);
447 			setSelectedMenuItems(selectedItems);
448 		}
449 	}

461 	/**
462 	 * Finds the selected items in a menu for a request.
463 	 *
464 	 * @param request the request being handled
465 	 * @param selectContainer the menu or sub-menu
466 	 * @param selections the current set of selections
467 	 */
468 	private void findSelections(final Request request, final MenuSelectContainer selectContainer,
469 			final List<MenuItemSelectable> selections) {
470 
471 		// Don't bother checking disabled or invisible containers
472 		if (!selectContainer.isVisible()
473 				|| (selectContainer instanceof Disableable && ((Disableable) selectContainer).isDisabled())) {
474 			return;
475 		}
476 
477 		// Get any selectable children of this container
478 		List<MenuItemSelectable> selectableItems = getSelectableItems(selectContainer);
479 
480 		// Now add the selections (if in the request)
481 		for (MenuItemSelectable selectableItem : selectableItems) {
482 			// Check if the item is on the request
483 			if (request.getParameter(selectableItem.getId() + ".selected") != null) {
484 				selections.add(selectableItem);
485 				if (SelectionMode.SINGLE.equals(selectContainer.getSelectionMode())) {
486 					// Only select the first item at this level.
487 					// We still need to check other levels of the menu for selection.
488 					break;
489 				}
490 			}
491 		}
492 
493 		// We need to recurse through and check for other selectable containers
494 		for (MenuItem item : selectContainer.getMenuItems()) {
495 			if (item instanceof MenuItemGroup) {
496 				for (MenuItem groupItem : ((MenuItemGroup) item).getMenuItems()) {
497 					if (groupItem instanceof MenuSelectContainer) {
498 						findSelections(request, (MenuSelectContainer) groupItem, selections);
499 					}
500 				}
501 			} else if (item instanceof MenuSelectContainer) {
502 				findSelections(request, (MenuSelectContainer) item, selections);
503 			}
504 		}
505 	}

507 	/**
508 	 * Retrieves the selectable items for the given container.
509 	 *
510 	 * @param selectContainer the component to search within.
511 	 * @return the list of selectable items for the given component. May be empty.
512 	 */
513 	private List<MenuItemSelectable> getSelectableItems(final MenuSelectContainer selectContainer) {
514 		List<MenuItemSelectable> result = new ArrayList<>(selectContainer.getMenuItems().size());
515 
516 		SelectionMode selectionMode = selectContainer.getSelectionMode();
517 
518 		for (MenuItem item : selectContainer.getMenuItems()) {
519 			if (item instanceof MenuItemGroup) {
520 				for (MenuItem groupItem : ((MenuItemGroup) item).getMenuItems()) {
521 					if (isSelectable(groupItem, selectionMode)) {
522 						result.add((MenuItemSelectable) groupItem);
523 					}
524 				}
525 			} else if (isSelectable(item, selectionMode)) {
526 				result.add((MenuItemSelectable) item);
527 			}
528 		}
529 
530 		return result;
531 	}

533 	/**
534 	 * Indicates whether the given menu item is selectable.
535 	 *
536 	 * @param item the menu item to check.
537 	 * @param selectionMode the select mode of the current menu/sub-menu
538 	 * @return true if the meu item is selectable, false otherwise.
539 	 */
540 	private boolean isSelectable(final MenuItem item, final SelectionMode selectionMode) {
541 
542 		if (!(item instanceof MenuItemSelectable) || !item.isVisible()
543 				|| (item instanceof Disableable && ((Disableable) item).isDisabled())) {
544 			return false;
545 		}
546 
547 		// SubMenus are only selectable in a column menu type
548 		if (item instanceof WSubMenu && !MenuType.COLUMN.equals(getType())) {
549 			return false;
550 		}
551 
552 		// Item is specificially set to selectable/unselectable
553 		Boolean itemSelectable = ((MenuItemSelectable) item).getSelectability();
554 		if (itemSelectable != null) {
555 			return itemSelectable;
556 		}
557 
558 		// Container has selection turned on
559 		return SelectionMode.SINGLE.equals(selectionMode) || SelectionMode.MULTIPLE.equals(selectionMode);
560 
561 	}

563 	/**
564 	 * @param items the list of menu items
565 	 * @param recurse true if recurse into child items that are menu containers
566 	 * @param container the current container
567 	 */
568 	private void getChildMenuItems(final List<MenuItem> items, final boolean recurse, final MenuContainer container) {
569 
570 		for (MenuItem item : container.getMenuItems()) {
571 			items.add(item);
572 			if (recurse && item instanceof MenuContainer) {
573 				getChildMenuItems(items, recurse, (MenuContainer) item);
574 			}
575 		}
576 	}
