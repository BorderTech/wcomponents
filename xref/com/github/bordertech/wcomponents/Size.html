3   /**
4    * Used to ensure consistent spacing of intra-component gaps and Margins.
5    * @author Mark Reeves
6    * @author John McGuiness
7    * @since 1.4.0
8    */
9   public enum Size {
10  	/**
11  	 * Explicit zero space. Used to remove innate margins (for example).
12  	 */
13  	ZERO("z"),
14  	/**
15  	 * Small space.
16  	 */
17  	SMALL("sm"),
18  	/**
19  	 * Medium space.
20  	 */
21  	MEDIUM("med"),
22  	/**
23  	 * Large space.
24  	 */
25  	LARGE("lg"),
26  	/**
27  	 * Very large space.
28  	 */
29  	XL("xl");
30  
31  	/**
32  	 * A usable representation of the Size.
33  	 */
34  	private final String size;
35  
36  	/**
37  	 *
38  	 * @param sz the size representation passed to the UI
39  	 */
40  	Size(final String sz) {
41  		size = sz;
42  	}
43  
44  	/**
45  	 *
46  	 * @return a string representation of the size
47  	 */
48  	@Override
49  	public String toString() {
50  		return size;
51  	}
52  }
