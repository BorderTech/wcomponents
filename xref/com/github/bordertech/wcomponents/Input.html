1   package com.github.bordertech.wcomponents;
2   
3   import com.github.bordertech.wcomponents.validation.ValidatingAction;
4   import com.github.bordertech.wcomponents.validator.FieldValidator;
5   import java.util.Iterator;
6   
7   /**
8    * Interface for components that correspond to an input field.
9    *
10   * @author Jonathan Austin
11   * @since 1.0.0
12   */
13  public interface Input extends WComponent, Disableable, Mandatable, DataBound, BeanBound, BeanProviderBound, Labelable {
14  
15  	// ================================
16  	// Validation
17  	/**
18  	 * Adds a validator to the input field. Validators are not automatically called on form submission. Using a
19  	 * {@link ValidatingAction} or an explicit call to {@link WComponent#validate(List)} will trigger validation on a
20  	 * WComponent tree.
21  	 *
22  	 * @param validator the validator to add.
23  	 */
24  	void addValidator(final FieldValidator validator);
25  
26  	/**
27  	 * @return the list of validators for this input field.
28  	 */
29  	Iterator<FieldValidator> getValidators();
30  
31  	// ================================
32  	// Action on change
33  	/**
34  	 * Sets the action that you want run if the input is changed by the user. The action command will be set to the new
35  	 * value of the input.
36  	 *
37  	 * @param actionOnChange the action to execute when the input is changed by the user.
38  	 */
39  	void setActionOnChange(final Action actionOnChange);
40  
41  	/**
42  	 * @return the action to execute when the input is changed by the user.
43  	 */
44  	Action getActionOnChange();
45  
46  	/**
47  	 * Returns the data object that has been associated with this input component, else null. For convenience, this data
48  	 * object is passed to the execute() method of the button's associated Action, in the ActionEvent parameter.
49  	 *
50  	 * @return the data object that has been associated with this input component.
51  	 */
52  	Object getActionObject();
53  
54  	/**
55  	 * Associates this input with a data object that can be easily accessed in the execute() method of the associated
56  	 * Action.
57  	 *
58  	 * @param data the data object.
59  	 */
60  	void setActionObject(final Object data);
61  
62  	/**
63  	 * Same as {#getValueAsString()}. This method exists simply to clarify the relationship between the input component,
64  	 * its Action, and the ActionEvent sent to the execute() method of the Action.
65  	 *
66  	 * @return the current selection, as a string.
67  	 */
68  	String getActionCommand();
69  
70  	// ================================
71  	// Default submit button
72  	/**
73  	 * Sets the button that should be submitted when the user hits enter key and cursor is inside this input field.
74  	 *
75  	 * @param defaultSubmitButton the default submit button for this field.
76  	 */
77  	void setDefaultSubmitButton(final WButton defaultSubmitButton);
78  
79  	/**
80  	 * @return the button that will be submitted if the user hits the enter key when the cursor is in this field.
81  	 */
82  	WButton getDefaultSubmitButton();
83  
84  	// ================================
85  	// Mandatory flag
86  	/**
87  	 * Set whether or not this input is mandatory, and customise the error message that will be displayed.
88  	 *
89  	 * @param mandatory true for mandatory, false for optional.
90  	 * @param message the message to display to the user on mandatory validation failure.
91  	 */
92  	void setMandatory(final boolean mandatory, final String message);
93  
94  	// ================================
95  	// ReadOnly flag
96  	/**
97  	 * Indicates whether the input is read only in the given context.
98  	 *
99  	 * @return true if the input is read only, otherwise false.
100 	 */
101 	boolean isReadOnly();
102 
103 	/**
104 	 * Sets whether the input is read only.
105 	 *
106 	 * @param readOnly if true, the input is read only. If false, it is editable.
107 	 */
108 	void setReadOnly(final boolean readOnly);
109 
110 	// ================================
111 	// Input Value
112 	/**
113 	 * Provide the value of the component on the Request.
114 	 * <p>
115 	 * If the component is not on the request, the components current value will be provided.
116 	 * </p>
117 	 *
118 	 * @param request the request being responded to.
119 	 * @return the value of this component on the Request, or its current state if it is not on the request.
120 	 */
121 	Object getRequestValue(final Request request);
122 
123 	/**
124 	 * Provide the value of the component returned by {@link #getData()} in the correct format.
125 	 * <p>
126 	 * If required, this method can convert the data into the correct type and also do any validation before the value
127 	 * is used.
128 	 * </p>
129 	 *
130 	 * @return the value of the component returned by {@link #getData()} in the correct format.
131 	 */
132 	Object getValue();
133 
134 	/**
135 	 * Retrieves a String representation of the input field's value.
136 	 *
137 	 * @return the String representation of the input field's value
138 	 */
139 	String getValueAsString();
140 
141 	/**
142 	 * Indicates whether this input field is empty. An input is considered empty if the value is null or an empty
143 	 * String.
144 	 *
145 	 * @return true if this input field is empty, false otherwise.
146 	 */
147 	boolean isEmpty();
148 
149 	/**
150 	 * Indicates if the input component was changed in the last request.
151 	 *
152 	 * @return true if the input was changed in the last request.
153 	 */
154 	boolean isChangedInLastRequest();
155 
156 }
