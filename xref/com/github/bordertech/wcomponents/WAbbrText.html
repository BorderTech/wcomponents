1   package com.github.bordertech.wcomponents;
2   
3   import com.github.bordertech.wcomponents.util.Factory;
4   import com.github.bordertech.wcomponents.util.LookupTable;
5   
6   /**
7    * The WAbbrText component represents an abbreviation or acronym and the full textual description for its (abbreviated)
8    * text content.
9    *
10   * @author Kishan Bisht, Mark Reeves
11   * @since 1.0.0
12   */
13  public class WAbbrText extends WText {
14  	/**
15  	 * The lookup table instance for this application.
16  	 */
17  	private static final LookupTable TABLE = Factory.newInstance(LookupTable.class);
18  
19  	/**
20  	 * <p>
21  	 * Creates an empty WAbbrText.
22  	 * </p>
23  	 * <p>
24  	 * An instance of WAbbrText created in this manner must have abbreviated display text and a toolTip (the full text
25  	 * represented by the abbreviation) set to be useful.
26  	 * </p>
27  	 */
28  	public WAbbrText() {
29  	}
30  
31  	/**
32  	 * <p>
33  	 * Creates a WAbbrText with the specified abbreviated display text.
34  	 * </p>
35  	 * <p>
36  	 * An instance of WAbbrText created in this manner must have a toolTip (the full text represented by the
37  	 * abbreviation) set to be useful.
38  	 * </p>
39  	 *
40  	 * @param text The abbreviated text to display
41  	 */
42  	public WAbbrText(final String text) {
43  		super(text);
44  	}
45  
46  	/**
47  	 * Creates a WAbbrText with the specified abbreviated display text and toolTip full expansion text.
48  	 *
49  	 * @param text the abbreviated text to display
50  	 * @param description the full text represented by the abbreviation
51  	 */
52  	public WAbbrText(final String text, final String description) {
53  		this(text);
54  		setToolTip(description);
55  	}
56  
57  	//================================
58  	// Attributes
59  	/**
60  	 * @return the expanded text represented by the abbreviation
61  	 * @deprecated as of WComponents 1.0.0, use {@link #getToolTip()} instead
62  	 */
63  	public String getAbbrText() {
64  		return getToolTip();
65  	}
66  
67  	/**
68  	 * Loads the abbreviated text component from the given code reference table entry using the entry's code as the
69  	 * visible (abbreviated) text and the entry's description as the abbreviation's expanded text toolTip.
70  	 *
71  	 * @param entry the CRT entry to use.
72  	 */
73  	public void setTextWithCode(final Object entry) {
74  		setText(TABLE.getCode(null, entry));
75  		setToolTip(TABLE.getDescription(null, entry));
76  	}
77  
78  	/**
79  	 * Sets the full text represented by the abbreviation.
80  	 *
81  	 * @param abbrText the full text (expansion) represented by the abbreviation
82  	 * @deprecated As of WComponents 1.0.0, use {@link #setToolTip(String, Serializable...)} instead.
83  	 */
84  	public void setAbbrText(final String abbrText) {
85  		setToolTip(abbrText);
86  	}
87  
88  	/**
89  	 * Loads the abbreviated text component from the given code reference table entry using the entry's description as
90  	 * the visible (abbreviated) text and the entry's code as the toolTip (expansion). You probably do not want this:
91  	 * you probably want {@link #setTextWithCode(java.lang.Object)}.
92  	 *
93  	 * @param entry the CRT entry to use
94  	 */
95  	public void setTextWithDesc(final Object entry) {
96  		setText(TABLE.getDescription(null, entry));
97  		setToolTip(TABLE.getCode(null, entry));
98  	}
99  
100 	/**
101 	 * Get a String representation of the component. This would normally only be required for debugging purposes.
102 	 *
103 	 * @return a String representation of this component
104 	 */
105 	@Override
106 	public String toString() {
107 		String text = getText();
108 		text = text == null ? "null" : '"' + text + '"';
109 
110 		String expandedText = getToolTip();
111 		expandedText = expandedText == null ? "null" : '"' + expandedText + '"';
112 
113 		return toString("text=" + text + ", toolTip=" + expandedText);
114 	}
115 }
