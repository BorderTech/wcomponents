1   package com.github.bordertech.wcomponents;
2   
3   import com.github.bordertech.wcomponents.util.Factory;
4   import com.github.bordertech.wcomponents.util.LookupTable;
5   
6   /**
7    * The WAbbrText component represents an abbreviation or acronym and the full textual description for its (abbreviated)
8    * text content.
9    *
10   * @author Kishan Bisht
11   * @author Mark Reeves
12   * @since 1.0.0
13   */
14  public class WAbbrText extends WText {
15  
16  	/**
17  	 * The lookup table instance for this application.
18  	 */
19  	private static final LookupTable TABLE = Factory.newInstance(LookupTable.class);
20  
21  	/**
22  	 * Creates an empty WAbbrText.
23  	 *
24  	 * An instance of WAbbrText created in this manner must have abbreviated display text and a toolTip (the full text
25  	 * represented by the abbreviation) set to be useful.
26  	 */
27  	public WAbbrText() {
28  	}
29  
30  	/**
31  	 * Creates a WAbbrText with the specified abbreviated display text.
32  	 *
33  	 * An instance of WAbbrText created in this manner must have a toolTip (the full text represented by the
34  	 * abbreviation) set to be useful.
35  	 *
36  	 * @param text The abbreviated text to display.
37  	 */
38  	public WAbbrText(final String text) {
39  		super(text);
40  	}
41  
42  	/**
43  	 * Creates a WAbbrText with the specified abbreviated display text and toolTip full expansion text.
44  	 *
45  	 * @param text The abbreviated text to display.
46  	 * @param description The full text represented by the abbreviation.
47  	 */
48  	public WAbbrText(final String text, final String description) {
49  		this(text);
50  		setToolTip(description);
51  	}
52  
53  	//================================
54  	// Attributes
55  	/**
56  	 * @return the expanded text represented by the abbreviation.
57  	 * @deprecated use getToolTip instead.
58  	 */
59  	public String getAbbrText() {
60  		return getToolTip();
61  	}
62  
63  	/**
64  	 * Sets the full text represented by the abbreviation.
65  	 *
66  	 * @param abbrText The full text represented by the abbreviation.
67  	 * @deprecated use setToolTip instead.
68  	 */
69  	public void setAbbrText(final String abbrText) {
70  		setToolTip(abbrText);
71  	}
72  
73  	/**
74  	 * Loads the abbreviated text component from the given code reference table entry.
75  	 *
76  	 * The display (abbreviated) text is set to the table entry's description. The toolTip is set to the table entry's
77  	 * code.
78  	 *
79  	 * @param entry the CRT entry to use.
80  	 */
81  	public void setTextWithDesc(final Object entry) {
82  		setText(TABLE.getDescription(null, entry));
83  		setToolTip(TABLE.getCode(null, entry));
84  	}
85  
86  	/**
87  	 * Loads the abbreviated text component from the given code reference table entry.
88  	 *
89  	 * The display (abbreviated) text is set to the table entry's code. The toolTip is set to the table entry's
90  	 * description.
91  	 *
92  	 * @param entry the CRT entry to use.
93  	 */
94  	public void setTextWithCode(final Object entry) {
95  		setText(TABLE.getCode(null, entry));
96  		setToolTip(TABLE.getDescription(null, entry));
97  	}
98  
99  	/**
100 	 * @return a String representation of this component, for debugging purposes.
101 	 */
102 	@Override
103 	public String toString() {
104 		String text = getText();
105 		text = text == null ? "null" : '"' + text + '"';
106 
107 		String expandedText = getToolTip();
108 		expandedText = expandedText == null ? "null" : '"' + expandedText + '"';
109 
110 		return toString("text=" + text + ", toolTip=" + expandedText);
111 	}
112 }
