58  	/**
59  	 * Creates a new WMessages.
60  	 *
61  	 * @param persistent if true, all messages are persisted between requests. If false, only validation messages are
62  	 * persisted between requests.
63  	 */
64  	public WMessages(final boolean persistent) {
65  		add(validationErrors);
66  
67  		if (!persistent) {
68  			// We don't want the message display components to retain any state information
69  			MutableContainer container = new DefaultTransientDataContainer();
70  			add(container);
71  
72  			container.add(errorMessages);
73  			container.add(warningMessages);
74  			container.add(infoMessages);
75  			container.add(successMessages);
76  		} else {
77  			add(errorMessages);
78  			add(warningMessages);
79  			add(infoMessages);
80  			add(successMessages);
81  		}
82  	}

84  	/**
85  	 * Retrieves the WMessages instance for the given component. The component tree is searched for an ancestor that
86  	 * implements the MessageContainer interface. If not found, a proxy is returned that searches for the
87  	 * MessageContainer each time a WMessages method is called. This allows developers to obtain a valid "instance"
88  	 * during e.g. constructors, where the component will not have been added to the tree yet.
89  	 *
90  	 * @param component the component to retrieve the WMessages instance for
91  	 * @return the WMessages instance for the given component.
92  	 */
93  	public static WMessages getInstance(final WComponent component) {
94  		MessageContainer container = getMessageContainer(component);
95  		WMessages messages = container != null ? container.getMessages() : null;
96  
97  		if (messages == null) {
98  			messages = new WMessagesProxy(component);
99  		}
100 
101 		return messages;
102 	}

104 	/**
105 	 * @return true if there are messages to display
106 	 */
107 	public boolean hasMessages() {
108 		return successMessages.hasMessages() || infoMessages.hasMessages() || warningMessages.
109 				hasMessages()
110 				|| errorMessages.hasMessages() || validationErrors.hasErrors();
111 	}

122 	/**
123 	 * Adds a message.
124 	 * <p>
125 	 * When setting <code>encodeText</code> to <code>false</code>, it then becomes the responsibility of the application
126 	 * to ensure that the text does not contain any characters which need to be escaped.
127 	 * </p>
128 	 * <p>
129 	 * <b>WARNING:</b> If you are using WMessageBox to display "user entered" or untrusted data, use of this method with
130 	 * <code>encodeText</code> set to <code>false</code> may result in security issues.
131 	 * </p>
132 	 *
133 	 * @param message the message to add
134 	 * @param encodeText true to encode the message, false to leave it unencoded.
135 	 */
136 	public void addMessage(final Message message, final boolean encodeText) {
137 		switch (message.getType()) {
138 			case Message.SUCCESS_MESSAGE:
139 				addMessage(successMessages, message, encodeText);
140 				break;
141 
142 			case Message.INFO_MESSAGE:
143 				addMessage(infoMessages, message, encodeText);
144 				break;
145 
146 			case Message.WARNING_MESSAGE:
147 				addMessage(warningMessages, message, encodeText);
148 				break;
149 
150 			case Message.ERROR_MESSAGE:
151 				addMessage(errorMessages, message, encodeText);
152 				break;
153 
154 			default:
155 				LOG.warn("Unknown message type: " + message.getType());
156 		}
157 	}

392 	 * Convenience method to add a message to a message box.
393 	 *
394 	 * @param box the message box to add the message to
395 	 * @param message the message to add
396 	 * @param encode true to encode the message, false otherwise.
397 	 */
398 	private void addMessage(final WMessageBox box, final Message message, final boolean encode) {
399 		String code = message.getMessage();
400 
401 		if (!Util.empty(code)) {
402 			box.addMessage(encode, code, message.getArgs());
403 		}
404 	}

488 	/**
489 	 * Searches the WComponent tree of the given component for an ancestor that implements the MessageContainer
490 	 * interface.
491 	 *
492 	 * @param component the component to return the Container for
493 	 * @return the nearest MessageContainer if found, null otherwise
494 	 */
495 	protected static MessageContainer getMessageContainer(final WComponent component) {
496 		for (WComponent c = component; c != null; c = c.getParent()) {
497 			if (c instanceof MessageContainer) {
498 				return (MessageContainer) c;
499 			}
500 		}
501 
502 		return null;
503 	}
