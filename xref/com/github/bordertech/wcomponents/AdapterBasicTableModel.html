7   /**
8    * Adapter to allow classes that implement the {@link BasicTableModel} interface to be used as the model for
9    * {@link WTable}.
10   * <p>
11   * {@link BasicTableModel} provides a basic interface for data that does not have a tree like structure (ie not
12   * expandable). Therefore, for data that is not expandable, it is recommended to use this adapter and
13   * {@link BasicTableModel} interface.
14   * </p>
15   *
16   * @author Jonathan Austin
17   * @since 1.0.0
18   */
19  public class AdapterBasicTableModel implements TableModel, Serializable {
20  
21  	/**
22  	 * The basic table model to adapt for the WTable.
23  	 */
24  	private final BasicTableModel model;
25  
26  	/**
27  	 * @param model the basic table model to adapt for the {@link WTable}
28  	 */
29  	public AdapterBasicTableModel(final BasicTableModel model) {
30  		this.model = model;
31  	}
32  
33  	/**
34  	 * @return the basic table model to adapt for the {@link WTable}
35  	 */
36  	public BasicTableModel getBacking() {
37  		return model;
38  	}
39  
40  	/**
41  	 * {@inheritDoc}
42  	 */
43  	@Override
44  	public Object getValueAt(final List<Integer> row, final int col) {
45  		return getBacking().getValueAt(getRowIndex(row), col);
46  	}
47  
48  	/**
49  	 * {@inheritDoc}
50  	 */
51  	@Override
52  	public boolean isCellEditable(final List<Integer> row, final int col) {
53  		return getBacking().isCellEditable(getRowIndex(row), col);
54  	}
55  
56  	/**
57  	 * {@inheritDoc}
58  	 */
59  	@Override
60  	public void setValueAt(final Object value, final List<Integer> row, final int col) {
61  		getBacking().setValueAt(value, getRowIndex(row), col);
62  	}
63  
64  	/**
65  	 * {@inheritDoc}
66  	 */
67  	@Override
68  	public boolean isSortable(final int col) {
69  		return getBacking().isSortable(col);
70  	}
71  
72  	/**
73  	 * {@inheritDoc}
74  	 */
75  	@Override
76  	public int[] sort(final int col, final boolean ascending) {
77  		return getBacking().sort(col, ascending);
78  	}
79  
80  	/**
81  	 * {@inheritDoc}
82  	 */
83  	@Override
84  	public boolean isDisabled(final List<Integer> row) {
85  		return getBacking().isDisabled(getRowIndex(row));
86  	}
87  
88  	/**
89  	 * {@inheritDoc}
90  	 */
91  	@Override
92  	public boolean isSelectable(final List<Integer> row) {
93  		return getBacking().isSelectable(getRowIndex(row));
94  	}
95

97  	 * @param row the row index
98  	 * @return false as data is not expandable
99  	 */
100 	@Override
101 	public boolean isExpandable(final List<Integer> row) {
102 		return false;
103 	}

138 	}

145 		return getBacking().getRowKey(getRowIndex(row));

148 	/**
149 	 * @param row the row index
150 	 * @return the first level index as the data is only one level.
151 	 */
152 	private int getRowIndex(final List<Integer> row) {
153 		return row.get(0);
154 	}

156 	/**
157 	 * <p>
158 	 * BasicTableModel provides a basic interface that can be adapted via {@link AdapterBasicTableModel} for
159 	 * {@link WTable}. This model is used for data that is not in a tree like structure (ie not expandable).
160 	 * </p>
161 	 * <p>
162 	 * As the data is not expandable, the interface only requires a single row index, instead of being like the
163 	 * {@link TableModel} interface that uses a list of indexes.
164 	 * </p>
165 	 * <p>
166 	 * Row and column indices for all methods are zero-based, and TableModels are not expected to perform
167 	 * bounds-checking.
168 	 * </p>
169 	 *
170 	 * @author Jonathan Austin
171 	 * @since 1.0.0
172 	 */
173 	public interface BasicTableModel {
174 
175 		/**
176 		 * Retrieves the value at the given row and column.
177 		 *
178 		 * @param row - the row index.
179 		 * @param col - the column index. Column of -1 indicates row has a renderer.
180 		 * @return the value at the given row and column.
181 		 */
182 		Object getValueAt(int row, int col);
183 
184 		/**
185 		 * Indicates whether the given cell is editable.
186 		 *
187 		 * @param row - the row index.
188 		 * @param col - the column index. Column of -1 indicates row has a renderer.
189 		 * @return true if the given cell is editable, false otherwise.
190 		 */
191 		boolean isCellEditable(int row, int col);
192 
193 		/**
194 		 * Sets the value at the given row and column.
195 		 *
196 		 * @param value the value to set.
197 		 * @param row - the row index.
198 		 * @param col - the column index.
199 		 */
200 		void setValueAt(Object value, int row, int col);
201 
202 		/**
203 		 * Indicates whether the model supports sorting by the given column.
204 		 *
205 		 * @param col the column index.
206 		 * @return true if the model is sortable by the given column, false otherwise.
207 		 */
208 		boolean isSortable(int col);
209 
210 		/**
211 		 * <p>
212 		 * Sorts the data by the given column. Any previous sorting should be disregarded.
213 		 * </p>
214 		 * <p>
215 		 * Data models must implement sorting in one of two ways.
216 		 * </p>
217 		 * <ol>
218 		 * <li>
219 		 * <p>
220 		 * If the data is accessible locally by the data model (ie. a sort won't result in a service call to obtain
221 		 * sorted data), then this method should not sort the actual data, but return a row-index mapping which the
222 		 * table will use to access the data. Row selection and expansion will be updated to use the new row indices.
223 		 * </p>
224 		 * <p>
225 		 * For example, if the data for the column is {"a", "b", "d", "c"}, then an ascending sort should return {0, 1,
226 		 * 3, 2}, and a descending sort {2, 3, 1, 0}.
227 		 * </p>
228 		 * </li>
229 		 * <li>
230 		 * <p>
231 		 * If the data is not accessible locally by the data model, or the model is otherwise unable to perform a
232 		 * mapping between old and new row indices, then the model should sort the actual data, and return null. In this
233 		 * case, the table will reset any row selection or expansion.
234 		 * </p>
235 		 * </li>
236 		 * </ol>
237 		 *
238 		 * @param col the column to sort on
239 		 * @param ascending true for an ascending sort, false for descending.
240 		 * @return the row indices in sort order, or null if row mappings can not be determined.
241 		 */
242 		int[] sort(int col, boolean ascending);
243 
244 		/**
245 		 * Indicates whether the given row is disabled.
246 		 *
247 		 * @param row the row index
248 		 * @return true if the row is disabled, false otherwise.
249 		 */
250 		boolean isDisabled(int row);
251 
252 		/**
253 		 * Indicates whether the given row is selectable.
254 		 *
255 		 * @param row the row index
256 		 * @return true if the row is disabled, false otherwise.
257 		 */
258 		boolean isSelectable(int row);
259 
260 		/**
261 		 * Retrieves the number of rows.
262 		 *
263 		 * @return the number of rows in the model for this level.
264 		 */
265 		int getRowCount();
266 
267 		/**
268 		 * Retrieves the key (ie bean) used to uniquely identify this row.
269 		 * <p>
270 		 * The usual implementation of this method would just return the row id passed in.
271 		 * </p>
272 		 * <p>
273 		 * However, if you are required to dynamically add/remove rows in the model, which would change the row index,
274 		 * then the implementation of this method needs to return an object that uniquely identifies this row.
275 		 * </p>
276 		 * <p>
277 		 * When rows have been added/removed to the model, the {@link WTable#handleDataChanged()} method on WTable needs
278 		 * to be called.
279 		 * </p>
280 		 *
281 		 * @param row the row index
282 		 * @return the key (ie bean) used to uniquely identify this row
283 		 */
284 		Object getRowKey(int row);
285 	}
