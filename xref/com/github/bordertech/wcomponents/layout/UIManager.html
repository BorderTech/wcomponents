24  /**
25   * The UIManager provides a mechanism for client applications to use different Renderers, without having to specify a
26   * renderer on a per-component basis.
27   *
28   * Renderers are assigned in the following order of preference:
29   * <ol>
30   * <li>
31   * An application-specific renderer for the component. This is configured using application parameters, e.g. for a WText
32   * component a Renderer would be specified as:
33   * <code>bordertech.wcomponents.UIManager.renderer.com.github.bordertech.wcomponents.WText=com.github.myApp.render.MyTextRenderer</code>
34   * </li>
35   * <li>
36   * A package implementation for the component, for the current RenderContext in use. This is controlled by the
37   * {@link RenderContext#getRenderPackage()} method, which specifies the package name. The UIManager expects that each
38   * renderer package includes a {@link RendererFactory} implementation which knows how to create a specific renderer for
39   * a component. The RendererFactory class for each package must be named "RendererFactory".
40   * </li>
41   * <li>
42   * As for the first two options, but for the class's parent class (then grand-parent etc., repeated until the
43   * {@link WComponent} class is reached).
44   * </li>
45   * </ol>
46   *
47   * @author Yiannis Paschalidis
48   * @since 1.0.0
49   */
50  public final class UIManager implements PropertyChangeListener {

103 	 * may have changed.
104 	 *
105 	 * @param evt ignored.
106 	 */
107 	@Override
108 	public void propertyChange(final PropertyChangeEvent evt) {

113 	/**
114 	 * Retrieves a renderer which can renderer the given component to the context.
115 	 *
116 	 * @param component the component to retrieve the renderer for.
117 	 * @param context the render context.
118 	 * @return an appropriate renderer for the component and context, or null if a suitable renderer could not be found.
119 	 */
120 	public static Renderer getRenderer(final WComponent component, final RenderContext context) {
121 		Class<? extends WComponent> clazz = component.getClass();
122 
123 		Duplet<String, Class<?>> key = new Duplet<String, Class<?>>(context.getRenderPackage(),
124 				clazz);
125 		Renderer renderer = INSTANCE.renderers.get(key);
126 
127 		if (renderer == null) {
128 			renderer = INSTANCE.findRenderer(component, key);
129 		} else if (renderer == NULL_RENDERER) {
130 			return null;
131 		}
132 
133 		return renderer;
134 	}
135 
136 	/**
137 	 * Retrieves a renderer which can renderer templates for the given context.
138 	 *
139 	 * @param context the render context.
140 	 * @return an appropriate renderer for the component and context, or null if a suitable renderer could not be found.
141 	 * @deprecated Use {@link WTemplate} instead.
142 	 */
143 	@Deprecated
144 	public static Renderer getTemplateRenderer(final RenderContext context) {
145 		String packageName = context.getRenderPackage();
146 		Renderer renderer = INSTANCE.templateRenderers.get(packageName);
147 
148 		if (renderer == null) {
149 			renderer = INSTANCE.findTemplateRenderer(packageName);
150 		} else if (renderer == NULL_RENDERER) {
151 			return null;
152 		}
153 
154 		return renderer;
155 	}

159 	 *
160 	 * @param packageName the package to retrieve the template renderer for.

166 		RendererFactory factory = INSTANCE.findRendererFactory(packageName);
167 		Renderer renderer = factory.getTemplateRenderer();
168 
169 		if (renderer == null) {
170 			templateRenderers.put(packageName, NULL_RENDERER);

172 			templateRenderers.put(packageName, renderer);

201 	/**
202 	 * Finds the layout for the given theme and component.
203 	 *
204 	 * @param component the WComponent class to find a manager for.
205 	 * @param key the component key to use for caching the renderer.
206 	 * @return the LayoutManager for the component.
207 	 */
208 	private synchronized Renderer findRenderer(final WComponent component,
209 			final Duplet<String, Class<?>> key) {
210 		LOG.info("Looking for layout for " + key.getSecond().getName() + " in " + key.getFirst());
211 
212 		Renderer renderer = findConfiguredRenderer(component, key.getFirst());
213 
214 		if (renderer == null) {
215 			renderers.put(key, NULL_RENDERER);
216 		} else {
217 			renderers.put(key, renderer);
218 		}
219 
220 		return renderer;
221 	}
222 
223 	/**
224 	 * Finds the renderer factory for the given package.
225 	 *
226 	 * @param packageName the package name to find the renderer factory for.
227 	 * @return the RendererFactory for the given package, or null if not found.
228 	 */
229 	private synchronized RendererFactory findRendererFactory(final String packageName) {
230 		RendererFactory factory = factoriesByPackage.get(packageName);
231 
232 		if (factory == null) {
233 			try {
234 				factory = (RendererFactory) Class.forName(packageName + ".RendererFactoryImpl").
235 						newInstance();
236 				factoriesByPackage.put(packageName, factory);
237 			} catch (Exception e) {
238 				throw new SystemException(
239 						"Failed to create layout manager factory for " + packageName, e);
240 			}
241 		}
242 
243 		return factory;
244 	}

246 	/**
247 	 * Attempts to find the configured renderer for the given output format and component.
248 	 *
249 	 * @param component the component to find a manager for.
250 	 * @param rendererPackage the package containing the renderers.
251 	 * @return the Renderer for the component, or null if there is no renderer defined.
252 	 */
253 	private Renderer findConfiguredRenderer(final WComponent component, final String rendererPackage) {
254 		Renderer renderer = null;
255 
256 		// We loop for each WComponent in the class hierarchy, as the
257 		// Renderer may have been specified at a higher level.
258 		for (Class<?> c = component.getClass(); renderer == null && c != null && !AbstractWComponent.class.
259 				equals(c); c = c.getSuperclass()) {
260 			String qualifiedClassName = c.getName();
261 
262 			// Is there an override for this class?
263 			String rendererName = ConfigurationProperties.getRendererOverride(qualifiedClassName);
264 
265 			if (rendererName != null) {
266 				renderer = createRenderer(rendererName);
267 
268 				if (renderer == null) {
269 					LOG.warn(
270 							"Layout Manager \"" + rendererName + "\" specified for " + qualifiedClassName + " was not found");
271 				} else {
272 					return renderer;
273 				}
274 			}
275 
276 			renderer = findRendererFactory(rendererPackage).getRenderer(c);
277 		}
278 
279 		return renderer;
280 	}

282 	/**
283 	 * Attempts to create a Renderer with the given name.
284 	 *
285 	 * @param rendererName the name of the Renderer
286 	 * @return a Renderer of the given type, or null if the class was not found.
287 	 */
288 	private static Renderer createRenderer(final String rendererName) {
289 		if (rendererName.endsWith(".vm")) {
290 			// This is a velocity template, so use a VelocityLayout
291 			return new VelocityRenderer(rendererName);
292 		}
293 
294 		try {
295 			Class<?> managerClass = Class.forName(rendererName);
296 			Object manager = managerClass.newInstance();
297 
298 			if (!(manager instanceof Renderer)) {
299 				throw new SystemException(rendererName + " is not a Renderer");
300 			}
301 
302 			return (Renderer) manager;
303 		} catch (ClassNotFoundException e) {
304 			// Legal - there might not a manager implementation in a given theme
305 			return null;
306 		} catch (InstantiationException e) {
307 			throw new SystemException("Failed to instantiate " + rendererName, e);
308 		} catch (IllegalAccessException e) {
309 			throw new SystemException("Failed to access " + rendererName, e);
310 		}
311 	}
