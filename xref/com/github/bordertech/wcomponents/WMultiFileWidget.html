72  			return null;

205 	/**
206 	 * Determines the file types accepted by this widget. Note that duplicates are not allowed and these are not case
207 	 * sensitive. A file type may be one of the following:
208 	 * <ul>
209 	 * <li>The string audio/* (Indicates that sound files are accepted.)</li>
210 	 * <li>The string video/* (Indicates that video files are accepted.)</li>
211 	 * <li>The string image/* (Indicates that image files are accepted.)</li>
212 	 * <li>A valid MIME type with no parameters (Indicates that files of the specified type are accepted.)</li>
213 	 * <li>A string whose first character is a "." (U+002E) character (Indicates that files with the specified file
214 	 * extension are accepted).</li>
215 	 * </ul>
216 	 *
217 	 * @param types The file types that will be accepted by the file input. Note that this is not additive, it will
218 	 * overwrite any previously set fileTypes. Pass null or and empty collection to clear all file types.
219 	 */
220 	public void setFileTypes(final Collection<String> types) {
221 		Set newFileTypes = null;
222 		MultiFileWidgetModel model = getOrCreateComponentModel();
223 		if (types != null) {
224 			newFileTypes = new HashSet<>();
225 			for (String fileType : types) {
226 				if (fileType != null) {
227 					String uniqueFileType = fileType.toLowerCase();  // Lowercase to avoid duplication
228 					if (isValid(uniqueFileType)) {
229 						newFileTypes.add(uniqueFileType);
230 					} else {
231 						throw new IllegalArgumentException("Not a valid filetype: " + fileType);
232 					}
233 				}
234 			}
235 		}
236 		model.fileTypes = newFileTypes;
237 	}

239 	/**
240 	 * Check that the file type SEEMS to be legit.
241 	 *
242 	 * @param fileType The file type to check
243 	 * @return true if the file type is valid.
244 	 */
245 	private boolean isValid(final String fileType) {
246 		boolean result = false;
247 		if (fileType != null && fileType.length() > 1) { // the shortest I can think of would be something like ".h"
248 			if (fileType.startsWith(".")) { // assume it's a file extension
249 				result = true;
250 			} else if (fileType.length() > 2 && fileType.indexOf('/') > 0) { // some imaginary mimetype like "a/*" would be at least 3 characters
251 				result = true;
252 			}
253 		}
254 		return result;
255 	}

538 	/**
539 	 * {@inheritDoc}
540 	 */
541 	@Override
542 	protected boolean beforeHandleRequest(final Request request) {
543 
544 		// Clear file upload ID (if set)
545 		setFileUploadRequestId(null);
546 
547 		// Check if is targeted request (thumbnail request or file content request)
548 		String targetParam = request.getParameter(Environment.TARGET_ID);
549 		boolean targetted = (targetParam != null && targetParam.equals(getTargetId()));
550 		if (targetted) {
551 			doHandleTargetedRequest(request);
552 			return false;
553 		}
554 
555 		// Check if AJAX trigger
556 		if (isCurrentAjaxTrigger()) {
557 			// Upload file request (check for multi part and file id)
558 			if (request.getParameter(FILE_UPLOAD_MULTI_PART_ID_KEY) != null && request.getFileItems(getId()) != null) {
559 				doHandleUploadRequest(request);
560 				return false;
561 			}
562 			// File in list selected by the user
563 			if (request.getParameter(FILE_UPLOAD_ID_KEY) != null) {
564 				doHandleFileAjaxActionRequest(request);
565 				return false;
566 			}
567 		}
568 
569 		return true;
570 	}

572 	/**
573 	 * Handle a file action AJAX request.
574 	 *
575 	 * @param request the request being processed
576 	 */
577 	protected void doHandleFileAjaxActionRequest(final Request request) {
578 		// Protect against client-side tampering of disabled components
579 		if (isDisabled()) {
580 			throw new SystemException("File widget is disabled.");
581 		}
582 
583 		// Check for file id
584 		String fileId = request.getParameter(FILE_UPLOAD_ID_KEY);
585 		if (fileId == null) {
586 			throw new SystemException("No file id provided for ajax action.");
587 		}
588 
589 		// Check valid file id
590 		FileWidgetUpload file = getFile(fileId);
591 		if (file == null) {
592 			throw new SystemException("Invalid file id [" + fileId + "].");
593 		}
594 
595 		// Run the action
596 		final Action action = getFileAjaxAction();
597 		if (action == null) {
598 			throw new SystemException("No action set for file ajax action request.");
599 		}
600 
601 		// Set the selected file id as the action object
602 		final ActionEvent event = new ActionEvent(this, "fileajax", fileId);
603 		Runnable later = new Runnable() {
604 			@Override
605 			public void run() {
606 				action.execute(event);
607 			}
608 		};
609 
610 		invokeLater(later);
611 	}

613 	/**
614 	 * Handle a targeted request. Can be a file upload, thumbnail request or file content request.
615 	 *
616 	 *
617 	 * @param request the request being processed
618 	 */
619 	protected void doHandleTargetedRequest(final Request request) {
620 
621 		// Check for file id
622 		String fileId = request.getParameter(FILE_UPLOAD_ID_KEY);
623 		if (fileId == null) {
624 			throw new SystemException("No file id provided for content request.");
625 		}
626 
627 		// Check valid file id
628 		FileWidgetUpload file = getFile(fileId);
629 		if (file == null) {
630 			throw new SystemException("Invalid file id [" + fileId + "].");
631 		}
632 
633 		// Check if thumb nail requested
634 		boolean thumbNail = request.getParameter(FILE_UPLOAD_THUMB_NAIL_KEY) != null;
635 		if (thumbNail) {
636 			doHandleThumbnailRequest(file);
637 		} else {
638 			doHandleFileContentRequest(file);
639 		}
640 	}

642 	/**
643 	 * The request is a targeted file upload request. Upload the file and respond with the file information.
644 	 *
645 	 * @param request the request being processed.
646 	 */
647 	protected void doHandleUploadRequest(final Request request) {
648 		// Protect against client-side tampering of disabled/read-only fields.
649 		if (isDisabled() || isReadOnly()) {
650 			throw new SystemException("File widget cannot be updated.");
651 		}
652 
653 		// Only process on a POST
654 		if (!"POST".equals(request.getMethod())) {
655 			throw new SystemException(
656 					"File widget cannot be updated by " + request.getMethod() + ".");
657 		}
658 
659 		// Check only one file item in the request
660 		FileItem[] items = request.getFileItems(getId());
661 		if (items.length > 1) {
662 			throw new SystemException("More than one file item received on the request.");
663 		}
664 
665 		// Check the client provided a fileID
666 		String fileId = request.getParameter(FILE_UPLOAD_MULTI_PART_ID_KEY);
667 		if (fileId == null) {
668 			throw new SystemException("No file id provided for file upload.");
669 		}
670 
671 		// Wrap the file item
672 		FileItemWrap wrap = new FileItemWrap(items[0]);
673 		FileWidgetUpload file = new FileWidgetUpload(fileId, wrap);
674 		addFile(file);
675 
676 		// Set the file id to be used ion the renderer
677 		setFileUploadRequestId(fileId);
678 		setNewUpload(true);
679 	}

711 		Image image = null;
712 		try {
713 			Dimension size = getThumbnailSize();

715 					getMimeType());

722 	/**
723 	 * Retrieves a URL for the uploaded file content.
724 	 *
725 	 * @param fileId the file id
726 	 * @return the URL to access the uploaded file.
727 	 */
728 	public String getFileUrl(final String fileId) {
729 		FileWidgetUpload file = getFile(fileId);
730 		if (file == null) {
731 			return null;
732 		}
733 
734 		Environment env = getEnvironment();
735 		Map<String, String> parameters = env.getHiddenParameters();
736 		parameters.put(Environment.TARGET_ID, getTargetId());
737 
738 		if (Util.empty(file.getFileCacheKey())) {
739 			// Add some randomness to the URL to prevent caching
740 			String random = WebUtilities.generateRandom();
741 			parameters.put(Environment.UNIQUE_RANDOM_PARAM, random);
742 		} else {
743 			// Remove step counter as not required for cached content
744 			parameters.remove(Environment.STEP_VARIABLE);
745 			parameters.remove(Environment.SESSION_TOKEN_VARIABLE);
746 			// Add the cache key
747 			parameters.put(Environment.CONTENT_CACHE_KEY, file.getFileCacheKey());
748 		}
749 
750 		// File id
751 		parameters.put(FILE_UPLOAD_ID_KEY, fileId);
752 
753 		// The targetable path needs to be configured for the portal environment.
754 		String url = env.getWServletPath();
755 
756 		// Note the last parameter. In javascript we don't want to encode "&".
757 		return WebUtilities.getPath(url, parameters, true);
758 	}

778 		Environment env = getEnvironment();

781 
782 		if (Util.empty(file.getThumbnailCacheKey())) {

784 			String random = WebUtilities.generateRandom();
785 			parameters.put(Environment.UNIQUE_RANDOM_PARAM, random);
786 		} else {
787 			// Remove step counter as not required for cached content
788 			parameters.remove(Environment.STEP_VARIABLE);
789 			parameters.remove(Environment.SESSION_TOKEN_VARIABLE);

793 
794 		// File id
795 		parameters.put(FILE_UPLOAD_ID_KEY, fileId);

807 	/**
808 	 * @param request the request to process
809 	 * @return the list of file selections
810 	 */
811 	private List<FileWidgetUpload> getNewSelection(final Request request) {
812 		// Get selected items (if any)
813 		String[] selectedParam = request.getParameterValues(getId() + ".selected");
814 		if (selectedParam == null) {
815 			selectedParam = new String[0];
816 		}
817 		List<String> selected = Arrays.asList(selectedParam);
818 
819 		List<FileWidgetUpload> newSelection = new ArrayList<>();
820 		List<FileWidgetUpload> currentSelection = getValue();
821 
822 		for (FileWidgetUpload file : currentSelection) {
823 			boolean select = selected.contains(file.getFileId());
824 			if (select) {
825 				newSelection.add(file);
826 			}
827 		}
828 
829 		return newSelection;
830 	}

841 	 * Selection lists are considered equal if they have the same items (order is not important). An empty list is
842 	 * considered equal to a null list.
843 	 *
844 	 * @param list1 the first list to check.
845 	 * @param list2 the second list to check.
846 	 * @return true if the lists are equal, false otherwise.
847 	 */
848 	private boolean selectionsEqual(final List<?> list1, final List<?> list2) {
849 		return (list1 == null || list1.isEmpty()) && (list2 == null || list2.isEmpty())
850 				|| (list1 != null && list2 != null && list1.size() == list2.size() && list1.
851 				containsAll(list2));
852 	}
