1   package com.github.bordertech.wcomponents.registry;
2   
3   import com.github.bordertech.wcomponents.WComponent;
4   import com.github.bordertech.wcomponents.util.SystemException;
5   import java.util.HashMap;
6   import java.util.Map;
7   import org.apache.commons.logging.Log;
8   import org.apache.commons.logging.LogFactory;
9   
10  /**
11   * A UIRegistry implementation that can load UIs using the key, where the keys are the fully qualified class names.
12   *
13   * @author Martin Shevchenko
14   * @since 1.0.0
15   */
16  public class UIRegistryClassLoaderImpl extends UIRegistry {
17  
18  	/**
19  	 * The logger instance for this class.
20  	 */
21  	private static final Log LOG = LogFactory.getLog(UIRegistryClassLoaderImpl.class);
22  
23  	/**
24  	 * The UI registry map.
25  	 */
26  	private final Map<String, Object> registry = new HashMap<>();
27  
28  	/**
29  	 * Registers the given user interface with the given key.
30  	 *
31  	 * @param key the registration key.
32  	 * @param ui the user interface to register.
33  	 */
34  	@Override
35  	public synchronized void register(final String key, final WComponent ui) {
36  		if (isRegistered(key)) {
37  			throw new SystemException("Cannot re-register a component. Key = " + key);
38  		}
39  
40  		registry.put(key, ui);
41  	}
42  
43  	/**
44  	 * Is there a user interface registered under the given key.
45  	 *
46  	 * @param key the registration key.
47  	 * @return true if there is a UI registered with the given key.
48  	 */
49  	@Override
50  	public synchronized boolean isRegistered(final String key) {
51  		return registry.containsKey(key);
52  	}
53  
54  	/**
55  	 * Retrieves the user interface that was registered with the given key. If the UI has not been registered, this
56  	 * attempts to load the UI using the key as a class name.
57  	 *
58  	 * @param key the registration key
59  	 * @return the UI for the given key. The UI may be newly created.
60  	 */
61  	@Override
62  	public synchronized WComponent getUI(final String key) {
63  		Object obj = registry.get(key);
64  
65  		if (obj == null) {
66  			// Looks like we haven't tried loading this UI yet, so do it now.
67  			obj = loadUI(key);
68  
69  			// Cache the result.
70  			// Note that the result could be a flag indicating that no UI could be load for the given key.
71  			registry.put(key, obj);
72  
73  			if (obj instanceof WComponent) {
74  				((WComponent) obj).setLocked(true);
75  			}
76  		}
77  
78  		if (obj instanceof WComponent) {
79  			LOG.debug("Returning cached WComponent. Key=" + key);
80  			return (WComponent) obj;
81  		} else {
82  			return null;
83  		}
84  	}
85  
86  	/**
87  	 * Attemps to load a ui using the key as a class name.
88  	 *
89  	 * @param key the registration key
90  	 * @return A WComponent if one could be loaded from the classpath, else it returns Boolean.FALSE.
91  	 */
92  	private static Object loadUI(final String key) {
93  		String classname = key.trim();
94  
95  		try {
96  			Class<?> clas = Class.forName(classname);
97  
98  			if (WComponent.class.isAssignableFrom(clas)) {
99  				Object instance = clas.newInstance();
100 				LOG.debug("WComponent successfully loaded with class name \"" + classname + "\".");
101 				return instance;
102 			} else {
103 				LOG.error("The resource with the name \"" + classname + "\" is not a WComponent.");
104 			}
105 		} catch (Exception ex) {
106 			LOG.error("Unable to load a WComponent using the resource name \"" + classname + "\"",
107 					ex);
108 		}
109 
110 		return Boolean.FALSE;
111 	}
112 }
