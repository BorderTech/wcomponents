5   /**
6    * <p>
7    * The RenderContext is used to render a component. It is passed into the {@link WComponent#paint(RenderContext)} method
8    * as an argument and also the specific {@link Renderer} instance used to render the component. Components themselves
9    * must not rely on knowledge of the context.</p>
10   *
11   * <p>
12   * The appropriate {@link Renderer} for each component is obtained from the {@link UIManager}, which calls the
13   * {@link #getRenderPackage()} method to obtain the package name which contains all of the renderers for a given type of
14   * context.</p>
15   *
16   * <p>
17   * RenderContext subclasses should contain additional data &amp; methods which are necessary to render the component in
18   * the context. For example, a Web/HTTP renderer may include a reference to the ServletResponse.</p>
19   *
20   * @author Yiannis Paschalidis
21   * @since 1.0.0
22   */
23  public interface RenderContext {
24  
25  	/**
26  	 * Return the java package name which contains the renderers for this renderer. This package must contain a public
27  	 * class named "RendererFactory", which implements the {@link RendererFactory} interface.
28  	 *
29  	 * @return the renderer package name.
30  	 */
31  	String getRenderPackage();
32  }
