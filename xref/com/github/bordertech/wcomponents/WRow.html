5   /**
6    * This is a layout component, to which you add {@link WColumn} components. The widths of all the columns added to the
7    * row should total 100.
8    *
9    * @author Ming Gao
10   * @author Yiannis Paschalidis
11   * @author Mark Reeves
12   */
13  public class WRow extends AbstractNamingContextContainer implements AjaxTarget, SubordinateTarget,
14  		Marginable {
15  
16  	/**
17  	 * The horizontal space between the columns in the row, measured in pixels.
18  	 */
19  	private final Size space;
20  
21  	/**
22  	 * For temporary backwards compatibility only.
23  	 */
24  	@Deprecated
25  	private final int gap;
26  
27  	/**
28  	 * For temporary backwards compatibility only.
29  	 * @param space the real space between items
30  	 * @param gap the requested space between items
31  	 */
32  	@Deprecated
33  	private WRow(final Size space, final int gap) {
34  		this.space = space;
35  		this.gap = gap;
36  	}
37  
38  	/**
39  	 * Creates a WRow.
40  	 */
41  	public WRow() {
42  		this(null);
43  	}
44  
45  	/**
46  	 * Create a WRow with the specified space between columns.
47  	 * @param space the space between columns in the row
48  	 */
49  	public WRow(final Size space) {
50  		this.space = space;
51  		this.gap = -1;
52  	}
53  
54  	/**
55  	 * Creates a WRow with the specified attributes.
56  	 *
57  	 * @param hgap the space between the columns in the row
58  	 * @deprecated use {@link #WRow(com.github.bordertech.wcomponents.Size)}
59  	 */
60  	@Deprecated
61  	public WRow(final int hgap) {
62  		this(SpaceUtil.intToSize(hgap), hgap);
63  	}
64  
65  	/**
66  	 * Adds the given column as a child of this component.
67  	 *
68  	 * @param column the column to add.
69  	 */
70  	public void add(final WColumn column) {
71  		super.add(column);
72  	}
73  
74  	/**
75  	 * Removes the given column from this components list of children.
76  	 *
77  	 * @param column the column to remove
78  	 */
79  	public void remove(final WColumn column) {
80  		super.remove(column);
81  	}
82  
83  	/**
84  	 * @return the horizontal space between the columns in the row, measured in pixels
85  	 * @deprecated use {@link #getGap()}
86  	 */
87  	@Deprecated
88  	public int getHgap() {
89  		return gap;
90  	}
91  
92  	/**
93  	 * @return the horizontal space between the columns in the row
94  	 */
95  	public Size getSpace() {
96  		return space;
97  	}
98  
99  	/**
100 	 * @return the space between the components added to the layout.
101 	 */
102 	@Deprecated
103 	public int getGap() {
104 		return gap;
105 	}

114

116 	 * {@inheritDoc}
117 	 */
