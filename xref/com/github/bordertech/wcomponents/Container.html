1   package com.github.bordertech.wcomponents;

3   import java.util.List;
4   
5   /**
6    * <p>
7    * The interface for WComponents which can contain child WComponents.
8    * </p>
9    * <p>
10   * A basic Container is often used to group related components together. Grouping components together is good for
11   * readability of the code and it is also how you create reusable chunks of UI.
12   * </p>
13   *
14   * @author Yiannis Paschalidis
15   * @since 1.0.0
16   */
17  public interface Container extends WComponent {
18  
19  	/**
20  	 * @return the number of child components currently contained within this component.
21  	 */
22  	int getChildCount();
23  
24  	/**
25  	 * Retrieves a child component by its index.
26  	 *
27  	 * @param index the index of the child component to be retrieved.
28  	 * @return the child component at the given index.
29  	 */
30  	WComponent getChildAt(int index);
31  
32  	/**
33  	 * Retrieves the index of the given child.
34  	 *
35  	 * @param childComponent the child component to retrieve the index for.
36  	 * @return the index of the given child component, or -1 if the component is not a child of this component.
37  	 */
38  	int getIndexOfChild(WComponent childComponent);
39  
40  	/**
41  	 * Retrieves a list of this Container's.
42  	 *
43           * @return an immutable list of this Container.
44  	 */
45  	List<WComponent> getChildren();
46  }
