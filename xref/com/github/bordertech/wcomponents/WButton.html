10  /**
11   * <p>
12   * A WButton is used to submit the contents of the form to the server. An {@link Action} can be associated with the
13   * button to execute application-specific code when the button is pressed.
14   * </p>
15   * <pre>
16   * // Create a button with the text &quot;Submit&quot;.
17   * WButton button = new WButton(&quot;Submit&quot;);
18   *
19   * // Set an action to run when the button is clicked.
20   * button.setAction(new Action() {
21   *   public void execute(final ActionEvent event) {
22   *     System.out.println(&quot;The button was clicked at &quot; + new Date());
23   *   }
24   * });
25   * </pre>
26   *
27   * @author James Gifford
28   * @since 1.0.0
29   */

128 	/**
129 	 * Override handleRequest in order to perform processing for this component. This implementation checks whether the
130 	 * button has been pressed in the request.
131 	 *
132 	 * @param request the request being responded to.
133 	 */
134 	@Override
135 	public void handleRequest(final Request request) {
136 		// Clear pressed
137 		clearPressed();
138 
139 		String requestValue = request.getParameter(getId());
140 		boolean pressed = "x".equals(requestValue);
141 
142 		// Only process on a POST
143 		if (pressed && !"POST".equals(request.getMethod())) {
144 			LOG.warn("Button pressed on a request that is not a POST. Will be ignored.");
145 			return;
146 		}
147 
148 		setPressed(pressed, request);
149 	}
150 
151 	/**
152 	 * Override preparePaintComponent to register an AJAX operation if this button is AJAX enabled.
153 	 *
154 	 * @param request the request being responded to
155 	 */
156 	@Override
157 	protected void preparePaintComponent(final Request request) {
158 		super.preparePaintComponent(request);
159 
160 		UIContext uic = UIContextHolder.getCurrent();
161 
162 		if (isAjax() && uic.getUI() != null) {
163 			AjaxTarget target = getAjaxTarget();
164 			AjaxHelper.registerComponent(target.getId(), getId());
165 		}
166 	}

186 	/**
187 	 * Sets whether this button is pressed. You probably do not want to invoke this manually, it is called from
188 	 * {@link #handleRequest}. If the button is pressed its Action is queued so it is invoked after the entire request
189 	 * has been handled.
190 	 *
191 	 * @param pressed true for pressed, false for not pressed
192 	 * @param request the Request that is being responded to
193 	 */
194 	protected void setPressed(final boolean pressed, final Request request) {
195 		if (pressed && isDisabled()) {
196 			// Protect against client-side tampering of disabled/read-only fields.
197 			LOG.warn("A disabled button has been triggered. " + getText() + ". " + getId());
198 			return;
199 		}
200 
201 		if (pressed != isPressed()) {
202 			getOrCreateComponentModel().isPressed = pressed;
203 		}
204 
205 		// If an action has been supplied then execute it, but only after
206 		// handle request has been performed on the entire component tree.
207 		if (pressed) {
208 			final Action action = getAction();
209 			if (action == null) {
210 				// Reset focus only if the current Request is an Ajax request.
211 				if (AjaxHelper.isCurrentAjaxTrigger(this)) {
212 					// Need to run the "afterActionExecute" as late as possible.
213 					Runnable later = new Runnable() {
214 						@Override
215 						public void run() {
216 							focusMe();
217 						}
218 					};
219 					invokeLater(later);
220 				}
221 			} else {
222 				final ActionEvent event = new ActionEvent(this, getActionCommand(),
223 						getActionObject());
224 
225 				Runnable later = new Runnable() {
226 					@Override
227 					public void run() {
228 						beforeActionExecute(request);
229 						action.execute(event);
230 						afterActionExecute(request);
231 					}
232 				};
233 
234 				invokeLater(later);
235 			}
236 		}
237 	}

254 	 */
255 	protected void afterActionExecute(final Request request) {
256 		// Only do focus if button is AJAX trigger
257 		if (AjaxHelper.isCurrentAjaxTrigger(this)) {

259 		}
260 	}

268 		}
