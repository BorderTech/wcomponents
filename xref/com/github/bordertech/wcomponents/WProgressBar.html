3   /**
4    * <p>
5    * WProgressBar is a component for displaying progress bars. The number of steps in the progress bar is configurable,
6    * and the progress bar's value can either be set manually, or sourced from an Integer bean.</p>
7    *
8    * <p>
9    * Methods are available to customise the progress bar's appearance, such as setting its size or whether to display the
10   * progress as a fraction or percentage.</p>
11   *
12   * @author Yiannis Paschalidis
13   * @since 1.0.0
14   */
15  public class WProgressBar extends WBeanComponent implements AjaxTarget, Labelable {
16  
17  	/**
18  	 * Progress bar types.
19  	 */
20  	public enum ProgressBarType {
21  		/**
22  		 * A normal progress bar.
23  		 */
24  		NORMAL,
25  		/**
26  		 * A smaller progress bar.
27  		 */
28  		SMALL
29  	}
30  
31  	/**
32  	 * Display unit types.
33  	 */
34  	public enum UnitType {
35  		/**
36  		 * Units will be displayed as a fraction, e.g. <code>33/100</code>.
37  		 */
38  		FRACTION,
39  		/**
40  		 * Units will be displayed as a percentage, e.g. <code>33%</code>.
41  		 */
42  		PERCENTAGE
43  	}
44  
45  	/**
46  	 * Creates a progress bar with the type of "normal" and a fraction unit type.
47  	 */
48  	public WProgressBar() {
49  		this(ProgressBarType.NORMAL, UnitType.FRACTION);
50  	}
51  
52  	/**
53  	 * Creates a progress bar with the type of "normal", a fraction unit type and the given maximum value.
54  	 *
55  	 * @param max the maximum value
56  	 */
57  	public WProgressBar(final int max) {
58  		this(ProgressBarType.NORMAL, UnitType.FRACTION, max);
59  	}
60  
61  	/**
62  	 * Creates a progress bar with the given bar and unit types.
63  	 *
64  	 * @param type the progress bar type.
65  	 * @param unitType the display unit type.
66  	 */
67  	public WProgressBar(final ProgressBarType type, final UnitType unitType) {
68  		setProgressBarType(type);
69  		setUnitType(unitType);
70  	}
71  
72  	/**
73  	 * Creates a progress bar with the given bar type, unit type and max value.
74  	 *
75  	 * @param type the progress bar type.
76  	 * @param unitType the display unit type.
77  	 * @param max the maximum value
78  	 */
79  	public WProgressBar(final ProgressBarType type, final UnitType unitType, final int max) {
80  		setProgressBarType(type);
81  		setUnitType(unitType);
82  		setMax(max);
83  	}
84  
85  	/**
86  	 * @return the maximum value of the progress bar.
87  	 */
88  	public int getMax() {
89  		return getComponentModel().max;
90  	}
91  
92  	/**
93  	 * Sets the maximum value of the progress bar.
94  	 *
95  	 * @param max the maximum allowable value.
96  	 */
97  	public void setMax(final int max) {
98  		getOrCreateComponentModel().max = max;
99  	}
100 
101 	/**
102 	 * Sets the value of the progress bar for the given context.
103 	 *
104 	 * @param value the progress bar value.
105 	 */
106 	public void setValue(final int value) {
107 		setData(value == 0 ? null : value);
108 	}
109 
110 	/**
111 	 * Retrieves the value of the progress bar.
112 	 *
113 	 * @return the progress bar's value for the context.
114 	 */
115 	public int getValue() {
116 		int max = getMax();
117 		Integer data = (Integer) getData();
118 		return data == null ? 0 : Math.max(0, Math.min(max, data));
119 	}
120 
121 	/**
122 	 * @return the progress bar type.
123 	 */
124 	public ProgressBarType getProgressBarType() {
125 		return getComponentModel().barType;
126 	}
127 
128 	/**
129 	 * Sets the progress bar type.
130 	 *
131 	 * @param type the progress bar type.
132 	 */
133 	public void setProgressBarType(final ProgressBarType type) {
134 		getOrCreateComponentModel().barType = type;
135 	}
136 
137 	/**
138 	 * Retrieves the display unit type.
139 	 *
140 	 * @return the unit type.
141 	 */
142 	public UnitType getUnitType() {
143 		return getComponentModel().unitType;
144 	}
145 
146 	/**
147 	 * Sets the display unit type.
148 	 *
149 	 * @param unitType the unit type.
150 	 */
151 	public void setUnitType(final UnitType unitType) {
152 		getOrCreateComponentModel().unitType = unitType;
153 	}
154 
155 	/**
156 	 * @return the progress bar text.
157 	 */
158 	public String getText() {
159 		return getComponentModel().text;
160 	}
161 
162 	/**
163 	 * Sets the progress bar text.
164 	 *
165 	 * @param text the text to set.
166 	 */
167 	public void setText(final String text) {
168 		getOrCreateComponentModel().text = text;
169 	}
170 
171 	@Override
172 	public String toString() {
173 		return toString(String.valueOf(getValue()) + '/' + getMax());
174 	}
175 
176 	/**
177 	 * Creates a new Component model appropriate for this component.
178 	 *
179 	 * @return a new WProgressBarModel.
180 	 */
181 	@Override
182 	protected WProgressBarModel newComponentModel() {
183 		return new WProgressBarModel();
184 	}
185 
186 	/**
187 	 * {@inheritDoc}
188 	 */
189 	@Override // For type safety only
190 	protected WProgressBarModel getOrCreateComponentModel() {
191 		return (WProgressBarModel) super.getOrCreateComponentModel();
192 	}
193 
194 	/**
195 	 * {@inheritDoc}
196 	 */
197 	@Override // For type safety only
198 	protected WProgressBarModel getComponentModel() {
199 		return (WProgressBarModel) super.getComponentModel();
200 	}
201 
202 	/**
203 	 * The component model that holds progress bar's state.
204 	 *
205 	 * @author Yiannis Paschalidis
206 	 */
207 	public static final class WProgressBarModel extends BeanAndProviderBoundComponentModel {
208 
209 		private ProgressBarType barType;
210 		private UnitType unitType;
211 		private int max;
212 		private String text;
213 	}
214 }
