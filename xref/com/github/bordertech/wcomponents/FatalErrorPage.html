1   package com.github.bordertech.wcomponents;
2   
3   import com.github.bordertech.wcomponents.servlet.WebXmlRenderContext;
4   import com.github.bordertech.wcomponents.util.Factory;
5   import com.github.bordertech.wcomponents.util.I18nUtilities;
6   import com.github.bordertech.wcomponents.util.error.SystemFailureMapper;
7   import java.io.PrintWriter;
8   
9   /**
10   * A component used to display a "Fatal" error page, when an unhandled error occurs.
11   *
12   * @author Martin Shevchenko
13   */
14  public class FatalErrorPage extends AbstractWComponent implements ErrorPage {
15  
16  	/**
17  	 * Indicates whether additional developer details should be included on the error page.
18  	 */
19  	private final boolean developerFriendly;
20  
21  	/**
22  	 * The unhandled error which caused this component to display.
23  	 */
24  	private final Throwable error;
25  
26  	/**
27  	 * Creates a FatalErrorPage.
28  	 *
29  	 * @param developerFriendly if true, additional technical details will be included.
30  	 * @param error the unhandled error.
31  	 */
32  	public FatalErrorPage(final boolean developerFriendly, final Throwable error) {
33  		this.developerFriendly = developerFriendly;
34  		this.error = error;
35  	}
36  
37  	/**
38  	 * @return a mapped version of the {@link #error}'s message.
39  	 */
40  	protected String getMessage() {
41  		SystemFailureMapper mapper = Factory.newInstance(SystemFailureMapper.class);
42  		String desc = I18nUtilities.format(null, mapper.toMessage(error));
43  		return desc;
44  	}
45  
46  	/**
47  	 * Renders this FatalErrorPage.
48  	 *
49  	 * @param renderContext the RenderContext to send the rendered output to.
50  	 */
51  	@Override
52  	protected void paintComponent(final RenderContext renderContext) {
53  		if (renderContext instanceof WebXmlRenderContext) {
54  			PrintWriter writer = ((WebXmlRenderContext) renderContext).getWriter();
55  			writer.println(getMessage());
56  
57  			if (developerFriendly) {
58  				writer.println("<pre style=\"background: lightgrey\">");
59  				writer.println("Additional details for the developer:");
60  				writer.println();
61  				error.printStackTrace(writer);
62  				writer.println("</pre>");
63  			}
64  		}
65  	}
66  }
