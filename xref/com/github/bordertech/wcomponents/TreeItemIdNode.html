1   package com.github.bordertech.wcomponents;
2   
3   import java.io.Serializable;
4   import java.util.ArrayList;
5   import java.util.Collections;
6   import java.util.List;
7   
8   /**
9    * A node that holds a tree item id. Used for custom tree item structures in {@link WTree}.
10   *
11   * @author Jonathan Austin
12   * @since 1.1.0
13   */
14  public class TreeItemIdNode implements Serializable {
15  
16  	/**
17  	 * The tree item id.
18  	 *
19  	 */
20  	private final String itemId;
21  	/**
22  	 * The list of this nodes children.
23  	 */
24  	private List<TreeItemIdNode> children;
25  	/**
26  	 * True if this node has children. Allows it be set true with out actually loading the children.
27  	 */
28  	private boolean hasChildren;
29  
30  	/**
31  	 * @param itemId the tree item id
32  	 */
33  	public TreeItemIdNode(final String itemId) {
34  		this.itemId = itemId;
35  	}
36  
37  	/**
38  	 *
39  	 * @return the tree item id
40  	 */
41  	public String getItemId() {
42  		return itemId;
43  	}
44  
45  	/**
46  	 * @param node the child node to add
47  	 */
48  	public void addChild(final TreeItemIdNode node) {
49  		if (children == null) {
50  			children = new ArrayList<>();
51  		}
52  		children.add(node);
53  		hasChildren = true;
54  	}
55  
56  	/**
57  	 * @return the list of child nodes
58  	 */
59  	public List<TreeItemIdNode> getChildren() {
60  		if (children == null) {
61  			return Collections.EMPTY_LIST;
62  		} else {
63  			return Collections.unmodifiableList(children);
64  		}
65  	}
66  
67  	/**
68  	 * @return true if this node has children
69  	 */
70  	public boolean hasChildren() {
71  		return hasChildren;
72  	}
73  
74  	/**
75  	 *
76  	 * @param hasChildren true if this node has children
77  	 */
78  	public void setHasChildren(final boolean hasChildren) {
79  		this.hasChildren = hasChildren;
80  		if (!hasChildren) {
81  			// Clear children
82  			children = null;
83  		}
84  	}
85  
86  }
