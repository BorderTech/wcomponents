1   package com.github.bordertech.wcomponents;
2   
3   import com.github.bordertech.wcomponents.util.InternalMessages;
4   import java.text.MessageFormat;
5   
6   /**
7    * <p>
8    * This component is a specialised version of a {@link WButton} that provides additional client-side functionality
9    * commonly associated with a "cancel" button.
10   * </p>
11   * <p>
12   * When a user presses the button, it displays a confirmation prompt before posting the form to the server.
13   * </p>
14   *
15   * <pre>
16   * WConfirmationButton button = new WConfirmationButton(&quot;Delete everything&quot;);
17   * button.setMessage(&quot;Are you really sure you want to delete everything?&quot;);
18   *
19   * // Set an action to run when the button is clicked.
20   * button.setAction(new Action()
21   * {
22   *     public void execute(ActionEvent event)
23   *     {
24   *         // (Code to delete everything goes here)
25   *     }
26   * });
27   * </pre>
28   *
29   * @author Martin Shevchenko
30   * @since 1.0.0
31   */
32  public class WConfirmationButton extends WButton {
33  
34  	/**
35  	 * Creates an empty confirmation button. An image or text must be provided separately.
36  	 */
37  	public WConfirmationButton() {
38  		// Default confirmation message
39  		setMessage(InternalMessages.DEFAULT_CONFIRMATION_PROMPT);
40  	}
41  
42  	/**
43  	 * Constructor. Sets the button text.
44  	 *
45  	 * @param text the button text
46  	 */
47  	public WConfirmationButton(final String text) {
48  		super(text);
49  		// Default confirmation message
50  		setMessage(InternalMessages.DEFAULT_CONFIRMATION_PROMPT);
51  	}
52  
53  	/**
54  	 * Constructor. Set the button text and the accesskey. The accesskey in combination with the alt key will activate
55  	 * the button.
56  	 *
57  	 * @param text the button text, using {@link MessageFormat} syntax.
58  	 * @param accessKey The access key.
59  	 */
60  	public WConfirmationButton(final String text, final char accessKey) {
61  		super(text);
62  		// Access key
63  		setAccessKey(accessKey);
64  		// Default confirmation message
65  		setMessage(InternalMessages.DEFAULT_CONFIRMATION_PROMPT);
66  	}
67  
68  }
