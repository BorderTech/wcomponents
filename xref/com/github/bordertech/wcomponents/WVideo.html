56  	/**
57  	 * This is used to indicate which playback controls to display for the video.
58  	 *
59  	 * <p>
60  	 * <strong>Note:</strong>
61  	 * Advancements in video support in clients since this API was first implemented means that most of this is now
62  	 * redundant. Under most circumstances the UI will display their native video controls. Where a particular WVideo
63  	 * does not have any source which is able to be played by the client then links to all sources will be provided.
64  	 * This enum is not worthless as the values NONE and PLAY_PAUSE are used to turn off native video controls in the
65  	 * client. The value NONE however causes major problems and is incompatible with autoplay for a11y reasons so it
66  	 * basically makes the media worthless. This enum may be replaced in the future with a simple boolean to trigger
67  	 * native controls or play/pause only (see https://github.com/BorderTech/wcomponents/issues/503).
68  	 * </p>
69  	 */
70  	public enum Controls {
71  		/**
72  		 * Do not display any controls: not recommended. May be incompatible with any of {@link #isAutoplay()} == true,
73  		 * {@link #isMuted()} == true or {@link #isLoop()} == true. If this value is set then the WVideo control
74  		 * <strong>MAY NOT WORK AT ALL</strong>.
75  		 * @deprecated since 1.1.1 as this is incompatible with WCAG requirements.
76  		 */
77  		NONE,
78  		/**
79  		 * Display all controls.
80  		 * @deprecated since 1.1.1 as themes use native video controls.
81  		 */
82  		ALL,
83  		/**
84  		 * A combined play/pause button.
85  		 */
86  		PLAY_PAUSE,
87  		/**
88  		 * Displays the "default" set of controls for the theme.
89  		 * @deprecated since 1.1.1 as themes use native video controls.
90  		 */
91  		DEFAULT,
92  		/**
93  		 * Displays the client's native set of controls.
94  		 */
95  		NATIVE
96  	}

387 	 * Creates dynamic URLs that the video clips can be loaded from. In fact the URL points to the main application
388 	 * servlet, but includes a non-null for the parameter associated with this WComponent (ie, its label). The
389 	 * handleRequest method below detects this when the browser requests a file.
390 	 *
391 	 * @return the urls to load the video files from, or null if there are no clips defined.
392 	 */
393 	public String[] getVideoUrls() {
394 		Video[] video = getVideo();
395 
396 		if (video == null || video.length == 0) {
397 			return null;
398 		}
399 
400 		String[] urls = new String[video.length];
401 
402 		// this variable needs to be set in the portlet environment.
403 		String url = getEnvironment().getWServletPath();
404 		Map<String, String> parameters = getBaseParameterMap();
405 
406 		for (int i = 0; i < urls.length; i++) {
407 			parameters.put(VIDEO_INDEX_REQUEST_PARAM_KEY, String.valueOf(i));
408 			urls[i] = WebUtilities.getPath(url, parameters, true);
409 		}
410 
411 		return urls;
412 	}

463 	/**
464 	 * Retrieves the base parameter map for serving content (videos + tracks).
465 	 *
466 	 * @return the base map for serving content.
467 	 */
468 	private Map<String, String> getBaseParameterMap() {
469 		Environment env = getEnvironment();
470 		Map<String, String> parameters = env.getHiddenParameters();
471 		parameters.put(Environment.TARGET_ID, getTargetId());
472 
473 		if (Util.empty(getCacheKey())) {
474 			// Add some randomness to the URL to prevent caching
475 			String random = WebUtilities.generateRandom();
476 			parameters.put(Environment.UNIQUE_RANDOM_PARAM, random);
477 		} else {
478 			// Remove step counter as not required for cached content
479 			parameters.remove(Environment.STEP_VARIABLE);
480 			parameters.remove(Environment.SESSION_TOKEN_VARIABLE);
481 			// Add the cache key
482 			parameters.put(Environment.CONTENT_CACHE_KEY, getCacheKey());
483 		}
484 
485 		return parameters;
486 	}

503 	/**
504 	 * When an video element is rendered to the client, the browser will make a second request to get the video content.
505 	 * The handleRequest method has been overridden to detect whether the request is the "content fetch" request by
506 	 * looking for the parameter that we encode in the content url.
507 	 *
508 	 * @param request the request being responded to.
509 	 */
510 	@Override
511 	public void handleRequest(final Request request) {
512 		super.handleRequest(request);
513 
514 
515 		String targ = request.getParameter(Environment.TARGET_ID);
516 		boolean contentReqested = (targ != null && targ.equals(getTargetId()));
517 
518 		if (contentReqested && request.getParameter(POSTER_REQUEST_PARAM_KEY) != null) {
519 			handlePosterRequest();
520 		}
521 
522 		if (isDisabled()) {
523 			return;
524 		}
525 
526 		if (contentReqested) {
527 			if (request.getParameter(VIDEO_INDEX_REQUEST_PARAM_KEY) != null) {
528 				handleVideoRequest(request);
529 			} else if (request.getParameter(TRACK_INDEX_REQUEST_PARAM_KEY) != null) {
530 				handleTrackRequest(request);
531 			}
532 		}
533 	}

535 	/**
536 	 * Handles a request for the poster.
537 	 */
538 	private void handlePosterRequest() {
539 		Image poster = getComponentModel().poster;
540 
541 		if (poster != null) {
542 			ContentEscape escape = new ContentEscape(poster);
543 			escape.setCacheable(!Util.empty(getCacheKey()));
544 			throw escape;
545 		} else {
546 			LOG.warn("Client requested non-existant poster");
547 		}
548 	}

550 	/**
551 	 * Handles a request for a video.
552 	 *
553 	 * @param request the request being responded to.
554 	 */
555 	private void handleVideoRequest(final Request request) {
556 		String videoRequested = request.getParameter(VIDEO_INDEX_REQUEST_PARAM_KEY);
557 		int videoFileIndex = 0;
558 
559 		try {
560 			videoFileIndex = Integer.parseInt(videoRequested);
561 		} catch (NumberFormatException e) {
562 			LOG.error("Failed to parse video index: " + videoFileIndex);
563 		}
564 
565 		Video[] video = getVideo();
566 
567 		if (video != null && videoFileIndex >= 0 && videoFileIndex < video.length) {
568 			ContentEscape escape = new ContentEscape(video[videoFileIndex]);
569 			escape.setCacheable(!Util.empty(getCacheKey()));
570 			throw escape;
571 		} else {
572 			LOG.warn("Client requested invalid video clip: " + videoFileIndex);
573 		}
574 	}
575 
576 	/**
577 	 * Handles a request for an auxillary track.
578 	 *
579 	 * @param request the request being responded to.
580 	 */
581 	private void handleTrackRequest(final Request request) {
582 		String trackRequested = request.getParameter(TRACK_INDEX_REQUEST_PARAM_KEY);
583 		int trackIndex = 0;
584 
585 		try {
586 			trackIndex = Integer.parseInt(trackRequested);
587 		} catch (NumberFormatException e) {
588 			LOG.error("Failed to parse track index: " + trackIndex);
589 		}
590 
591 		Track[] tracks = getTracks();
592 
593 		if (tracks != null && trackIndex >= 0 && trackIndex < tracks.length) {
594 			ContentEscape escape = new ContentEscape(tracks[trackIndex]);
595 			escape.setCacheable(!Util.empty(getCacheKey()));
596 			throw escape;
597 		} else {
598 			LOG.warn("Client requested invalid track: " + trackIndex);
599 		}
600 	}
