113 	/**
114 	 * {@inheritDoc}
115 	 */
116 	@Override
117 	public String getNamingContextId() {
118 		boolean append = isAppendID();
119 
120 		if (!append) {
121 			// Check if this is the top level name context
122 			NamingContextable top = WebUtilities.getParentNamingContext(this);
123 			if (top != null) {
124 				// Not top context, so always append
125 				append = true;
126 			}
127 		}
128 
129 		if (append) {
130 			return getId();
131 		} else {
132 			return "";
133 		}
134 	}

193 	/**
194 	 * Add a custom javaScript resource that is to be loaded and used by the application.
195 	 *
196 	 * @param resource the JavaScript resource
197 	 */
198 	public void addJsResource(final ApplicationResource resource) {
199 		WApplicationModel model = getOrCreateComponentModel();
200 		if (model.jsResources == null) {
201 			model.jsResources = new ArrayList<>();
202 		} else if (model.jsResources.contains(resource)) {
203 			return;
204 		}
205 		model.jsResources.add(resource);
206 		MemoryUtil.checkSize(model.jsResources.size(), this.getClass().getSimpleName());
207 	}

410 	/**
411 	 * Holds the details of CSS and JS custom resources.
412 	 */
413 	public static class ApplicationResource implements Serializable {
414 
415 		/**
416 		 * Unique resource id.
417 		 */
418 		private final String resourceId;
419 
420 		/**
421 		 * Resource URL.
422 		 */
423 		private final String url;
424 
425 		/**
426 		 * Internal resource.
427 		 */
428 		private final InternalResource resource;
429 
430 		/**
431 		 * @param url URL to a resource
432 		 */
433 		public ApplicationResource(final String url) {
434 			if (Util.empty(url)) {
435 				throw new IllegalArgumentException("A URL must be provided.");
436 			}
437 			this.resourceId = "url:" + url;
438 			this.url = url;
439 			this.resource = null;
440 		}
441 
442 		/**
443 		 * @param resource internal resource
444 		 */
445 		public ApplicationResource(final InternalResource resource) {
446 			if (resource == null) {
447 				throw new IllegalArgumentException("A resource must be provided.");
448 			}
449 			this.resourceId = "file:" + resource.getResourceName();
450 			this.url = null;
451 			this.resource = resource;
452 		}
453 
454 		/**
455 		 * @return the resource id
456 		 */
457 		public String getResourceId() {
458 			return resourceId;
459 		}
460 
461 		/**
462 		 * @return the resource URL
463 		 */
464 		public String getUrl() {
465 			return url;
466 		}
467 
468 		/**
469 		 *
470 		 * @return the internal resource
471 		 */
472 		public InternalResource getResource() {
473 			return resource;
474 		}
475 
476 		/**
477 		 * @return the target URL of the resource
478 		 */
479 		public String getTargetUrl() {
480 			return url == null ? resource.getTargetUrl() : url;
481 		}
482 
483 		/**
484 		 * {@inheritDoc}
485 		 */
486 		@Override
487 		public boolean equals(final Object obj) {
488 			return obj instanceof ApplicationResource && resourceId.equals(((ApplicationResource) obj).getResourceId());
489 		}
490 
491 		/**
492 		 * {@inheritDoc}
493 		 */
494 		@Override
495 		public int hashCode() {
496 			return resourceId.hashCode();
497 		}
498 	}
