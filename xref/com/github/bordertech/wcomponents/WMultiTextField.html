14  /**
15   * <p>
16   * The WMultiTextField component allows multiple text input elements to be generated dynamically on the client without
17   * requiring the page to be reloaded. This component takes its appearance and attributes from a regular text input but
18   * allows for one or more items of text to be entered via the add link adjacent to the right of the text input.
19   * </p>
20   * <p>
21   * This component is useful in instances where the user needs to enter one or more text items into the interfaces for a
22   * particular field. For instance, the user may have one or more aliases that need to be entered into the system.
23   * </p>
24   * <p>
25   * The following attributes can be set on WMultiTextField:
26   * </p>
27   * <ul>
28   * <li>MaxInputs: The maximum number of text inputs the user can add to the component. Client-side functionality will
29   * stop users adding more than the allowable number of inputs via the UI. This class chops off any excess inputs if an
30   * attempt is made to add them programmatically.</li>
31   * <li>MaxLength: The maximum length of the text fields rendered to allow user text entry.</li>
32   * <li>Columns: The size of the text fields rendered to allow user text entry.</li>
33   * </ul>
34   *
35   * @author Christina Harris
36   * @author Jonathan Austin
37   * @since 1.0.0
38   */
39  public class WMultiTextField extends AbstractInput implements AjaxTrigger, AjaxTarget, SubordinateTarget, MultiInputComponent {

62  	/**
63  	 * {@inheritDoc}
64  	 */
65  	@Override
66  	public String[] getValue() {
67  		Object data = getData();
68  		if (data == null) {
69  			return null;
70  		}
71  
72  		String[] array = null;
73  
74  		// Array data
75  		if (data instanceof String[]) {
76  			array = (String[]) data;
77  		} else if (data instanceof List) { // List data
78  			List<?> list = (List<?>) data;
79  			array = new String[list.size()];
80  
81  			for (int i = 0; i < list.size(); i++) {
82  				Object item = list.get(i);
83  				array[i] = item == null ? "" : item.toString();
84  			}
85  		} else { // Object
86  			array = new String[]{data.toString()};
87  		}
88  
89  		return removeEmptyStrings(array);
90  	}

92  	/**
93  	 * {@inheritDoc}
94  	 */
95  	@Override
96  	public void setData(final Object data) {
97  		String[] inputs = (String[]) data;
98  		String[] parsed = removeEmptyStrings(inputs);
99  		int maxInputs = getMaxInputs();
100 
101 		if (maxInputs > 0 && parsed != null && parsed.length > maxInputs) {
102 			LOG.warn("Attempt made to set [" + parsed.length
103 					+ "] inputs on WMultiTextField but the maximum allowable is [" + maxInputs + "]. Only the first ["
104 					+ maxInputs + "] inputs will be included.");
105 
106 			String[] choppedInputs = new String[maxInputs];
107 			System.arraycopy(parsed, 0, choppedInputs, 0, maxInputs);
108 			super.setData(choppedInputs);
109 		} else {
110 			super.setData(parsed);
111 		}
112 	}

239 	 * The string is a comma seperated list of the string inputs.
240 	 *
241 	 * @return A string concatenation of the string inputs.
242 	 */
243 	@Override
244 	public String getValueAsString() {
245 		String result = null;
246 
247 		String[] inputs = getValue();
248 
249 		if (inputs != null && inputs.length > 0) {
250 			StringBuffer stringValues = new StringBuffer();
251 
252 			for (int i = 0; i < inputs.length; i++) {
253 				if (i > 0) {
254 					stringValues.append(", ");
255 				}
256 
257 				stringValues.append(inputs[i]);
258 			}
259 
260 			result = stringValues.toString();
261 
262 		}
263 
264 		return result;
265 	}

268 	 * Set the inputs based on the incoming request. The text input values are set as an array of strings on the
269 	 * parameter with this name {@link #getName()}. Any empty strings will be ignored.

271 	 * @param request the current request.
272 	 * @return true if the inputs have changed, otherwise return false
273 	 */
274 	@Override
275 	protected boolean doHandleRequest(final Request request) {
276 		String[] values = getRequestValue(request);
277 		String[] current = getValue();
278 
279 		boolean changed = !Arrays.equals(values, current);
280 
281 		if (changed) {
282 			setData(values);
283 		}
284 
285 		return changed;
286 	}

301 	/**
302 	 * Helper that removes empty/null string from the <code>original</code> string array.
303 	 * <p>
304 	 * Will treat an empty array the same as null.
305 	 * </p>
306 	 *
307 	 * @param originals The string array from which the null/empty strings should be removed from.
308 	 * @return Array of non empty strings from the <code>original</code> string array.
309 	 */
310 	private String[] removeEmptyStrings(final String[] originals) {
311 		if (originals == null) {
312 			return null;
313 		} else {
314 			List<String> parsed = new ArrayList<>();
315 
316 			for (String original : originals) {
317 				if (original != null && original.length() > 0) {
318 					parsed.add(original);
319 				}
320 			}
321 			// Treat empty the same as null
322 			return parsed.isEmpty() ? null : parsed.toArray(new String[parsed.size()]);
323 		}
324 	}

326 	/**
327 	 * Override validateComponent to perform further validation.
328 	 *
329 	 * @param diags the list into which any validation diagnostics are added.
330 	 */
331 	@Override
332 	protected void validateComponent(final List<Diagnostic> diags) {
333 		super.validateComponent(diags);
334 
335 		String[] values = getValue();
336 
337 		if (values != null) {
338 			int max = getMaxLength();
339 			int min = getMinLength();
340 			Pattern pattern = getComponentModel().pattern;
341 
342 			for (String value : values) {
343 				boolean invalid = false;
344 
345 				if (Util.empty(value)) {
346 					continue;
347 				}
348 
349 				// Minimum Length
350 				if (min > 0 && value.length() < min) {
351 					diags.add(createErrorDiagnostic(
352 							InternalMessages.DEFAULT_VALIDATION_ERROR_MIN_LENGTH, this,
353 							String.valueOf(min)));
354 					invalid = true;
355 				}
356 
357 				// Maximum Length
358 				if (max > 0 && value.length() > max) {
359 					diags.add(createErrorDiagnostic(
360 							InternalMessages.DEFAULT_VALIDATION_ERROR_MAX_LENGTH, this,
361 							String.valueOf(max)));
362 					invalid = true;
363 				}
364 
365 				// Pattern
366 				if (pattern != null) {
367 					Matcher matcher = pattern.matcher(value);
368 					if (!matcher.matches()) {
369 						diags.add(createErrorDiagnostic(
370 								InternalMessages.DEFAULT_VALIDATION_ERROR_INVALID_PATTERN, this));
371 						invalid = true;
372 					}
373 				}
374 
375 				// Only report the first invalid item
376 				if (invalid) {
377 					break;
378 				}
379 			}
380 		}
381 	}
