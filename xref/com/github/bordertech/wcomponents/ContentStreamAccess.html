3   import java.io.IOException;
4   import java.io.InputStream;

6   /**
7    * <p>
8    * This interface enables access to arbitrary document content such as a PDF. It could be fetched from the database or
9    * it could be generated on the fly.</p>
10   *
11   * <p>
12   * This extension of {@link ContentAccess} should be used to serve up large binary content (e.g. video), rather than
13   * requiring the entire byte array to be held in memory.</p>
14   *
15   * @author Yiannis Paschalidis
16   * @since 1.0.0
17   */
18  public interface ContentStreamAccess extends ContentAccess {
19  
20  	/**
21  	 * Retrieves the stream that make up the document content.
22  	 *
23  	 * @return the stream containing the document content.
24  	 * @throws IOException if there is an error retrieving the content.
25  	 */
26  	InputStream getStream() throws IOException;
27  }
