41  	 *
42  	 * @param renderContext the renderContext to send output to.
43  	 */
44  	@Override
45  	protected void afterPaint(final RenderContext renderContext) {
46  		super.afterPaint(renderContext);
47  
48  		if (this.isPressed() && renderContext instanceof WebXmlRenderContext) {
49  			PrintWriter writer = ((WebXmlRenderContext) renderContext).getWriter();
50  
51  			StringBuffer temp = dumpAll();
52  			writer.println("<br/><br/>");
53  			writer.println(temp);
54  		}
55  	}

57  	/**
58  	 * Dumps all the profiling information in textual format to a StringBuffer.
59  	 *
60  	 * @return the profiling information in textual format.
61  	 */
62  	public StringBuffer dumpAll() {
63  		WComponent root = WebUtilities.getTop(this);
64  
65  		Map<String, GroupData> compTallyByClass = new TreeMap<>();
66  
67  		GroupData compDataOverall = new GroupData();
68  
69  		StringBuffer out = new StringBuffer();
70  
71  		String className = root.getClass().getName();
72  
73  		out.append("<strong>The root of the WComponent tree is:</strong><br/>").append(className).append(
74  				"<br/>");
75  
76  		tally(root, compTallyByClass, compDataOverall, out);
77  
78  		out.append("<strong>WComponent usage overall:</strong><br/> ");
79  		out.append(compDataOverall.total).append(" WComponent(s) in the WComponent tree.<br/>");
80  
81  		if (compDataOverall.total > 0) {
82  			out.append("<strong>WComponent usage by class:</strong><br/>");
83  
84  			for (Map.Entry<String, GroupData> entry : compTallyByClass.entrySet()) {
85  				className = entry.getKey();
86  				GroupData dataForClass = entry.getValue();
87  				out.append(' ').append(dataForClass.total).append("  ").append(className).append(
88  						"<br/>");
89  			}
90  
91  			out.append("<br/><hr/>");
92  		}
93  
94  		processUic(out);
95  
96  		return out;
97  	}

99  	/**
100 	 * @param component the component
101 	 * @param map the group data
102 	 * @param compDataOverall the overall group data
103 	 * @param out the buffer
104 	 */
105 	private void tally(final WComponent component, final Map<String, GroupData> map,
106 			final GroupData compDataOverall, final StringBuffer out) {
107 		String classKey = component.getClass().getName();
108 		GroupData dataForClass = map.get(classKey);
109 
110 		if (dataForClass == null) {
111 			dataForClass = new GroupData();
112 			map.put(classKey, dataForClass);
113 		}
114 
115 		dataForClass.total++;
116 		compDataOverall.total++;
117 
118 		if (component instanceof Container) {
119 			int count = ((Container) component).getChildCount();
120 
121 			for (int i = 0; i < count; i++) {
122 				WComponent child = ((Container) component).getChildAt(i);
123 				tally(child, map, compDataOverall, out);
124 			}
125 		}
126 	}

128 	/**
129 	 * @param out the current buffer
130 	 */
131 	private void processUic(final StringBuffer out) {
132 		Set components = UIContextHolder.getCurrent().getComponents();
133 
134 		// Variables that will contain the stats collected.
135 		GroupData dataOverall = new GroupData();
136 		Map<String, GroupData> tallyByClass = new TreeMap<>();
137 		Map<String, ProfileData> profileByClass = new TreeMap<>();
138 		Set<WComponent> topComponents = new HashSet<>();
139 
140 		out.append("\n<strong>UIContext details:\n</strong>"
141 				+ "\n<br/>\n<table border=\"1\">\n<tr>\n<th>Class Name</th>\n"
142 				+ "\n<th>ID</th>\n"
143 				+ "\n<th>Is Serialisable</th>\n"
144 				+ "\n<th>Size</th></tr>\n");
145 
146 		for (Iterator iter = components.iterator(); iter.hasNext();) {
147 			WComponent comp = (WComponent) iter.next();
148 			// processData(comp);
149 
150 			String className = comp.getClass().getName();
151 
152 			ProfileData profileForClass = profileByClass.get(className);
153 
154 			if (!comp.isDefaultState()) {
155 				ProfileDetailsData profile = new ProfileDetailsData();
156 				profile.id = comp.getId();
157 
158 				profile.isSerializable = true;
159 				profile.size = 0;
160 
161 				out.append("\n<tr>\n<td>");
162 				out.append(className);
163 				out.append("</td>\n\n<td>");
164 				out.append(profile.id);
165 				out.append("</td>\n\n<td>");
166 				out.append(profile.isSerializable);
167 				out.append("</td>\n\n<td>");
168 				out.append(profile.size);
169 				out.append("</td></tr>\n");
170 
171 				if (profileForClass == null) {
172 					profileForClass = new ProfileData();
173 					profileByClass.put(className, profileForClass);
174 				}
175 
176 				profileForClass.profiles.add(profile);
177 			}
178 
179 			GroupData dataForClass = tallyByClass.get(className);
180 
181 			if (dataForClass == null) {
182 				dataForClass = new GroupData();
183 				tallyByClass.put(className, dataForClass);
184 			}
185 
186 			dataForClass.total++;
187 			dataOverall.total++;
188 
189 			if (comp.isDefaultState()) {
190 				dataForClass.unnecessaryCount++;
191 				dataOverall.unnecessaryCount++;
192 			}
193 
194 			WComponent top = WebUtilities.getTop(comp);
195 			topComponents.add(top);
196 		}
197 
198 		out.append("\n</table>\n<br/><hr/>"
199 				+ "<strong>WComponent session usage overall:</strong>\n<br/> ");
200 
201 		out.append(dataOverall.total).append(" WComponent(s) storing data in the session.\n<br/>");
202 
203 		if (dataOverall.total > 0) {
204 			out.append('[')
205 					.append(dataOverall.unnecessaryCount)
206 					.append("] WComponent(s) in default state. WComponents in default state do not need to store data in the session.<br/>"
207 							+ " Therefore, only ").append(
208 							dataOverall.total - dataOverall.unnecessaryCount)
209 					.append(" WComponent(s) actually need(s) to store data in the session.<br/>");
210 
211 			out.append("<strong>WComponent session usage by class:</strong><br/>");
212 
213 			for (Map.Entry<String, GroupData> entry : tallyByClass.entrySet()) {
214 				String className = entry.getKey();
215 				GroupData dataForClass = entry.getValue();
216 
217 				out.append(' ').append(dataForClass.total).append(" [").append(
218 						dataForClass.unnecessaryCount).append("] ")
219 						.append(className).append("<br/>");
220 			}
221 
222 			out.append("Found ").append(topComponents.size()).append(" top component(s).<br/><hr/>");
223 		}
224 	}
