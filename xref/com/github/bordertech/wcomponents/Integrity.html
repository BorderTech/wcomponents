1   package com.github.bordertech.wcomponents;
2   
3   import com.github.bordertech.wcomponents.util.ConfigurationProperties;
4   import org.apache.commons.logging.LogFactory;
5   
6   /**
7    * <p>
8    * A convenience class to raise integrity issues. An integrity issue is usually issued on misconfiguration of a
9    * WComponent.</p>
10   *
11   * <p>
12   * If the <code>sfp.web.integrity.terminate.mode</code> parameter is set, an {@link IntegrityException} will be thrown,
13   * otherwise a warning message will be logged.<p>
14   *
15   * @author Martin Shevchenko
16   * @since 1.0.0
17   */
18  public final class Integrity {
19  
20  	/**
21  	 * Prevent instantiation of this class.
22  	 */
23  	private Integrity() {
24  	}
25  
26  	/**
27  	 * Raises an integrity issue.
28  	 *
29  	 * @param comp the component to issue the exception for.
30  	 * @param message the integrity message to issue.
31  	 */
32  	public static void issue(final WComponent comp, final String message) {
33  		String debugMessage = message + ' ' + comp;
34  
35  		if (ConfigurationProperties.getIntegrityErrorMode()) {
36  			throw new IntegrityException(debugMessage);
37  		} else {
38  			LogFactory.getLog(Integrity.class).warn(debugMessage);
39  		}
40  	}
41  }
