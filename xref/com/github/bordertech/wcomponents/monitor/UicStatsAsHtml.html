12  /**
13   * A utility class which writes out UicStats statistics in HTML format.
14   *
15   * @author Martin Shevchenko
16   * @since 1.0.0
17   */
18  public final class UicStatsAsHtml {
19  
20  	/**
21  	 * No instance methods here.
22  	 */
23  	private UicStatsAsHtml() {
24  	}
25  
26  	/**
27  	 * Writes out the given statistics in HTML format.
28  	 *
29  	 * @param writer the writer to write to.
30  	 * @param stats the stats to write.
31  	 */
32  	public static void write(final PrintWriter writer, final UicStats stats) {
33  		writer.println("<dl>");
34  
35  		writer.print("<dt>Total root wcomponents found in UIC</dt>");
36  		writer.println("<dd>" + stats.getRootWCs().size() + "</dd>");
37  
38  		writer.print("<dt>Size of UIC (by serialization)</dt>");
39  		writer.println("<dd>" + stats.getOverallSerializedSize() + "</dd>");
40  
41  		writer.print("<dt>UI</dt>");
42  		writer.println("<dd>" + stats.getUI().getClass().getName() + "</dd>");
43  
44  		writer.println("</dl>");
45  
46  		for (Iterator<WComponent> it = stats.getWCsAnalysed(); it.hasNext();) {
47  			WComponent comp = it.next();
48  			Map<WComponent, UicStats.Stat> treeStats = stats.getWCTreeStats(comp);
49  
50  			writer.println("<br /><strong>Analysed component:</strong> " + comp);
51  			writer.println("<br /><strong>Number of components in tree:</strong> " + treeStats.size());
52  
53  			writeHeader(writer);
54  			writeProfileForTree(writer, treeStats);
55  			writeFooter(writer);
56  		}
57  	}
58  
59  	/**
60  	 * Writes the stats for a single component.
61  	 *
62  	 * @param writer the writer to write the stats to.
63  	 * @param treeStats the stats for the component.
64  	 */
65  	private static void writeProfileForTree(final PrintWriter writer,
66  			final Map<WComponent, UicStats.Stat> treeStats) {
67  		// Copy all the stats into a list so we can sort and cull.
68  		List<UicStats.Stat> statList = new ArrayList<>(treeStats.values());
69  
70  		Comparator<UicStats.Stat> comparator = new Comparator<UicStats.Stat>() {
71  			@Override
72  			public int compare(final UicStats.Stat stat1, final UicStats.Stat stat2) {
73  				if (stat1.getModelState() > stat2.getModelState()) {
74  					return -1;
75  				} else if (stat1.getModelState() < stat2.getModelState()) {
76  					return 1;
77  				} else {
78  					int diff = stat1.getClassName().compareTo(stat2.getClassName());
79  
80  					if (diff == 0) {
81  						diff = stat1.getName().compareTo(stat2.getName());
82  					}
83  
84  					return diff;
85  				}
86  			}
87  		};
88  
89  		Collections.sort(statList, comparator);
90  
91  		for (int i = 0; i < statList.size(); i++) {
92  			UicStats.Stat stat = statList.get(i);
93  			writeRow(writer, stat);
94  		}
95  	}
96  
97  	/**
98  	 * Writes the stats header HTML.
99  	 *
100 	 * @param writer the writer to write the header to
101 	 */
102 	private static void writeHeader(final PrintWriter writer) {
103 		writer.println("<table>");
104 		writer.println("<thead>");
105 		writer.print("<tr>");
106 		writer.print("<th>Class</th>");
107 		writer.print("<th>Model</th>");
108 		writer.print("<th>Size</th>");
109 		writer.print("<th>Ref.</th>");
110 		writer.print("<th>Name</th>");
111 		writer.print("<th>Comment</th>");
112 		writer.println("</tr>");
113 		writer.println("</thead>");
114 	}
115 
116 	/**
117 	 * Writes the stats footer HTML.
118 	 *
119 	 * @param writer the writer to write the footer to
120 	 */
121 	private static void writeFooter(final PrintWriter writer) {
122 		writer.println("</table>");
123 	}
124 
125 	/**
126 	 * Writes a row containing a single stat.
127 	 *
128 	 * @param writer the writer to write the row to.
129 	 * @param stat the stat to write.
130 	 */
131 	private static void writeRow(final PrintWriter writer, final UicStats.Stat stat) {
132 		writer.print("<tr>");
133 		writer.print("<td>" + stat.getClassName() + "</td>");
134 		writer.print("<td>" + stat.getModelStateAsString() + "</td>");
135 
136 		if (stat.getSerializedSize() > 0) {
137 			writer.print("<td>" + stat.getSerializedSize() + "</td>");
138 		} else {
139 			writer.print("<td>&#160;</td>");
140 		}
141 
142 		writer.print("<td>" + stat.getRef() + "</td>");
143 		writer.print("<td>" + stat.getName() + "</td>");
144 
145 		if (stat.getComment() == null) {
146 			writer.print("<td>&#160;</td>");
147 		} else {
148 			writer.print("<td>" + stat.getComment() + "</td>");
149 		}
150 
151 		writer.println("</tr>");
152 	}
153 }
