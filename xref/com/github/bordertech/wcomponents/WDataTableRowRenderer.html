84  	/**
85  	 * <p>
86  	 * The preparePaintComponent method has been overridden to ensure that expanded row renderers have been correctly
87  	 * initialised.</p>
88  	 *
89  	 * <p>
90  	 * Expanded row renderers are lazily instantiated and added to the shared structure as needed. This means for the
91  	 * first use of a renderer, it will not have been part of the WComponent tree, and would not have had its
92  	 * preparePaintComponent called. We therefore add the renderer to the tree here, and manually call its
93  	 * preparePaint.</p>
94  	 *
95  	 * @param request the Request being responded to.
96  	 */
97  	@Override
98  	protected void preparePaintComponent(final Request request) {
99  		super.preparePaintComponent(request);
100 
101 		Class<? extends WComponent> rowRendererClass = getRowRendererClass();
102 
103 		if (rowRendererClass != null && !expandedRenderers.containsKey(rowRendererClass)) {
104 			getExpandedTreeNodeRenderer(rowRendererClass).preparePaint(request);
105 		}
106 	}

108 	/**
109 	 * <p>
110 	 * This is called to lazily add expanded renderers as necessary. To save memory, only one instance of a renderer
111 	 * class is ever added to the row renderer instance. The RendererWrapper ensures that data binding occurs at the
112 	 * right time.</p>
113 	 *
114 	 * @param rendererClass the renderer class.
115 	 * @return the expanded renderer for the given row.
116 	 */
117 	public WComponent getExpandedTreeNodeRenderer(final Class<? extends WComponent> rendererClass) {
118 		if (rendererClass == null) {
119 			return null;
120 		}
121 
122 		// If we already have an additional renderer for this class, return it.
123 		// This is synchronized, as it updates the shared model using locking/unlocking, which is a bit dodgy.
124 		synchronized (expandedRenderers) {
125 			WComponent renderer = expandedRenderers.get(rendererClass);
126 
127 			if (renderer != null) {
128 				return renderer;
129 			}
130 
131 			// Not found, create a new instance of the given class
132 			renderer = new RendererWrapper(this, rendererClass, -1);
133 			expandedRenderers.put(rendererClass, renderer);
134 			setLocked(false);
135 			add(renderer);
136 			setLocked(true);
137 
138 			return renderer;
139 		}
140 	}

143 	 * @param uic the SubUIContext to determine the row index for
144 	 * @return the row index for the specified SubUIContext.
145 	 */
146 	private int getRowIndex(final SubUIContext uic) {
147 		// SubUIContext does not cater for pagination
148 		return (Integer) table.getRepeater().getBeanList().get(uic.getRowIndex());
149 	}

151 	/**
152 	 * Retrieves the renderer class for a given row.
153 	 *
154 	 * @return the renderer class for a given row (if it's an expanded row), otherwise null.
155 	 */
156 	private Class<? extends WComponent> getRowRendererClass() {
157 		UIContext uic = UIContextHolder.getCurrent();
158 
159 		if (table.getDataModel() instanceof TreeTableDataModel && uic instanceof SubUIContext) {
160 			int row = getRowIndex((SubUIContext) uic);
161 			TreeTableDataModel dataModel = (TreeTableDataModel) table.getDataModel();
162 
163 			TableTreeNode node = dataModel.getNodeAtLine(row);
164 			return node.getRendererClass();
165 		}
166 
167 		return null;
168 	}

170 	/**
171 	 * The renderer wrapper is responsible for ensuring that the renderer is only used when needed (ie. it is only
172 	 * involved in processing of certain rows), and ensuring that data is passed to / from the renderer when required.
173 	 */
174 	private static final class RendererWrapper extends WBeanContainer implements BeanProvider {
175 
176 		/**
177 		 * The table that this wrapper belongs to.
178 		 */
179 		private final WDataTableRowRenderer rowRenderer;
180 
181 		/**
182 		 * The actual renderer used to render the column/expanded content.
183 		 */
184 		private final WComponent renderer;
185 
186 		/**
187 		 * The index of the column which this wrapper renders, or -1 if this is an expended renderer.
188 		 */
189 		private final int columnIndex;
190 
191 		/**
192 		 * Creates a RendererWrapper.
193 		 *
194 		 * @param rowRenderer the row renderer.
195 		 * @param rendererClass the column/expanded content renderer class.
196 		 * @param columnIndex the index of the column renderer, or -1 if the renderer is for expanded content.
197 		 */
198 		private RendererWrapper(final WDataTableRowRenderer rowRenderer,
199 				final Class<? extends WComponent> rendererClass, final int columnIndex) {
200 			this.rowRenderer = rowRenderer;
201 			this.columnIndex = columnIndex;
202 			WComponent rendererComponent = null;
203 
204 			// Not found, create a new instance of the given class
205 			try {
206 				rendererComponent = rendererClass.newInstance();
207 
208 				if (rendererComponent instanceof BeanProviderBound) {
209 					((BeanProviderBound) rendererComponent).setBeanProvider(this);
210 				}
211 
212 				add(rendererComponent);
213 			} catch (Exception e) {
214 				LOG.error("Failed to instantiate renderer: " + rendererClass.getName(), e);
215 				setVisible(false);
216 			}
217 
218 			renderer = rendererComponent;
219 		}
220 
221 		/**
222 		 * Creates a RendererWrapper.
223 		 *
224 		 * @param rowRenderer the row renderer.
225 		 * @param renderer the column/expanded content renderer component.
226 		 * @param columnIndex the index of the column renderer, or -1 if the renderer is for expanded content.
227 		 */
228 		private RendererWrapper(final WDataTableRowRenderer rowRenderer, final WComponent renderer,
229 				final int columnIndex) {
230 			this.rowRenderer = rowRenderer;
231 			this.columnIndex = columnIndex;
232 			this.renderer = renderer;
233 
234 			if (renderer instanceof BeanProviderBound) {
235 				((BeanProviderBound) renderer).setBeanProvider(this);
236 			}
237 
238 			add(renderer);
239 		}

241 		/**
242 		 * Column Renderers are invisible for expanded rows with an expanded renderer set.
243 		 *
244 		 * @return true if this component is visible, false if invisible.
245 		 */
246 		@Override
247 		public boolean isVisible() {
248 			if (!super.isVisible()) {
249 				return false;
250 			}
251 
252 			Class<? extends WComponent> rendererClass = rowRenderer.getRowRendererClass();
253 
254 			if (columnIndex >= 0) {
255 				// column renderer
256 				return rendererClass == null;
257 			} else {
258 				// expanded content renderer
259 				return renderer != null && Util.equals(renderer.getClass(), rendererClass);
260 			}
261 		}

263 		/**
264 		 * Provides data to a component rendering a column.
265 		 *
266 		 * @param beanProviderBound the component rendering the column.
267 		 * @return a bean value for component that is rendering the specified row/column
268 		 */
269 		@Override
270 		public Object getBean(final BeanProviderBound beanProviderBound) {
271 			UIContext uic = UIContextHolder.getCurrent();
272 
273 			// Make sure we have the correct uic
274 			while (uic instanceof SubUIContext && !((SubUIContext) uic).isInContext(
275 					(WComponent) beanProviderBound)) {
276 				uic = ((SubUIContext) uic).getParentContext();
277 			}
278 
279 			if (!(uic instanceof SubUIContext)) {
280 				LOG.error("Unable to handle UIContext type: " + uic.getClass().getName());
281 				return null;
282 			}
283 
284 			int row = rowRenderer.getRowIndex((SubUIContext) uic);
285 
286 			if (columnIndex < 0) {
287 				// This is not a column renderer, it must be an additional row renderer
288 				TreeTableDataModel dataModel = (TreeTableDataModel) rowRenderer.table.getDataModel();
289 				return dataModel.getNodeAtLine(row).getData();
290 			} else {
291 				return rowRenderer.table.getDataModel().getValueAt(row, columnIndex);
292 			}
293 		}
294 
295 		/**
296 		 * Some renderers may not be bean provider bound, or not bean-aware. We need to make sure that the data is set
297 		 * correctly for these columns.
298 		 *
299 		 * @param request the request being responded to.
300 		 */
301 		@Override
302 		protected void preparePaintComponent(final Request request) {
303 			super.preparePaintComponent(request);
304 
305 			TableDataModel model = rowRenderer.table.getDataModel();
306 			UIContext uic = UIContextHolder.getCurrent();
307 			int row = rowRenderer.getRowIndex((SubUIContext) uic);
308 
309 			// Update input read-only status
310 			if (renderer instanceof Input) {
311 				((Input) renderer).setReadOnly(!model.isCellEditable(row, columnIndex));
312 			}
313 
314 			if (!(renderer instanceof BeanProviderBound)) {
315 				Object bean = null;
316 
317 				if (columnIndex >= 0) {
318 					bean = model.getValueAt(row, columnIndex);
319 				} else if (model instanceof TreeTableDataModel) {
320 					TreeTableDataModel treeModel = (TreeTableDataModel) model;
321 					bean = treeModel.getNodeAtLine(row).getData();
322 				}
323 
324 				if (renderer instanceof BeanBound) {
325 					((BeanBound) renderer).setBean(bean);
326 				} else if (renderer instanceof DataBound) {
327 					((DataBound) renderer).setData(bean);
328 				}
329 			}
330 		}
331 	}

333 	/**
334 	 * Adds a column to the renderer. This method is called by {@link WDataTable} to keep the renderer's and table's
335 	 * columns in sync.
336 	 *
337 	 * @param column the column to add.
338 	 * @param columnIndex the index of the column.
339 	 */
340 	void addColumn(final WTableColumn column, final int columnIndex) {
341 		WComponent renderer = column.getRenderer();
342 
343 		if (renderer == null) {
344 			Class<? extends WComponent> rendererClass = column.getRendererClass();
345 
346 			if (!(BeanProviderBound.class.isAssignableFrom(rendererClass))
347 					&& !(BeanBound.class.isAssignableFrom(rendererClass))
348 					&& !(DataBound.class.isAssignableFrom(rendererClass))) {
349 				throw new IllegalArgumentException(
350 						"Column renderers must be BeanProvider-, Bean- or Data-Bound");
351 			}
352 
353 			add(new RendererWrapper(this, rendererClass, columnIndex));
354 		} else {
355 			if (!(renderer instanceof BeanProviderBound)
356 					&& !(renderer instanceof BeanBound)
357 					&& !(renderer instanceof DataBound)) {
358 				throw new IllegalArgumentException(
359 						"Column renderers must be BeanProvider-, Bean- or Data-Bound");
360 			}
361 
362 			add(new RendererWrapper(this, renderer, columnIndex));
363 		}
364 	}
