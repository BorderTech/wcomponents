1   package com.github.bordertech.wcomponents.file;
2   
3   import java.io.IOException;
4   import java.io.InputStream;
5   import org.apache.commons.fileupload.FileItem;
6   
7   /**
8    * A {@link File} implementation that is backed by a {@link FileItem}.
9    *
10   * @author Christina Harris
11   * @since 1.0.0
12   */
13  public class FileItemWrap implements File {
14  
15  	/**
16  	 * The file item backing this instance.
17  	 */
18  	private final FileItem backing;
19  
20  	/**
21  	 * Constructor. Set the backing FileItem.
22  	 *
23  	 * @param item The FileItem that backs this DefaultFileImpl.
24  	 */
25  	public FileItemWrap(final FileItem item) {
26  		backing = item;
27  	}
28  
29  	/**
30  	 * The name of the file as supplied by the client. Depending on the client this may or may not include the full path
31  	 * to the file.
32  	 *
33  	 * @return The name of the file.
34  	 */
35  	@Override
36  	public String getName() {
37  		return backing.getName();
38  	}
39  
40  	/**
41  	 * @return the byte content of the file.
42  	 */
43  	@Override
44  	public byte[] getBytes() {
45  		return backing.get();
46  	}
47  
48  	/**
49  	 * Returns the name of the file.
50  	 *
51  	 * @return file name.
52  	 */
53  	@Override
54  	public String getDescription() {
55  		return backing.getName();
56  	}
57  
58  	/**
59  	 * Returns the content type passed by the browser or <code>null</code> if not defined.
60  	 *
61  	 * @return The content type passed by the browser or <code>null</code> if not defined.
62  	 */
63  	@Override
64  	public String getMimeType() {
65  		String contentType = backing.getContentType();
66  
67  		// IE6 (and apparently IE7) are broken and send the MIME type "image/pjpeg"
68  		// regardless of whether the image is a progressive jpeg or not. Sending the
69  		// incorrect MIME type back can cause display issues. It is always safe to
70  		// change it to just be "image/jpeg".
71  		if ("image/pjpeg".equals(contentType)) {
72  			contentType = "image/jpeg";
73  		}
74  
75  		return contentType;
76  	}
77  
78  	/**
79  	 * Returns an {@link java.io.InputStream InputStream} that can be used to retrieve the contents of the file.
80  	 *
81  	 * @return An {@link java.io.InputStream InputStream} that can be used to retrieve the contents of the file.
82  	 * @throws IOException unable to access file.
83  	 */
84  	@Override
85  	public InputStream getInputStream() throws IOException {
86  		return backing.getInputStream();
87  	}
88  
89  	/**
90  	 * Returns the size of the file item.
91  	 *
92  	 * @return The size of the file item, in bytes.
93  	 */
94  	@Override
95  	public long getSize() {
96  		return backing.getSize();
97  	}
98  
99  	/**
100 	 * Constructs a {@link java.io.File} instance using the name of the file, and invokes {@link java.io.File#getName()}
101 	 * to retrieve the name of the file minus its extension.
102 	 *
103 	 * @return The name of the file or directory denoted by this abstract pathname, or the empty string if this
104 	 * pathname's name sequence is empty
105 	 */
106 	@Override
107 	public String getFileName() {
108 		return new java.io.File(getName()).getName();
109 	}
110 
111 	/**
112 	 * Returns a string representation of this object.
113 	 *
114 	 * @return a string representation of this object.
115 	 */
116 	@Override
117 	public String toString() {
118 		return "name=" + backing.getName();
119 	}
120 }
