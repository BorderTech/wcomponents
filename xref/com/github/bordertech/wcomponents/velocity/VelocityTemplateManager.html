12  @Deprecated
13  public final class VelocityTemplateManager {
14  
15  	/**
16  	 * Look for templates under this directory in the classpath.
17  	 */
18  	private static final String VELOCITY_PREFIX = "";
19  
20  	/**
21  	 * Remove this prefix from full classnames before appending to the above prefix.
22  	 */
23  	private static final String PATH_REMOVE = "";
24  
25  	/**
26  	 * The logger instance for this class.
27  	 */
28  	private static final Log LOG = LogFactory.getLog(VelocityTemplateManager.class);
29  
30  	/**
31  	 * Hide the constructor as there are no instance methods.
32  	 */
33  	private VelocityTemplateManager() {
34  	}
35  
36  	/**
37  	 * Map the given class to a velocity template resource name.
38  	 * <p>
39  	 * For instance, com.github.bordertech.wcomponents.WTextField maps to the template in
40  	 * com/github/bordertech/wcomponents/WTextField.vm
41  	 * </p>
42  	 *
43  	 * @param clazz is the Class to find the resource name for.
44  	 * @return the template resource name for the given class.
45  	 */
46  	public static String toTemplateResourceName(final Class clazz) {
47  		String url = getTemplatePath(clazz);
48  		url = url.replace('.', '/');
49  		int index = url.lastIndexOf('$');
50  
51  		if (index != -1) {
52  			int first = url.lastIndexOf('/');
53  			String classs = url.substring(index + 1, url.length());
54  			String dir = url.substring(0, first + 1);
55  			url = dir + classs;
56  		}
57  
58  		url = VELOCITY_PREFIX + url + ".vm";
59  		// Internalize to avoid having heaps of copies of identical strings around.
60  		url = url.intern();
61  		LOG.debug("vm template : " + url);
62  		return url;
63  	}
64  
65  	/**
66  	 * @param clazz a Class from which to retrieve the Class Name
67  	 * @return String which is the class name without the path
68  	 */
69  	private static String getTemplatePath(final Class clazz) {
70  		String name = clazz.getName();
71  		LOG.debug("Full ClassPath : " + name);
72  		name = name.substring(PATH_REMOVE.length(), name.length());
73  		LOG.debug("Classname : " + name);
74  		return name;
75  	}
76  }
