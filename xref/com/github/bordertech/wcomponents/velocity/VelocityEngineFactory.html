1   package com.github.bordertech.wcomponents.velocity;
2   
3   import com.github.bordertech.wcomponents.util.Config;
4   import com.github.bordertech.wcomponents.util.ConfigurationProperties;
5   import com.github.bordertech.wcomponents.util.SystemException;
6   import java.io.PrintWriter;
7   import java.io.StringWriter;
8   import java.util.Properties;
9   import org.apache.commons.logging.Log;
10  import org.apache.commons.logging.LogFactory;
11  import org.apache.velocity.app.VelocityEngine;
12  import org.apache.velocity.runtime.RuntimeConstants;
13  
14  /**
15   * A wrapper for the creation of a (singleton) VelocityEngine, using runtime {@link Config configuration} parameters for
16   * configuration.
17   *
18   * @author James Gifford
19   * @since 1.0.0
20   */
21  public final class VelocityEngineFactory {
22  
23  	/**
24  	 * The logger instance for this class.
25  	 */
26  	private static final Log LOG = LogFactory.getLog(VelocityEngineFactory.class);
27  
28  	/**
29  	 * The singleton VelocityEngine instance associated with this factory.
30  	 */
31  	private static VelocityEngine engine;
32  
33  	/**
34  	 * Prevent instantiation of this class.
35  	 */
36  	private VelocityEngineFactory() {
37  	}
38  
39  	/**
40  	 * <p>
41  	 * Returns the VelocityEngine associated with this factory. If this is the first time we are using the engine,
42  	 * create it and initialise it.</p>
43  	 *
44  	 * <p>
45  	 * Note that velocity engines are hugely resource intensive, so we don't want too many of them. For the time being
46  	 * we have a single instance stored as a static variable. This would only be a problem if the VelocityLayout class
47  	 * ever wanted to use different engine configurations (unlikely).</p>
48  	 *
49  	 * @return the VelocityEngine associated with this factory.
50  	 */
51  	public static synchronized VelocityEngine getVelocityEngine() {
52  		if (engine == null) {
53  			String fileTemplates = ConfigurationProperties.getVelocityFileTemplates();
54  			boolean cacheTemplates = ConfigurationProperties.getVelocityCacheTemplates();
55  
56  			VelocityEngine newEngine = new VelocityEngine();
57  			Properties props = new Properties();
58  
59  			// Configure the velocity template differently according to whether we are in
60  			// "source mode" or not
61  			if (fileTemplates != null && !"".equals(fileTemplates)) {
62  				// Source mode
63  				LOG.info("Velocity engine running in source mode from " + fileTemplates);
64  
65  				props.setProperty("resource.loader", "file,class");
66  				props.setProperty("file.resource.loader.path", fileTemplates);
67  				props.setProperty("file.resource.loader.cache", "false");
68  				props.setProperty("file.resource.loader.modificationCheckInterval", "2");
69  
70  				props.setProperty("class.resource.loader.cache", "false");
71  				props.setProperty("class.resource.loader.modificationCheckInterval", "2");
72  
73  				props.setProperty("class.resource.loader.class",
74  						"org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader");
75  			} else {
76  				String cache = String.valueOf(cacheTemplates);
77  				props.setProperty("class.resource.loader.cache", cache);
78  				props.setProperty("resource.loader", "class");
79  				props.setProperty("class.resource.loader.class",
80  						"org.apache.velocity.runtime.resource.loader.ClasspathResourceLoader");
81  			}
82  
83  			// Setup commons logging for velocity
84  			props.setProperty(RuntimeConstants.RUNTIME_LOG_LOGSYSTEM_CLASS,
85  					"com.github.bordertech.wcomponents.velocity.VelocityLogger");
86  
87  			// Set up access to the common velocity macros.
88  			props.setProperty(RuntimeConstants.VM_LIBRARY, ConfigurationProperties.getVelocityMacroLibrary());
89  
90  			try {
91  				if (LOG.isInfoEnabled()) {
92  					// Dump properties
93  					StringWriter writer = new StringWriter();
94  					props.list(new PrintWriter(writer));
95  					LOG.info("Configuring velocity with the following properties...\n" + writer);
96  				}
97  
98  				newEngine.init(props);
99  			} catch (Exception ex) {
100 				throw new SystemException("Failed to configure VelocityEngine", ex);
101 			}
102 
103 			engine = newEngine;
104 		}
105 
106 		return engine;
107 	}
108 }
