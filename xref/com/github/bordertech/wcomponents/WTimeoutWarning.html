5   /**
6    * WTimeoutWarning provides a mechanism to pass a session timeout period in seconds to the client. This can then be used
7    * to indicate to the user when their session is about to expire.
8    * <p>
9    * This has two purposes: firstly it provides a means to minimise data loss by users by indicating that they should
10   * submit their work before the session expires; and secondly it is a WCAG requirement that users be given a warning of
11   * the end of a time-limited process and a means to extend that limit.
12   * </p>
13   * <p>
14   * If the timeout period is "zero", then use the http session timeout interval will be used. If a user's session does
15   * not expire then an existing WTimeoutWarning can have its timeout set to -1. This will prevent painting the
16   * WTimeoutWarning. This value is the equivalent of a http session time out value of -1 which indicates that the session
17   * does not time out.
18   * </p>
19   *
20   * @author Mark Reeves
21   * @since 1.0.0
22   */
23  public class WTimeoutWarning extends AbstractWComponent {
24  
25  	/**
26  	 * The number of seconds before timeout in which a warning is shown to the user. The default of 20 is the smallest
27  	 * amount of time (in seconds) a warning can be set to according to the WCAG 2.0
28  	 */
29  	private static final int MINIMUM_WARNING = 20;
30  
31  	/**
32  	 * Creates a WTimeoutWarning using default timeout and warning settings.
33  	 */
34  	public WTimeoutWarning() {
35  		// use the defaults for timeout and warning
36  	}
37  
38  	/**
39  	 * Creates a timeout indicator with both a timeout period and warning period
40  	 * <p>
41  	 * If the timeout period is "zero", then use the http session timeout interval will be used.
42  	 * </p>
43  	 *
44  	 * @param timeoutPeriod The timeout in seconds. This should be slightly less than the shortest timeout period in the
45  	 * entire application stack, not just the web server session timeout.
46  	 * @param warningPeriod The time in seconds before the time out occurs in which the warning indicator should show.
47  	 * There is a minimum for this which is determined by the Web Content Accessibility Guidelines 2.0.
48  	 */
49  	public WTimeoutWarning(final int timeoutPeriod, final int warningPeriod) {
50  		setTimeoutPeriod(timeoutPeriod);
51  		setWarningPeriod(warningPeriod);
52  	}
53  
54  	/**
55  	 * Creates a timeout indicator with a timeout period and uses the default warning period.
56  	 * <p>
57  	 * If the timeout period is "zero", then use the http session timeout interval will be used.
58  	 * </p>
59  	 *
60  	 * @param timeoutPeriod The timeout in seconds.
61  	 */
62  	public WTimeoutWarning(final int timeoutPeriod) {
63  		setTimeoutPeriod(timeoutPeriod);
64  	}
65  
66  	/**
67  	 * Set a timeout indicator for a specified number of seconds.
68  	 * <p>
69  	 * If the timeout period is "zero", then the session timeout interval will be used.
70  	 * </p>
71  	 *
72  	 * @param timeoutPeriod the timeout in seconds.
73  	 */
74  	public void setTimeoutPeriod(final int timeoutPeriod) {
75  		getOrCreateComponentModel().timeoutPeriod = timeoutPeriod;
76  	}
77  
78  	/**
79  	 * Set a pre-timeout warning to a specified number of seconds. This should be no less than the WCAG minimum of 20.
80  	 * <p>
81  	 * Setting the warning period to exactly ZERO (0) allows the warning time to be determined in the client layer.
82  	 * </p>
83  	 *
84  	 * @param warningPeriod The timeout warning will show this many seconds before the timeout as indicated by the
85  	 * timeoutPeriod.
86  	 */
87  	public void setWarningPeriod(final int warningPeriod) {
88  		if (0 != warningPeriod && warningPeriod < MINIMUM_WARNING) {
89  			throw new IllegalArgumentException("Warning period must be at least " + String.valueOf(
90  					MINIMUM_WARNING));
91  		}
92  		getOrCreateComponentModel().warningPeriod = warningPeriod;
93  	}
94  
95  	/**
96  	 * Get the indicated timeout period.
97  	 *
98  	 * @return The timeout period in seconds.
99  	 */
100 	public int getTimeoutPeriod() {
101 		return getComponentModel().timeoutPeriod;
102 	}
103 
104 	/**
105 	 * Get the warning period.
106 	 *
107 	 * @return The warning period in seconds.
108 	 */
109 	public int getWarningPeriod() {
110 		return getComponentModel().warningPeriod;
111 	}
112 
113 	/**
114 	 * Initialise by setting the timeout to the HTTPSession maxInactiveInterval if the timeoutPeriod has not yet been
115 	 * set. {@inheritDoc}
116 	 */
117 	@Override
118 	protected void preparePaintComponent(final Request request) {
119 		super.preparePaintComponent(request);
120 
121 		if (!isInitialised()) {
122 			// If the timeout period is "zero", then use the session timeout interval.
123 			if (getTimeoutPeriod() == 0) {
124 				int max = request.getMaxInactiveInterval();
125 				this.setTimeoutPeriod(max);
126 			}
127 			setInitialised(true);
128 		}
129 	}
130 
131 	/**
132 	 * {@inheritDoc}
133 	 */
134 	@Override
135 	protected TimeoutWarningModel newComponentModel() {
136 		return new TimeoutWarningModel();
137 	}
138 
139 	/**
140 	 * {@inheritDoc}
141 	 */
142 	@Override
143 	protected TimeoutWarningModel getComponentModel() {
144 		return (TimeoutWarningModel) super.getComponentModel();
145 	}
146 
147 	/**
148 	 * {@inheritDoc}
149 	 */
150 	@Override
151 	protected TimeoutWarningModel getOrCreateComponentModel() {
152 		return (TimeoutWarningModel) super.getOrCreateComponentModel();
153 	}
154 
155 	/**
156 	 * Extends ComponentModel to provide timeout and warning periods to be set.
157 	 */
158 	public static class TimeoutWarningModel extends ComponentModel {
159 
160 		/**
161 		 * The session timeout in seconds.
162 		 */
163 		private int timeoutPeriod = ConfigurationProperties.getTimeoutPeriod();
164 
165 		/**
166 		 * The warning period in seconds. If this is set it must be at least 20.
167 		 */
168 		private int warningPeriod = ConfigurationProperties.getTimeoutWarningPeriod();
169 	}
