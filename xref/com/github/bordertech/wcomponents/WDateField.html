104 	/**
105 	 * Attempts to convert the given object to a date. Throws a SystemException on error.
106 	 *
107 	 * @param data the data to convert.
108 	 * @return the converted date, or null if <code>data</code> was null/empty.
109 	 */
110 	private Date convertDate(final Object data) {
111 		if (data == null) {
112 			return null;
113 		} else if (data instanceof Date) {
114 			return (Date) data;
115 		} else if (data instanceof Long) {
116 			return new Date((Long) data);
117 		} else if (data instanceof Calendar) {
118 			return ((Calendar) data).getTime();
119 		} else if (data instanceof String) {
120 			try {
121 				SimpleDateFormat sdf = new SimpleDateFormat(INTERNAL_DATE_FORMAT);
122 				sdf.setLenient(lenient);
123 				return sdf.parse((String) data);
124 			} catch (ParseException e) {
125 				throw new SystemException("Could not convert String data [" + data + "] to a date.");
126 			}
127 		}
128 
129 		throw new SystemException("Cannot convert data type " + data.getClass() + " to a date.");
130 	}

164 	/**
165 	 * {@inheritDoc}
166 	 */
167 	@Override
168 	protected boolean doHandleRequest(final Request request) {
169 		// Valid Date on the request
170 		Date dateValue = getRequestValue(request);
171 
172 		// Text entered by the user (An empty string is treated as null)
173 		String value = request.getParameter(getId());
174 		String text = Util.empty(value) ? null : value;
175 
176 		// Current Date
177 		Date currentDate = getValue();
178 
179 		boolean changed;
180 
181 		// If a "valid" date value has not been entered, then check if the "user text" has changed
182 		if (dateValue == null) {
183 			// User entered text
184 			changed = !Util.equals(text, getText()) || currentDate != null;
185 		} else {
186 			// Valid Date
187 			changed = !Util.equals(dateValue, currentDate);
188 		}
189 
190 		if (changed) {
191 			boolean valid = dateValue != null || text == null;
192 			handleRequestValue(dateValue, valid, text);
193 		}
194 
195 		return changed;
196 	}

198 	/**
199 	 * Set the request value.
200 	 *
201 	 * @param value the date value
202 	 * @param valid true if valid value
203 	 * @param text the user text
204 	 */
205 	protected void handleRequestValue(final Date value, final boolean valid, final String text) {
206 		// As setData() clears the text value (if valid), this must be called first so it can be set after
207 		setData(value);
208 		DateFieldModel model = getOrCreateComponentModel();
209 		model.validDate = valid;
210 		model.text = text;
211 	}

213 	/**
214 	 * {@inheritDoc}
215 	 */
216 	@Override
217 	public Date getRequestValue(final Request request) {
218 		if (isPresent(request)) {
219 			// User entered a valid date
220 			String dateParam = request.getParameter(getId() + "-date");
221 			if (dateParam == null) {
222 				return null;
223 			}
224 
225 			// Check the date is not empty and correct length
226 			if (Util.empty(dateParam) || dateParam.length() != INTERNAL_DATE_FORMAT.length()) {
227 				LOG.warn("Date parameter is not the valid length of " + INTERNAL_DATE_FORMAT.
228 						length() + " characters ("
229 						+ dateParam + ") and will be treated as null");
230 				return null;
231 			}
232 
233 			// Check it is valid
234 			try {
235 				SimpleDateFormat sdf = new SimpleDateFormat(INTERNAL_DATE_FORMAT);
236 				sdf.setLenient(lenient);
237 				return sdf.parse(dateParam);
238 			} catch (ParseException e) {
239 				LOG.warn(
240 						"Date parameter could not be parsed (" + dateParam + ") and will be treated as null",
241 						e);
242 				return null;
243 			}
244 		} else {
245 			return getValue();
246 		}
247 	}

273 	/**
274 	 * <p>
275 	 * Performs validation of the date. Validation ensures that the entered date is between the minimum/maximum values
276 	 * (if applicable).
277 	 * </p>
278 	 * <p>
279 	 * Subclasses can override this method to perform more specific validation.
280 	 * </p>
281 	 *
282 	 * @param diags the list into which any validation diagnostics are added.
283 	 */
284 	protected void validateDate(final List<Diagnostic> diags) {
285 		Date value = getValue();
286 		if (value == null) {
287 			return;
288 		}
289 
290 		Date min = getMinDate();
291 		Date max = getMaxDate();
292 
293 		if (min != null && value.compareTo(min) < 0) {
294 			diags.add(createErrorDiagnostic(
295 					InternalMessages.DEFAULT_VALIDATION_ERROR_DATE_AFTER_OR_EQUAL, this, min));
296 		}
297 
298 		if (max != null && value.compareTo(max) > 0) {
299 			diags.add(createErrorDiagnostic(
300 					InternalMessages.DEFAULT_VALIDATION_ERROR_DATE_BEFORE_OR_EQUAL, this, max));
301 		}
302 	}
