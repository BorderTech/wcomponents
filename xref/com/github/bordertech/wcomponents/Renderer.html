3   import java.io.Serializable;

5   /**
6    * <p>
7    * Handles rendering of WComponents. This interface is technology independant, but implementations will be tied to a
8    * particular technology, e.g. web-based HTML/XML. Each WComponent will only use one renderer per rendering technology.
9    *
10   * <p>
11   * If a WComponent has no renderer, it just renders each of its child components in turn.</p>
12   *
13   * <p>
14   * All renderers must be thread-safe, as only a single instance of the renderer is created by the UIManager. It is
15   * recommended that renderers do not keep any state information, to avoid the performance penality of having to
16   * synchronize methods.</p>
17   *
18   * @author Yiannis Paschalidis
19   * @since 1.0.0
20   */
21  public interface Renderer extends Serializable {
22  
23  	/**
24  	 * Renders the component.
25  	 *
26  	 * @param component the component to paint
27  	 * @param renderContext the context for rendering.
28  	 */
29  	void render(WComponent component, RenderContext renderContext);
30  }
