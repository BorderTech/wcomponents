5   /**
6    * This interface is used to integrate a container (e.g. WServlet, WPortlet) with a ContainerHelper (e.g.
7    * WPortletHelper).
8    *
9    * @author Martin Shevchenko
10   * @since 1.0.0
11   * @deprecated no longer used
12   */
13  @Deprecated
14  public interface WComponentContainer {
15  
16  	/**
17  	 * Subclasses may override this method.
18  	 * <p>
19  	 * This method will be called internally once per request/response cycle.
20  	 * </p>
21  	 * <p>
22  	 * It is up to the subclass to determine the life cycle of the returned component. Should the same instance be
23  	 * returned and shared by all sessions, or should there be a new instance per session.</p>
24  	 *
25  	 * <p>
26  	 * A good way to always return the same instance is to use the UIRegistry. E.g.
27  	 * </p>
28  	 * <pre>
29  	 *   return UIRegistry.getInstance().getUI(KitchenSink.class.getName());
30  	 * </pre>
31  	 *
32  	 * @param nativeRequest the native request being responded to, for example an HttpServletRequest or PortletRequest.
33  	 *
34  	 * @return the top-level WComponent which defines the UI.
35  	 */
36  	WComponent getUI(Object nativeRequest);
37  
38  	/**
39  	 * The interceptor component returned here is used to wrap the html fragment generated by the UI component. Portlets
40  	 * and Servlets need to perform different wrapping of the UI component.
41  	 *
42  	 * @param nativeRequest the native request being responded to, for example an HttpServletRequest or PortletRequest.
43  	 *
44  	 * @return the top-level interceptor in the chain.
45  	 */
46  	InterceptorComponent createInterceptorChain(Object nativeRequest);
47  }
