8   /**
9    * Adds headline entries to a page. Headline entries are intended to only be added once to a page, but in a portal
10   * environment this proves difficult. This interceptor is currently used on a per-portlet basis which is not really what
11   * we want. It seems to work for now.
12   *
13   * @author Martin Shevchenko
14   */
15  public class HeadLineInterceptor extends InterceptorComponent {
16  
17  	/**
18  	 * Override preparePaint in order to perform processing specific to this interceptor. Any old headers are cleared
19  	 * out before preparePaint is called on the main UI.
20  	 *
21  	 * @param request the request being responded to.
22  	 */
23  	@Override
24  	public void preparePaint(final Request request) {
25  		// Headers
26  		// The WHeaders comes from the root WComponent and is a
27  		// mechanism for WComponents to add their own headers
28  		// (eg more JavaScript references).
29  
30  		Headers headers = this.getUI().getHeaders();
31  		headers.reset();
32  
33  		super.preparePaint(request);
34  	}
35  
36  	/**
37  	 * Override paint in order to perform processing specific to this interceptor. This implementation is responsible
38  	 * for rendering the headlines for the UI.
39  	 *
40  	 * @param renderContext the renderContext to send the output to.
41  	 */
42  	@Override
43  	public void paint(final RenderContext renderContext) {
44  		if (renderContext instanceof WebXmlRenderContext) {
45  			PageContentHelper.addAllHeadlines(((WebXmlRenderContext) renderContext).getWriter(),
46  					getUI().getHeaders());
47  		}
48  
49  		getBackingComponent().paint(renderContext);
50  	}
51  }
