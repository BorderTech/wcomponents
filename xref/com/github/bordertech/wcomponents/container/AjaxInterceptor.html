20  /**
21   * This {@link InterceptorComponent} looks for an ajax 'trigger' parameter in the request. The trigger parameter
22   * identifies the {@link WAjaxControl} to be handled by the request. In the paint phase the interceptor will only paint
23   * the targeted components to the response.
24   *
25   * @see WAjaxControl
26   * @author Christina Harris
27   * @since 1.0.0
28   */
29  public class AjaxInterceptor extends InterceptorComponent {
30  
31  	/**
32  	 * The logger instance for this class.
33  	 */
34  	private static final Log LOG = LogFactory.getLog(AjaxInterceptor.class);
35  
36  	/**
37  	 * {@inheritDoc}
38  	 */
39  	@Override
40  	public void serviceRequest(final Request request) {
41  		String triggerId = request.getParameter(WServlet.AJAX_TRIGGER_PARAM_NAME);
42  
43  		AjaxOperation ajaxOperation = AjaxHelper.getCurrentOperation();
44  		if (ajaxOperation == null) {
45  			throw new IllegalStateException(
46  					"No AJAX operation available for trigger " + triggerId + ".");
47  		}
48  
49  		ComponentWithContext triggerWithContext = AjaxHelper.getCurrentTriggerAndContext();
50  		if (triggerWithContext == null) {
51  			throw new IllegalStateException(
52  					"No component/context available for AJAX trigger " + triggerId + ".");
53  		}
54  
55  		UIContext uic = UIContextHolder.getCurrent();
56  
57  		// Reset the focus for this new request.
58  		uic.setFocussed(null, null);
59  
60  		// We've hit the action phase, so we do want focus on this app.
61  		uic.setFocusRequired(true);
62  
63  		// Process trigger only
64  		if (isProcessTriggerOnly(triggerWithContext, ajaxOperation)) {
65  			// Get user context
66  			UIContext tuic = triggerWithContext.getContext();
67  			UIContextHolder.pushContext(tuic);
68  			try {
69  				WComponent trigger = triggerWithContext.getComponent();
70  				trigger.serviceRequest(request);
71  				// Manually invoke laters as the InvokeLaters in the service request is not run due to the trigger
72  				// having a "parent"
73  				tuic.doInvokeLaters();
74  			} finally {
75  				UIContextHolder.popContext();
76  			}
77  		} else if ("GET".equals(request.getMethod())) { // GET only supports the above scenarios
78  			throw new IllegalStateException(
79  					"GET is not supported for the AJAX trigger " + triggerId + ".");
80  		} else {
81  			// service the request
82  			super.serviceRequest(request);
83  		}
84  	}
85  
86  	/**
87  	 * Paints the targeted ajax regions. The format of the response is an agreement between the server and the client
88  	 * side JavaScript handling our ajax response.
89  	 *
90  	 * @param renderContext the renderContext to send the output to.
91  	 */
92  	@Override
93  	public void paint(final RenderContext renderContext) {
94  		AjaxOperation operation = AjaxHelper.getCurrentOperation();
95  		if (operation == null) {
96  			// the request attribute that we place in the ui context in the action phase can't be null
97  			throw new SystemException(
98  					"Can't paint AJAX response. Couldn't find the expected reference to the AjaxOperation.");
99  		}
100 
101 		if (operation.getTargetContainerId() != null) {
102 			paintContainerResponse(renderContext, operation);
103 		} else {
104 			paintResponse(renderContext, operation);
105 		}
106 	}
107 
108 	/**
109 	 * Paint the ajax container response.
110 	 *
111 	 * @param renderContext the render context
112 	 * @param operation the ajax operation
113 	 */
114 	private void paintContainerResponse(final RenderContext renderContext,
115 			final AjaxOperation operation) {
116 		WebXmlRenderContext webRenderContext = (WebXmlRenderContext) renderContext;
117 		XmlStringBuilder xml = webRenderContext.getWriter();
118 
119 		// Get trigger's context
120 		ComponentWithContext trigger = AjaxHelper.getCurrentTriggerAndContext();
121 		if (trigger == null) {
122 			throw new SystemException("No context available for trigger " + operation.getTriggerId());
123 		}
124 
125 		xml.appendTagOpen("ui:ajaxtarget");
126 		xml.appendAttribute("id", operation.getTargetContainerId());
127 		xml.appendAttribute("action", AjaxOperation.AjaxAction.REPLACE_CONTENT.getDesc());
128 		xml.appendClose();
129 
130 		// Paint targets - Assume targets are in the same context as the trigger
131 		UIContextHolder.pushContext(trigger.getContext());
132 		try {
133 			for (String targetId : operation.getTargets()) {
134 				ComponentWithContext target;
135 				if (targetId.equals(operation.getTriggerId())) {
136 					target = trigger;
137 				} else {
138 					target = WebUtilities.getComponentById(targetId, true);
139 					if (target == null) {
140 						LOG.warn("Could not find ajax target to render [" + targetId + "]");
141 						continue;
142 					}
143 				}
144 				target.getComponent().paint(renderContext);
145 			}
146 		} finally {
147 			UIContextHolder.popContext();
148 		}
149 
150 		xml.appendEndTag("ui:ajaxtarget");
151 	}

153 	/**
154 	 * Paint the ajax response.
155 	 *
156 	 * @param renderContext the render context
157 	 * @param operation the ajax operation
158 	 */
159 	private void paintResponse(final RenderContext renderContext, final AjaxOperation operation) {
160 		WebXmlRenderContext webRenderContext = (WebXmlRenderContext) renderContext;
161 		XmlStringBuilder xml = webRenderContext.getWriter();
162 
163 		// Get trigger's context
164 		ComponentWithContext trigger = AjaxHelper.getCurrentTriggerAndContext();
165 		if (trigger == null) {
166 			throw new SystemException("No context available for trigger " + operation.getTriggerId());
167 		}
168 
169 		for (String targetId : operation.getTargets()) {
170 			ComponentWithContext target;
171 			if (targetId.equals(operation.getTriggerId())) {
172 				target = trigger;
173 			} else {
174 				target = WebUtilities.getComponentById(targetId, true);
175 				if (target == null) {
176 					LOG.warn("Could not find ajax target to render [" + targetId + "]");
177 					continue;
178 				}
179 			}
180 
181 			UIContextHolder.pushContext(target.getContext());
182 			try {
183 				xml.appendTagOpen("ui:ajaxtarget");
184 				xml.appendAttribute("id", targetId);
185 				xml.appendAttribute("action", operation.getAction().getDesc());
186 				xml.appendClose();
187 
188 				target.getComponent().paint(renderContext);
189 
190 				xml.appendEndTag("ui:ajaxtarget");
191 			} finally {
192 				UIContextHolder.popContext();
193 			}
194 		}
195 	}

197 	/**
198 	 * Check if process this trigger only.
199 	 *
200 	 * @param triggerWithContext the trigger with its context
201 	 * @param operation current ajax operation
202 	 * @return true if process this trigger only
203 	 */
204 	private boolean isProcessTriggerOnly(final ComponentWithContext triggerWithContext,
205 			final AjaxOperation operation) {
206 
207 		// Target container implies only process the trigger or is Internal Ajax
208 		if (operation.getTargetContainerId() != null || operation.isInternalAjaxRequest()) {
209 			return true;
210 		}
211 
212 		WComponent trigger = triggerWithContext.getComponent();
213 
214 		// Check if trigger is a polling AJAX control
215 		if (trigger instanceof WAjaxControl) {
216 			// Get user context
217 			UIContext uic = triggerWithContext.getContext();
218 			UIContextHolder.pushContext(uic);
219 			try {
220 				WAjaxControl ajax = (WAjaxControl) trigger;
221 				// Is a polling region so only process trigger
222 				if (ajax.getDelay() > 0) {
223 					return true;
224 				}
225 			} finally {
226 				UIContextHolder.popContext();
227 			}
228 		}
229 
230 		return false;
231 	}
