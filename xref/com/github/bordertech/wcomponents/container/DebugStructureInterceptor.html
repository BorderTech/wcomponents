14  import java.util.List;

16  /**
17   * <p>
18   * An Interceptor used to output structural debugging information.</p>
19   *
20   * @author Yiannis Paschalidis
21   * @since 1.0.0
22   */
23  public class DebugStructureInterceptor extends InterceptorComponent {
24  
25  	/**
26  	 * Override paint to render additional information for debugging purposes.
27  	 *
28  	 * @param renderContext the renderContext to send the output to.
29  	 */
30  	@Override
31  	public void paint(final RenderContext renderContext) {
32  		super.paint(renderContext);
33  
34  		if (!DebugUtil.isDebugFeaturesEnabled() || !(renderContext instanceof WebXmlRenderContext)) {
35  			return;
36  		}
37  
38  		XmlStringBuilder xml = ((WebXmlRenderContext) renderContext).getWriter();
39  
40  		xml.appendTag("ui:debug");
41  		writeDebugInfo(getUI(), xml);
42  		xml.appendEndTag("ui:debug");
43  	}
44  
45  	/**
46  	 * Writes debugging information for the given component.
47  	 *
48  	 * @param component the component to write debugging information for.
49  	 * @param xml the writer to send the debug output to.
50  	 */
51  	protected void writeDebugInfo(final WComponent component, final XmlStringBuilder xml) {
52  		if (component != null && (component.isVisible() || component instanceof WInvisibleContainer)) {
53  			xml.appendTagOpen("ui:debugInfo");
54  			xml.appendAttribute("for", component.getId());
55  			xml.appendAttribute("class", component.getClass().getName());
56  			xml.appendOptionalAttribute("type", getType(component));
57  			xml.appendClose();
58  
59  			xml.appendTagOpen("ui:debugDetail");
60  			xml.appendAttribute("key", "defaultState");
61  			xml.appendAttribute("value", component.isDefaultState());
62  			xml.appendEnd();
63  
64  			xml.appendEndTag("ui:debugInfo");
65  
66  			if (component instanceof WRepeater) {
67  				// special case for WRepeaters - we must paint the info for each row.
68  				WRepeater repeater = (WRepeater) component;
69  
70  				List<UIContext> contexts = repeater.getRowContexts();
71  
72  				for (int i = 0; i < contexts.size(); i++) {
73  					UIContextHolder.pushContext(contexts.get(i));
74  
75  					try {
76  						writeDebugInfo(repeater.getRepeatedComponent(i), xml);
77  					} finally {
78  						UIContextHolder.popContext();
79  					}
80  				}
81  			} else if (component instanceof WCardManager) {
82  				writeDebugInfo(((WCardManager) component).getVisible(), xml);
83  			} else if (component instanceof Container) {
84  				final int size = ((Container) component).getChildCount();
85  
86  				for (int i = 0; i < size; i++) {
87  					writeDebugInfo(((Container) component).getChildAt(i), xml);
88  				}
89  			}
90  		}
91  	}
92  
93  	/**
94  	 * Tries to return the type of component.
95  	 *
96  	 * @param component the component to determine the type of.
97  	 * @return the component type, or null if the given component is not a core component.
98  	 */
99  	private String getType(final WComponent component) {
100 		for (Class<?> clazz = component.getClass(); clazz != null && WComponent.class.
101 				isAssignableFrom(clazz); clazz = clazz.getSuperclass()) {
102 			if ("com.github.bordertech.wcomponents".equals(clazz.getPackage().getName())) {
103 				return clazz.getName();
104 			}
105 		}
106 
107 		return null;
108 	}
109 }
