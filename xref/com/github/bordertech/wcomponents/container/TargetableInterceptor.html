11  
12  /**
13   * This interceptor component checks to see if a parameter named {@link Environment#TARGET_ID} exists on the request. If
14   * the parameter is found then this interceptor replaces the top level ui component with the {@link Targetable}
15   * component with the same id as the parameter value.
16   *
17   * @author Christina Harris
18   * @since 1.0.0
19   */
20  public class TargetableInterceptor extends InterceptorComponent {
21  
22  	/**
23  	 * The id of the targeted component.
24  	 */
25  	private String targetId = null;
26  
27  	/**
28  	 * {@inheritDoc}
29  	 */
30  	@Override
31  	public void serviceRequest(final Request request) {
32  		// Check if this is a targeted request
33  		targetId = request.getParameter(Environment.TARGET_ID);
34  		if (targetId == null) {
35  			throw new SystemException("No target id request parameter");
36  		}
37  
38  		ComponentWithContext target = WebUtilities.getComponentById(targetId, true);
39  		if (target == null) {
40  			throw new SystemException("No target component found for id " + targetId);
41  		}
42  
43  		// go straight to the target component.
44  		attachUI(target.getComponent());
45  
46  		UIContextHolder.pushContext(target.getContext());
47  		try {
48  			super.serviceRequest(request);
49  		} finally {
50  			UIContextHolder.popContext();
51  		}
52  	}
53  
54  	/**
55  	 * {@inheritDoc}
56  	 */
57  	@Override
58  	public void preparePaint(final Request request) {
59  		ComponentWithContext target = WebUtilities.getComponentById(targetId, true);
60  		if (target == null) {
61  			throw new SystemException("No target component found for id " + targetId);
62  		}
63  
64  		UIContextHolder.pushContext(target.getContext());
65  		try {
66  			super.preparePaint(request);
67  		} finally {
68  			UIContextHolder.popContext();
69  		}
70  	}
71  
72  	/**
73  	 * {@inheritDoc}
74  	 */
75  	@Override
76  	public void paint(final RenderContext renderContext) {
77  		ComponentWithContext target = WebUtilities.getComponentById(targetId, true);
78  		if (target == null) {
79  			throw new SystemException("No target component found for id " + targetId);
80  		}
81  
82  		UIContextHolder.pushContext(target.getContext());
83  		try {
84  			super.paint(renderContext);
85  		} finally {
86  			UIContextHolder.popContext();
87  		}
88  	}
89  
90  }
