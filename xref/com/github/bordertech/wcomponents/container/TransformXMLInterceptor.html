36  /**
37   * This interceptor is used to perform server-side XSLT so that HTML is delivered to the client instead of XML. This
38   * works by buffering the response in memory and then transforming it before sending the response to the client. This
39   * will use more memory and CPU on the server. If this becomes a problem it may be better to perform the transform on an
40   * appliance (or the client).
41   *
42   * @author Rick Brown
43   * @since 1.0.0
44   */
45  public class TransformXMLInterceptor extends InterceptorComponent {
46  
47  	/**
48  	 * Support for no-xslt agents.
49  	 */
50  	private static final String NO_XSLT_FLAG = "wcnoxslt";
51  
52  	/**
53  	 * The logger instance for this class.
54  	 */
55  	private static final Log LOG = LogFactory.getLog(TransformXMLInterceptor.class);
56  
57  	/**
58  	 * The theme XSLT resource name.
59  	 */
60  	private static final String RESOURCE_NAME = ThemeUtil.getThemeBase() + "xslt/" + ThemeUtil.getThemeXsltName();
61  
62  	/**
63  	 * The XSLT cached templates.
64  	 */
65  	private static final Templates TEMPLATES = initTemplates();
66  
67  	/**
68  	 * If true then server side XSLT will be ignored regardless of the configuration property. This is to account for
69  	 * user agents that cannot handle HTML, yes such a thing exists.
70  	 */
71  	private boolean doTransform = true;
72  
73  	/**
74  	 * Override preparePaint in order to perform processing specific to this interceptor.
75  	 *
76  	 * @param request the request being responded to.
77  	 */
78  	@Override
79  	public void preparePaint(final Request request) {
80  		if (doTransform && request instanceof ServletRequest) {
81  			HttpServletRequest httpServletRequest = ((ServletRequest) request).getBackingRequest();
82  			String userAgentString = httpServletRequest.getHeader("User-Agent");
83  			/* It is possible to opt out on a case by case basis by setting a flag on the ua string.
84  			 * This helps custom user agents that do not support HTML as well as facilitating debugging.
85  			 */
86  			if (userAgentString != null && userAgentString.contains(NO_XSLT_FLAG)) {
87  				doTransform = false;
88  			}
89  		}
90  		super.preparePaint(request);
91  	}
92  
93  	/**
94  	 * Override paint to perform XML to HTML transformation.
95  	 *
96  	 * @param renderContext the renderContext to send the output to.
97  	 */
98  	@Override
99  	public void paint(final RenderContext renderContext) {
100 
101 		if (!doTransform) {
102 			super.paint(renderContext);
103 			return;
104 		}
105 
106 		if (!(renderContext instanceof WebXmlRenderContext)) {
107 			LOG.warn("Unable to transform a " + renderContext);
108 			super.paint(renderContext);
109 			return;
110 		}
111 		LOG.debug("Transform XML Interceptor: Start");
112 
113 		UIContext uic = UIContextHolder.getCurrent();
114 
115 		// Set up a render context to buffer the XML payload.
116 		StringWriter xmlBuffer = new StringWriter();
117 		PrintWriter xmlWriter = new PrintWriter(xmlBuffer);
118 
119 		WebXmlRenderContext xmlContext = new WebXmlRenderContext(xmlWriter, uic.getLocale());
120 
121 		super.paint(xmlContext);  // write the XML to the buffer
122 
123 		// Get a handle to the true PrintWriter.
124 		WebXmlRenderContext webRenderContext = (WebXmlRenderContext) renderContext;
125 		PrintWriter writer = webRenderContext.getWriter();
126 
127 		/*
128 		 * Switch the response content-type to HTML.
129 		 * In theory the transformation could be to ANYTHING (not just HTML) so perhaps it would make more sense to
130 		 *    write a new interceptor "ContentTypeInterceptor" which attempts to sniff payloads and choose the correct
131 		 *    content-type. This is exactly the kind of thing IE6 loved to do, so perhaps it's a bad idea.
132 		 */
133 		Response response = getResponse();
134 		response.setContentType(WebUtilities.CONTENT_TYPE_HTML);
135 
136 		String xml = xmlBuffer.toString();
137 		if (isAllowCorruptCharacters() && !Util.empty(xml)) {
138 			// Remove illegal HTML characters from the content before transforming it.
139 			xml = removeCorruptCharacters(xml);
140 		}
141 		// Double encode template tokens in the XML
142 		xml = WebUtilities.doubleEncodeBrackets(xml);
143 		// Setup temp writer
144 		StringWriter tempBuffer = new StringWriter();
145 		PrintWriter tempWriter = new PrintWriter(tempBuffer);
146 		// Perform the transformation and write the result.
147 		transform(xml, uic, tempWriter);
148 		// Decode Double Encoded Brackets
149 		String tempResp = tempBuffer.toString();
150 		tempResp = WebUtilities.doubleDecodeBrackets(tempResp);
151 		// Write response
152 		writer.write(tempResp);
153 
154 		LOG.debug("Transform XML Interceptor: Finished");
155 	}
156 
157 	/**
158 	 * Transform the UI XML to HTML using the correct XSLT from the classpath.
159 	 *
160 	 * @param xml The XML to transform.
161 	 * @param uic The UIContext used to determine variables such as locale.
162 	 * @param writer The result of the transformation will be written to this writer.
163 	 */
164 	private void transform(final String xml, final UIContext uic, final PrintWriter writer) {
165 
166 		Transformer transformer = newTransformer();
167 		Source inputXml;
168 		try {
169 			inputXml = new StreamSource(new ByteArrayInputStream(xml.getBytes("utf-8")));
170 			StreamResult result = new StreamResult(writer);
171 			transformer.transform(inputXml, result);
172 		} catch (UnsupportedEncodingException | TransformerException ex) {
173 			throw new SystemException("Could not transform xml", ex);
174 		}
175 	}
176 
177 	/**
178 	 * Creates a new Transformer instance using cached XSLT Templates. There will be one cached template. Transformer
179 	 * instances are not thread-safe and cannot be reused (they can after the transformation is complete).
180 	 *
181 	 * @return A new Transformer instance.
182 	 */
183 	private static Transformer newTransformer() {
184 
185 		if (TEMPLATES == null) {
186 			throw new IllegalStateException("TransformXMLInterceptor not initialized.");
187 		}
188 
189 		try {
190 			return TEMPLATES.newTransformer();
191 		} catch (TransformerConfigurationException ex) {
192 			throw new SystemException("Could not create transformer for " + RESOURCE_NAME, ex);
193 		}
194 	}
195 
196 	/**
197 	 * Statically initialize the XSLT templates that are cached for all future transforms.
198 	 *
199 	 * @return the XSLT Templates.
200 	 */
201 	private static Templates initTemplates() {
202 		try {
203 			URL xsltURL = ThemeUtil.class.getResource(RESOURCE_NAME);
204 			if (xsltURL != null) {
205 				Source xsltSource = new StreamSource(xsltURL.openStream(), xsltURL.toExternalForm());
206 				TransformerFactory factory = new net.sf.saxon.TransformerFactoryImpl();
207 				Templates templates = factory.newTemplates(xsltSource);
208 				LOG.debug("Generated XSLT templates for: " + RESOURCE_NAME);
209 				return templates;
210 			} else {
211 				// Server-side XSLT enabled but theme resource not on classpath.
212 				throw new IllegalStateException(RESOURCE_NAME + " not on classpath");
213 			}
214 		} catch (IOException | TransformerConfigurationException ex) {
215 			throw new SystemException("Could not create transformer for " + RESOURCE_NAME, ex);
216 		}
217 	}
218 
219 	/**
220 	 * @return true if allow corrupt characters in XSLT processing.
221 	 */
222 	private static boolean isAllowCorruptCharacters() {
223 		return ConfigurationProperties.getXsltAllowCorruptCharacters();
224 	}
225 
226 	/**
227 	 * Remove bad characters in XML.
228 	 *
229 	 * @param input The String to escape.
230 	 * @return the clean string
231 	 */
232 	private static String removeCorruptCharacters(final String input) {
233 		if (Util.empty(input)) {
234 			return input;
235 		}
236 		return ESCAPE_BAD_XML10.translate(input);
237 	}
238 
239 	/**
240 	 * Translator object for escaping XML 1.0.
241 	 *
242 	 * While {@link #escapeXml10(String)} is the expected method of use, this object allows the XML escaping
243 	 * functionality to be used as the foundation for a custom translator.
244 	 */
245 	private static final CharSequenceTranslator ESCAPE_BAD_XML10
246 			= new AggregateTranslator(
247 					new LookupTranslator(
248 							new String[][]{
249 								{"\u000b", ""},
250 								{"\u000c", ""},
251 								{"\ufffe", ""},
252 								{"\uffff", ""}
253 							}),
254 					WebUtilities.NumericEntityIgnorer.between(0x00, 0x08),
255 					WebUtilities.NumericEntityIgnorer.between(0x0e, 0x1f),
256 					WebUtilities.NumericEntityIgnorer.between(0x7f, 0x9f)
257 			);
