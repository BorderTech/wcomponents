73  		if (webComponent instanceof WComponent) {
74  			// No interceptor supplied but we need one to make things simple.
75  			// Dummy up a pass-through/do-nothing interceptor.
76  			InterceptorComponent inter = new InterceptorComponent();
77  			inter.setBackingComponent(webComponent);
78  			setInterceptor(inter);
79  		} else if (webComponent instanceof InterceptorComponent) {
80  			setInterceptor((InterceptorComponent) webComponent);
81  		} else {
82  			throw new IllegalArgumentException("Unexpected extension of WebComponent supplied.");
83  		}
84  	}

109 	 * Prepare the user context for this request.
110 	 * <p>
111 	 * This must be called before processAction and render.
112 	 * </p>
113 	 *
114 	 * @return the user context
115 	 */
116 	public UIContext prepareUserContext() {
117 		UIContext uic = getUIContext();
118 		if (requestImpliesNew() || uic == null) {
119 			LOG.debug("Preparing a new session");
120 			setNewConversation(Boolean.TRUE);
121 			uic = createUIContext();
122 			setUIContext(uic);
123 		} else {
124 			setNewConversation(Boolean.FALSE);
125 			// In Development mode, simulate running in a cluster
126 			cycleUIContext();
127 		}
128 		return uic;
129 	}

131 	/**
132 	 * Support standard processing of the action phase of a request.
133 	 *
134 	 * @throws IOException if there is an IO error on writing a response.
135 	 */
136 	public void processAction() throws IOException {
137 		if (isDisposed()) {
138 			LOG.error("Skipping action phase. Attempt to reuse disposed ContainerHelper instance");
139 			return;
140 		}
141 
142 		try {
143 			// Check user context has been prepared
144 			if (getNewConversation() == null) {
145 				throw new IllegalStateException(
146 						"User context has not been prepared before the action phase");
147 			}
148 
149 			prepareAction();
150 
151 			UIContext uic = getUIContext();
152 			if (uic == null) {
153 				throw new IllegalStateException("No user context set for the action phase.");
154 			}
155 
156 			UIContextHolder.pushContext(uic);
157 
158 			// Make sure maps are cleared up
159 			uic.clearScratchMap();
160 			uic.clearRequestScratchMap();
161 
162 			prepareRequest();
163 
164 			Request req = getRequest();
165 			getInterceptor().attachResponse(getResponse());
166 			getInterceptor().serviceRequest(req);
167 
168 			if (req.isLogout()) {
169 				handleLogout();
170 				dispose();
171 			}
172 		} catch (ActionEscape esc) {
173 			LOG.debug("ActionEscape performed.");
174 			// Action escapes must be handled in the action phase and then
175 			// do nothing if they reach the render phase (which they will in
176 			// the servlet implementation)
177 			handleEscape(esc);
178 			dispose();
179 		} catch (Escape esc) {
180 			LOG.debug("Escape performed during action phase.");
181 			// We can't handle the escape until the render phase.
182 		} catch (Throwable t) {
183 			// We try not to let any exception propagate to container.
184 			String message = "Caught exception during action phase.";
185 			LOG.error(message, t);
186 
187 			// We can't handle the error until the render phase.
188 			propogateError(t);
189 		} finally {
190 			UIContextHolder.reset();
191 		}
192 	}
193 
194 	/**
195 	 * Support standard processing of the render phase of a request.
196 	 *
197 	 * @throws IOException IO Exception
198 	 */
199 	public void render() throws IOException {
200 		if (isDisposed()) {
201 			LOG.debug("Skipping render phase.");
202 			return;
203 		}
204 
205 		try {
206 			// Check user context has been prepared
207 			if (getNewConversation() == null) {
208 				throw new IllegalStateException(
209 						"User context has not been prepared before the render phase");
210 			}
211 
212 			prepareRender();
213 
214 			UIContext uic = getUIContext();
215 			if (uic == null) {
216 				throw new IllegalStateException("No user context set for the render phase.");
217 			}
218 
219 			UIContextHolder.pushContext(uic);
220 			prepareRequest();
221 
222 			// Handle errors from the action phase now.
223 			if (havePropogatedError()) {
224 				handleError(getPropogatedError());
225 				return;
226 			}
227 
228 			WComponent uiComponent = getUI();
229 
230 			if (uiComponent == null) {
231 				throw new SystemException("No UI Component exists.");
232 			}
233 
234 			Environment environment = uiComponent.getEnvironment();
235 
236 			if (environment == null) {
237 				throw new SystemException("No WEnvironment exists.");
238 			}
239 
240 			getInterceptor().attachResponse(getResponse());
241 			getInterceptor().preparePaint(getRequest());
242 
243 			String contentType = getUI().getHeaders().getContentType();
244 			Response response = getResponse();
245 			response.setContentType(contentType);
246 
247 			addGenericHeaders(uic, getUI());
248 
249 			PrintWriter writer = getPrintWriter();
250 			getInterceptor().paint(new WebXmlRenderContext(writer, uic.getLocale()));
251 
252 			// The following only matters for a Portal context
253 			String title = uiComponent instanceof WApplication ? ((WApplication) uiComponent).
254 					getTitle() : null;
255 
256 			if (title != null) {
257 				setTitle(title);
258 			}
259 		} catch (Escape esc) {
260 			LOG.debug("Escape performed during render phase.");
261 			handleEscape(esc);
262 		} catch (Throwable t) {
263 			// We try not to let any exception propagate to container.
264 			String message = "Caught exception during render phase.";
265 			LOG.error(message, t);
266 			handleError(t);
267 		} finally {
268 			UIContextHolder.reset();
269 			dispose();
270 		}
271 	}

358 	/**
359 	 * Prepare the session for the current request.
360 	 */
361 	protected void prepareRequest() {
362 		LOG.debug("Preparing for request by adding headers and environment to top wcomponent");
363 
364 		// Configure the UIContext to handle this request.
365 		UIContext uiContext = getUIContext();
366 
367 		// Add WEnvironment if not already done.
368 		// If the component is new, then it will not have a WEnvironment yet.
369 		Environment env;
370 
371 		if (uiContext.isDummyEnvironment()) {
372 			env = createEnvironment();
373 			uiContext.setEnvironment(env);
374 		} else {
375 			env = uiContext.getEnvironment();
376 		}
377 
378 		// Update the environment for the current phase of the request
379 		// processing.
380 		updateEnvironment(env);
381 
382 		// Prepare an implementation of a wcomponent Request suitable to the
383 		// type of
384 		// container we are running in.
385 		if (getRequest() == null) {
386 			setRequest(createRequest());
387 		}
388 
389 		// Update the wcomponent Request for the current phase of the request
390 		// processing.
391 		updateRequest(getRequest());
392 	}

556 	/**
557 	 * Propogates an error from the action phase to the render phase.
558 	 *
559 	 * @param error the error to propogate.
560 	 */
561 	private void propogateError(final Throwable error) {
562 		// Unhandled runtime exceptions from action phase
563 		// must be remembered for subsequent renders
564 		if (getRequest() == null) {
565 			LOG.error("Unable to remember error from action phase beyond this request");
566 		} else {
567 			LOG.debug("Remembering error from action phase");
568 			getRequest().setAttribute(ACTION_ERROR_KEY, error);
569 		}
570 	}

572 	/**
573 	 * Indicates whether there is an error which has been propogated from the action to the render phase.
574 	 *
575 	 * @return true if there is a propogated escape, false otherwise.
576 	 */
577 	private boolean havePropogatedError() {
578 		Request req = getRequest();
579 		return req != null && req.getAttribute(ACTION_ERROR_KEY) != null;
580 	}

582 	/**
583 	 * Retrieves an error which has been propogated from the action to the render phase.
584 	 *
585 	 * @return the propogated escape, or null if there isn't one.
586 	 */
587 	private Throwable getPropogatedError() {
588 		Request req = getRequest();
589 
590 		if (req != null) {
591 			return (Throwable) req.getAttribute(ACTION_ERROR_KEY);
592 		}
593 
594 		return null;
595 	}

597 	/**
598 	 * Last resort error handling.
599 	 *
600 	 * @param error the error to handle.
601 	 * @throws IOException if there is an error writing the error HTML.
602 	 */
603 	public void handleError(final Throwable error) throws IOException {
604 
605 		LOG.debug("Start handleError...");
606 
607 		// Should the session be removed upon error?
608 		boolean terminate = ConfigurationProperties.getTerminateSessionOnError();
609 		// If we are unfriendly, terminate the session
610 		if (terminate) {
611 			invalidateSession();
612 		}
613 
614 		// Are we in developer friendly error mode?
615 		boolean friendly = ConfigurationProperties.getDeveloperErrorHandling();
616 
617 		FatalErrorPageFactory factory = Factory.newInstance(FatalErrorPageFactory.class);
618 		WComponent errorPage = factory.createErrorPage(friendly, error);
619 
620 		String html = renderErrorPageToHTML(errorPage);
621 
622 		// Setup the response
623 		Response response = getResponse();
624 		response.setContentType(WebUtilities.CONTENT_TYPE_HTML);
625 
626 		// Make sure not cached
627 		getResponse().setHeader("Cache-Control", CacheType.NO_CACHE.getSettings());
628 		getResponse().setHeader("Pragma", "no-cache");
629 		getResponse().setHeader("Expires", "-1");
630 
631 		getPrintWriter().println(html);
632 
633 		LOG.debug("End handleError");
634 	}

636 	/**
637 	 * Render the error page component to HTML.
638 	 *
639 	 * @param errorPage the error page component
640 	 * @return the error page as HTML
641 	 */
642 	protected String renderErrorPageToHTML(final WComponent errorPage) {
643 
644 		// Check if using the default error page
645 		boolean defaultErrorPage = errorPage instanceof FatalErrorPage;
646 
647 		String html = null;
648 
649 		// If not default implementation of error page, Transform error page to HTML
650 		if (!defaultErrorPage) {
651 			// Set UIC and Environment (Needed for Theme Paths)
652 			UIContext uic = new UIContextImpl();
653 			uic.setEnvironment(createEnvironment());
654 			UIContextHolder.pushContext(uic);
655 			try {
656 				html = WebUtilities.renderWithTransformToHTML(errorPage);
657 			} catch (Exception e) {
658 				LOG.warn("Could not transform error page.", e);
659 			} finally {
660 				UIContextHolder.popContext();
661 			}
662 		}
663 
664 		// Not transformed. So just render.
665 		if (html == null) {
666 			UIContextHolder.pushContext(new UIContextImpl());
667 			try {
668 				html = WebUtilities.render(errorPage);
669 			} catch (Exception e) {
670 				LOG.warn("Could not render error page.", e);
671 				html = "System error occurred but could not render error page.";
672 			} finally {
673 				UIContextHolder.popContext();
674 			}
675 		}
676 
677 		return html;
678 	}
