4   import com.github.bordertech.wcomponents.Request;
5   import com.github.bordertech.wcomponents.UIContext;
6   import com.github.bordertech.wcomponents.UIContextDebugWrapper;
7   import com.github.bordertech.wcomponents.UIContextHolder;
8   import org.apache.commons.logging.Log;
9   import org.apache.commons.logging.LogFactory;
10  
11  /**
12   * This interceptor removes component model objects that are no longer needed from the UIContext. This is not essential
13   * other than to keep web server memory usage down.
14   *
15   * @author Martin Shevchenko
16   * @since 1.0.0
17   */
18  public class ContextCleanupInterceptor extends InterceptorComponent {
19  
20  	/**
21  	 * The logger instance for this class.
22  	 */
23  	private static final Log LOG = LogFactory.getLog(ContextCleanupInterceptor.class);
24  
25  	/**
26  	 * Override serviceRequest to clear out the targetable index from the last request. And clear out the scratch map
27  	 * after the request has been processed (and before painting occurs).
28  	 *
29  	 * @param request the request being responded to.
30  	 */
31  	@Override
32  	public void serviceRequest(final Request request) {
33  		LOG.debug("Before Service Request - Clearing targetable index.");
34  		UIContext uic = UIContextHolder.getCurrent();
35  
36  		super.serviceRequest(request);
37  		// Clear phase scope scratch map
38  		LOG.debug("After Service Request - Clearing scratch map with phase scope.");
39  		uic.clearScratchMap();
40  	}
41  
42  	/**
43  	 * Override paint to clear out the scratch map and component models which are no longer necessary.
44  	 *
45  	 * @param renderContext the renderContext to send the output to.
46  	 */
47  	@Override
48  	public void paint(final RenderContext renderContext) {
49  		super.paint(renderContext);
50  		UIContext uic = UIContextHolder.getCurrent();
51  
52  		if (LOG.isDebugEnabled()) {
53  			UIContextDebugWrapper debugWrapper = new UIContextDebugWrapper(uic);
54  			LOG.debug("Session usage after paint:\n" + debugWrapper);
55  		}
56  
57  		LOG.debug("Performing session tidy up of WComponents (any WComponents disconnected from the active top component will not be tidied up.");
58  		getUI().tidyUpUIContextForTree();
59  
60  		LOG.debug("After paint - Clearing scratch maps.");
61  		uic.clearScratchMap();
62  		uic.clearRequestScratchMap();
63  	}
64  }
