9   import java.io.PrintWriter;
10  
11  /**
12   * WhitespaceFilterInterceptor is an interceptor that removes HTML comments and redundant white space from HTML output.
13   *
14   * For local testing/troubleshooting, the filter can be disabled by setting the following {@link Config configuration}
15   * parameter:
16   * <pre>
17   * com.github.bordertech.wcomponents.container.WhitespaceFilterInterceptor.enabled=false
18   * </pre>
19   *
20   * @author Yiannis Paschalidis
21   * @since 1.0.0
22   */
23  public class WhitespaceFilterInterceptor extends InterceptorComponent {
24  
25  	/**
26  	 * Paints the component.
27  	 *
28  	 * @param renderContext the renderContext to send the output to.
29  	 */
30  	@Override
31  	public void paint(final RenderContext renderContext) {
32  		boolean enabled = ConfigurationProperties.getWhitespaceFilter();
33  
34  		if (enabled && renderContext instanceof WebXmlRenderContext) {
35  			PrintWriter writer = ((WebXmlRenderContext) renderContext).getWriter();
36  			writer = new WhiteSpaceFilterPrintWriter(writer);
37  
38  			WebXmlRenderContext filteredContext = new WebXmlRenderContext(writer, UIContextHolder.
39  					getCurrent().getLocale());
40  			filteredContext.getWriter().turnIndentingOff();
41  			super.paint(filteredContext);
42  		} else {
43  			super.paint(renderContext);
44  		}
45  	}
46  }
