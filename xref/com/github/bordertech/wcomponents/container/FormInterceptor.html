3   import com.github.bordertech.wcomponents.RenderContext;
4   import com.github.bordertech.wcomponents.Request;
5   import com.github.bordertech.wcomponents.UIContext;
6   import com.github.bordertech.wcomponents.UIContextHolder;
7   import com.github.bordertech.wcomponents.util.ConfigurationProperties;
8   
9   /**
10   * This interceptor is used as the top-level component in both Portlet and Servlet environments.
11   *
12   * @author Martin Shevchenko
13   * @since 1.0.0
14   */
15  public class FormInterceptor extends InterceptorComponent {
16  
17  	/**
18  	 * Override serviceRequest in order to perform processing specific to this interceptor.
19  	 *
20  	 * @param request the request being responded to.
21  	 */
22  	@Override
23  	public void serviceRequest(final Request request) {
24  		// Reset the focus for this new request.
25  		UIContext uic = UIContextHolder.getCurrent();
26  		uic.setFocussed(null, null);
27  
28  		// We've hit the action phase, so we do want focus on this app.
29  		uic.setFocusRequired(true);
30  
31  		super.serviceRequest(request);
32  	}
33  
34  	/**
35  	 * Override preparePaint in order to perform processing specific to this interceptor.
36  	 *
37  	 * @param request the request being responded to.
38  	 */
39  	@Override
40  	public void preparePaint(final Request request) {
41  		// Reset the form encoding type before painting the content.
42  		UIContext uic = UIContextHolder.getCurrent();
43  		uic.getEnvironment().setFormEncType(null);
44  
45  		super.preparePaint(request);
46  	}
47  
48  	/**
49  	 * Override paint in order to perform processing specific to this interceptor.
50  	 *
51  	 * @param renderContext the renderContext to send the output to.
52  	 */
53  	@Override
54  	public void paint(final RenderContext renderContext) {
55  		getBackingComponent().paint(renderContext);
56  
57  		// We don't want to remember the focus for the next render because on
58  		// a multi portlet page, we'd end up with multiple portlets trying to
59  		// set the focus.
60  		UIContext uic = UIContextHolder.getCurrent();
61  
62  		if (uic.isFocusRequired()) {
63  			boolean sticky = ConfigurationProperties.getStickyFocus();
64  
65  			if (!sticky) {
66  				uic.setFocussed(null, null);
67  				uic.setFocusRequired(false);
68  			}
69  		}
70  	}
71  }
