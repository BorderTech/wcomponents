11  import java.util.Map;
12  
13  /**
14   * This interceptor setups the AJAX operation details.
15   *
16   * @author Jonathan Austin
17   * @since 1.0.0
18   */
19  public class AjaxSetupInterceptor extends InterceptorComponent {
20  
21  	/**
22  	 * Setup the AJAX operation details.
23  	 *
24  	 * @param request the request being serviced.
25  	 */
26  	@Override
27  	public void serviceRequest(final Request request) {
28  		// Get trigger id
29  		String triggerId = request.getParameter(WServlet.AJAX_TRIGGER_PARAM_NAME);
30  		if (triggerId == null) {
31  			throw new SystemException("No AJAX trigger id to on request");
32  		}
33  
34  		// Get AJAX operations
35  		Map<String, AjaxOperation> operations = (Map<String, AjaxOperation>) request
36  				.getSessionAttribute(AjaxHelper.AJAX_OPERATIONS_SESSION_KEY);
37  		if (operations == null) {
38  			throw new SystemException(
39  					"No AJAX operations have been registered. Trigger " + triggerId + ".");
40  		}
41  
42  		// Get Operation for this trigger
43  		AjaxOperation ajaxOperation = operations.get(triggerId);
44  		if (ajaxOperation == null) {
45  			throw new SystemException(
46  					"No AJAX operation has been registered for trigger " + triggerId + ".");
47  		}
48  
49  		// Find the Component for this trigger
50  		ComponentWithContext trigger = WebUtilities.getComponentById(ajaxOperation.getTriggerId(),
51  				true);
52  		if (trigger == null) {
53  			throw new SystemException("No component found for AJAX trigger " + triggerId + ".");
54  		}
55  
56  		// Set current operation
57  		AjaxHelper.setCurrentOperationDetails(ajaxOperation, trigger);
58  
59  		// Process Service Request
60  		super.serviceRequest(request);
61  	}
62  
63  	/**
64  	 * {@inheritDoc}
65  	 */
66  	@Override
67  	public void paint(final RenderContext renderContext) {
68  		try {
69  			super.paint(renderContext);
70  		} finally {
71  			AjaxHelper.clearCurrentOperationDetails();
72  		}
73  	}
74  }
