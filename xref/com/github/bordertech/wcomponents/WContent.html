6   /**
7    * <p>
8    * This wcomponent enables the display of arbitrary document content from a WComponent application. This component does
9    * not by default display anything until the {@link #display()} method is called. It then causes the browser to access
10   * the content through the WContentHelperServlet.</p>
11   *
12   * <p>
13   * By default, the browser will display the content in a new window, but the {@link #setDisplayMode(DisplayMode)} method
14   * can be used to display the content inline or prompt the user to save the content to a file.</p>
15   *
16   * @author Martin Shevchenko
17   * @since 1.0.0
18   */
19  public class WContent extends AbstractWComponent implements Targetable {

27  		 */
28  		OPEN_NEW_WINDOW,

169 	/**
170 	 * Retrieves a dynamic URL which this targetable component can be accessed from.
171 	 *
172 	 * @return the URL to access this targetable component.
173 	 */
174 	public String getUrl() {
175 		ContentAccess content = getContentAccess();
176 
177 		String mode = DisplayMode.PROMPT_TO_SAVE.equals(getDisplayMode()) ? "attach" : "inline";
178 
179 		// Check for a "static" resource
180 		if (content instanceof InternalResource) {
181 			String url = ((InternalResource) content).getTargetUrl();
182 			// This magic parameter is a work-around to the loading indicator becoming
183 			// "stuck" in certain browsers.
184 			// It is also used by the static resource handler to set the correct headers
185 			url = url + "&" + URL_CONTENT_MODE_PARAMETER_KEY + "=" + mode;
186 			return url;
187 		}
188 
189 		Environment env = getEnvironment();
190 		Map<String, String> parameters = env.getHiddenParameters();
191 		parameters.put(Environment.TARGET_ID, getTargetId());
192 
193 		if (Util.empty(getCacheKey())) {
194 			// Add some randomness to the URL to prevent caching
195 			String random = WebUtilities.generateRandom();
196 			parameters.put(Environment.UNIQUE_RANDOM_PARAM, random);
197 		} else {
198 			// Remove step counter as not required for cached content
199 			parameters.remove(Environment.STEP_VARIABLE);
200 			parameters.remove(Environment.SESSION_TOKEN_VARIABLE);
201 			// Add the cache key
202 			parameters.put(Environment.CONTENT_CACHE_KEY, getCacheKey());
203 		}
204 
205 		// This magic parameter is a work-around to the loading indicator becoming
206 		// "stuck" in certain browsers. It is only read by the theme.
207 		parameters.put(URL_CONTENT_MODE_PARAMETER_KEY, mode);
208 
209 		// The targetable path needs to be configured for the portal environment.
210 		String url = env.getWServletPath();
211 
212 		// Note the last parameter. In javascript we don't want to encode "&".
213 		return WebUtilities.getPath(url, parameters, true);
214 	}

216 	/**
217 	 * <p>
218 	 * Override handleRequest in order to perform processing specific to this component.</p>
219 	 *
220 	 * <p>
221 	 * When the new browser window for the document content is opened, it will make another request to fetch the
222 	 * content. It is that situation we are trying to detect and handle here.</p>
223 	 *
224 	 * @param request the request being responded to.
225 	 */
226 	@Override
227 	public void handleRequest(final Request request) {
228 		resetDisplayRequested();
229 
230 		// Has this component been targeted to return the document content?
231 		// Look in the request for the target parameter and see if it's for us.
232 		// The target parameter is encoded into the url in the javascript that
233 		// was rendered by this component to open the new browser window.
234 		String targ = request.getParameter(Environment.TARGET_ID);
235 		boolean contentReqested = targ != null && targ.equals(getTargetId());
236 
237 		if (contentReqested) {
238 			ContentEscape escape = new ContentEscape(getContentAccess());
239 			escape.setCacheable(!Util.empty(getCacheKey()));
240 			escape.setDisplayInline(getDisplayMode() != DisplayMode.PROMPT_TO_SAVE);
241 			throw escape;
242 		}
243 	}
